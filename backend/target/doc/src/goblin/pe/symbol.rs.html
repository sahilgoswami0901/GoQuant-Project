161/// A COFF symbol.
162///
163/// Unwind information for this function can be loaded with [`ExceptionData::get_unwind_info`].
164///
165/// [`ExceptionData::get_unwind_info`]: struct.ExceptionData.html#method.get_unwind_info
166#[repr(C)]
167#[derive(Debug, Copy, Clone, PartialEq, Default, Pread, Pwrite, IOread, IOwrite, SizeWith)]
168pub struct Symbol {
169    /// The name of the symbol.
170    ///
171    /// An array of 8 bytes is used if the name is not more than 8 bytes long.
172    /// This array is padded with nulls on the right if the name is less than 8 bytes long.
173    ///
174    /// For longer names, the first 4 bytes are all zeros, and the second 4 bytes
175    /// are an offset into the string table.
176    pub name: [u8; 8],
177    /// The value that is associated with the symbol.
178    ///
179    /// The interpretation of this field depends on `section_number` and
180    /// `storage_class`. A typical meaning is the relocatable address.
181    pub value: u32,
182    /// A one-based index into the section table. Zero and negative values have special meanings.
183    pub section_number: i16,
184    /// A number that represents type.
185    ///
186    /// Microsoft tools set this field to 0x20 (function) or 0x0 (not a function).
187    pub typ: u16,
188    /// An enumerated value that represents storage class.
189    pub storage_class: u8,
190    /// The number of auxiliary symbol table entries that follow this record.
191    ///
192    /// Each auxiliary record is the same size as a standard symbol-table record (18 bytes),
193    /// but rather than define a new symbol, the auxiliary record gives additional information
194    /// on the last symbol defined.
195    pub number_of_aux_symbols: u8,
196}
197
198impl Symbol {
199    /// Parse the symbol at the given offset.
200    ///
201    /// If the symbol has an inline name, then also returns a reference to the name's
202    /// location in `bytes`.
203    pub fn parse<'a>(bytes: &'a [u8], offset: usize) -> error::Result<(Option<&'a str>, Symbol)> {
204        let symbol = bytes.pread::<Symbol>(offset)?;
205        let name = if symbol.name[0] != 0 {
206            bytes
207                .pread_with(offset, ctx::StrCtx::DelimiterUntil(0, 8))
208                .ok()
209        } else {
210            None
211        };
212        Ok((name, symbol))
213    }
214
215    /// Returns the symbol name.
216    ///
217    /// This may be a reference to an inline name in the symbol, or to
218    /// a strtab entry.
219    pub fn name<'a>(&'a self, strtab: &'a strtab::Strtab) -> error::Result<&'a str> {
220        if let Some(offset) = self.name_offset() {
221            strtab.get_at(offset as usize).ok_or_else(|| {
222                error::Error::Malformed(format!("Invalid Symbol name offset {:#x}", offset))
223            })
224        } else {
225            Ok(self.name.pread(0)?)
226        }
227    }
228
229    /// Return the strtab offset of the symbol name.
230    ///
231    /// Returns `None` if the name is inline.
232    pub fn name_offset(&self) -> Option<u32> {
233        // Symbol offset starts at the strtable's length, so let's adjust it
234        let length_field_size = core::mem::size_of::<u32>() as u32;
235
236        if self.name[0] == 0 {
237            self.name
238                .pread_with(4, scroll::LE)
239                .ok()
240                .map(|offset: u32| offset - length_field_size)
241        } else {
242            None
243        }
244    }
245
246    /// Set the strtab offset of the symbol name.
247    pub fn set_name_offset(&mut self, offset: u32) {
248        self.name[..4].copy_from_slice(&[0; 4]);
249        self.name.pwrite_with(offset, 4, scroll::LE).unwrap();
250    }
251
252    /// Return the base type of the symbol.
253    ///
254    /// This type uses the `IMAGE_SYM_TYPE_*` definitions.
255    pub fn base_type(&self) -> u16 {
256        self.typ & IMAGE_SYM_TYPE_MASK
257    }
258
259    /// Return the derived type of the symbol.
260    ///
261    /// This type uses the `IMAGE_SYM_DTYPE_*` definitions.
262    pub fn derived_type(&self) -> u16 {
263        self.typ >> IMAGE_SYM_DTYPE_SHIFT
264    }
265
266    /// Return true for function definitions.
267    ///
268    /// These symbols use `AuxFunctionDefinition` for auxiliary symbol records.
269    pub fn is_function_definition(&self) -> bool {
270        self.storage_class == IMAGE_SYM_CLASS_EXTERNAL
271            && self.derived_type() == IMAGE_SYM_DTYPE_FUNCTION
272            && self.section_number > 0
273    }
274
275    /// Return true for weak external symbols.
276    ///
277    /// These symbols use `AuxWeakExternal` for auxiliary symbol records.
278    pub fn is_weak_external(&self) -> bool {
279        self.storage_class == IMAGE_SYM_CLASS_WEAK_EXTERNAL
280    }
281
282    /// Return true for file symbol records.
283    ///
284    /// The auxiliary records contain the name of the source code file.
285    pub fn is_file(&self) -> bool {
286        self.storage_class == IMAGE_SYM_CLASS_FILE
287    }
288
289    /// Return true for section definitions.
290    ///
291    /// These symbols use `AuxSectionDefinition` for auxiliary symbol records.
292    pub fn is_section_definition(&self) -> bool {
293        self.storage_class == IMAGE_SYM_CLASS_STATIC && self.number_of_aux_symbols > 0
294    }
295}

414/// A COFF symbol table.
415pub struct SymbolTable<'a> {
416    symbols: &'a [u8],
417}
418
419impl<'a> SymbolTable<'a> {
420    /// Parse a COFF symbol table at the given offset.
421    ///
422    /// The offset and number of symbols should be from the COFF header.
423    pub fn parse(bytes: &'a [u8], offset: usize, number: usize) -> error::Result<SymbolTable<'a>> {
424        let symbols = bytes.pread_with(offset, Self::size(number))?;
425        Ok(SymbolTable { symbols })
426    }
427
428    /// Get the size in bytes of the symbol table.
429    pub fn size(number: usize) -> usize {
430        number * COFF_SYMBOL_SIZE
431    }
432
433    /// Get the symbol at the given index.
434    ///
435    /// If the symbol has an inline name, then also returns a reference to the name's
436    /// location in `bytes`.
437    pub fn get(&self, index: usize) -> Option<(Option<&'a str>, Symbol)> {
438        let offset = index * COFF_SYMBOL_SIZE;
439        Symbol::parse(self.symbols, offset).ok()
440    }
441
442    /// Get the auxiliary symbol record for a function definition.
443    pub fn aux_function_definition(&self, index: usize) -> Option<AuxFunctionDefinition> {
444        let offset = index * COFF_SYMBOL_SIZE;
445        self.symbols.pread(offset).ok()
446    }
447
448    /// Get the auxiliary symbol record for a `.bf` or `.ef` symbol record.
449    pub fn aux_begin_and_end_function(&self, index: usize) -> Option<AuxBeginAndEndFunction> {
450        let offset = index * COFF_SYMBOL_SIZE;
451        self.symbols.pread(offset).ok()
452    }
453
454    /// Get the auxiliary symbol record for a weak external.
455    pub fn aux_weak_external(&self, index: usize) -> Option<AuxWeakExternal> {
456        let offset = index * COFF_SYMBOL_SIZE;
457        self.symbols.pread(offset).ok()
458    }
459
460    /// Get the file name from the auxiliary symbol record for a file symbol record.
461    pub fn aux_file(&self, index: usize, number: usize) -> Option<&'a str> {
462        let offset = index * COFF_SYMBOL_SIZE;
463        let length = number * COFF_SYMBOL_SIZE;
464        self.symbols
465            .pread_with(offset, ctx::StrCtx::DelimiterUntil(0, length))
466            .ok()
467    }
468
469    /// Get the auxiliary symbol record for a section definition.
470    pub fn aux_section_definition(&self, index: usize) -> Option<AuxSectionDefinition> {
471        let offset = index * COFF_SYMBOL_SIZE;
472        self.symbols.pread(offset).ok()
473    }
474
475    /// Return an iterator for the COFF symbols.
476    ///
477    /// This iterator skips over auxiliary symbol records.
478    pub fn iter(&self) -> SymbolIterator<'a> {
479        SymbolIterator {
480            index: 0,
481            symbols: self.symbols,
482        }
483    }
484}
485
486impl<'a> Debug for SymbolTable<'a> {
487    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
488        fmt.debug_struct("SymbolTable")
489            .field("symbols", &self.iter().collect::<Vec<_>>())
490            .finish()
491    }
492}

494/// An iterator for COFF symbols.
495///
496/// This iterator skips over auxiliary symbol records.
497#[derive(Default)]
498pub struct SymbolIterator<'a> {
499    index: usize,
500    symbols: &'a [u8],
501}
502
503impl<'a> Iterator for SymbolIterator<'a> {
504    type Item = (usize, Option<&'a str>, Symbol);
505    fn next(&mut self) -> Option<Self::Item> {
506        let offset = self.index * COFF_SYMBOL_SIZE;
507        if offset >= self.symbols.len() {
508            None
509        } else {
510            let index = self.index;
511            let (name, symbol) = Symbol::parse(self.symbols, offset).ok()?;
512            self.index += 1 + symbol.number_of_aux_symbols as usize;
513            Some((index, name, symbol))
514        }
515    }
