13pub fn is_in_range(rva: usize, r1: usize, r2: usize) -> bool {
14    r1 <= rva && rva < r2
15}

18#[inline]
19fn aligned_pointer_to_raw_data(pointer_to_raw_data: usize) -> usize {
20    const PHYSICAL_ALIGN: usize = 0x1ff;
21    pointer_to_raw_data & !PHYSICAL_ALIGN
22}

24#[inline]
25fn section_read_size(section: &section_table::SectionTable, file_alignment: u32) -> usize {
26    fn round_size(size: usize) -> usize {
27        const PAGE_MASK: usize = 0xfff;
28        (size + PAGE_MASK) & !PAGE_MASK
29    }
30
31    // Paraphrased from https://reverseengineering.stackexchange.com/a/4326 (by Peter Ferrie).
32    //
33    // Handles the corner cases such as mis-aligned pointers (round down) and sizes (round up)
34    // Further rounding corner cases:
35    // - the physical pointer should be rounded down to a multiple of 512, regardless of the value in the header
36    // - the read size is rounded up by using a combination of the file alignment and 4kb
37    // - the virtual size is always rounded up to a multiple of 4kb, regardless of the value in the header.
38    //
39    // Reference C implementation:
40    //
41    // long pointerToRaw = section.get(POINTER_TO_RAW_DATA);
42    // long alignedpointerToRaw = pointerToRaw & ~0x1ff;
43    // long sizeOfRaw = section.get(SIZE_OF_RAW_DATA);
44    // long readsize = ((pointerToRaw + sizeOfRaw) + filealign - 1) & ~(filealign - 1)) - alignedpointerToRaw;
45    // readsize = min(readsize, (sizeOfRaw + 0xfff) & ~0xfff);
46    // long virtsize = section.get(VIRTUAL_SIZE);
47    //
48    // if (virtsize)
49    // {
50    //     readsize = min(readsize, (virtsize + 0xfff) & ~0xfff);
51    // }
52
53    let file_alignment = file_alignment as usize;
54    let size_of_raw_data = section.size_of_raw_data as usize;
55    let virtual_size = section.virtual_size as usize;
56    let read_size = {
57        let read_size =
58            ((section.pointer_to_raw_data as usize + size_of_raw_data + file_alignment - 1)
59                & !(file_alignment - 1))
60                - aligned_pointer_to_raw_data(section.pointer_to_raw_data as usize);
61        cmp::min(read_size, round_size(size_of_raw_data))
62    };
63
64    if virtual_size == 0 {
65        read_size
66    } else {
67        cmp::min(read_size, round_size(virtual_size))
68    }
69}

71fn rva2offset(rva: usize, section: &section_table::SectionTable) -> usize {
72    (rva - section.virtual_address as usize)
73        + aligned_pointer_to_raw_data(section.pointer_to_raw_data as usize)
74}

76fn is_in_section(rva: usize, section: &section_table::SectionTable, file_alignment: u32) -> bool {
77    let section_rva = section.virtual_address as usize;
78    is_in_range(
79        rva,
80        section_rva,
81        section_rva + section_read_size(section, file_alignment),
82    )
83}

85pub fn find_offset(
86    rva: usize,
87    sections: &[section_table::SectionTable],
88    file_alignment: u32,
89    opts: &options::ParseOptions,
90) -> Option<usize> {
91    if opts.resolve_rva {
92        for (i, section) in sections.iter().enumerate() {
93            debug!(
94                "Checking {} for {:#x} âˆˆ {:#x}..{:#x}",
95                section.name().unwrap_or(""),
96                rva,
97                section.virtual_address,
98                section.virtual_address + section.virtual_size
99            );
100            if is_in_section(rva, &section, file_alignment) {
101                let offset = rva2offset(rva, &section);
102                debug!(
103                    "Found in section {}({}), remapped into offset {:#x}",
104                    section.name().unwrap_or(""),
105                    i,
106                    offset
107                );
108                return Some(offset);
109            }
110        }
111        None
112    } else {
113        Some(rva)
114    }
115}

117pub fn find_offset_or(
118    rva: usize,
119    sections: &[section_table::SectionTable],
120    file_alignment: u32,
121    opts: &options::ParseOptions,
122    msg: &str,
123) -> error::Result<usize> {
124    find_offset(rva, sections, file_alignment, opts)
125        .ok_or_else(|| error::Error::Malformed(msg.to_string()))
126}

128pub fn try_name<'a>(
129    bytes: &'a [u8],
130    rva: usize,
131    sections: &[section_table::SectionTable],
132    file_alignment: u32,
133    opts: &options::ParseOptions,
134) -> error::Result<&'a str> {
135    match find_offset(rva, sections, file_alignment, opts) {
136        Some(offset) => Ok(bytes.pread::<&str>(offset)?),
137        None => Err(error::Error::Malformed(format!(
138            "Cannot find name from rva {:#x} in sections: {:?}",
139            rva, sections
140        ))),
141    }
142}

162pub fn get_data_with_opts<'a, T>(
163    bytes: &'a [u8],
164    sections: &[section_table::SectionTable],
165    directory: DataDirectory,
166    file_alignment: u32,
167    opts: &options::ParseOptions,
168) -> error::Result<T>
169where
170    T: scroll::ctx::TryFromCtx<'a, scroll::Endian, Error = scroll::Error>,
171{
172    let rva = directory.virtual_address as usize;
173    let offset = find_offset(rva, sections, file_alignment, opts)
174        .ok_or_else(|| error::Error::Malformed(directory.virtual_address.to_string()))?;
175    let result: T = bytes.pread_with(offset, scroll::LE)?;
176    Ok(result)
177}
