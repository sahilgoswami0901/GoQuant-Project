63        PT_DYNAMIC => "PT_DYNAMIC",
64        PT_INTERP => "PT_INTERP",
65        PT_NOTE => "PT_NOTE",

71        PT_GNU_EH_FRAME => "PT_GNU_EH_FRAME",
72        PT_GNU_STACK => "PT_GNU_STACK",
73        PT_GNU_RELRO => "PT_GNU_RELRO",

81    }

93    /// A unified ProgramHeader - convertable to and from 32-bit and 64-bit variants
94    pub struct ProgramHeader {
95        pub p_type  : u32,
96        pub p_flags : u32,
97        pub p_offset: u64,
98        pub p_vaddr : u64,
99        pub p_paddr : u64,
100        pub p_filesz: u64,
101        pub p_memsz : u64,
102        pub p_align : u64,
103    }
104
105    impl ProgramHeader {
106        /// Return the size of the underlying program header, given a `Ctx`
107        #[inline]
108        pub fn size(ctx: Ctx) -> usize {
109            use scroll::ctx::SizeWith;
110            Self::size_with(&ctx)
111        }
112        /// Create a new `PT_LOAD` ELF program header
113        pub fn new() -> Self {
114            ProgramHeader {
115                p_type  : PT_LOAD,
116                p_flags : 0,
117                p_offset: 0,
118                p_vaddr : 0,
119                p_paddr : 0,
120                p_filesz: 0,
121                p_memsz : 0,
122                //TODO: check if this is true for 32-bit pt_load
123                p_align : 2 << 20,
124            }
125        }
126        /// Returns this program header's file offset range
127        pub fn file_range(&self) -> Range<usize> {
128            self.p_offset as usize..self.p_offset.saturating_add(self.p_filesz) as usize
129        }
130        /// Returns this program header's virtual memory range
131        pub fn vm_range(&self) -> Range<usize> {
132            self.p_vaddr as usize..self.p_vaddr.saturating_add(self.p_memsz) as usize
133        }
134        /// Sets the executable flag
135        pub fn executable(&mut self) {
136            self.p_flags |= PF_X;
137        }
138        /// Sets the write flag
139        pub fn write(&mut self) {
140            self.p_flags |= PF_W;
141        }
142        /// Sets the read flag
143        pub fn read(&mut self) {
144            self.p_flags |= PF_R;
145        }
146        /// Whether this program header is executable
147        pub fn is_executable(&self) -> bool {
148            self.p_flags & PF_X != 0
149        }
150        /// Whether this program header is readable
151        pub fn is_read(&self) -> bool {
152            self.p_flags & PF_R != 0
153        }
154        /// Whether this program header is writable
155        pub fn is_write(&self) -> bool {
156            self.p_flags & PF_W != 0
157        }
158        #[cfg(feature = "endian_fd")]
159        pub fn parse(bytes: &[u8], mut offset: usize, count: usize, ctx: Ctx) -> crate::error::Result<Vec<ProgramHeader>> {
160            use scroll::Pread;
161            // Sanity check to avoid OOM
162            if count > bytes.len() / Self::size(ctx) {
163                return Err(crate::error::Error::BufferTooShort(count, "program headers"));
164            }
165            let mut program_headers = Vec::with_capacity(count);
166            for _ in 0..count {
167                let phdr = bytes.gread_with(&mut offset, ctx)?;
168                program_headers.push(phdr);
169            }
170            Ok(program_headers)
171        }
172    }

189    impl ctx::SizeWith<Ctx> for ProgramHeader {
190        fn size_with(ctx: &Ctx) -> usize {
191            match ctx.container {
192                Container::Little => {
193                    program_header32::SIZEOF_PHDR
194                },
195                Container::Big => {
196                    program_header64::SIZEOF_PHDR
197                },
198            }
199        }
200    }
201
202    impl<'a> ctx::TryFromCtx<'a, Ctx> for ProgramHeader {
203        type Error = crate::error::Error;
204        fn try_from_ctx(bytes: &'a [u8], Ctx { container, le}: Ctx) -> result::Result<(Self, usize), Self::Error> {
205            use scroll::Pread;
206            let res = match container {
207                Container::Little => {
208                    (bytes.pread_with::<program_header32::ProgramHeader>(0, le)?.into(), program_header32::SIZEOF_PHDR)
209                },
210                Container::Big => {
211                    (bytes.pread_with::<program_header64::ProgramHeader>(0, le)?.into(), program_header64::SIZEOF_PHDR)
212                }
213            };
214            Ok(res)
215        }
216    }
217
218    impl ctx::TryIntoCtx<Ctx> for ProgramHeader {
219        type Error = crate::error::Error;
220        fn try_into_ctx(self, bytes: &mut [u8], Ctx {container, le}: Ctx) -> result::Result<usize, Self::Error> {
221            use scroll::Pwrite;
222            match container {
223                Container::Little => {
224                    let phdr: program_header32::ProgramHeader = self.into();
225                    Ok(bytes.pwrite_with(phdr, 0, le)?)
226                },
227                Container::Big => {
228                    let phdr: program_header64::ProgramHeader = self.into();
229                    Ok(bytes.pwrite_with(phdr, 0, le)?)
230                }
231            }
232        }
233    }

311        impl ProgramHeader {
312            #[cfg(feature = "endian_fd")]
313            pub fn parse(
314                bytes: &[u8],
315                mut offset: usize,
316                count: usize,
317                ctx: ::scroll::Endian,
318            ) -> Result<Vec<ProgramHeader>> {
319                use scroll::Pread;
320                let mut program_headers = vec![ProgramHeader::default(); count];
321                let offset = &mut offset;
322                bytes.gread_inout_with(offset, &mut program_headers, ctx)?;
323                Ok(program_headers)
324            }
325
326            #[cfg(feature = "alloc")]
327            pub fn from_bytes(bytes: &[u8], phnum: usize) -> Vec<ProgramHeader> {
328                let mut phdrs = vec![ProgramHeader::default(); phnum];
329                phdrs
330                    .copy_from_bytes(bytes)
331                    .expect("buffer is too short for given number of entries");
332                phdrs
333            }
334
335            /// # Safety
336            ///
337            /// This function creates a `ProgramHeader` directly from a raw pointer
338            pub unsafe fn from_raw_parts<'a>(
339                phdrp: *const ProgramHeader,
340                phnum: usize,
341            ) -> &'a [ProgramHeader] {
342                slice::from_raw_parts(phdrp, phnum)
343            }
344
345            #[cfg(feature = "std")]
346            pub fn from_fd(fd: &mut File, offset: u64, count: usize) -> Result<Vec<ProgramHeader>> {
347                let mut phdrs = vec![ProgramHeader::default(); count];
348                fd.seek(Start(offset))?;
349                unsafe {
350                    fd.read_exact(plain::as_mut_bytes(&mut *phdrs))?;
351                }
352                Ok(phdrs)
353            }
354        }
