1macro_rules! elf_section_header {
2    ($size:ident) => {
3        // XXX: Do not import scroll traits here.
4        // See: https://github.com/rust-lang/rust/issues/65090#issuecomment-538668155
5
6        #[repr(C)]
7        #[derive(Copy, Clone, Eq, PartialEq, Default)]
8        #[cfg_attr(
9            feature = "alloc",
10            derive(scroll::Pread, scroll::Pwrite, scroll::SizeWith)
11        )]
12        /// Section Headers are typically used by humans and static linkers for additional information or how to relocate the object
13        ///
14        /// **NOTE** section headers are strippable from a binary without any loss of portability/executability; _do not_ rely on them being there!
15        pub struct SectionHeader {
16            /// Section name (string tbl index)
17            pub sh_name: u32,
18            /// Section type
19            pub sh_type: u32,
20            /// Section flags
21            pub sh_flags: $size,
22            /// Section virtual addr at execution
23            pub sh_addr: $size,
24            /// Section file offset
25            pub sh_offset: $size,
26            /// Section size in bytes
27            pub sh_size: $size,
28            /// Link to another section
29            pub sh_link: u32,
30            /// Additional section information
31            pub sh_info: u32,
32            /// Section alignment
33            pub sh_addralign: $size,
34            /// Entry size if section holds table
35            pub sh_entsize: $size,
36        }
37
38        use plain;
39        // Declare that this is a plain type.
40        unsafe impl plain::Plain for SectionHeader {}
41

408    impl SectionHeader {
409        /// Return the size of the underlying program header, given a `container`
410        #[inline]
411        pub fn size(ctx: Ctx) -> usize {
412            use scroll::ctx::SizeWith;
413            Self::size_with(&ctx)
414        }
415        pub fn new() -> Self {
416            SectionHeader {
417                sh_name: 0,
418                sh_type: SHT_PROGBITS,
419                sh_flags: u64::from(SHF_ALLOC),
420                sh_addr: 0,
421                sh_offset: 0,
422                sh_size: 0,
423                sh_link: 0,
424                sh_info: 0,
425                sh_addralign: 2 << 8,
426                sh_entsize: 0,
427            }
428        }
429        /// Returns this section header's file offset range,
430        /// if the section occupies space in fhe file.
431        pub fn file_range(&self) -> Option<Range<usize>> {
432            // Sections with type SHT_NOBITS have no data in the file itself,
433            // they only exist in memory.
434            if self.sh_type == SHT_NOBITS {
435                None
436            } else {
437                Some(self.sh_offset as usize..(self.sh_offset as usize).saturating_add(self.sh_size as usize))
438            }
439        }
440        /// Returns this section header's virtual memory range
441        pub fn vm_range(&self) -> Range<usize> {
442            self.sh_addr as usize..(self.sh_addr as usize).saturating_add(self.sh_size as usize)
443        }
444        /// Parse `count` section headers from `bytes` at `offset`, using the given `ctx`
445        #[cfg(feature = "endian_fd")]
446        pub fn parse(bytes: &[u8], mut offset: usize, mut count: usize, ctx: Ctx) -> error::Result<Vec<SectionHeader>> {
447            use scroll::Pread;
448            // Zero offset means no section headers, not even the null section header.
449            if offset == 0 {
450                return Ok(Vec::new());
451            }
452            let empty_sh = bytes.gread_with::<SectionHeader>(&mut offset, ctx)?;
453            if count == 0 as usize {
454                // Zero count means either no section headers if offset is also zero (checked
455                // above), or the number of section headers overflows SHN_LORESERVE, in which
456                // case the count is stored in the sh_size field of the null section header.
457                count = empty_sh.sh_size as usize;
458            }
459
460            // Sanity check to avoid OOM
461            if count > bytes.len() / Self::size(ctx) {
462                return Err(error::Error::BufferTooShort(count, "section headers"));
463            }
464            let mut section_headers = Vec::with_capacity(count);
465            section_headers.push(empty_sh);
466            for _ in 1..count {
467                let shdr = bytes.gread_with(&mut offset, ctx)?;
468                section_headers.push(shdr);
469            }
470            Ok(section_headers)
471        }
472        pub fn check_size(&self, size: usize) -> error::Result<()> {
473            if self.sh_type == SHT_NOBITS {
474                return Ok(());
475            }
476            let (end, overflow) = self.sh_offset.overflowing_add(self.sh_size);
477            if overflow || end > size as u64 {
478                let message = format!("Section {} size ({}) + offset ({}) is out of bounds. Overflowed: {}",
479                    self.sh_name, self.sh_offset, self.sh_size, overflow);
480                return Err(error::Error::Malformed(message));
481            }
482            let (_, overflow) = self.sh_addr.overflowing_add(self.sh_size);
483            if overflow {
484                let message = format!("Section {} size ({}) + addr ({}) is out of bounds. Overflowed: {}",
485                    self.sh_name, self.sh_addr, self.sh_size, overflow);
486                return Err(error::Error::Malformed(message));
487            }
488            Ok(())
489        }
490        pub fn is_relocation(&self) -> bool {
491            self.sh_type == SHT_RELA
492        }
493        pub fn is_executable(&self) -> bool {
494            self.is_alloc() && self.sh_flags as u32 & SHF_EXECINSTR == SHF_EXECINSTR
495        }
496        pub fn is_writable(&self) -> bool {
497            self.is_alloc() && self.sh_flags as u32 & SHF_WRITE == SHF_WRITE
498        }
499        pub fn is_alloc(&self) -> bool {
500            self.sh_flags as u32 & SHF_ALLOC == SHF_ALLOC
501        }
502    }
