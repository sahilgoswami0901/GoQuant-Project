1//! A client for subscribing to messages from the RPC server.
2//!
3//! The [`PubsubClient`] implements [Solana WebSocket event
4//! subscriptions][spec].
5//!
6//! [spec]: https://solana.com/docs/rpc/websocket
7//!
8//! This is a nonblocking (async) API. For a blocking API use the synchronous
9//! client in [`crate::pubsub_client`].
10//!
11//! A single `PubsubClient` client may be used to subscribe to many events via
12//! subscription methods like [`PubsubClient::account_subscribe`]. These methods
13//! return a [`PubsubClientResult`] of a pair, the first element being a
14//! [`BoxStream`] of subscription-specific [`RpcResponse`]s, the second being an
15//! unsubscribe closure, an asynchronous function that can be called and
16//! `await`ed to unsubscribe.
17//!
18//! Note that `BoxStream` contains an immutable reference to the `PubsubClient`
19//! that created it. This makes `BoxStream` not `Send`, forcing it to stay in
20//! the same task as its `PubsubClient`. `PubsubClient` though is `Send` and
21//! `Sync`, and can be shared between tasks by putting it in an `Arc`. Thus
22//! one viable pattern to creating multiple subscriptions is:
23//!
24//! - create an `Arc<PubsubClient>`
25//! - spawn one task for each subscription, sharing the `PubsubClient`.
26//! - in each task:
27//!   - create a subscription
28//!   - send the `UnsubscribeFn` to another task to handle shutdown
29//!   - loop while receiving messages from the subscription
30//!
31//! This pattern is illustrated in the example below.
32//!
33//! By default the [`block_subscribe`] and [`vote_subscribe`] events are
34//! disabled on RPC nodes. They can be enabled by passing
35//! `--rpc-pubsub-enable-block-subscription` and
36//! `--rpc-pubsub-enable-vote-subscription` to `agave-validator`. When these
37//! methods are disabled, the RPC server will return a "Method not found" error
38//! message.
39//!
40//! [`block_subscribe`]: https://docs.rs/solana-rpc/latest/solana_rpc/rpc_pubsub/trait.RpcSolPubSub.html#tymethod.block_subscribe
41//! [`vote_subscribe`]: https://docs.rs/solana-rpc/latest/solana_rpc/rpc_pubsub/trait.RpcSolPubSub.html#tymethod.vote_subscribe
42//!
43//! # Examples
44//!
45//! Demo two async `PubsubClient` subscriptions with clean shutdown.
46//!
47//! This spawns a task for each subscription type, each of which subscribes and
48//! sends back a ready message and an unsubscribe channel (closure), then loops
49//! on printing messages. The main task then waits for user input before
50//! unsubscribing and waiting on the tasks.
51//!
52//! ```
53//! use anyhow::Result;
54//! use futures_util::StreamExt;
55//! use solana_pubsub_client::nonblocking::pubsub_client::PubsubClient;
56//! use std::sync::Arc;
57//! use tokio::io::AsyncReadExt;
58//! use tokio::sync::mpsc::unbounded_channel;
59//!
60//! pub async fn watch_subscriptions(
61//!     websocket_url: &str,
62//! ) -> Result<()> {
63//!
64//!     // Subscription tasks will send a ready signal when they have subscribed.
65//!     let (ready_sender, mut ready_receiver) = unbounded_channel::<()>();
66//!
67//!     // Channel to receive unsubscribe channels (actually closures).
68//!     // These receive a pair of `(Box<dyn FnOnce() -> BoxFuture<'static, ()> + Send>), &'static str)`,
69//!     // where the first is a closure to call to unsubscribe, the second is the subscription name.
70//!     let (unsubscribe_sender, mut unsubscribe_receiver) = unbounded_channel::<(_, &'static str)>();
71//!
72//!     // The `PubsubClient` must be `Arc`ed to share it across tasks.
73//!     let pubsub_client = Arc::new(PubsubClient::new(websocket_url).await?);
74//!
75//!     let mut join_handles = vec![];
76//!
77//!     join_handles.push(("slot", tokio::spawn({
78//!         // Clone things we need before moving their clones into the `async move` block.
79//!         //
80//!         // The subscriptions have to be made from the tasks that will receive the subscription messages,
81//!         // because the subscription streams hold a reference to the `PubsubClient`.
82//!         // Otherwise we would just subscribe on the main task and send the receivers out to other tasks.
83//!
84//!         let ready_sender = ready_sender.clone();
85//!         let unsubscribe_sender = unsubscribe_sender.clone();
86//!         let pubsub_client = Arc::clone(&pubsub_client);
87//!         async move {
88//!             let (mut slot_notifications, slot_unsubscribe) =
89//!                 pubsub_client.slot_subscribe().await?;
90//!
91//!             // With the subscription started,
92//!             // send a signal back to the main task for synchronization.
93//!             ready_sender.send(()).expect("channel");
94//!
95//!             // Send the unsubscribe closure back to the main task.
96//!             unsubscribe_sender.send((slot_unsubscribe, "slot"))
97//!                 .map_err(|e| format!("{}", e)).expect("channel");
98//!
99//!             // Drop senders so that the channels can close.
100//!             // The main task will receive until channels are closed.
101//!             drop((ready_sender, unsubscribe_sender));
102//!
103//!             // Do something with the subscribed messages.
104//!             // This loop will end once the main task unsubscribes.
105//!             while let Some(slot_info) = slot_notifications.next().await {
106//!                 println!("------------------------------------------------------------");
107//!                 println!("slot pubsub result: {:?}", slot_info);
108//!             }
109//!
110//!             // This type hint is necessary to allow the `async move` block to use `?`.
111//!             Ok::<_, anyhow::Error>(())
112//!         }
113//!     })));
114//!
115//!     join_handles.push(("root", tokio::spawn({
116//!         let ready_sender = ready_sender.clone();
117//!         let unsubscribe_sender = unsubscribe_sender.clone();
118//!         let pubsub_client = Arc::clone(&pubsub_client);
119//!         async move {
120//!             let (mut root_notifications, root_unsubscribe) =
121//!                 pubsub_client.root_subscribe().await?;
122//!
123//!             ready_sender.send(()).expect("channel");
124//!             unsubscribe_sender.send((root_unsubscribe, "root"))
125//!                 .map_err(|e| format!("{}", e)).expect("channel");
126//!             drop((ready_sender, unsubscribe_sender));
127//!
128//!             while let Some(root) = root_notifications.next().await {
129//!                 println!("------------------------------------------------------------");
130//!                 println!("root pubsub result: {:?}", root);
131//!             }
132//!
133//!             Ok::<_, anyhow::Error>(())
134//!         }
135//!     })));
136//!
137//!     // Drop these senders so that the channels can close
138//!     // and their receivers return `None` below.
139//!     drop(ready_sender);
140//!     drop(unsubscribe_sender);
141//!
142//!     // Wait until all subscribers are ready before proceeding with application logic.
143//!     while let Some(_) = ready_receiver.recv().await { }
144//!
145//!     // Do application logic here.
146//!
147//!     // Wait for input or some application-specific shutdown condition.
148//!     tokio::io::stdin().read_u8().await?;
149//!
150//!     // Unsubscribe from everything, which will shutdown all the tasks.
151//!     while let Some((unsubscribe, name)) = unsubscribe_receiver.recv().await {
152//!         println!("unsubscribing from {}", name);
153//!         unsubscribe().await
154//!     }
155//!
156//!     // Wait for the tasks.
157//!     for (name, handle) in join_handles {
158//!         println!("waiting on task {}", name);
159//!         if let Ok(Err(e)) = handle.await {
160//!             println!("task {} failed: {}", name, e);
161//!         }
162//!     }
163//!
164//!     Ok(())
165//! }
166//! # Ok::<(), anyhow::Error>(())
167//! ```

251type UnsubscribeFn = Box<dyn FnOnce() -> BoxFuture<'static, ()> + Send>;
252type SubscribeResponseMsg =
253    Result<(mpsc::UnboundedReceiver<Value>, UnsubscribeFn), PubsubClientError>;
254type SubscribeRequestMsg = (String, Value, oneshot::Sender<SubscribeResponseMsg>);
255type SubscribeResult<'a, T> = PubsubClientResult<(BoxStream<'a, T>, UnsubscribeFn)>;
256type RequestMsg = (
257    String,
258    Value,

262/// A client for subscribing to messages from the RPC server.
263///
264/// See the [module documentation][self].
265#[derive(Debug)]
266pub struct PubsubClient {
267    subscribe_sender: mpsc::UnboundedSender<SubscribeRequestMsg>,
268    request_sender: mpsc::UnboundedSender<RequestMsg>,
269    shutdown_sender: oneshot::Sender<()>,
270    node_version: RwLock<Option<semver::Version>>,
271    ws: JoinHandle<PubsubClientResult>,
272}
273
274impl PubsubClient {
275    pub async fn new(url: &str) -> PubsubClientResult<Self> {
276        let url = Url::parse(url)?;
277        let (ws, _response) = connect_async(url)
278            .await
279            .map_err(PubsubClientError::ConnectionError)?;
280
281        let (subscribe_sender, subscribe_receiver) = mpsc::unbounded_channel();
282        let (request_sender, request_receiver) = mpsc::unbounded_channel();
283        let (shutdown_sender, shutdown_receiver) = oneshot::channel();
284
285        Ok(Self {
286            subscribe_sender,
287            request_sender,
288            shutdown_sender,
289            node_version: RwLock::new(None),
290            ws: tokio::spawn(PubsubClient::run_ws(
291                ws,
292                subscribe_receiver,
293                request_receiver,
294                shutdown_receiver,
295            )),
296        })
297    }
298
299    pub async fn shutdown(self) -> PubsubClientResult {
300        let _ = self.shutdown_sender.send(());
301        self.ws.await.unwrap() // WS future should not be cancelled or panicked
302    }

304    pub async fn set_node_version(&self, version: semver::Version) -> Result<(), ()> {
305        let mut w_node_version = self.node_version.write().await;
306        *w_node_version = Some(version);
307        Ok(())
308    }
309
310    async fn get_node_version(&self) -> PubsubClientResult<semver::Version> {
311        let r_node_version = self.node_version.read().await;
312        if let Some(version) = &*r_node_version {
313            Ok(version.clone())
314        } else {
315            drop(r_node_version);
316            let mut w_node_version = self.node_version.write().await;
317            let node_version = self.get_version().await?;
318            *w_node_version = Some(node_version.clone());
319            Ok(node_version)
320        }
321    }
322
323    async fn get_version(&self) -> PubsubClientResult<semver::Version> {
324        let (response_sender, response_receiver) = oneshot::channel();
325        self.request_sender
326            .send(("getVersion".to_string(), Value::Null, response_sender))
327            .map_err(|err| PubsubClientError::ConnectionClosed(err.to_string()))?;
328        let result = response_receiver
329            .await
330            .map_err(|err| PubsubClientError::ConnectionClosed(err.to_string()))??;
331        let node_version: RpcVersionInfo = serde_json::from_value(result)?;
332        let node_version = semver::Version::parse(&node_version.solana_core).map_err(|e| {
333            PubsubClientError::RequestFailed {
334                reason: format!("failed to parse cluster version: {e}"),
335                message: "getVersion".to_string(),
336            }
337        })?;
338        Ok(node_version)
339    }
340
341    async fn subscribe<'a, T>(&self, operation: &str, params: Value) -> SubscribeResult<'a, T>
342    where
343        T: DeserializeOwned + Send + 'a,
344    {
345        let (response_sender, response_receiver) = oneshot::channel();
346        self.subscribe_sender
347            .send((operation.to_string(), params, response_sender))
348            .map_err(|err| PubsubClientError::ConnectionClosed(err.to_string()))?;
349
350        let (notifications, unsubscribe) = response_receiver
351            .await
352            .map_err(|err| PubsubClientError::ConnectionClosed(err.to_string()))??;
353        Ok((
354            UnboundedReceiverStream::new(notifications)
355                .filter_map(|value| ready(serde_json::from_value::<T>(value).ok()))
356                .boxed(),
357            unsubscribe,
358        ))
359    }

535    async fn run_ws(
536        mut ws: WebSocketStream<MaybeTlsStream<TcpStream>>,
537        mut subscribe_receiver: mpsc::UnboundedReceiver<SubscribeRequestMsg>,
538        mut request_receiver: mpsc::UnboundedReceiver<RequestMsg>,
539        mut shutdown_receiver: oneshot::Receiver<()>,
540    ) -> PubsubClientResult {
541        let mut request_id: u64 = 0;
542
543        let mut requests_subscribe = BTreeMap::new();
544        let mut requests_unsubscribe = BTreeMap::<u64, oneshot::Sender<()>>::new();
545        let mut other_requests = BTreeMap::new();
546        let mut subscriptions = BTreeMap::new();
547        let (unsubscribe_sender, mut unsubscribe_receiver) = mpsc::unbounded_channel();
548
549        loop {
550            tokio::select! {
551                // Send close on shutdown signal
552                _ = (&mut shutdown_receiver) => {
553                    let frame = CloseFrame { code: CloseCode::Normal, reason: "".into() };
554                    ws.send(Message::Close(Some(frame))).await?;
555                    ws.flush().await?;
556                    break;
557                },
558                // Send `Message::Ping` each 10s if no any other communication
559                () = sleep(Duration::from_secs(10)) => {
560                    ws.send(Message::Ping(Vec::new())).await?;
561                },
562                // Read message for subscribe
563                Some((operation, params, response_sender)) = subscribe_receiver.recv() => {
564                    request_id += 1;
565                    let method = format!("{operation}Subscribe");
566                    let text = json!({"jsonrpc":"2.0","id":request_id,"method":method,"params":params}).to_string();
567                    ws.send(Message::Text(text)).await?;
568                    requests_subscribe.insert(request_id, (operation, response_sender));
569                },
570                // Read message for unsubscribe
571                Some((operation, sid, response_sender)) = unsubscribe_receiver.recv() => {
572                    subscriptions.remove(&sid);
573                    request_id += 1;
574                    let method = format!("{operation}Unsubscribe");
575                    let text = json!({"jsonrpc":"2.0","id":request_id,"method":method,"params":[sid]}).to_string();
576                    ws.send(Message::Text(text)).await?;
577                    requests_unsubscribe.insert(request_id, response_sender);
578                },
579                // Read message for other requests
580                Some((method, params, response_sender)) = request_receiver.recv() => {
581                    request_id += 1;
582                    let text = json!({"jsonrpc":"2.0","id":request_id,"method":method,"params":params}).to_string();
583                    ws.send(Message::Text(text)).await?;
584                    other_requests.insert(request_id, response_sender);
585                }
586                // Read incoming WebSocket message
587                next_msg = ws.next() => {
588                    let msg = match next_msg {
589                        Some(msg) => msg?,
590                        None => break,
591                    };
592                    trace!("ws.next(): {:?}", &msg);
593
594                    // Get text from the message
595                    let text = match msg {
596                        Message::Text(text) => text,
597                        Message::Binary(_data) => continue, // Ignore
598                        Message::Ping(data) => {
599                            ws.send(Message::Pong(data)).await?;
600                            continue
601                        },
602                        Message::Pong(_data) => continue,
603                        Message::Close(_frame) => break,
604                        Message::Frame(_frame) => continue,
605                    };
606
607
608                    let mut json: Map<String, Value> = serde_json::from_str(&text)?;
609
610                    // Subscribe/Unsubscribe response, example:
611                    // `{"jsonrpc":"2.0","result":5308752,"id":1}`
612                    if let Some(id) = json.get("id") {
613                        let id = id.as_u64().ok_or_else(|| {
614                            PubsubClientError::SubscribeFailed { reason: "invalid `id` field".into(), message: text.clone() }
615                        })?;
616
617                        let err = json.get("error").map(|error_object| {
618                            match serde_json::from_value::<RpcErrorObject>(error_object.clone()) {
619                                Ok(rpc_error_object) => {
620                                    format!("{} ({})",  rpc_error_object.message, rpc_error_object.code)
621                                }
622                                Err(err) => format!(
623                                    "Failed to deserialize RPC error response: {} [{}]",
624                                    serde_json::to_string(error_object).unwrap(),
625                                    err
626                                )
627                            }
628                        });
629
630                        if let Some(response_sender) = other_requests.remove(&id) {
631                            match err {
632                                Some(reason) => {
633                                    let _ = response_sender.send(Err(PubsubClientError::RequestFailed { reason, message: text.clone()}));
634                                },
635                                None => {
636                                    let json_result = json.get("result").ok_or_else(|| {
637                                        PubsubClientError::RequestFailed { reason: "missing `result` field".into(), message: text.clone() }
638                                    })?;
639                                    if response_sender.send(Ok(json_result.clone())).is_err() {
640                                        break;
641                                    }
642                                }
643                            }
644                        } else if let Some(response_sender) = requests_unsubscribe.remove(&id) {
645                            let _ = response_sender.send(()); // do not care if receiver is closed
646                        } else if let Some((operation, response_sender)) = requests_subscribe.remove(&id) {
647                            match err {
648                                Some(reason) => {
649                                    let _ = response_sender.send(Err(PubsubClientError::SubscribeFailed { reason, message: text.clone()}));
650                                },
651                                None => {
652                                    // Subscribe Id
653                                    let sid = json.get("result").and_then(Value::as_u64).ok_or_else(|| {
654                                        PubsubClientError::SubscribeFailed { reason: "invalid `result` field".into(), message: text.clone() }
655                                    })?;
656
657                                    // Create notifications channel and unsubscribe function
658                                    let (notifications_sender, notifications_receiver) = mpsc::unbounded_channel();
659                                    let unsubscribe_sender = unsubscribe_sender.clone();
660                                    let unsubscribe = Box::new(move || async move {
661                                        let (response_sender, response_receiver) = oneshot::channel();
662                                        // do nothing if ws already closed
663                                        if unsubscribe_sender.send((operation, sid, response_sender)).is_ok() {
664                                            let _ = response_receiver.await; // channel can be closed only if ws is closed
665                                        }
666                                    }.boxed());
667
668                                    if response_sender.send(Ok((notifications_receiver, unsubscribe))).is_err() {
669                                        break;
670                                    }
671                                    subscriptions.insert(sid, notifications_sender);
672                                }
673                            }
674                        } else {
675                            error!("Unknown request id: {}", id);
676                            break;
677                        }
678                        continue;
679                    }
680
681                    // Notification, example:
682                    // `{"jsonrpc":"2.0","method":"logsNotification","params":{"result":{...},"subscription":3114862}}`
683                    if let Some(Value::Object(params)) = json.get_mut("params") {
684                        if let Some(sid) = params.get("subscription").and_then(Value::as_u64) {
685                            let mut unsubscribe_required = false;
686
687                            if let Some(notifications_sender) = subscriptions.get(&sid) {
688                                if let Some(result) = params.remove("result") {
689                                    if notifications_sender.send(result).is_err() {
690                                        unsubscribe_required = true;
691                                    }
692                                }
693                            } else {
694                                unsubscribe_required = true;
695                            }
696
697                            if unsubscribe_required {
698                                if let Some(Value::String(method)) = json.remove("method") {
699                                    if let Some(operation) = method.strip_suffix("Notification") {
700                                        let (response_sender, _response_receiver) = oneshot::channel();
701                                        let _ = unsubscribe_sender.send((operation.to_string(), sid, response_sender));
702                                    }
703                                }
704                            }
705                        }
706                    }
707                }
708            }
709        }
710
711        Ok(())
712    }
713}
