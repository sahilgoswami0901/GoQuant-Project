1//! Helper functions to generate split zero-knowledge proofs for confidential
2//! transfers in the Confidential Transfer Extension.
3//!
4//! The logic in this submodule should belong to the `solana-zk-token-sdk` and
5//! will be removed with the next upgrade to the Solana program.
6
7use crate::{
8    extension::confidential_transfer::{
9        ciphertext_extraction::{transfer_amount_source_ciphertext, SourceDecryptHandles},
10        processor::verify_and_split_deposit_amount,
11        *,
12    },
13    solana_zk_token_sdk::{
14        encryption::{
15            auth_encryption::{AeCiphertext, AeKey},
16            elgamal::{DecryptHandle, ElGamalCiphertext, ElGamalKeypair, ElGamalPubkey},
17            grouped_elgamal::GroupedElGamal,
18            pedersen::Pedersen,
19        },
20        instruction::{
21            transfer::TransferAmountCiphertext, BatchedGroupedCiphertext2HandlesValidityProofData,
22            BatchedRangeProofU128Data, CiphertextCommitmentEqualityProofData,
23        },
24        zk_token_elgamal::ops::subtract_with_lo_hi,
25    },
26};
27
28/// The main logic to create the three split proof data for a transfer.
29pub fn transfer_split_proof_data(
30    current_available_balance: &ElGamalCiphertext,
31    current_decryptable_available_balance: &AeCiphertext,
32    transfer_amount: u64,
33    source_elgamal_keypair: &ElGamalKeypair,
34    aes_key: &AeKey,
35    destination_elgamal_pubkey: &ElGamalPubkey,
36    auditor_elgamal_pubkey: Option<&ElGamalPubkey>,
37) -> Result<
38    (
39        CiphertextCommitmentEqualityProofData,
40        BatchedGroupedCiphertext2HandlesValidityProofData,
41        BatchedRangeProofU128Data,
42        SourceDecryptHandles,
43    ),
44    TokenError,
45> {
46    let default_auditor_pubkey = ElGamalPubkey::default();
47    let auditor_elgamal_pubkey = auditor_elgamal_pubkey.unwrap_or(&default_auditor_pubkey);
48
49    // Split the transfer amount into the low and high bit components.
50    let (transfer_amount_lo, transfer_amount_hi) =
51        verify_and_split_deposit_amount(transfer_amount)?;
52
53    // Encrypt the `lo` and `hi` transfer amounts.
54    let (transfer_amount_grouped_ciphertext_lo, transfer_amount_opening_lo) =
55        TransferAmountCiphertext::new(
56            transfer_amount_lo,
57            source_elgamal_keypair.pubkey(),
58            destination_elgamal_pubkey,
59            auditor_elgamal_pubkey,
60        );
61
62    let (transfer_amount_grouped_ciphertext_hi, transfer_amount_opening_hi) =
63        TransferAmountCiphertext::new(
64            transfer_amount_hi,
65            source_elgamal_keypair.pubkey(),
66            destination_elgamal_pubkey,
67            auditor_elgamal_pubkey,
68        );
69
70    // Decrypt the current available balance at the source
71    let current_decrypted_available_balance = current_decryptable_available_balance
72        .decrypt(aes_key)
73        .ok_or(TokenError::AccountDecryption)?;
74
75    // Compute the remaining balance at the source
76    let new_decrypted_available_balance = current_decrypted_available_balance
77        .checked_sub(transfer_amount)
78        .ok_or(TokenError::InsufficientFunds)?;
79
80    // Create a new Pedersen commitment for the remaining balance at the source
81    let (new_available_balance_commitment, new_source_opening) =
82        Pedersen::new(new_decrypted_available_balance);
83
84    // Compute the remaining balance at the source as ElGamal ciphertexts
85    let transfer_amount_source_ciphertext_lo =
86        transfer_amount_source_ciphertext(&transfer_amount_grouped_ciphertext_lo.into());
87    let transfer_amount_source_ciphertext_hi =
88        transfer_amount_source_ciphertext(&transfer_amount_grouped_ciphertext_hi.into());
89
90    let current_available_balance = (*current_available_balance).into();
91    let new_available_balance_ciphertext = subtract_with_lo_hi(
92        &current_available_balance,
93        &transfer_amount_source_ciphertext_lo,
94        &transfer_amount_source_ciphertext_hi,
95    )
96    .ok_or(TokenError::CiphertextArithmeticFailed)?;
97    let new_available_balance_ciphertext: ElGamalCiphertext = new_available_balance_ciphertext
98        .try_into()
99        .map_err(|_| TokenError::MalformedCiphertext)?;
100
101    // generate equality proof data
102    let equality_proof_data = CiphertextCommitmentEqualityProofData::new(
103        source_elgamal_keypair,
104        &new_available_balance_ciphertext,
105        &new_available_balance_commitment,
106        &new_source_opening,
107        new_decrypted_available_balance,
108    )
109    .map_err(|_| TokenError::ProofGeneration)?;
110
111    // create source decrypt handle
112    let source_decrypt_handle_lo =
113        DecryptHandle::new(source_elgamal_keypair.pubkey(), &transfer_amount_opening_lo);
114    let source_decrypt_handle_hi =
115        DecryptHandle::new(source_elgamal_keypair.pubkey(), &transfer_amount_opening_hi);
116
117    let source_decrypt_handles = SourceDecryptHandles {
118        lo: source_decrypt_handle_lo.into(),
119        hi: source_decrypt_handle_hi.into(),
120    };
121
122    // encrypt the transfer amount under the destination and auditor ElGamal public
123    // key
124    let transfer_amount_destination_auditor_ciphertext_lo = GroupedElGamal::encrypt_with(
125        [destination_elgamal_pubkey, auditor_elgamal_pubkey],
126        transfer_amount_lo,
127        &transfer_amount_opening_lo,
128    );
129    let transfer_amount_destination_auditor_ciphertext_hi = GroupedElGamal::encrypt_with(
130        [destination_elgamal_pubkey, auditor_elgamal_pubkey],
131        transfer_amount_hi,
132        &transfer_amount_opening_hi,
133    );
134
135    // generate ciphertext validity data
136    let ciphertext_validity_proof_data = BatchedGroupedCiphertext2HandlesValidityProofData::new(
137        destination_elgamal_pubkey,
138        auditor_elgamal_pubkey,
139        &transfer_amount_destination_auditor_ciphertext_lo,
140        &transfer_amount_destination_auditor_ciphertext_hi,
141        transfer_amount_lo,
142        transfer_amount_hi,
143        &transfer_amount_opening_lo,
144        &transfer_amount_opening_hi,
145    )
146    .map_err(|_| TokenError::ProofGeneration)?;
147
148    // generate range proof data
149    const REMAINING_BALANCE_BIT_LENGTH: usize = 64;
150    const TRANSFER_AMOUNT_LO_BIT_LENGTH: usize = 16;
151    const TRANSFER_AMOUNT_HI_BIT_LENGTH: usize = 32;
152    const PADDING_BIT_LENGTH: usize = 16;
153
154    let (padding_commitment, padding_opening) = Pedersen::new(0_u64);
155
156    let range_proof_data = BatchedRangeProofU128Data::new(
157        vec![
158            &new_available_balance_commitment,
159            transfer_amount_grouped_ciphertext_lo.get_commitment(),
160            transfer_amount_grouped_ciphertext_hi.get_commitment(),
161            &padding_commitment,
162        ],
163        vec![
164            new_decrypted_available_balance,
165            transfer_amount_lo,
166            transfer_amount_hi,
167            0,
168        ],
169        vec![
170            REMAINING_BALANCE_BIT_LENGTH,
171            TRANSFER_AMOUNT_LO_BIT_LENGTH,
172            TRANSFER_AMOUNT_HI_BIT_LENGTH,
173            PADDING_BIT_LENGTH,
174        ],
175        vec![
176            &new_source_opening,
177            &transfer_amount_opening_lo,
178            &transfer_amount_opening_hi,
179            &padding_opening,
180        ],
181    )
182    .map_err(|_| TokenError::ProofGeneration)?;
183
184    Ok((
185        equality_proof_data,
186        ciphertext_validity_proof_data,
187        range_proof_data,
188        source_decrypt_handles,
189    ))
190}
