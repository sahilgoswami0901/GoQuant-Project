36/// Processes an [InitializeMint] instruction.
37fn process_initialize_mint(
38    accounts: &[AccountInfo],
39    authority: &OptionalNonZeroPubkey,
40    auto_approve_new_account: PodBool,
41    auditor_encryption_pubkey: &OptionalNonZeroElGamalPubkey,
42) -> ProgramResult {
43    let account_info_iter = &mut accounts.iter();
44    let mint_info = next_account_info(account_info_iter)?;
45
46    check_program_account(mint_info.owner)?;
47    let mint_data = &mut mint_info.data.borrow_mut();
48    let mut mint = StateWithExtensionsMut::<Mint>::unpack_uninitialized(mint_data)?;
49    let confidential_transfer_mint = mint.init_extension::<ConfidentialTransferMint>(true)?;
50
51    confidential_transfer_mint.authority = *authority;
52    confidential_transfer_mint.auto_approve_new_accounts = auto_approve_new_account;
53    confidential_transfer_mint.auditor_elgamal_pubkey = *auditor_encryption_pubkey;
54
55    Ok(())
56}

58/// Processes an [UpdateMint] instruction.
59fn process_update_mint(
60    accounts: &[AccountInfo],
61    auto_approve_new_account: PodBool,
62    auditor_encryption_pubkey: &OptionalNonZeroElGamalPubkey,
63) -> ProgramResult {
64    let account_info_iter = &mut accounts.iter();
65    let mint_info = next_account_info(account_info_iter)?;
66    let authority_info = next_account_info(account_info_iter)?;
67
68    check_program_account(mint_info.owner)?;
69    let mint_data = &mut mint_info.data.borrow_mut();
70    let mut mint = StateWithExtensionsMut::<Mint>::unpack(mint_data)?;
71    let confidential_transfer_mint = mint.get_extension_mut::<ConfidentialTransferMint>()?;
72    let maybe_confidential_transfer_mint_authority: Option<Pubkey> =
73        confidential_transfer_mint.authority.into();
74    let confidential_transfer_mint_authority =
75        maybe_confidential_transfer_mint_authority.ok_or(TokenError::NoAuthorityExists)?;
76
77    if !authority_info.is_signer {
78        return Err(ProgramError::MissingRequiredSignature);
79    }
80
81    if confidential_transfer_mint_authority != *authority_info.key {
82        return Err(TokenError::OwnerMismatch.into());
83    }
84
85    confidential_transfer_mint.auto_approve_new_accounts = auto_approve_new_account;
86    confidential_transfer_mint.auditor_elgamal_pubkey = *auditor_encryption_pubkey;
87    Ok(())
88}

90/// Processes a [ConfigureAccount] instruction.
91fn process_configure_account(
92    program_id: &Pubkey,
93    accounts: &[AccountInfo],
94    decryptable_zero_balance: &DecryptableBalance,
95    maximum_pending_balance_credit_counter: &PodU64,
96    proof_instruction_offset: i64,
97) -> ProgramResult {
98    let account_info_iter = &mut accounts.iter();
99    let token_account_info = next_account_info(account_info_iter)?;
100    let mint_info = next_account_info(account_info_iter)?;
101
102    // zero-knowledge proof certifies that the supplied ElGamal public key is valid
103    let proof_context =
104        verify_configure_account_proof(account_info_iter, proof_instruction_offset)?;
105
106    let authority_info = next_account_info(account_info_iter)?;
107    let authority_info_data_len = authority_info.data_len();
108
109    check_program_account(token_account_info.owner)?;
110    let token_account_data = &mut token_account_info.data.borrow_mut();
111    let mut token_account = StateWithExtensionsMut::<Account>::unpack(token_account_data)?;
112
113    if token_account.base.mint != *mint_info.key {
114        return Err(TokenError::MintMismatch.into());
115    }
116
117    Processor::validate_owner(
118        program_id,
119        &token_account.base.owner,
120        authority_info,
121        authority_info_data_len,
122        account_info_iter.as_slice(),
123    )?;
124
125    check_program_account(mint_info.owner)?;
126    let mint_data = &mut mint_info.data.borrow();
127    let mint = StateWithExtensions::<Mint>::unpack(mint_data)?;
128    let confidential_transfer_mint = mint.get_extension::<ConfidentialTransferMint>()?;
129
130    // Note: The caller is expected to use the `Reallocate` instruction to ensure
131    // there is sufficient room in their token account for the new
132    // `ConfidentialTransferAccount` extension
133    let confidential_transfer_account =
134        token_account.init_extension::<ConfidentialTransferAccount>(false)?;
135    confidential_transfer_account.approved = confidential_transfer_mint.auto_approve_new_accounts;
136    confidential_transfer_account.elgamal_pubkey = proof_context.pubkey;
137    confidential_transfer_account.maximum_pending_balance_credit_counter =
138        *maximum_pending_balance_credit_counter;
139
140    // The all-zero ciphertext [0; 64] is a valid encryption of zero
141    confidential_transfer_account.pending_balance_lo = EncryptedBalance::zeroed();
142    confidential_transfer_account.pending_balance_hi = EncryptedBalance::zeroed();
143    confidential_transfer_account.available_balance = EncryptedBalance::zeroed();
144
145    confidential_transfer_account.decryptable_available_balance = *decryptable_zero_balance;
146    confidential_transfer_account.allow_confidential_credits = true.into();
147    confidential_transfer_account.pending_balance_credit_counter = 0.into();
148    confidential_transfer_account.expected_pending_balance_credit_counter = 0.into();
149    confidential_transfer_account.actual_pending_balance_credit_counter = 0.into();
150    confidential_transfer_account.allow_non_confidential_credits = true.into();
151
152    // if the mint is extended for fees, then initialize account for confidential
153    // transfer fees
154    if mint.get_extension::<TransferFeeConfig>().is_ok() {
155        let confidential_transfer_fee_amount =
156            token_account.init_extension::<ConfidentialTransferFeeAmount>(false)?;
157        confidential_transfer_fee_amount.withheld_amount = EncryptedWithheldAmount::zeroed();
158    }
159
160    Ok(())
161}

163/// Processes an [ApproveAccount] instruction.
164fn process_approve_account(accounts: &[AccountInfo]) -> ProgramResult {
165    let account_info_iter = &mut accounts.iter();
166    let token_account_info = next_account_info(account_info_iter)?;
167    let mint_info = next_account_info(account_info_iter)?;
168    let authority_info = next_account_info(account_info_iter)?;
169
170    check_program_account(token_account_info.owner)?;
171    let token_account_data = &mut token_account_info.data.borrow_mut();
172    let mut token_account = StateWithExtensionsMut::<Account>::unpack(token_account_data)?;
173
174    if *mint_info.key != token_account.base.mint {
175        return Err(TokenError::MintMismatch.into());
176    }
177
178    check_program_account(mint_info.owner)?;
179    let mint_data = &mint_info.data.borrow_mut();
180    let mint = StateWithExtensions::<Mint>::unpack(mint_data)?;
181    let confidential_transfer_mint = mint.get_extension::<ConfidentialTransferMint>()?;
182    let maybe_confidential_transfer_mint_authority: Option<Pubkey> =
183        confidential_transfer_mint.authority.into();
184    let confidential_transfer_mint_authority =
185        maybe_confidential_transfer_mint_authority.ok_or(TokenError::NoAuthorityExists)?;
186
187    if authority_info.is_signer && *authority_info.key == confidential_transfer_mint_authority {
188        let confidential_transfer_state =
189            token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
190        confidential_transfer_state.approved = true.into();
191        Ok(())
192    } else {
193        Err(ProgramError::MissingRequiredSignature)
194    }
195}

197/// Processes an [EmptyAccount] instruction.
198fn process_empty_account(
199    program_id: &Pubkey,
200    accounts: &[AccountInfo],
201    proof_instruction_offset: i64,
202) -> ProgramResult {
203    let account_info_iter = &mut accounts.iter();
204    let token_account_info = next_account_info(account_info_iter)?;
205
206    // zero-knowledge proof certifies that the available balance ciphertext holds
207    // the balance of 0.
208    let proof_context = verify_empty_account_proof(account_info_iter, proof_instruction_offset)?;
209
210    let authority_info = next_account_info(account_info_iter)?;
211    let authority_info_data_len = authority_info.data_len();
212
213    check_program_account(token_account_info.owner)?;
214    let token_account_data = &mut token_account_info.data.borrow_mut();
215    let mut token_account = StateWithExtensionsMut::<Account>::unpack(token_account_data)?;
216
217    Processor::validate_owner(
218        program_id,
219        &token_account.base.owner,
220        authority_info,
221        authority_info_data_len,
222        account_info_iter.as_slice(),
223    )?;
224
225    let confidential_transfer_account =
226        token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
227
228    // Check that the encryption public key and ciphertext associated with the
229    // confidential extension account are consistent with those that were
230    // actually used to generate the zkp.
231    if confidential_transfer_account.elgamal_pubkey != proof_context.pubkey {
232        msg!("Encryption public-key mismatch");
233        return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
234    }
235    if confidential_transfer_account.available_balance != proof_context.ciphertext {
236        msg!("Available balance mismatch");
237        return Err(ProgramError::InvalidInstructionData);
238    }
239    confidential_transfer_account.available_balance = EncryptedBalance::zeroed();
240
241    // check that all balances are all-zero ciphertexts
242    confidential_transfer_account.closable()?;
243
244    Ok(())
245}

247/// Processes a [Deposit] instruction.
248#[cfg(feature = "zk-ops")]
249fn process_deposit(
250    program_id: &Pubkey,
251    accounts: &[AccountInfo],
252    amount: u64,
253    expected_decimals: u8,
254) -> ProgramResult {
255    let account_info_iter = &mut accounts.iter();
256    let token_account_info = next_account_info(account_info_iter)?;
257    let mint_info = next_account_info(account_info_iter)?;
258    let authority_info = next_account_info(account_info_iter)?;
259    let authority_info_data_len = authority_info.data_len();
260
261    check_program_account(mint_info.owner)?;
262    let mint_data = &mint_info.data.borrow_mut();
263    let mint = StateWithExtensions::<Mint>::unpack(mint_data)?;
264
265    if expected_decimals != mint.base.decimals {
266        return Err(TokenError::MintDecimalsMismatch.into());
267    }
268
269    if mint.get_extension::<NonTransferable>().is_ok() {
270        return Err(TokenError::NonTransferable.into());
271    }
272
273    check_program_account(token_account_info.owner)?;
274    let token_account_data = &mut token_account_info.data.borrow_mut();
275    let mut token_account = StateWithExtensionsMut::<Account>::unpack(token_account_data)?;
276
277    Processor::validate_owner(
278        program_id,
279        &token_account.base.owner,
280        authority_info,
281        authority_info_data_len,
282        account_info_iter.as_slice(),
283    )?;
284
285    if token_account.base.is_frozen() {
286        return Err(TokenError::AccountFrozen.into());
287    }
288
289    if token_account.base.mint != *mint_info.key {
290        return Err(TokenError::MintMismatch.into());
291    }
292
293    // Wrapped SOL deposits are not supported because lamports cannot be vanished.
294    assert!(!token_account.base.is_native());
295
296    token_account.base.amount = token_account
297        .base
298        .amount
299        .checked_sub(amount)
300        .ok_or(TokenError::Overflow)?;
301    token_account.pack_base();
302
303    let confidential_transfer_account =
304        token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
305    confidential_transfer_account.valid_as_destination()?;
306
307    // A deposit amount must be a 48-bit number
308    let (amount_lo, amount_hi) = verify_and_split_deposit_amount(amount)?;
309
310    // Prevent unnecessary ciphertext arithmetic syscalls if `amount_lo` or
311    // `amount_hi` is zero
312    if amount_lo > 0 {
313        confidential_transfer_account.pending_balance_lo =
314            syscall::add_to(&confidential_transfer_account.pending_balance_lo, amount_lo)
315                .ok_or(TokenError::CiphertextArithmeticFailed)?;
316    }
317    if amount_hi > 0 {
318        confidential_transfer_account.pending_balance_hi =
319            syscall::add_to(&confidential_transfer_account.pending_balance_hi, amount_hi)
320                .ok_or(TokenError::CiphertextArithmeticFailed)?;
321    }
322
323    confidential_transfer_account.increment_pending_balance_credit_counter()?;
324
325    Ok(())
326}

328/// Verifies that a deposit amount is a 48-bit number and returns the least
329/// significant 16 bits and most significant 32 bits of the amount.
330#[cfg(feature = "zk-ops")]
331pub fn verify_and_split_deposit_amount(amount: u64) -> Result<(u64, u64), TokenError> {
332    if amount > MAXIMUM_DEPOSIT_TRANSFER_AMOUNT {
333        return Err(TokenError::MaximumDepositAmountExceeded);
334    }
335    let deposit_amount_lo = amount & (u16::MAX as u64);
336    let deposit_amount_hi = amount.checked_shr(u16::BITS).unwrap();
337    Ok((deposit_amount_lo, deposit_amount_hi))
338}

340/// Processes a [Withdraw] instruction.
341#[cfg(feature = "zk-ops")]
342fn process_withdraw(
343    program_id: &Pubkey,
344    accounts: &[AccountInfo],
345    amount: u64,
346    expected_decimals: u8,
347    new_decryptable_available_balance: DecryptableBalance,
348    proof_instruction_offset: i64,
349) -> ProgramResult {
350    let account_info_iter = &mut accounts.iter();
351    let token_account_info = next_account_info(account_info_iter)?;
352    let mint_info = next_account_info(account_info_iter)?;
353
354    // zero-knowledge proof certifies that the account has enough available balance
355    // to withdraw the amount.
356    let proof_context = verify_withdraw_proof(account_info_iter, proof_instruction_offset)?;
357
358    let authority_info = next_account_info(account_info_iter)?;
359    let authority_info_data_len = authority_info.data_len();
360
361    check_program_account(mint_info.owner)?;
362    let mint_data = &mint_info.data.borrow_mut();
363    let mint = StateWithExtensions::<Mint>::unpack(mint_data)?;
364
365    if expected_decimals != mint.base.decimals {
366        return Err(TokenError::MintDecimalsMismatch.into());
367    }
368
369    if mint.get_extension::<NonTransferable>().is_ok() {
370        return Err(TokenError::NonTransferable.into());
371    }
372
373    check_program_account(token_account_info.owner)?;
374    let token_account_data = &mut token_account_info.data.borrow_mut();
375    let mut token_account = StateWithExtensionsMut::<Account>::unpack(token_account_data)?;
376
377    Processor::validate_owner(
378        program_id,
379        &token_account.base.owner,
380        authority_info,
381        authority_info_data_len,
382        account_info_iter.as_slice(),
383    )?;
384
385    if token_account.base.is_frozen() {
386        return Err(TokenError::AccountFrozen.into());
387    }
388
389    if token_account.base.mint != *mint_info.key {
390        return Err(TokenError::MintMismatch.into());
391    }
392
393    // Wrapped SOL withdrawals are not supported because lamports cannot be
394    // apparated.
395    assert!(!token_account.base.is_native());
396
397    let confidential_transfer_account =
398        token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
399    confidential_transfer_account.valid_as_source()?;
400
401    // Check that the encryption public key associated with the confidential
402    // extension is consistent with the public key that was actually used to
403    // generate the zkp.
404    if confidential_transfer_account.elgamal_pubkey != proof_context.pubkey {
405        return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
406    }
407
408    // Prevent unnecessary ciphertext arithmetic syscalls if the withdraw amount is
409    // zero
410    if amount > 0 {
411        confidential_transfer_account.available_balance =
412            syscall::subtract_from(&confidential_transfer_account.available_balance, amount)
413                .ok_or(TokenError::CiphertextArithmeticFailed)?;
414    }
415    // Check that the final available balance ciphertext is consistent with the
416    // actual ciphertext for which the zero-knowledge proof was generated for.
417    if confidential_transfer_account.available_balance != proof_context.final_ciphertext {
418        return Err(TokenError::ConfidentialTransferBalanceMismatch.into());
419    }
420
421    confidential_transfer_account.decryptable_available_balance = new_decryptable_available_balance;
422    token_account.base.amount = token_account
423        .base
424        .amount
425        .checked_add(amount)
426        .ok_or(TokenError::Overflow)?;
427    token_account.pack_base();
428
429    Ok(())
430}

432/// Processes a [Transfer] or [TransferWithSplitProofs] instruction.
433#[allow(clippy::too_many_arguments)]
434#[cfg(feature = "zk-ops")]
435fn process_transfer(
436    program_id: &Pubkey,
437    accounts: &[AccountInfo],
438    new_source_decryptable_available_balance: DecryptableBalance,
439    proof_instruction_offset: i64,
440    split_proof_context_state_accounts: bool,
441    no_op_on_uninitialized_split_context_state: bool,
442    close_split_context_state_on_execution: bool,
443    source_decrypt_handles: &SourceDecryptHandles,
444) -> ProgramResult {
445    let account_info_iter = &mut accounts.iter();
446    let source_account_info = next_account_info(account_info_iter)?;
447    let mint_info = next_account_info(account_info_iter)?;
448    let destination_token_account_info = next_account_info(account_info_iter)?;
449
450    check_program_account(mint_info.owner)?;
451    let mint_data = &mint_info.data.borrow_mut();
452    let mint = StateWithExtensions::<Mint>::unpack(mint_data)?;
453
454    if mint.get_extension::<NonTransferable>().is_ok() {
455        return Err(TokenError::NonTransferable.into());
456    }
457    let confidential_transfer_mint = mint.get_extension::<ConfidentialTransferMint>()?;
458
459    // A `Transfer` instruction must be accompanied by a zero-knowledge proof
460    // instruction that certify the validity of the transfer amounts. The kind
461    // of zero-knowledge proof instruction depends on whether a transfer incurs
462    // a fee or not.
463    //   - If the mint is not extended for fees or the instruction is for a
464    //     self-transfer, then
465    //   transfer fee is not required.
466    //   - If the mint is extended for fees and the instruction is not a
467    //     self-transfer, then
468    //   transfer fee is required.
469    if mint.get_extension::<TransferFeeConfig>().is_err() {
470        // Transfer fee is not required. Decode the zero-knowledge proof as
471        // `TransferData`.
472        //
473        // The zero-knowledge proof certifies that:
474        //   1. the transfer amount is encrypted in the correct form
475        //   2. the source account has enough balance to send the transfer amount
476        let maybe_proof_context = verify_transfer_proof(
477            account_info_iter,
478            proof_instruction_offset,
479            split_proof_context_state_accounts,
480            no_op_on_uninitialized_split_context_state,
481            close_split_context_state_on_execution,
482            source_decrypt_handles,
483        )?;
484        // If `maybe_proof_context` is `None`, then this means that
485        // `no_op_on_uninitialized_split_context_state` is true and a required context
486        // state account is not yet initialized. Even if this is the case, we
487        // follow through with the rest of the transfer logic to perform all the
488        // necessary checks for a transfer to be safe.
489
490        // If `close_split_context_state_on_execution` is `true`, then the source
491        // account authority info is located after the lamport destination,
492        // context state authority, and zk token proof program account infos.
493        // Flush out these account infos.
494        if close_split_context_state_on_execution && maybe_proof_context.is_none() {
495            let _lamport_destination_account_info = next_account_info(account_info_iter)?;
496            let _context_state_authority_info = next_account_info(account_info_iter)?;
497            let _zk_token_proof_program_info = next_account_info(account_info_iter)?;
498        }
499
500        let authority_info = next_account_info(account_info_iter)?;
501
502        // Check that the auditor encryption public key associated wth the confidential
503        // mint is consistent with what was actually used to generate the zkp.
504        if let Some(ref proof_context) = maybe_proof_context {
505            if !confidential_transfer_mint
506                .auditor_elgamal_pubkey
507                .equals(&proof_context.transfer_pubkeys.auditor)
508            {
509                return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
510            }
511        }
512
513        process_source_for_transfer(
514            program_id,
515            source_account_info,
516            mint_info,
517            authority_info,
518            account_info_iter.as_slice(),
519            maybe_proof_context.as_ref(),
520            new_source_decryptable_available_balance,
521        )?;
522
523        process_destination_for_transfer(
524            destination_token_account_info,
525            mint_info,
526            maybe_proof_context.as_ref(),
527        )?;
528
529        if maybe_proof_context.is_none() {
530            msg!(
531                "Context states not fully initialized: returning with no op; transfer is NOT yet
532            executed"
533            );
534        }
535    } else {
536        // Transfer fee is required.
537        let transfer_fee_config = mint.get_extension::<TransferFeeConfig>()?;
538        let fee_parameters = transfer_fee_config.get_epoch_fee(Clock::get()?.epoch);
539
540        // Decode the zero-knowledge proof as `TransferWithFeeData`.
541        //
542        // The zero-knowledge proof certifies that:
543        //   1. the transfer amount is encrypted in the correct form
544        //   2. the source account has enough balance to send the transfer amount
545        //   3. the transfer fee is computed correctly and encrypted in the correct form
546        let maybe_proof_context = verify_transfer_with_fee_proof(
547            account_info_iter,
548            proof_instruction_offset,
549            split_proof_context_state_accounts,
550            no_op_on_uninitialized_split_context_state,
551            close_split_context_state_on_execution,
552            source_decrypt_handles,
553            fee_parameters,
554        )?;
555
556        // If `maybe_proof_context` is `None`, then this means that
557        // `no_op_on_uninitialized_split_context_state` is true and a required context
558        // state account is not yet initialized. Even if this is the case, we
559        // follow through with the rest of the transfer with fee logic to
560        // perform all the necessary checks to be safe.
561
562        // If `close_split_context_state_on_execution` is `true`, then the source
563        // account authority info is located after the lamport destination,
564        // context state authority, and zk token proof program account infos.
565        // Flush out these account infos.
566        if close_split_context_state_on_execution && maybe_proof_context.is_none() {
567            let _lamport_destination_account_info = next_account_info(account_info_iter)?;
568            let _context_state_authority_info = next_account_info(account_info_iter)?;
569            let _zk_token_proof_program_info = next_account_info(account_info_iter)?;
570        }
571
572        let authority_info = next_account_info(account_info_iter)?;
573
574        // Check that the encryption public keys associated with the mint confidential
575        // transfer and confidential transfer fee extensions are consistent with
576        // the keys that were used to generate the zkp.
577        if let Some(ref proof_context) = maybe_proof_context {
578            if !confidential_transfer_mint
579                .auditor_elgamal_pubkey
580                .equals(&proof_context.transfer_with_fee_pubkeys.auditor)
581            {
582                return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
583            }
584        }
585
586        let confidential_transfer_fee_config =
587            mint.get_extension::<ConfidentialTransferFeeConfig>()?;
588
589        // Check that the withdraw withheld authority ElGamal public key in the mint is
590        // consistent with what was used to generate the zkp.
591        if let Some(ref proof_context) = maybe_proof_context {
592            if proof_context
593                .transfer_with_fee_pubkeys
594                .withdraw_withheld_authority
595                != confidential_transfer_fee_config.withdraw_withheld_authority_elgamal_pubkey
596            {
597                return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
598            }
599        }
600
601        process_source_for_transfer_with_fee(
602            program_id,
603            source_account_info,
604            mint_info,
605            authority_info,
606            account_info_iter.as_slice(),
607            maybe_proof_context.as_ref(),
608            new_source_decryptable_available_balance,
609        )?;
610
611        let is_self_transfer = source_account_info.key == destination_token_account_info.key;
612        process_destination_for_transfer_with_fee(
613            destination_token_account_info,
614            mint_info,
615            maybe_proof_context.as_ref(),
616            is_self_transfer,
617        )?;
618
619        if maybe_proof_context.is_none() {
620            msg!(
621                "Context state not fully initialized: returning with no op; transfer is NOT yet executed"
622            );
623        }
624    }
625
626    Ok(())
627}

629#[cfg(feature = "zk-ops")]
630fn process_source_for_transfer(
631    program_id: &Pubkey,
632    source_account_info: &AccountInfo,
633    mint_info: &AccountInfo,
634    authority_info: &AccountInfo,
635    signers: &[AccountInfo],
636    maybe_proof_context: Option<&TransferProofContextInfo>,
637    new_source_decryptable_available_balance: DecryptableBalance,
638) -> ProgramResult {
639    check_program_account(source_account_info.owner)?;
640    let authority_info_data_len = authority_info.data_len();
641    let token_account_data = &mut source_account_info.data.borrow_mut();
642    let mut token_account = StateWithExtensionsMut::<Account>::unpack(token_account_data)?;
643
644    Processor::validate_owner(
645        program_id,
646        &token_account.base.owner,
647        authority_info,
648        authority_info_data_len,
649        signers,
650    )?;
651
652    if token_account.base.is_frozen() {
653        return Err(TokenError::AccountFrozen.into());
654    }
655
656    if token_account.base.mint != *mint_info.key {
657        return Err(TokenError::MintMismatch.into());
658    }
659
660    let confidential_transfer_account =
661        token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
662    confidential_transfer_account.valid_as_source()?;
663
664    if let Some(proof_context) = maybe_proof_context {
665        // Check that the source encryption public key is consistent with what was
666        // actually used to generate the zkp.
667        if proof_context.transfer_pubkeys.source != confidential_transfer_account.elgamal_pubkey {
668            return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
669        }
670
671        let source_transfer_amount_lo =
672            transfer_amount_source_ciphertext(&proof_context.ciphertext_lo);
673        let source_transfer_amount_hi =
674            transfer_amount_source_ciphertext(&proof_context.ciphertext_hi);
675
676        let new_source_available_balance = syscall::subtract_with_lo_hi(
677            &confidential_transfer_account.available_balance,
678            &source_transfer_amount_lo,
679            &source_transfer_amount_hi,
680        )
681        .ok_or(TokenError::CiphertextArithmeticFailed)?;
682
683        // Check that the computed available balance is consistent with what was
684        // actually used to generate the zkp on the client side.
685        if new_source_available_balance != proof_context.new_source_ciphertext {
686            return Err(TokenError::ConfidentialTransferBalanceMismatch.into());
687        }
688
689        confidential_transfer_account.available_balance = new_source_available_balance;
690        confidential_transfer_account.decryptable_available_balance =
691            new_source_decryptable_available_balance;
692    }
693
694    Ok(())
695}

697#[cfg(feature = "zk-ops")]
698fn process_destination_for_transfer(
699    destination_token_account_info: &AccountInfo,
700    mint_info: &AccountInfo,
701    maybe_transfer_proof_context_info: Option<&TransferProofContextInfo>,
702) -> ProgramResult {
703    check_program_account(destination_token_account_info.owner)?;
704    let destination_token_account_data = &mut destination_token_account_info.data.borrow_mut();
705    let mut destination_token_account =
706        StateWithExtensionsMut::<Account>::unpack(destination_token_account_data)?;
707
708    if destination_token_account.base.is_frozen() {
709        return Err(TokenError::AccountFrozen.into());
710    }
711
712    if destination_token_account.base.mint != *mint_info.key {
713        return Err(TokenError::MintMismatch.into());
714    }
715
716    if memo_required(&destination_token_account) {
717        check_previous_sibling_instruction_is_memo()?;
718    }
719
720    let destination_confidential_transfer_account =
721        destination_token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
722    destination_confidential_transfer_account.valid_as_destination()?;
723
724    if let Some(proof_context) = maybe_transfer_proof_context_info {
725        if proof_context.transfer_pubkeys.destination
726            != destination_confidential_transfer_account.elgamal_pubkey
727        {
728            return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
729        }
730
731        let destination_ciphertext_lo =
732            transfer_amount_destination_ciphertext(&proof_context.ciphertext_lo);
733        let destination_ciphertext_hi =
734            transfer_amount_destination_ciphertext(&proof_context.ciphertext_hi);
735
736        destination_confidential_transfer_account.pending_balance_lo = syscall::add(
737            &destination_confidential_transfer_account.pending_balance_lo,
738            &destination_ciphertext_lo,
739        )
740        .ok_or(TokenError::CiphertextArithmeticFailed)?;
741
742        destination_confidential_transfer_account.pending_balance_hi = syscall::add(
743            &destination_confidential_transfer_account.pending_balance_hi,
744            &destination_ciphertext_hi,
745        )
746        .ok_or(TokenError::CiphertextArithmeticFailed)?;
747
748        destination_confidential_transfer_account.increment_pending_balance_credit_counter()?;
749    }
750
751    Ok(())
752}

825#[cfg(feature = "zk-ops")]
826fn process_destination_for_transfer_with_fee(
827    destination_token_account_info: &AccountInfo,
828    mint_info: &AccountInfo,
829    maybe_proof_context: Option<&TransferWithFeeProofContextInfo>,
830    is_self_transfer: bool,
831) -> ProgramResult {
832    check_program_account(destination_token_account_info.owner)?;
833    let destination_token_account_data = &mut destination_token_account_info.data.borrow_mut();
834    let mut destination_token_account =
835        StateWithExtensionsMut::<Account>::unpack(destination_token_account_data)?;
836
837    if destination_token_account.base.is_frozen() {
838        return Err(TokenError::AccountFrozen.into());
839    }
840
841    if destination_token_account.base.mint != *mint_info.key {
842        return Err(TokenError::MintMismatch.into());
843    }
844
845    if memo_required(&destination_token_account) {
846        check_previous_sibling_instruction_is_memo()?;
847    }
848
849    let destination_confidential_transfer_account =
850        destination_token_account.get_extension_mut::<ConfidentialTransferAccount>()?;
851    destination_confidential_transfer_account.valid_as_destination()?;
852
853    if let Some(proof_context) = maybe_proof_context {
854        if proof_context.transfer_with_fee_pubkeys.destination
855            != destination_confidential_transfer_account.elgamal_pubkey
856        {
857            return Err(TokenError::ConfidentialTransferElGamalPubkeyMismatch.into());
858        }
859
860        let destination_transfer_amount_lo =
861            transfer_amount_destination_ciphertext(&proof_context.ciphertext_lo);
862        let destination_transfer_amount_hi =
863            transfer_amount_destination_ciphertext(&proof_context.ciphertext_hi);
864
865        destination_confidential_transfer_account.pending_balance_lo = syscall::add(
866            &destination_confidential_transfer_account.pending_balance_lo,
867            &destination_transfer_amount_lo,
868        )
869        .ok_or(TokenError::CiphertextArithmeticFailed)?;
870
871        destination_confidential_transfer_account.pending_balance_hi = syscall::add(
872            &destination_confidential_transfer_account.pending_balance_hi,
873            &destination_transfer_amount_hi,
874        )
875        .ok_or(TokenError::CiphertextArithmeticFailed)?;
876
877        destination_confidential_transfer_account.increment_pending_balance_credit_counter()?;
878
879        // process transfer fee
880        if !is_self_transfer {
881            // Decode lo and hi fee amounts encrypted under the destination encryption
882            // public key
883            let destination_fee_lo =
884                fee_amount_destination_ciphertext(&proof_context.fee_ciphertext_lo);
885            let destination_fee_hi =
886                fee_amount_destination_ciphertext(&proof_context.fee_ciphertext_hi);
887
888            // Subtract the fee amount from the destination pending balance
889            destination_confidential_transfer_account.pending_balance_lo = syscall::subtract(
890                &destination_confidential_transfer_account.pending_balance_lo,
891                &destination_fee_lo,
892            )
893            .ok_or(TokenError::CiphertextArithmeticFailed)?;
894            destination_confidential_transfer_account.pending_balance_hi = syscall::subtract(
895                &destination_confidential_transfer_account.pending_balance_hi,
896                &destination_fee_hi,
897            )
898            .ok_or(TokenError::CiphertextArithmeticFailed)?;
899
900            // Decode lo and hi fee amounts encrypted under the withdraw authority
901            // encryption public key
902            let withdraw_withheld_authority_fee_lo =
903                fee_amount_withdraw_withheld_authority_ciphertext(&proof_context.fee_ciphertext_lo);
904            let withdraw_withheld_authority_fee_hi =
905                fee_amount_withdraw_withheld_authority_ciphertext(&proof_context.fee_ciphertext_hi);
906
907            let destination_confidential_transfer_fee_amount =
908                destination_token_account.get_extension_mut::<ConfidentialTransferFeeAmount>()?;
909
910            // Add the fee amount to the destination withheld fee
911            destination_confidential_transfer_fee_amount.withheld_amount = syscall::add_with_lo_hi(
912                &destination_confidential_transfer_fee_amount.withheld_amount,
913                &withdraw_withheld_authority_fee_lo,
914                &withdraw_withheld_authority_fee_hi,
915            )
916            .ok_or(TokenError::CiphertextArithmeticFailed)?;
917        }
918    }
919
920    Ok(())
921}

1036#[allow(dead_code)]
1037pub(crate) fn process_instruction(
1038    program_id: &Pubkey,
1039    accounts: &[AccountInfo],
1040    input: &[u8],
1041) -> ProgramResult {
1042    check_program_account(program_id)?;
1043
1044    match decode_instruction_type(input)? {
1045        ConfidentialTransferInstruction::InitializeMint => {
1046            msg!("ConfidentialTransferInstruction::InitializeMint");
1047            let data = decode_instruction_data::<InitializeMintData>(input)?;
1048            process_initialize_mint(
1049                accounts,
1050                &data.authority,
1051                data.auto_approve_new_accounts,
1052                &data.auditor_elgamal_pubkey,
1053            )
1054        }
1055        ConfidentialTransferInstruction::UpdateMint => {
1056            msg!("ConfidentialTransferInstruction::UpdateMint");
1057            let data = decode_instruction_data::<UpdateMintData>(input)?;
1058            process_update_mint(
1059                accounts,
1060                data.auto_approve_new_accounts,
1061                &data.auditor_elgamal_pubkey,
1062            )
1063        }
1064        ConfidentialTransferInstruction::ConfigureAccount => {
1065            msg!("ConfidentialTransferInstruction::ConfigureAccount");
1066            let data = decode_instruction_data::<ConfigureAccountInstructionData>(input)?;
1067            process_configure_account(
1068                program_id,
1069                accounts,
1070                &data.decryptable_zero_balance,
1071                &data.maximum_pending_balance_credit_counter,
1072                data.proof_instruction_offset as i64,
1073            )
1074        }
1075        ConfidentialTransferInstruction::ApproveAccount => {
1076            msg!("ConfidentialTransferInstruction::ApproveAccount");
1077            process_approve_account(accounts)
1078        }
1079        ConfidentialTransferInstruction::EmptyAccount => {
1080            msg!("ConfidentialTransferInstruction::EmptyAccount");
1081            let data = decode_instruction_data::<EmptyAccountInstructionData>(input)?;
1082            process_empty_account(program_id, accounts, data.proof_instruction_offset as i64)
1083        }
1084        ConfidentialTransferInstruction::Deposit => {
1085            msg!("ConfidentialTransferInstruction::Deposit");
1086            #[cfg(feature = "zk-ops")]
1087            {
1088                let data = decode_instruction_data::<DepositInstructionData>(input)?;
1089                process_deposit(program_id, accounts, data.amount.into(), data.decimals)
1090            }
1091            #[cfg(not(feature = "zk-ops"))]
1092            Err(ProgramError::InvalidInstructionData)
1093        }
1094        ConfidentialTransferInstruction::Withdraw => {
1095            msg!("ConfidentialTransferInstruction::Withdraw");
1096            #[cfg(feature = "zk-ops")]
1097            {
1098                let data = decode_instruction_data::<WithdrawInstructionData>(input)?;
1099                process_withdraw(
1100                    program_id,
1101                    accounts,
1102                    data.amount.into(),
1103                    data.decimals,
1104                    data.new_decryptable_available_balance,
1105                    data.proof_instruction_offset as i64,
1106                )
1107            }
1108            #[cfg(not(feature = "zk-ops"))]
1109            Err(ProgramError::InvalidInstructionData)
1110        }
1111        ConfidentialTransferInstruction::Transfer => {
1112            msg!("ConfidentialTransferInstruction::Transfer");
1113            #[cfg(feature = "zk-ops")]
1114            {
1115                let data = decode_instruction_data::<TransferInstructionData>(input)?;
1116                process_transfer(
1117                    program_id,
1118                    accounts,
1119                    data.new_source_decryptable_available_balance,
1120                    data.proof_instruction_offset as i64,
1121                    false,
1122                    false,
1123                    false,
1124                    &SourceDecryptHandles::zeroed(),
1125                )
1126            }
1127            #[cfg(not(feature = "zk-ops"))]
1128            Err(ProgramError::InvalidInstructionData)
1129        }
1130        ConfidentialTransferInstruction::ApplyPendingBalance => {
1131            msg!("ConfidentialTransferInstruction::ApplyPendingBalance");
1132            #[cfg(feature = "zk-ops")]
1133            {
1134                process_apply_pending_balance(
1135                    program_id,
1136                    accounts,
1137                    decode_instruction_data::<ApplyPendingBalanceData>(input)?,
1138                )
1139            }
1140            #[cfg(not(feature = "zk-ops"))]
1141            {
1142                Err(ProgramError::InvalidInstructionData)
1143            }
1144        }
1145        ConfidentialTransferInstruction::DisableConfidentialCredits => {
1146            msg!("ConfidentialTransferInstruction::DisableConfidentialCredits");
1147            process_allow_confidential_credits(program_id, accounts, false)
1148        }
1149        ConfidentialTransferInstruction::EnableConfidentialCredits => {
1150            msg!("ConfidentialTransferInstruction::EnableConfidentialCredits");
1151            process_allow_confidential_credits(program_id, accounts, true)
1152        }
1153        ConfidentialTransferInstruction::DisableNonConfidentialCredits => {
1154            msg!("ConfidentialTransferInstruction::DisableNonConfidentialCredits");
1155            process_allow_non_confidential_credits(program_id, accounts, false)
1156        }
1157        ConfidentialTransferInstruction::EnableNonConfidentialCredits => {
1158            msg!("ConfidentialTransferInstruction::EnableNonConfidentialCredits");
1159            process_allow_non_confidential_credits(program_id, accounts, true)
1160        }
1161        ConfidentialTransferInstruction::TransferWithSplitProofs => {
1162            msg!("ConfidentialTransferInstruction::TransferWithSplitProofs");
1163            #[cfg(feature = "zk-ops")]
1164            {
1165                let data =
1166                    decode_instruction_data::<TransferWithSplitProofsInstructionData>(input)?;
1167                process_transfer(
1168                    program_id,
1169                    accounts,
1170                    data.new_source_decryptable_available_balance,
1171                    0,
1172                    true,
1173                    data.no_op_on_uninitialized_split_context_state.into(),
1174                    data.close_split_context_state_on_execution.into(),
1175                    &data.source_decrypt_handles,
