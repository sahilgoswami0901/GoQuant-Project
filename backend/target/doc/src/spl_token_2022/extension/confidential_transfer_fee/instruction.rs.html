32/// Confidential Transfer extension instructions
33#[cfg_attr(feature = "serde-traits", derive(Serialize, Deserialize))]
34#[cfg_attr(feature = "serde-traits", serde(rename_all = "camelCase"))]
35#[derive(Clone, Copy, Debug, TryFromPrimitive, IntoPrimitive)]
36#[repr(u8)]
37pub enum ConfidentialTransferFeeInstruction {
38    /// Initializes confidential transfer fees for a mint.
39    ///
40    /// The `ConfidentialTransferFeeInstruction::InitializeConfidentialTransferFeeConfig`
41    /// instruction requires no signers and MUST be included within the same
42    /// Transaction as `TokenInstruction::InitializeMint`. Otherwise another
43    /// party can initialize the configuration.
44    ///
45    /// The instruction fails if the `TokenInstruction::InitializeMint`
46    /// instruction has already executed for the mint.
47    ///
48    /// Accounts expected by this instruction:
49    ///
50    ///   0. `[writable]` The SPL Token mint.
51    ///
52    /// Data expected by this instruction:
53    ///   `InitializeConfidentialTransferFeeConfigData`
54    InitializeConfidentialTransferFeeConfig,
55
56    /// Transfer all withheld confidential tokens in the mint to an account.
57    /// Signed by the mint's withdraw withheld tokens authority.
58    ///
59    /// The withheld confidential tokens are aggregated directly into the
60    /// destination available balance.
61    ///
62    /// In order for this instruction to be successfully processed, it must be
63    /// accompanied by the `VerifyCiphertextCiphertextEquality` instruction
64    /// of the `zk_token_proof` program in the same transaction or the
65    /// address of a context state account for the proof must be provided.
66    ///
67    /// Accounts expected by this instruction:
68    ///
69    ///   * Single owner/delegate
70    ///   0. `[writable]` The token mint. Must include the `TransferFeeConfig`
71    ///      extension.
72    ///   1. `[writable]` The fee receiver account. Must include the
73    ///      `TransferFeeAmount` and `ConfidentialTransferAccount` extensions.
74    ///   2. `[]` Instructions sysvar if `VerifyCiphertextCiphertextEquality` is
75    ///      included in the same transaction or context state account if
76    ///      `VerifyCiphertextCiphertextEquality` is pre-verified into a context
77    ///      state account.
78    ///   3. `[signer]` The mint's `withdraw_withheld_authority`.
79    ///
80    ///   * Multisignature owner/delegate
81    ///   0. `[writable]` The token mint. Must include the `TransferFeeConfig`
82    ///      extension.
83    ///   1. `[writable]` The fee receiver account. Must include the
84    ///      `TransferFeeAmount` and `ConfidentialTransferAccount` extensions.
85    ///   2. `[]` Instructions sysvar if `VerifyCiphertextCiphertextEquality` is
86    ///      included in the same transaction or context state account if
87    ///      `VerifyCiphertextCiphertextEquality` is pre-verified into a context
88    ///      state account.
89    ///   3. `[]` The mint's multisig `withdraw_withheld_authority`.
90    ///   4. ..3+M `[signer]` M signer accounts.
91    ///
92    /// Data expected by this instruction:
93    ///   WithdrawWithheldTokensFromMintData
94    WithdrawWithheldTokensFromMint,
95
96    /// Transfer all withheld tokens to an account. Signed by the mint's
97    /// withdraw withheld tokens authority. This instruction is susceptible
98    /// to front-running. Use `HarvestWithheldTokensToMint` and
99    /// `WithdrawWithheldTokensFromMint` as an alternative.
100    ///
101    /// The withheld confidential tokens are aggregated directly into the
102    /// destination available balance.
103    ///
104    /// Note on front-running: This instruction requires a zero-knowledge proof
105    /// verification instruction that is checked with respect to the account
106    /// state (the currently withheld fees). Suppose that a withdraw
107    /// withheld authority generates the
108    /// `WithdrawWithheldTokensFromAccounts` instruction along with a
109    /// corresponding zero-knowledge proof for a specified set of accounts,
110    /// and submits it on chain. If the withheld fees at any
111    /// of the specified accounts change before the
112    /// `WithdrawWithheldTokensFromAccounts` is executed on chain, the
113    /// zero-knowledge proof will not verify with respect to the new state,
114    /// forcing the transaction to fail.
115    ///
116    /// If front-running occurs, then users can look up the updated states of
117    /// the accounts, generate a new zero-knowledge proof and try again.
118    /// Alternatively, withdraw withheld authority can first move the
119    /// withheld amount to the mint using `HarvestWithheldTokensToMint` and
120    /// then move the withheld fees from mint to a specified destination
121    /// account using `WithdrawWithheldTokensFromMint`.
122    ///
123    /// In order for this instruction to be successfully processed, it must be
124    /// accompanied by the `VerifyWithdrawWithheldTokens` instruction of the
125    /// `zk_token_proof` program in the same transaction or the address of a
126    /// context state account for the proof must be provided.
127    ///
128    /// Accounts expected by this instruction:
129    ///
130    ///   * Single owner/delegate
131    ///   0. `[]` The token mint. Must include the `TransferFeeConfig`
132    ///      extension.
133    ///   1. `[writable]` The fee receiver account. Must include the
134    ///      `TransferFeeAmount` and `ConfidentialTransferAccount` extensions.
135    ///   2. `[]` Instructions sysvar if `VerifyCiphertextCiphertextEquality` is
136    ///      included in the same transaction or context state account if
137    ///      `VerifyCiphertextCiphertextEquality` is pre-verified into a context
138    ///      state account.
139    ///   3. `[signer]` The mint's `withdraw_withheld_authority`.
140    ///   4. ..3+N `[writable]` The source accounts to withdraw from.
141    ///
142    ///   * Multisignature owner/delegate
143    ///   0. `[]` The token mint. Must include the `TransferFeeConfig`
144    ///      extension.
145    ///   1. `[writable]` The fee receiver account. Must include the
146    ///      `TransferFeeAmount` and `ConfidentialTransferAccount` extensions.
147    ///   2. `[]` Instructions sysvar if `VerifyCiphertextCiphertextEquality` is
148    ///      included in the same transaction or context state account if
149    ///      `VerifyCiphertextCiphertextEquality` is pre-verified into a context
150    ///      state account.
151    ///   3. `[]` The mint's multisig `withdraw_withheld_authority`.
152    ///   4. ..4+M `[signer]` M signer accounts.
153    ///   4+M+1. ..4+M+N `[writable]` The source accounts to withdraw from.
154    ///
155    /// Data expected by this instruction:
156    ///   WithdrawWithheldTokensFromAccountsData
157    WithdrawWithheldTokensFromAccounts,
158
159    /// Permissionless instruction to transfer all withheld confidential tokens
160    /// to the mint.
161    ///
162    /// Succeeds for frozen accounts.
163    ///
164    /// Accounts provided should include both the `TransferFeeAmount` and
165    /// `ConfidentialTransferAccount` extension. If not, the account is skipped.
166    ///
167    /// Accounts expected by this instruction:
168    ///
169    ///   0. `[writable]` The mint.
170    ///   1. ..1+N `[writable]` The source accounts to harvest from.
171    ///
172    /// Data expected by this instruction:
173    ///   None
174    HarvestWithheldTokensToMint,
175
176    /// Configure a confidential transfer fee mint to accept harvested
177    /// confidential fees.
178    ///
179    /// Accounts expected by this instruction:
180    ///
181    ///   * Single owner/delegate
182    ///   0. `[writable]` The token mint.
183    ///   1. `[signer]` The confidential transfer fee authority.
184    ///
185    ///   *Multisignature owner/delegate
186    ///   0. `[writable]` The token mint.
187    ///   1. `[]` The confidential transfer fee multisig authority,
188    ///   2. `[signer]` Required M signer accounts for the SPL Token Multisig
189    ///      account.
190    ///
191    /// Data expected by this instruction:
192    ///   None
193    EnableHarvestToMint,
194
195    /// Configure a confidential transfer fee mint to reject any harvested
196    /// confidential fees.
197    ///
198    /// Accounts expected by this instruction:
199    ///
200    ///   * Single owner/delegate
201    ///   0. `[writable]` The token mint.
202    ///   1. `[signer]` The confidential transfer fee authority.
203    ///
204    ///   *Multisignature owner/delegate
205    ///   0. `[writable]` The token mint.
206    ///   1. `[]` The confidential transfer fee multisig authority,
207    ///   2. `[signer]` Required M signer accounts for the SPL Token Multisig
208    ///      account.
209    ///
210    /// Data expected by this instruction:
211    ///   None
212    DisableHarvestToMint,
213}

287/// Create an inner `WithdrawWithheldTokensFromMint` instruction
288///
289/// This instruction is suitable for use with a cross-program `invoke`
290pub fn inner_withdraw_withheld_tokens_from_mint(
291    token_program_id: &Pubkey,
292    mint: &Pubkey,
293    destination: &Pubkey,
294    new_decryptable_available_balance: &DecryptableBalance,
295    authority: &Pubkey,
296    multisig_signers: &[&Pubkey],
297    proof_data_location: ProofLocation<CiphertextCiphertextEqualityProofData>,
298) -> Result<Instruction, ProgramError> {
299    check_program_account(token_program_id)?;
300    let mut accounts = vec![
301        AccountMeta::new(*mint, false),
302        AccountMeta::new(*destination, false),
303    ];
304
305    let proof_instruction_offset = match proof_data_location {
306        ProofLocation::InstructionOffset(proof_instruction_offset, _) => {
307            accounts.push(AccountMeta::new_readonly(sysvar::instructions::id(), false));
308            proof_instruction_offset.into()
309        }
310        ProofLocation::ContextStateAccount(context_state_account) => {
311            accounts.push(AccountMeta::new_readonly(*context_state_account, false));
312            0
313        }
314    };
315
316    accounts.push(AccountMeta::new_readonly(
317        *authority,
318        multisig_signers.is_empty(),
319    ));
320
321    for multisig_signer in multisig_signers.iter() {
322        accounts.push(AccountMeta::new_readonly(**multisig_signer, true));
323    }
324
325    Ok(encode_instruction(
326        token_program_id,
327        accounts,
328        TokenInstruction::ConfidentialTransferFeeExtension,
329        ConfidentialTransferFeeInstruction::WithdrawWithheldTokensFromMint,
330        &WithdrawWithheldTokensFromMintData {
331            proof_instruction_offset,
332            new_decryptable_available_balance: *new_decryptable_available_balance,
333        },
334    ))
335}

337/// Create an `WithdrawWithheldTokensFromMint` instruction
338pub fn withdraw_withheld_tokens_from_mint(
339    token_program_id: &Pubkey,
340    mint: &Pubkey,
341    destination: &Pubkey,
342    new_decryptable_available_balance: &DecryptableBalance,
343    authority: &Pubkey,
344    multisig_signers: &[&Pubkey],
345    proof_data_location: ProofLocation<CiphertextCiphertextEqualityProofData>,
346) -> Result<Vec<Instruction>, ProgramError> {
347    let mut instructions = vec![inner_withdraw_withheld_tokens_from_mint(
348        token_program_id,
349        mint,
350        destination,
351        new_decryptable_available_balance,
352        authority,
353        multisig_signers,
354        proof_data_location,
355    )?];
356
357    if let ProofLocation::InstructionOffset(proof_instruction_offset, proof_data) =
358        proof_data_location
359    {
360        // This constructor appends the proof instruction right after the
361        // `WithdrawWithheldTokensFromMint` instruction. This means that the proof
362        // instruction offset must be always be 1. To use an arbitrary proof
363        // instruction offset, use the
364        // `inner_withdraw_withheld_tokens_from_mint` constructor.
365        let proof_instruction_offset: i8 = proof_instruction_offset.into();
366        if proof_instruction_offset != 1 {
367            return Err(TokenError::InvalidProofInstructionOffset.into());
368        }
369        instructions.push(verify_ciphertext_ciphertext_equality(None, proof_data));
370    };
371
372    Ok(instructions)
373}

399            accounts.push(AccountMeta::new_readonly(sysvar::instructions::id(), false));
400            proof_instruction_offset.into()
401        }
402        ProofLocation::ContextStateAccount(context_state_account) => {
403            accounts.push(AccountMeta::new_readonly(*context_state_account, false));
404            0
405        }
406    };
407
408    accounts.push(AccountMeta::new_readonly(
409        *authority,
410        multisig_signers.is_empty(),
411    ));
412
413    for multisig_signer in multisig_signers.iter() {
414        accounts.push(AccountMeta::new_readonly(**multisig_signer, true));
415    }
416
417    for source in sources.iter() {
418        accounts.push(AccountMeta::new(**source, false));
419    }
420
421    Ok(encode_instruction(
422        token_program_id,
423        accounts,
424        TokenInstruction::ConfidentialTransferFeeExtension,
425        ConfidentialTransferFeeInstruction::WithdrawWithheldTokensFromAccounts,

428            num_token_accounts,
429            new_decryptable_available_balance: *new_decryptable_available_balance,
430        },
