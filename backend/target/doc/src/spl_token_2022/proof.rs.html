13/// Decodes the proof context data associated with a zero-knowledge proof
14/// instruction.
15pub fn decode_proof_instruction_context<T: Pod + ZkProofData<U>, U: Pod>(
16    expected: ProofInstruction,
17    instruction: &Instruction,
18) -> Result<&U, ProgramError> {
19    if instruction.program_id != zk_token_proof_program::id()
20        || ProofInstruction::instruction_type(&instruction.data) != Some(expected)
21    {
22        msg!("Unexpected proof instruction");
23        return Err(ProgramError::InvalidInstructionData);
24    }
25
26    ProofInstruction::proof_data::<T, U>(&instruction.data)
27        .map(ZkProofData::context_data)
28        .ok_or(ProgramError::InvalidInstructionData)
29}

31/// A proof location type meant to be used for arguments to instruction
32/// constructors.
33#[derive(Clone, Copy)]
34pub enum ProofLocation<'a, T> {
35    /// The proof is included in the same transaction of a corresponding
36    /// token-2022 instruction.
37    InstructionOffset(NonZeroI8, &'a T),
38    /// The proof is pre-verified into a context state account.
39    ContextStateAccount(&'a Pubkey),
40}

42/// Instruction options for when using split context state accounts
43#[derive(Clone, Copy)]
44pub struct SplitContextStateAccountsConfig {
45    /// If true, execute no op when an associated split proof context state
46    /// account is not initialized. Otherwise, fail on an uninitialized
47    /// context state account.
48    pub no_op_on_uninitialized_split_context_state: bool,
49    /// Close associated context states after a complete execution of the
50    /// transfer instruction.
51    pub close_split_context_state_on_execution: bool,
52}
