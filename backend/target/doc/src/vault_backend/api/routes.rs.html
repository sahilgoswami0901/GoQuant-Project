1//! # API Route Configuration
2//!
3//! This module sets up all the HTTP routes for the API.
4
5use actix_web::web;
6
7use super::handlers;
8
9/// Configure all API routes.
10///
11/// This function is called from main.rs to set up
12/// all the endpoint routes.
13///
14/// ## Route Structure
15///
16/// ```text
17/// /
18/// ├── /health              GET - Health check
19/// └── /vault
20///     ├── /initialize      POST - Create vault
21///     ├── /deposit         POST - Deposit USDT
22///     ├── /withdraw        POST - Withdraw USDT
23///     ├── /balance/:user   GET - Get balance
24///     ├── /transactions/:user  GET - Transaction history
25///     └── /tvl             GET - Total Value Locked
26/// ```
27pub fn configure_routes(cfg: &mut web::ServiceConfig) {
28    cfg
29        // Root endpoint - API information
30        .route("/", web::get().to(handlers::api_info))
31        
32        // Health check endpoint
33        .route("/health", web::get().to(handlers::health_check))
34        
35        // Vault endpoints
36        .service(
37            web::scope("/vault")
38                // Initialize a new vault
39                .route("/initialize", web::post().to(handlers::initialize_vault))
40                
41                // Deposit USDT into vault
42                .route("/deposit", web::post().to(handlers::deposit))
43                
44                // Withdraw USDT from vault
45                .route("/withdraw", web::post().to(handlers::withdraw))
46                
47                // Lock collateral (internal - position manager)
48                .route("/lock-collateral", web::post().to(handlers::lock_collateral))
49                
50                // Unlock collateral (internal - position manager)
51                .route("/unlock-collateral", web::post().to(handlers::unlock_collateral))
52                
53                // Transfer collateral (internal - settlement/liquidation)
54                .route("/transfer-collateral", web::post().to(handlers::transfer_collateral))
55                
56                // Get vault balance
57                .route("/balance/{user}", web::get().to(handlers::get_balance))
58                
59                // Get transaction history
60                .route(
61                    "/transactions/{user}",
62                    web::get().to(handlers::get_transactions),
63                )
64                
65                // Get Total Value Locked
66                .route("/tvl", web::get().to(handlers::get_tvl))
67                
68                // Mint test USDT (devnet only)
69                .route("/mint-usdt", web::post().to(handlers::mint_usdt))
70        );
71}
