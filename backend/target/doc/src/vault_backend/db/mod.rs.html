1//! # Database Module
2//!
3//! This module handles all database operations for the Collateral Vault backend.
4//! We use PostgreSQL for storing:
5//!
6//! - Vault account records (cached from blockchain)
7//! - Transaction history (deposits, withdrawals, locks, etc.)
8//! - Balance snapshots (for analytics)
9//! - Reconciliation logs (for auditing)
10//!
11//! ## Why Store Off-Chain?
12//!
13//! Even though all vault data is on the Solana blockchain, we store it
14//! locally for several reasons:
15//!
16//! 1. **Speed**: Database queries are ~10ms, blockchain queries are ~200ms
17//! 2. **History**: Blockchain doesn't store full transaction history easily
18//! 3. **Analytics**: Complex queries are impossible on-chain
19//! 4. **Reliability**: Works even if Solana RPC is slow/unavailable
20//!
21//! ## Architecture
22//!
23//! ```text
24//! ┌─────────────────────────────────────────────────────────────────┐
25//! │                      DATABASE LAYER                              │
26//! │                                                                  │
27//! │  ┌──────────────────────────────────────────────────────────┐   │
28//! │  │                   Connection Pool                         │   │
29//! │  │                  (deadpool-postgres)                    │   │
30//! │  └──────────────────────────────────────────────────────────┘   │
31//! │                              │                                   │
32//! │         ┌────────────────────┼────────────────────┐             │
33//! │         ▼                    ▼                    ▼             │
34//! │  ┌────────────┐      ┌────────────┐       ┌────────────┐       │
35//! │  │  Vaults    │      │Transactions│       │ Snapshots  │       │
36//! │  │  Table     │      │   Table    │       │   Table    │       │
37//! │  └────────────┘      └────────────┘       └────────────┘       │
38//! └─────────────────────────────────────────────────────────────────┘
39//! ```

49/// Database-related errors.
50#[derive(Error, Debug)]
51pub enum DatabaseError {
52    /// Failed to connect to the database
53    #[error("Database connection failed: {0}")]
54    ConnectionError(String),
55
56    /// Query execution failed
57    #[error("Query failed: {0}")]
58    QueryError(#[from] tokio_postgres::Error),
59
60    /// Migration failed
61    #[error("Migration failed: {0}")]
62    MigrationError(String),
63
64    /// Record not found
65    #[error("Record not found: {0}")]
66    NotFound(String),
67
68    /// Invalid configuration
69    #[error("Invalid configuration: {0}")]
70    ConfigError(String),
71}

73/// Database connection wrapper.
74///
75/// This struct wraps the connection pool and provides
76/// methods for common database operations.
77///
78/// ## Connection Pooling
79///
80/// We use deadpool-postgres for connection pooling.
81/// This efficiently manages database connections.
82///
83/// ## Usage
84///
85/// ```rust,ignore
86/// let db = Database::connect("postgres://...").await?;
87/// let vault = queries::get_vault_by_owner(db.pool(), "pubkey").await?;
88/// ```
89#[derive(Clone)]
90pub struct Database {
91    /// The connection pool
92    pool: Pool,
93}
94
95impl Database {
96    /// Connect to the PostgreSQL database.
97    ///
98    /// Creates a connection pool with sensible defaults:
99    /// - Max 10 connections
100    /// - 30 second connection timeout
101    ///
102    /// ## Arguments
103    ///
104    /// * `database_url` - PostgreSQL connection string
105    ///
106    /// ## Returns
107    ///
108    /// * `Ok(Database)` - Connected successfully
109    /// * `Err(DatabaseError)` - Connection failed
110    ///
111    /// ## Example
112    ///
113    /// ```rust,ignore
114    /// let db = Database::connect("postgres://postgres:password@localhost/vault").await?;
115    /// ```
116    pub async fn connect(database_url: &str) -> Result<Self, DatabaseError> {
117        info!("Connecting to database...");
118
119        // Parse the connection string using tokio_postgres::Config
120        let tokio_config = database_url.parse::<TokioConfig>()
121            .map_err(|e| DatabaseError::ConfigError(format!("Invalid database URL: {}", e)))?;
122
123        // Convert to deadpool config
124        let mut config = Config::new();
125        
126        if let Some(dbname) = tokio_config.get_dbname() {
127            config.dbname = Some(dbname.to_string());
128        }
129        if let Some(user) = tokio_config.get_user() {
130            config.user = Some(user.to_string());
131        }
132        if let Some(password) = tokio_config.get_password() {
133            // Password is &[u8], convert to String
134            config.password = Some(String::from_utf8_lossy(password).to_string());
135        }
136        if let Some(host) = tokio_config.get_hosts().first() {
137            if let tokio_postgres::config::Host::Tcp(host_str) = host {
138                config.host = Some(host_str.clone());
139            }
140        }
141        if let Some(port) = tokio_config.get_ports().first() {
142            config.port = Some(*port);
143        }
144        
145        // Set pool size
146        config.pool = Some(deadpool_postgres::PoolConfig {
147            max_size: 10,
148            ..Default::default()
149        });
150
151        // Create pool
152        let pool = config
153            .create_pool(Some(Runtime::Tokio1), NoTls)
154            .map_err(|e| DatabaseError::ConnectionError(e.to_string()))?;
155
156        // Test connection
157        let client = pool.get().await
158            .map_err(|e| DatabaseError::ConnectionError(e.to_string()))?;
159
160        // Simple query to verify connection
161        client.query("SELECT 1", &[]).await
162            .map_err(|e| DatabaseError::ConnectionError(e.to_string()))?;
163
164        info!("Database connection established");
165
166        Ok(Self { pool })
167    }

169    /// Run database migrations.
170    ///
171    /// Migrations are SQL scripts that set up and update the database schema.
172    /// They're run in order and tracked so each only runs once.
173    ///
174    /// ## Migration Files
175    ///
176    /// Located in `migrations/` directory:
177    /// ```text
178    /// migrations/
179    /// ├── 001_initial_schema.sql
180    /// └── ...
181    /// ```
182    ///
183    /// ## Example
184    ///
185    /// ```rust,ignore
186    /// db.run_migrations().await?;
187    /// ```
188    pub async fn run_migrations(&self) -> Result<(), DatabaseError> {
189        info!("Running database migrations...");
190
191        let client = self.pool.get().await
192            .map_err(|e| DatabaseError::ConnectionError(e.to_string()))?;
193
194        // Get current working directory for debugging
195        let current_dir = std::env::current_dir()
196            .map(|p| p.to_string_lossy().to_string())
197            .unwrap_or_else(|_| "unknown".to_string());
198        debug!("Current working directory: {}", current_dir);
199
200        // Read migration file (try multiple possible paths)
201        let migration_paths = [
202            "migrations/001_initial_schema.sql",
203            "../migrations/001_initial_schema.sql",
204            "backend/migrations/001_initial_schema.sql",
205            "./backend/migrations/001_initial_schema.sql",
206        ];
207
208        let mut migration_sql = None;
209        let mut found_path = None;
210        for path in &migration_paths {
211            match std::fs::read_to_string(path) {
212                Ok(content) => {
213                    migration_sql = Some(content);
214                    found_path = Some(path.to_string());
215                    info!("Found migration file at: {}", path);
216                    break;
217                }
218                Err(e) => {
219                    debug!("Tried path '{}': {}", path, e);
220                }
221            }
222        }
223
224        let migration_sql = migration_sql
225            .ok_or_else(|| {
226                error!("Could not find migration file. Current dir: {}. Tried paths: {:?}", current_dir, migration_paths);
227                DatabaseError::MigrationError(
228                    format!("Could not find migration file. Current directory: {}. Tried paths: {:?}", current_dir, migration_paths)
229                )
230            })?;
231
232        info!("Found migration file at: {}, executing SQL...", found_path.unwrap_or_default());
233
234        // Clean SQL: Remove comments but preserve structure
235        // We need to be careful with $$ delimiters in functions
236        let cleaned_sql: String = migration_sql
237            .lines()
238            .map(|line| {
239                // Remove single-line comments (-- comments)
240                // But preserve lines that are part of function definitions
241                if let Some(comment_pos) = line.find("--") {
242                    // Don't remove if it's inside a $$ block (simple heuristic)
243                    if line.trim().starts_with("--") {
244                        String::new() // Full comment line, remove it
245                    } else {
246                        line[..comment_pos].trim().to_string()
247                    }
248                } else {
249                    line.trim().to_string()
250                }
251            })
252            .filter(|line| !line.is_empty())
253            .collect::<Vec<_>>()
254            .join("\n");
255
256        info!("Executing migration SQL ({} bytes)...", cleaned_sql.len());
257
258        // Execute the entire SQL file as one batch
259        // PostgreSQL's batch_execute handles multiple statements correctly,
260        // including functions with $$ delimiters
261        match client.batch_execute(&cleaned_sql).await {
262            Ok(_) => {
263                info!("Migrations completed successfully");
264                Ok(())
265            }
266            Err(e) => {
267                // Extract detailed error information
268                let error_msg = e.to_string();
269                let error_code = e.code();
270                let error_detail = e.as_db_error()
271                    .and_then(|db_err| db_err.detail())
272                    .unwrap_or("No detail available");
273                let error_hint = e.as_db_error()
274                    .and_then(|db_err| db_err.hint())
275                    .unwrap_or("No hint available");
276                
277                error!("Migration execution error:");
278                error!("  Error: {}", error_msg);
279                if let Some(code) = error_code {
280                    error!("  Code: {}", code.code());
281                }
282                error!("  Detail: {}", error_detail);
283                error!("  Hint: {}", error_hint);
284                
285                // Check if error is about objects already existing
286                // PostgreSQL error codes:
287                // 42P07 = duplicate_table
288                // 42710 = duplicate_object (for functions, triggers, etc.)
289                let is_duplicate_error = error_code
290                    .map(|code| {
291                        let code_str = code.code();
292                        code_str == "42P07" || code_str == "42710"
293                    })
294                    .unwrap_or(false);
295                
296                if error_msg.contains("already exists") || 
297                   error_msg.contains("duplicate") ||
298                   (error_msg.contains("relation") && error_msg.contains("already exists")) ||
299                   is_duplicate_error {
300                    warn!("Some database objects may already exist (error code: {:?}). This is OK if migrations were run before.", 
301                          error_code.map(|c| c.code()));
302                    info!("Migrations completed (some objects may already exist)");
303                    Ok(())
304                } else {
305                    // Show first 1000 chars of SQL for debugging
306                    let sql_preview: String = cleaned_sql
307                        .chars()
308                        .take(1000)
309                        .collect();
310                    error!("SQL preview (first 1000 chars):\n{}", sql_preview);
311                    
312                    Err(DatabaseError::MigrationError(format!(
313                        "Migration execution failed:\n  Error: {}\n  Code: {:?}\n  Detail: {}\n  Hint: {}\n  SQL size: {} bytes",
314                        error_msg,
315                        error_code.map(|c| c.code()),
316                        error_detail,
317                        error_hint,
318                        cleaned_sql.len()
319                    )))
320                }
321            }
322        }
323    }

325    /// Get a reference to the connection pool.
326    ///
327    /// Use this when you need direct access to the pool
328    /// for custom queries.
329    pub fn pool(&self) -> &Pool {
330        &self.pool
331    }
332}
