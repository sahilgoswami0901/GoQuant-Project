1//! # WebSocket Module
2//!
3//! This module provides real-time updates to connected clients via WebSocket.
4//!
5//! ## Features
6//!
7//! - Real-time balance updates
8//! - Transaction notifications
9//! - TVL updates
10//! - Lock/unlock event notifications
11//!
12//! ## Connection Flow
13//!
14//! ```text
15//! 1. Client connects to /ws/:user
16//!              ↓
17//! 2. Server authenticates (optional)
18//!              ↓
19//! 3. Server subscribes to user's vault events
20//!              ↓
21//! 4. Events are pushed as they occur:
22//!    - balance_update
23//!    - transaction_confirmed
24//!    - collateral_locked
25//!    - collateral_unlocked
26//! ```
27//!
28//! ## Message Format
29//!
30//! All messages are JSON:
31//!
32//! ```json
33//! {
34//!     "event": "balance_update",
35//!     "data": {
36//!         "totalBalance": 1000000000,
37//!         "lockedBalance": 200000000,
38//!         "availableBalance": 800000000
39//!     },
40//!     "timestamp": "2024-01-15T12:00:00Z"
41//! }
42//! ```

56/// WebSocket event types.
57#[derive(Debug, Clone, Serialize, Deserialize)]
58#[serde(rename_all = "snake_case")]
59pub enum WsEventType {
60    /// Vault balance was updated.
61    BalanceUpdate,
62    /// Transaction was confirmed.
63    TransactionConfirmed,
64    /// Collateral was locked for a position.
65    CollateralLocked,
66    /// Collateral was unlocked.
67    CollateralUnlocked,
68    /// System health update.
69    HealthUpdate,
70    /// TVL update.
71    TvlUpdate,
72    /// Ping/pong for keepalive.
73    Ping,
74    /// Error message.
75    Error,
76}

78/// WebSocket message wrapper.
79#[derive(Debug, Clone, Serialize, Deserialize)]
80#[serde(rename_all = "camelCase")]
81pub struct WsMessage<T> {
82    /// Event type.
83    pub event: WsEventType,
84    /// Event data.
85    pub data: T,
86    /// Timestamp.
87    pub timestamp: chrono::DateTime<Utc>,
88}

91    /// Create a new WebSocket message.
92    pub fn new(event: WsEventType, data: T) -> Self {
93        Self {
94            event,
95            data,
96            timestamp: Utc::now(),
97        }
98    }
99
100    /// Serialize to JSON string.
101    pub fn to_json(&self) -> Result<String, serde_json::Error> {
102        serde_json::to_string(self)
103    }

148/// WebSocket connection registry.
149/// 
150/// Tracks active WebSocket sessions per user and provides
151/// methods to send messages to specific users or broadcast to all.
152#[derive(Clone)]
153pub struct WsRegistry {
154    /// Map of user pubkey -> broadcast sender
155    /// Each user can have multiple connections (multiple tabs/devices)
156    sessions: Arc<Mutex<HashMap<String, Vec<broadcast::Sender<String>>>>>,
157}
158
159impl WsRegistry {
160    /// Create a new WebSocket registry.
161    pub fn new() -> Self {
162        Self {
163            sessions: Arc::new(Mutex::new(HashMap::new())),
164        }
165    }
166
167    /// Register a new WebSocket connection for a user.
168    /// Returns a receiver that will receive messages for this user.
169    pub async fn register(&self, user: String) -> broadcast::Receiver<String> {
170        let mut sessions = self.sessions.lock().await;
171        let (tx, rx) = broadcast::channel(100); // Buffer up to 100 messages
172        
173        sessions
174            .entry(user.clone())
175            .or_insert_with(Vec::new)
176            .push(tx);
177        
178        info!("Registered WebSocket for user: {} (total connections: {})", 
179            user, sessions.get(&user).map(|v| v.len()).unwrap_or(0));
180        
181        rx
182    }
183
184    /// Unregister a WebSocket connection for a user.
185    /// Note: Senders are automatically cleaned up when receivers are dropped.
186    /// We just remove entries that have no active receivers.
187    pub async fn unregister(&self, user: &str) {
188        let mut sessions = self.sessions.lock().await;
189        if let Some(senders) = sessions.get_mut(user) {
190            // Remove senders with no active receivers
191            senders.retain(|tx| tx.receiver_count() > 0);
192            
193            if senders.is_empty() {
194                sessions.remove(user);
195                info!("Unregistered all WebSocket connections for user: {}", user);
196            } else {
197                info!("Unregistered one WebSocket connection for user: {} (remaining: {})", 
198                    user, senders.len());
199            }
200        }
201    }
202
203    /// Send a message to a specific user.
204    ///
205    /// All active WebSocket connections for that user will receive the message.
206    /// If the user has no active connections, this is not an error (returns Ok).
207    ///
208    /// ## Arguments
209    ///
210    /// * `user` - User's public key
211    /// * `event` - Event type
212    /// * `data` - Event data (must implement Serialize)
213    ///
214    /// ## Returns
215    ///
216    /// * `Ok(())` - Message sent (or user not connected, which is not an error)
217    /// * `Err(String)` - Failed to serialize message
218    pub async fn send_to_user<T: Serialize>(
219        &self,
220        user: &str,
221        event: WsEventType,
222        data: T,
223    ) -> Result<(), String> {
224        let message = WsMessage::new(event, data);
225        let json = message.to_json()
226            .map_err(|e| format!("Failed to serialize message: {}", e))?;
227
228        let sessions = self.sessions.lock().await;
229        
230        if let Some(senders) = sessions.get(user) {
231            let mut sent_count = 0;
232            let mut errors = Vec::new();
233            
234            for sender in senders.iter() {
235                match sender.send(json.clone()) {
236                    Ok(_) => sent_count += 1,
237                    Err(e) => {
238                        errors.push(e.to_string());
239                    }
240                }
241            }
242            
243            if sent_count > 0 {
244                debug!("Sent message to user {} ({} connections)", user, sent_count);
245            }
246            
247            if !errors.is_empty() {
248                warn!("Failed to send to some connections for user {}: {:?}", user, errors);
249            }
250            
251            Ok(())
252        } else {
253            debug!("No active WebSocket connections for user: {}", user);
254            Ok(()) // Not an error - user just not connected
255        }
256    }
257
258/// Broadcast a message to all connected users.
259///
260    /// Sends the message to all active WebSocket connections across all users.
261    ///
262    /// ## Arguments
263    ///
264    /// * `event` - Event type
265    /// * `data` - Event data (must implement Serialize)
266    ///
267    /// ## Returns
268    ///
269    /// * `Ok(())` - Message broadcasted
270    /// * `Err(String)` - Failed to serialize message
271    pub async fn broadcast<T: Serialize>(
272        &self,
273        event: WsEventType,
274        data: T,
275    ) -> Result<(), String> {
276        let message = WsMessage::new(event, data);
277        let json = message.to_json()
278            .map_err(|e| format!("Failed to serialize message: {}", e))?;
279
280        let sessions = self.sessions.lock().await;
281        let mut total_sent = 0;
282        
283        for (_user, senders) in sessions.iter() {
284            for sender in senders.iter() {
285                if sender.send(json.clone()).is_ok() {
286                    total_sent += 1;
287                }
288            }
289        }
290        
291        info!("Broadcasted message to {} connections", total_sent);
292        Ok(())
293    }
294
295    /// Get the number of active connections for a user.
296    pub async fn connection_count(&self, user: &str) -> usize {
297        let sessions = self.sessions.lock().await;
298        sessions.get(user).map(|v| v.len()).unwrap_or(0)
299    }
300
301    /// Get total number of active connections across all users.
302    pub async fn total_connections(&self) -> usize {
303        let sessions = self.sessions.lock().await;
304        sessions.values().map(|v| v.len()).sum()
305    }
306}

321/// WebSocket connection handler.
322///
323/// Handles the initial WebSocket handshake and then manages
324/// the bidirectional communication.
325///
326/// ## Endpoint
327///
328/// `GET /ws/:user`
329///
330/// ## Path Parameters
331///
332/// - `user` - User's wallet public key to subscribe to
333///
334/// ## Example (JavaScript)
335///
336/// ```javascript
337/// const ws = new WebSocket('ws://localhost:8080/ws/7xKt9Fj2...');
338///
339/// ws.onmessage = (event) => {
340///     const message = JSON.parse(event.data);
341///     console.log('Event:', message.event);
342///     console.log('Data:', message.data);
343/// };
344/// ```
345pub async fn websocket_handler(
346    req: HttpRequest,
347    path: web::Path<String>,
348    body: web::Payload,
349    state: web::Data<Arc<AppState>>,
350) -> Result<HttpResponse, actix_web::Error> {
351    let user = path.into_inner();
352    info!("WebSocket connection request for user: {}", user);
353
354    // Upgrade to WebSocket connection
355    let (response, mut session, mut msg_stream) = actix_ws::handle(&req, body)?;
356
357    // Register this connection
358    let ws_registry = state.ws_registry.clone();
359    let mut rx = ws_registry.register(user.clone()).await;
360
361    // Spawn a task to handle the WebSocket
362    let user_clone = user.clone();
363    let ws_registry_clone = ws_registry.clone();
364    
365    actix_rt::spawn(async move {
366        info!("WebSocket connected for user: {}", user_clone);
367
368        // Send initial connection confirmation
369        let welcome = WsMessage::new(
370            WsEventType::HealthUpdate,
371            serde_json::json!({
372                "status": "connected",
373                "user": user_clone,
374                "message": "You will receive real-time updates for your vault"
375            }),
376        );
377        
378        if let Ok(json) = welcome.to_json() {
379            if let Err(e) = session.text(json).await {
380                error!("Failed to send welcome message: {}", e);
381            }
382        }
383
384        // Spawn a task to forward messages from registry to WebSocket
385        let mut session_clone = session.clone();
386        let user_for_task = user_clone.clone();
387        actix_rt::spawn(async move {
388            while let Ok(msg) = rx.recv().await {
389                if let Err(e) = session_clone.text(msg).await {
390                    error!("Failed to send message to {}: {}", user_for_task, e);
391                    break;
392        }
393            }
394        });
395
396        // Handle incoming messages
397        while let Some(Ok(msg)) = msg_stream.next().await {
398            match msg {
399                Message::Ping(bytes) => {
400                    debug!("Received ping from {}", user_clone);
401                    let _ = session.pong(&bytes).await;
402                }
403                Message::Pong(_) => {
404                    debug!("Received pong from {}", user_clone);
405                }
406                Message::Text(text) => {
407                    debug!("Received text from {}: {}", user_clone, text);
408                    
409                    // Handle incoming messages (e.g., subscription requests)
410                    // For now, we just echo back
411                    let response = WsMessage::new(
412                        WsEventType::Ping,
413                        serde_json::json!({ 
414                            "received": text.to_string(),
415                            "message": "WebSocket is active and listening for events"
416                        }),
417                    );
418                    
419                    if let Ok(json) = response.to_json() {
420                        let _ = session.text(json).await;
421                    }
422                }
423                Message::Binary(_) => {
424                    warn!("Received unexpected binary message from {}", user_clone);
425                }
426                Message::Close(reason) => {
427                    info!("WebSocket closed for {}: {:?}", user_clone, reason);
428                    break;
429                }
430                _ => {}
431            }
432        }
433
434        // Unregister on disconnect
435        ws_registry_clone.unregister(&user_clone).await;
436        info!("WebSocket disconnected for user: {}", user_clone);
437    });
438
439    Ok(response)
440}
