1//! # Solana Client Module
2//!
3//! This module provides a client for interacting with the Solana blockchain.
4//! It wraps the Solana SDK and provides high-level methods for vault operations.
5//!
6//! ## Responsibilities
7//!
8//! - Connect to Solana RPC
9//! - Fetch vault account data
10//! - Submit transactions
11//! - Subscribe to account changes
12//! - Monitor transaction confirmations
13//!
14//! ## Connection Types
15//!
16//! | Type | Use Case |
17//! |------|----------|
18//! | HTTP RPC | Queries, transaction submission |
19//! | WebSocket | Real-time event subscriptions |
20//!
21//! ## Account Data Flow
22//!
23//! ```text
24//! 1. Backend requests vault data
25//!              ↓
26//! 2. SolanaClient.get_vault_account()
27//!              ↓
28//! 3. Solana RPC returns raw account data
29//!              ↓
30//! 4. Deserialize Anchor account structure
31//!              ↓
32//! 5. Return typed VaultAccountData
33//! ```

159    /// Execute an RPC operation with retry logic.
160    ///
161    /// Retries up to 4 attempts (initial attempt + 3 retries) with exponential backoff
162    /// on network errors. Uses a 10-second timeout per attempt.
163    /// This helps reduce error rates when connecting to api.devnet.solana.com.
164    ///
165    /// ## Arguments
166    ///
167    /// * `operation` - Async closure that performs the RPC operation
168    ///
169    /// ## Returns
170    ///
171    /// * `Ok(T)` - Operation succeeded (on any attempt)
172    /// * `Err(String)` - Operation failed after all retries (4 total attempts)
173    async fn retry_rpc_operation<F, Fut, T>(&self, mut operation: F) -> Result<T, String>
174    where
175        F: FnMut() -> Fut,
176        Fut: std::future::Future<Output = Result<T, String>>,
177    {
178        const MAX_RETRIES: u32 = 3;
179        const INITIAL_DELAY_MS: u64 = 200;
180        const TIMEOUT_SECS: u64 = 10;
181
182        for attempt in 0..=MAX_RETRIES {
183            match timeout(Duration::from_secs(TIMEOUT_SECS), operation()).await {
184                Ok(Ok(result)) => {
185                    if attempt > 0 {
186                        info!("RPC operation succeeded after {} retries", attempt);
187                    }
188                    return Ok(result);
189                }
190                Ok(Err(e)) => {
191                    // RPC call completed but returned an error
192                    if attempt < MAX_RETRIES {
193                        let delay_ms = INITIAL_DELAY_MS * (1 << attempt); // Exponential backoff: 200ms, 400ms, 800ms, 1600ms, 3200ms
194                        debug!("RPC operation failed (attempt {}): {}. Retrying in {}ms...", 
195                            attempt + 1, e, delay_ms);
196                        tokio::time::sleep(Duration::from_millis(delay_ms)).await;
197                        continue;
198                    } else {
199                        warn!("RPC operation failed after {} attempts: {}", MAX_RETRIES + 1, e);
200                        return Err(e);
201                    }
202                }
203                Err(_) => {
204                    // Timeout
205                    if attempt < MAX_RETRIES {
206                        let delay_ms = INITIAL_DELAY_MS * (1 << attempt);
207                        debug!("RPC operation timed out (attempt {}). Retrying in {}ms...", 
208                            attempt + 1, delay_ms);
209                        tokio::time::sleep(Duration::from_millis(delay_ms)).await;
210                        continue;
211                    } else {
212                        warn!("RPC operation timed out after {} attempts ({}s timeout)", 
213                            MAX_RETRIES + 1, TIMEOUT_SECS);
214                        return Err(format!("RPC operation timed out after {} attempts", MAX_RETRIES + 1));
215                    }
216                }
217            }
218        }
219
220        Err("RPC operation failed after all retries".to_string())
221    }

223    /// Check if the Solana RPC is healthy.
224    ///
225    /// Makes a simple request to verify connectivity with timeout and retry logic.
226    ///
227    /// ## Returns
228    ///
229    /// * `Ok(true)` - RPC is responding
230    /// * `Ok(false)` - RPC is not healthy (after retries)
231    /// * `Err(...)` - Connection error
232    pub async fn get_health(&self) -> Result<bool, String> {
233        const MAX_RETRIES: u32 = 2;
234        const TIMEOUT_SECS: u64 = 5;
235        
236        // Retry logic with exponential backoff
237        for attempt in 0..=MAX_RETRIES {
238            match timeout(
239                Duration::from_secs(TIMEOUT_SECS),
240                self.get_slot()
241            ).await {
242                Ok(Ok(_)) => {
243                    // Success - reset any previous warnings
244                    if attempt > 0 {
245                        info!("Solana RPC health check recovered after {} retries", attempt);
246                    }
247                    return Ok(true);
248                }
249                Ok(Err(e)) => {
250                    // RPC call succeeded but returned an error
251                    if attempt < MAX_RETRIES {
252                        let delay_ms = 500 * (attempt + 1) as u64;
253                        debug!("Solana RPC health check attempt {} failed: {}. Retrying in {}ms...", 
254                            attempt + 1, e, delay_ms);
255                        tokio::time::sleep(Duration::from_millis(delay_ms)).await;
256                        continue;
257                    } else {
258                        // Only warn on final failure, not every single timeout
259                        warn!("Solana RPC health check failed after {} attempts: {}", 
260                            MAX_RETRIES + 1, e);
261                        return Ok(false);
262                    }
263                }
264                Err(_) => {
265                    // Timeout
266                    if attempt < MAX_RETRIES {
267                        let delay_ms = 500 * (attempt + 1) as u64;
268                        debug!("Solana RPC health check timed out (attempt {}). Retrying in {}ms...", 
269                            attempt + 1, delay_ms);
270                        tokio::time::sleep(Duration::from_millis(delay_ms)).await;
271                        continue;
272                    } else {
273                        // Only warn on final failure
274                        warn!("Solana RPC health check timed out after {} attempts ({}s timeout)", 
275                            MAX_RETRIES + 1, TIMEOUT_SECS);
276                        return Ok(false);
277                    }
278                }
279            }
280        }
281        
282        Ok(false)
283    }

285    /// Get the current slot.
286    ///
287    /// The slot is Solana's block number equivalent.
288    pub async fn get_slot(&self) -> Result<u64, String> {
289        use actix_web::web;
290        
291        let rpc_url = self.rpc_url.clone();
292        self.retry_rpc_operation(|| {
293            let rpc_url = rpc_url.clone();
294            async move {
295                let client = RpcClient::new_with_commitment(
296                    rpc_url,
297                    CommitmentConfig::confirmed(),
298                );
299                web::block(move || client.get_slot())
300                    .await
301                    .map_err(|e| format!("Failed to execute blocking task: {}", e))?
302                    .map_err(|e| format!("Failed to get slot: {}", e))
303            }
304        }).await
305    }

337    /// Get vault account data for a user.
338    ///
339    /// Fetches the vault PDA account and deserializes it.
340    ///
341    /// ## Arguments
342    ///
343    /// * `owner` - The vault owner's public key (base58)
344    ///
345    /// ## Returns
346    ///
347    /// * `Ok(Some(VaultAccountData))` - Vault found and deserialized
348    /// * `Ok(None)` - Vault doesn't exist
349    /// * `Err(...)` - RPC or deserialization error
350    ///
351    /// ## Example
352    ///
353    /// ```rust,ignore
354    /// let vault = client.get_vault_account("7xKt9Fj2...").await?;
355    /// if let Some(v) = vault {
356    ///     println!("Balance: {} USDT", v.total_balance / 1_000_000);
357    /// }
358    /// ```
359    pub async fn get_vault_account(
360        &self,
361        owner: &str,
362    ) -> Result<Option<VaultAccountData>, String> {
363        debug!("Fetching vault account for: {}", owner);
364
365        let owner_pubkey = Pubkey::from_str(owner)
366            .map_err(|e| format!("Invalid owner pubkey: {}", e))?;
367
368        // Derive the vault PDA
369        let (vault_pda, _bump) = Pubkey::find_program_address(
370            &[b"vault", owner_pubkey.as_ref()],
371            &self.program_id,
372        );
373
374        debug!("Vault PDA: {}", vault_pda);
375
376        let rpc_url = self.rpc_url.clone();
377        let vault_pda_clone = vault_pda;
378        let program_id = self.program_id;
379
380        // Fetch the account with retry logic
381        use actix_web::web;
382        match self.retry_rpc_operation(|| {
383            let rpc_url = rpc_url.clone();
384            let vault_pda = vault_pda_clone;
385            async move {
386                let client = RpcClient::new_with_commitment(
387                    rpc_url,
388                    CommitmentConfig::confirmed(),
389                );
390                web::block(move || client.get_account(&vault_pda))
391                    .await
392                    .map_err(|e| format!("Failed to spawn blocking task: {}", e))?
393                    .map_err(|e| format!("Failed to fetch vault: {}", e))
394            }
395        }).await {
396            Ok(account) => {
397                // Deserialize the account data
398                // Note: In production, use Anchor's account deserialization
399                let vault_data = self.deserialize_vault_account(&account.data, &vault_pda)?;
400                Ok(Some(vault_data))
401            }
402            Err(e) => {
403                // Check if it's a "not found" error
404                if e.contains("AccountNotFound") {
405                    debug!("Vault not found for: {}", owner);
406                    Ok(None)
407                } else {
408                    Err(e)
409                }
410            }
411        }
412    }

414    /// Deserialize vault account data.
415    ///
416    /// Parses the raw account bytes into a VaultAccountData struct.
417    ///
418    /// ## Account Structure (Anchor)
419    ///
420    /// ```text
421    /// Offset | Size | Field
422    /// -------|------|------
423    /// 0      | 8    | Anchor discriminator
424    /// 8      | 32   | owner
425    /// 40     | 32   | token_account
426    /// 72     | 8    | total_balance
427    /// 80     | 8    | locked_balance
428    /// 88     | 8    | available_balance
429    /// 96     | 8    | total_deposited
430    /// 104    | 8    | total_withdrawn
431    /// 112    | 8    | created_at (i64)
432    /// 120    | 1    | bump
433    /// ```
434    fn deserialize_vault_account(
435        &self,
436        data: &[u8],
437        vault_address: &Pubkey,
438    ) -> Result<VaultAccountData, String> {
439        if data.len() < 121 {
440            return Err("Account data too short".to_string());
441        }
442
443        // Skip 8-byte Anchor discriminator
444        let data = &data[8..];
445
446        // Parse fields
447        // owner: [0..32]
448        // token_account: [32..64]
449        let token_account = Pubkey::try_from(&data[32..64])
450            .map_err(|_| "Invalid token account")?;
451
452        // total_balance: [64..72]
453        let total_balance = u64::from_le_bytes(
454            data[64..72].try_into().map_err(|_| "Invalid total_balance")?
455        );
456
457        // locked_balance: [72..80]
458        let locked_balance = u64::from_le_bytes(
459            data[72..80].try_into().map_err(|_| "Invalid locked_balance")?
460        );
461
462        // available_balance: [80..88]
463        let available_balance = u64::from_le_bytes(
464            data[80..88].try_into().map_err(|_| "Invalid available_balance")?
465        );
466
467        // total_deposited: [88..96]
468        let total_deposited = u64::from_le_bytes(
469            data[88..96].try_into().map_err(|_| "Invalid total_deposited")?
470        );
471
472        // total_withdrawn: [96..104]
473        let total_withdrawn = u64::from_le_bytes(
474            data[96..104].try_into().map_err(|_| "Invalid total_withdrawn")?
475        );
476
477        // created_at: [104..112]
478        let created_at_timestamp = i64::from_le_bytes(
479            data[104..112].try_into().map_err(|_| "Invalid created_at")?
480        );
481
482        let created_at = Utc.timestamp_opt(created_at_timestamp, 0)
483            .single()
484            .unwrap_or_else(Utc::now);
485
486        Ok(VaultAccountData {
487            vault_address: vault_address.to_string(),
488            token_account: token_account.to_string(),
489            total_balance,
490            locked_balance,
491            available_balance,
492            total_deposited,
493            total_withdrawn,
494            created_at,
495        })
496    }

498    /// Get the token balance of an account.
499    ///
500    /// Fetches the actual SPL token balance from a token account.
501    ///
502    /// ## Arguments
503    ///
504    /// * `token_account` - The token account address (base58)
505    ///
506    /// ## Returns
507    ///
508    /// Token balance in smallest units.
509    pub async fn get_token_balance(
510        &self,
511        token_account: &str,
512    ) -> Result<u64, String> {
513        let token_pubkey = Pubkey::from_str(token_account)
514            .map_err(|e| format!("Invalid token account: {}", e))?;
515
516        let rpc_url = self.rpc_url.clone();
517        let token_pubkey_clone = token_pubkey;
518
519        use actix_web::web;
520        let balance = self.retry_rpc_operation(|| {
521            let rpc_url = rpc_url.clone();
522            let token_pubkey = token_pubkey_clone;
523            async move {
524                let client = RpcClient::new_with_commitment(
525                    rpc_url,
526                    CommitmentConfig::confirmed(),
527                );
528                web::block(move || client.get_token_account_balance(&token_pubkey))
529                    .await
530                    .map_err(|e| format!("Failed to execute blocking task: {}", e))?
531                    .map_err(|e| format!("Failed to get token balance: {}", e))
532            }
533        }).await?;
534
535        let amount = balance
536            .amount
537            .parse::<u64>()
538            .map_err(|e| format!("Invalid balance amount: {}", e))?;
539
540        Ok(amount)
541    }

543    /// Check if a transaction is confirmed.
544    ///
545    /// ## Arguments
546    ///
547    /// * `signature` - Transaction signature (base58)
548    ///
549    /// ## Returns
550    ///
551    /// * `Ok(true)` - Transaction is confirmed
552    /// * `Ok(false)` - Transaction not found or not confirmed
553    /// * `Err(...)` - RPC error
554    pub async fn is_transaction_confirmed(
555        &self,
556        signature: &str,
557    ) -> Result<bool, String> {
558        let sig = Signature::from_str(signature)
559            .map_err(|e| format!("Invalid signature: {}", e))?;
560
561        let rpc_url = self.rpc_url.clone();
562        let sig_clone = sig;
563
564        use actix_web::web;
565        match self.retry_rpc_operation(|| {
566            let rpc_url = rpc_url.clone();
567            let sig = sig_clone;
568            async move {
569                let client = RpcClient::new_with_commitment(
570                    rpc_url,
571                    CommitmentConfig::confirmed(),
572                );
573                web::block(move || client.get_signature_status(&sig))
574                    .await
575                    .map_err(|e| format!("Failed to execute blocking task: {}", e))?
576                    .map_err(|e| format!("Failed to get signature status: {}", e))
577            }
578        }).await {
579            Ok(Some(status)) => {
580                match status {
581                    Ok(_) => Ok(true),
582                    Err(_) => Ok(false),
583                }
584            }
585            Ok(None) => Ok(false),
586            Err(e) => Err(format!("Failed to check signature: {}", e)),
587        }
588    }
