1//! # Transaction Builder Service
2//!
3//! The TransactionBuilder creates Solana transactions for vault operations.
4//! It handles all the complexity of building properly formatted transactions.
5//!
6//! ## Responsibilities
7//!
8//! - Build initialize_vault transactions
9//! - Build deposit transactions
10//! - Build withdrawal transactions
11//! - Build lock/unlock collateral transactions
12//! - Handle SPL Token account setup
13//! - Set appropriate compute budgets
14//!
15//! ## Transaction Structure
16//!
17//! A Solana transaction consists of:
18//!
19//! ```text
20//! Transaction
21//! ├── Recent Blockhash (for expiration)
22//! ├── Fee Payer (who pays transaction fees)
23//! └── Instructions[]
24//!     ├── Compute Budget (optional, for priority)
25//!     └── Program Instruction (the actual operation)
26//!         ├── Program ID
27//!         ├── Accounts[]
28//!         └── Data (serialized instruction args)
29//! ```
30//!
31//! ## Unsigned vs Signed Transactions
32//!
33//! This builder creates **unsigned** transactions. The flow is:
34//!
35//! ```text
36//! 1. Backend builds unsigned transaction
37//!              ↓
38//! 2. Send to frontend (base64 encoded)
39//!              ↓
40//! 3. User signs with their wallet
41//!              ↓
42//! 4. Frontend submits signed transaction to Solana
43//! ```
44//!
45//! This ensures the user's private key never leaves their wallet.

61/// Errors that can occur when building transactions.
62#[derive(Debug, thiserror::Error)]
63pub enum TransactionBuilderError {
64    /// Invalid public key format.
65    #[error("Invalid public key: {0}")]
66    InvalidPubkey(String),
67
68    /// Failed to get recent blockhash.
69    #[error("Failed to get blockhash: {0}")]
70    BlockhashError(String),
71
72    /// Failed to serialize transaction.
73    #[error("Serialization error: {0}")]
74    SerializationError(String),
75
76    /// Solana client error.
77    #[error("Solana error: {0}")]
78    #[allow(dead_code)]
79    SolanaError(String),
80}

82/// The Transaction Builder service.
83///
84/// Creates properly formatted Solana transactions for
85/// vault operations.
86///
87/// ## Usage
88///
89/// ```rust,ignore
90/// let builder = TransactionBuilder::new(solana_client, config);
91///
92/// // Build a deposit transaction
93/// let tx = builder.build_deposit("user_pubkey", 100_000_000).await?;
94///
95/// // tx is base64-encoded, send to frontend for signing
96/// ```
97#[derive(Clone)]
98pub struct TransactionBuilder {
99    /// Solana RPC client.
100    solana: SolanaClient,
101
102    /// Application configuration.
103    #[allow(dead_code)]
104    config: AppConfig,
105
106    /// The vault program ID.
107    program_id: Pubkey,
108
109    /// USDT token mint.
110    usdt_mint: Pubkey,
111}

135    /// Build an initialize_vault transaction.
136    ///
137    /// Creates a new vault PDA and associated token account for the user.
138    ///
139    /// ## Arguments
140    ///
141    /// * `user_pubkey` - User's wallet public key (base58)
142    ///
143    /// ## Returns
144    ///
145    /// * `Ok((tx_base64, vault_pda))` - Unsigned transaction and vault address
146    /// * `Err(...)` - Failed to build transaction
147    ///
148    /// ## Accounts Required
149    ///
150    /// 1. User (signer, payer)
151    /// 2. Vault PDA (will be created)
152    /// 3. USDT Mint
153    /// 4. Vault Token Account (will be created)
154    /// 5. System Program
155    /// 6. Token Program
156    /// 7. Associated Token Program
157    pub async fn build_initialize_vault(
158        &self,
159        user_pubkey: &str,
160    ) -> Result<(String, String), TransactionBuilderError> {
161        info!("Building initialize_vault transaction for: {}", user_pubkey);
162
163        let user = Pubkey::from_str(user_pubkey)
164            .map_err(|_| TransactionBuilderError::InvalidPubkey(user_pubkey.to_string()))?;
165
166        // Derive the vault PDA
167        let (vault_pda, _bump) = Pubkey::find_program_address(
168            &[b"vault", user.as_ref()],
169            &self.program_id,
170        );
171
172        // Derive the vault authority PDA
173        let (_vault_authority, _) = Pubkey::find_program_address(
174            &[b"vault_authority"],
175            &self.program_id,
176        );
177
178        // Get the vault's associated token account
179        let vault_token_account = spl_associated_token_account::get_associated_token_address(
180            &vault_pda,
181            &self.usdt_mint,
182        );
183
184        // Build the instruction
185        // Note: In production, you'd use Anchor's IDL to build this
186        // For now, we'll create a placeholder instruction structure
187        let instruction = Instruction {
188            program_id: self.program_id,
189            accounts: vec![
190                AccountMeta::new(user, true),           // user (signer, payer)
191                AccountMeta::new(vault_pda, false),     // vault (PDA, will init)
192                AccountMeta::new_readonly(self.usdt_mint, false), // usdt_mint
193                AccountMeta::new(vault_token_account, false), // vault_token_account
194                AccountMeta::new_readonly(system_program::id(), false),
195                AccountMeta::new_readonly(spl_token::id(), false),
196                AccountMeta::new_readonly(
197                    spl_associated_token_account::id(),
198                    false,
199                ),
200            ],
201            // Data would be the serialized Anchor instruction
202            // For production: use anchor_client to build proper instruction data
203            data: vec![0], // Placeholder: "initialize_vault" discriminator
204        };
205
206        // Get recent blockhash
207        let _blockhash = self
208            .solana
209            .get_recent_blockhash()
210            .await
211            .map_err(|e| TransactionBuilderError::BlockhashError(e.to_string()))?;
212
213        // Build transaction
214        let message = Message::new(&[instruction], Some(&user));
215        let transaction = Transaction::new_unsigned(message);
216
217        // Serialize to base64
218        let tx_bytes = bincode::serialize(&transaction)
219            .map_err(|e| TransactionBuilderError::SerializationError(e.to_string()))?;
220        let tx_base64 = BASE64.encode(&tx_bytes);
221
222        debug!("Built initialize_vault tx, vault PDA: {}", vault_pda);
223
224        Ok((tx_base64, vault_pda.to_string()))
225    }

239    pub async fn build_deposit(
240        &self,
241        user_pubkey: &str,

243    ) -> Result<String, TransactionBuilderError> {
244        info!("Building deposit transaction: {} USDT for {}", 
245            amount as f64 / 1_000_000.0, user_pubkey);
246
247        let user = Pubkey::from_str(user_pubkey)
248            .map_err(|_| TransactionBuilderError::InvalidPubkey(user_pubkey.to_string()))?;
249
250        // Derive PDAs
251        let (vault_pda, _) = Pubkey::find_program_address(
252            &[b"vault", user.as_ref()],
253            &self.program_id,
254        );
255
256        let (vault_authority, _) = Pubkey::find_program_address(
257            &[b"vault_authority"],
258            &self.program_id,
259        );
260
261        // Get token accounts
262        let user_token_account = spl_associated_token_account::get_associated_token_address(
263            &user,
264            &self.usdt_mint,
265        );
266
267        let vault_token_account = spl_associated_token_account::get_associated_token_address(
268            &vault_pda,
269            &self.usdt_mint,
270        );
271
272        // Build instruction
273        let instruction = Instruction {
274            program_id: self.program_id,
275            accounts: vec![
276                AccountMeta::new(user, true),               // user (signer)
277                AccountMeta::new(vault_pda, false),         // vault
278                AccountMeta::new(user_token_account, false), // user_token_account
279                AccountMeta::new(vault_token_account, false), // vault_token_account
280                AccountMeta::new_readonly(vault_authority, false), // vault_authority
281                AccountMeta::new_readonly(spl_token::id(), false), // token_program
282            ],
283            // Placeholder: "deposit" discriminator + amount
284            data: Self::build_deposit_data(amount),
285        };
286
287        // Get recent blockhash
288        let _blockhash = self
289            .solana
290            .get_recent_blockhash()
291            .await
292            .map_err(|e| TransactionBuilderError::BlockhashError(e.to_string()))?;
293
294        // Build and serialize transaction
295        let message = Message::new(&[instruction], Some(&user));
296        let transaction = Transaction::new_unsigned(message);
297
298        let tx_bytes = bincode::serialize(&transaction)
299            .map_err(|e| TransactionBuilderError::SerializationError(e.to_string()))?;
300        
301        Ok(BASE64.encode(&tx_bytes))
302    }

508    /// Build a transfer_collateral transaction.
509    ///
510    /// Transfers collateral from one vault to another.
511    /// Only callable by authorized programs (liquidation engine, settlement relayer).
512    ///
513    /// ## Arguments
514    ///
515    /// * `from_pubkey` - Source vault owner
516    /// * `to_pubkey` - Destination vault owner
517    /// * `amount` - Amount to transfer (in smallest units, 6 decimals)
518    /// * `reason` - Transfer reason (0=settlement, 1=liquidation, 2=fee)
519    /// * `authority_pubkey` - Public key of the authorized program (liquidation engine) that will sign
520    ///
521    /// ## Returns
522    ///
523    /// Base64-encoded unsigned transaction (must be signed by the authority)
524    pub async fn build_transfer_collateral(
525        &self,
526        from_pubkey: &str,
527        to_pubkey: &str,
528        amount: u64,
529        reason: u8,
530        authority_pubkey: &str,
531    ) -> Result<String, TransactionBuilderError> {
532        info!("Building transfer_collateral transaction: {} USDT from {} to {} (authority: {})", 
533            amount as f64 / 1_000_000.0, from_pubkey, to_pubkey, authority_pubkey);
534
535        let from_user = Pubkey::from_str(from_pubkey)
536            .map_err(|_| TransactionBuilderError::InvalidPubkey(from_pubkey.to_string()))?;
537        let to_user = Pubkey::from_str(to_pubkey)
538            .map_err(|_| TransactionBuilderError::InvalidPubkey(to_pubkey.to_string()))?;
539
540        let authority = Pubkey::from_str(authority_pubkey)
541            .map_err(|_| TransactionBuilderError::InvalidPubkey(authority_pubkey.to_string()))?;
542
543        // Derive PDAs
544        let (from_vault_pda, _) = Pubkey::find_program_address(
545            &[b"vault", from_user.as_ref()],
546            &self.program_id,
547        );
548
549        let (to_vault_pda, _) = Pubkey::find_program_address(
550            &[b"vault", to_user.as_ref()],
551            &self.program_id,
552        );
553
554        let (vault_authority, _) = Pubkey::find_program_address(
555            &[b"vault_authority"],
556            &self.program_id,
557        );
558
559        // Get token accounts
560        let from_token_account = spl_associated_token_account::get_associated_token_address(
561            &from_vault_pda,
562            &self.usdt_mint,
563        );
564
565        let to_token_account = spl_associated_token_account::get_associated_token_address(
566            &to_vault_pda,
567            &self.usdt_mint,
568        );
569
570        // Build instruction
571        // The authority (liquidation engine) is the signer
572        let instruction = Instruction {
573            program_id: self.program_id,
574            accounts: vec![
575                AccountMeta::new(authority, true),              // authority (signer - liquidation engine)
576                AccountMeta::new(from_vault_pda, false),        // from_vault
577                AccountMeta::new(from_token_account, false),     // from_token_account
578                AccountMeta::new(to_vault_pda, false),          // to_vault
579                AccountMeta::new(to_token_account, false),      // to_token_account
580                AccountMeta::new_readonly(vault_authority, false), // vault_authority
581                AccountMeta::new_readonly(spl_token::id(), false), // token_program
582            ],
583            data: Self::build_transfer_data(amount, reason),
584        };
585
586        let _blockhash = self
587            .solana
588            .get_recent_blockhash()
589            .await
590            .map_err(|e| TransactionBuilderError::BlockhashError(e.to_string()))?;
591
592        // Authority (liquidation engine) is the fee payer
593        let message = Message::new(&[instruction], Some(&authority));
594        let transaction = Transaction::new_unsigned(message);
595
596        let tx_bytes = bincode::serialize(&transaction)
597            .map_err(|e| TransactionBuilderError::SerializationError(e.to_string()))?;
598        
599        Ok(BASE64.encode(&tx_bytes))
600    }

606    /// Build instruction data for deposit.
607    /// 
608    /// Uses the correct Anchor discriminator from the IDL.
609    /// Discriminator: [242, 35, 198, 137, 82, 225, 242, 182] (0xf223c68952e1f2b6)
610    fn build_deposit_data(amount: u64) -> Vec<u8> {
611        // Anchor discriminator for "deposit" instruction
612        // This is the first 8 bytes of sha256("global:deposit")
613        let discriminator = vec![242, 35, 198, 137, 82, 225, 242, 182];
614        let mut data = discriminator;
615        // Append amount as little-endian u64 (8 bytes)
616        data.extend_from_slice(&amount.to_le_bytes());
617        data
618    }
