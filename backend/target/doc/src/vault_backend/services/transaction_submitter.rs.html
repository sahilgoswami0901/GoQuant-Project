1//! # Transaction Submitter Service
2//!
3//! This service signs and submits Solana transactions automatically.
4//! For user operations (deposit/withdraw), requires the user's keypair.

6use std::fs;
7use solana_sdk::{
8    signature::Keypair,
9    transaction::Transaction,
10};
11use solana_client::rpc_client::RpcClient;
12use base64::{Engine as _, engine::general_purpose::STANDARD as BASE64};
13use tracing::{info, error};

15use crate::config::AppConfig;

17/// Errors that can occur when submitting transactions.
18#[derive(Debug, thiserror::Error)]
19pub enum TransactionSubmitterError {
20    /// Failed to load keypair.
21    #[error("Failed to load keypair: {0}")]
22    KeypairError(String),
23
24    /// Failed to decode transaction.
25    #[error("Failed to decode transaction: {0}")]
26    DecodeError(String),
27
28    /// Failed to get recent blockhash.
29    #[error("Failed to get blockhash: {0}")]
30    BlockhashError(String),
31
32    /// RPC error.
33    #[error("RPC error: {0}")]
34    RpcError(String),
35}
36
37/// Transaction Submitter service.
38///
39/// Signs and submits Solana transactions automatically.
40#[derive(Clone)]
41pub struct TransactionSubmitter {
42    /// Application configuration.
43    config: AppConfig,
44}
45
46impl TransactionSubmitter {
47    /// Create a new TransactionSubmitter.
48    pub fn new(config: AppConfig) -> Self {
49        Self { config }
50    }

52    /// Sign and submit a transaction using a keypair.
53    ///
54    /// ## Arguments
55    ///
56    /// * `unsigned_tx_base64` - Base64-encoded unsigned transaction
57    /// * `keypair_path` - Path to keypair file to sign with
58    ///
59    /// ## Returns
60    ///
61    /// Transaction signature if successful.
62
63    pub async fn sign_and_submit(
64        &self,
65        unsigned_tx_base64: &str,
66        keypair_path: &str,
67    ) -> Result<String, TransactionSubmitterError> {
68        use actix_web::web;
69
70        info!("Signing and submitting transaction...");
71
72        // Load keypair
73        let keypair_path_expanded = shellexpand::full(keypair_path)
74            .map_err(|e| TransactionSubmitterError::KeypairError(format!("Invalid path: {}", e)))?;
75        
76        let keypair_bytes: Vec<u8> = serde_json::from_str(
77            &fs::read_to_string(keypair_path_expanded.as_ref())
78                .map_err(|e| TransactionSubmitterError::KeypairError(format!("Failed to read keypair: {}", e)))?
79        )
80        .map_err(|e| TransactionSubmitterError::KeypairError(format!("Failed to parse keypair: {}", e)))?;
81
82        // Decode transaction
83        let tx_bytes = BASE64.decode(unsigned_tx_base64)
84            .map_err(|e| TransactionSubmitterError::DecodeError(e.to_string()))?;
85        
86        let mut transaction: Transaction = bincode::deserialize(&tx_bytes)
87            .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to deserialize: {}", e)))?;
88
89        // Submit transaction
90        let rpc_url = self.config.solana_rpc_url.clone();
91        let keypair_bytes_clone = keypair_bytes.clone();
92        let tx_bytes = bincode::serialize(&transaction)
93            .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to serialize: {}", e)))?;
94
95        let signature = tokio::task::spawn_blocking(move || {
96            let rpc_client = RpcClient::new(rpc_url);
97            let keypair = Keypair::from_bytes(&keypair_bytes_clone)
98                .map_err(|e| TransactionSubmitterError::KeypairError(format!("Failed to recreate keypair: {}", e)))?;
99
100            // Get recent blockhash (transaction might have expired, so get fresh one)
101            let recent_blockhash = rpc_client.get_latest_blockhash()
102                .map_err(|e| TransactionSubmitterError::BlockhashError(e.to_string()))?;
103
104            // Deserialize and update transaction with fresh blockhash and sign
105            let mut transaction: Transaction = bincode::deserialize(&tx_bytes)
106                .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to deserialize: {}", e)))?;
107            
108            transaction.sign(&[&keypair], recent_blockhash);
109
110            // Submit transaction
111            let signature = rpc_client.send_and_confirm_transaction(&transaction)
112                .map_err(|e| TransactionSubmitterError::RpcError(format!("Failed to submit: {}", e)))?;
113
114            Ok::<String, TransactionSubmitterError>(signature.to_string())
115        }).await
116        .map_err(|e| TransactionSubmitterError::RpcError(format!("Blocking task failed: {}", e)))?
117        .map_err(|e| TransactionSubmitterError::RpcError(e.to_string()))?;
118
119        info!("✅ Transaction submitted: {}", signature);
120        Ok(signature)
121    }

123    /// Sign a transaction without submitting it.
124    ///
125    /// ## Arguments
126    ///
127    /// * `unsigned_tx_base64` - Base64-encoded unsigned transaction
128    /// * `keypair_path` - Path to keypair file to sign with
129    ///
130    /// ## Returns
131    ///
132    /// Tuple of (signed_transaction_base64, transaction_signature)
133    pub async fn sign_transaction(
134        &self,
135        unsigned_tx_base64: &str,
136        keypair_path: &str,
137    ) -> Result<(String, String), TransactionSubmitterError> {
138        info!("Signing transaction with keypair: {}", keypair_path);
139
140        // Load keypair
141        let keypair_path_expanded = shellexpand::full(keypair_path)
142            .map_err(|e| TransactionSubmitterError::KeypairError(format!("Invalid path: {}", e)))?;
143        
144        let keypair_bytes: Vec<u8> = serde_json::from_str(
145            &fs::read_to_string(keypair_path_expanded.as_ref())
146                .map_err(|e| TransactionSubmitterError::KeypairError(format!("Failed to read keypair: {}", e)))?
147        )
148        .map_err(|e| TransactionSubmitterError::KeypairError(format!("Failed to parse keypair: {}", e)))?;
149
150        // Decode transaction
151        let tx_bytes = BASE64.decode(unsigned_tx_base64)
152            .map_err(|e| TransactionSubmitterError::DecodeError(e.to_string()))?;
153
154        // Sign transaction
155        let rpc_url = self.config.solana_rpc_url.clone();
156        let keypair_bytes_clone = keypair_bytes.clone();
157
158        let signed_tx_base64 = tokio::task::spawn_blocking(move || {
159            let rpc_client = RpcClient::new(rpc_url);
160            let keypair = Keypair::from_bytes(&keypair_bytes_clone)
161                .map_err(|e| TransactionSubmitterError::KeypairError(format!("Failed to recreate keypair: {}", e)))?;
162
163            // Get recent blockhash (transaction might have expired, so get fresh one)
164            let recent_blockhash = rpc_client.get_latest_blockhash()
165                .map_err(|e| TransactionSubmitterError::BlockhashError(e.to_string()))?;
166
167            // Deserialize transaction
168            let mut transaction: Transaction = bincode::deserialize(&tx_bytes)
169                .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to deserialize: {}", e)))?;
170
171            // Update transaction with fresh blockhash and sign
172            transaction.sign(&[&keypair], recent_blockhash);
173
174            // Get the signature from the signed transaction
175            let signature = transaction.signatures[0].to_string();
176
177            // Serialize signed transaction
178            let signed_tx_bytes = bincode::serialize(&transaction)
179                .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to serialize signed tx: {}", e)))?;
180
181            Ok::<(String, String), TransactionSubmitterError>((BASE64.encode(&signed_tx_bytes), signature))
182        }).await
183        .map_err(|e| TransactionSubmitterError::RpcError(format!("Blocking task failed: {}", e)))?
184        .map_err(|e| TransactionSubmitterError::RpcError(e.to_string()))?;
185
186        info!("✅ Transaction signed with signature: {}", signed_tx_base64.1);
187        Ok(signed_tx_base64)
188    }

190    /// Submit an already-signed transaction.
191    ///
192    /// Uses `send_and_confirm_transaction` which waits for confirmation.
193    /// Runs in a blocking task (`tokio::task::spawn_blocking`) to avoid blocking the async runtime.
194    ///
195    /// ## Arguments
196    ///
197    /// * `signed_tx_base64` - Base64-encoded signed transaction
198    ///
199    /// ## Returns
200    ///
201    /// Transaction signature (as string) if successful.
202    pub async fn submit_signed_transaction(
203        &self,
204        signed_tx_base64: &str,
205    ) -> Result<String, TransactionSubmitterError> {
206        use actix_web::web;
207
208        info!("Submitting signed transaction...");
209
210        // Decode transaction
211        let tx_bytes = BASE64.decode(signed_tx_base64)
212            .map_err(|e| TransactionSubmitterError::DecodeError(e.to_string()))?;
213        
214        let transaction: Transaction = bincode::deserialize(&tx_bytes)
215            .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to deserialize: {}", e)))?;
216
217        // Submit transaction
218        let rpc_url = self.config.solana_rpc_url.clone();
219        let tx_bytes = bincode::serialize(&transaction)
220            .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to serialize: {}", e)))?;
221
222        let signature = tokio::task::spawn_blocking(move || {
223            let rpc_client = RpcClient::new(rpc_url);
224            let transaction: Transaction = bincode::deserialize(&tx_bytes)
225                .map_err(|e| TransactionSubmitterError::DecodeError(format!("Failed to deserialize: {}", e)))?;
226
227            // Submit transaction
228            let signature = rpc_client.send_and_confirm_transaction(&transaction)
229                .map_err(|e| TransactionSubmitterError::RpcError(format!("Failed to submit: {}", e)))?;
230
231            Ok::<String, TransactionSubmitterError>(signature.to_string())
232        }).await
233        .map_err(|e| TransactionSubmitterError::RpcError(format!("Blocking task failed: {}", e)))?
234        .map_err(|e| TransactionSubmitterError::RpcError(e.to_string()))?;
235
236        info!("✅ Transaction submitted: {}", signature);
237        Ok(signature)
238    }
