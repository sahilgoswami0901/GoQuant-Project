1//! **light-poseidon** is a [Poseidon](https://eprint.iacr.org/2019/458) hash
2//! implementation in Rust created for [Light Protocol](https://www.lightprotocol.com/).
3//!
4//! # Parameters
5//!
6//! The library provides pre-generated parameters over the BN254 curve, however
7//! it can work with any parameters provided as long as developers take care
8//! of generating the round constants.
9//!
10//! Parameters provided by the library are:
11//!
12//! * *x^5* S-boxes
13//! * width - *2 ≤ t ≤ 13*
14//! * inputs - *1 ≤ n ≤ 12*
15//! * 8 full rounds and partial rounds depending on *t*: *[56, 57, 56, 60, 60, 63, 64, 63, 60, 66, 60, 65]*
16//!
17//! The parameters can be generated with:
18//!
19//! ```bash
20//! cargo xtask generate-poseidon-parameters
21//! ````
22//!
23//! # Output type
24//!
25//! [`Poseidon`](crate::Poseidon) type implements two traits which serve the purpose
26//! of returning the calculated hash in different representations:
27//!
28//! * [`PoseidonBytesHasher`](crate::PoseidonBytesHasher) with the
29//!   `hash_bytes_be` and `hash_bytes_le` methods which returns a byte array.
30//! * [`PoseidonHasher`](crate::PoseidonHasher) with the `hash` method which returns
31//!   [`ark_ff::PrimeField`](ark_ff::PrimeField). Might be useful if you want
32//!   to immediately process the result with an another library which works with
33//!   [`ark_ff::PrimeField`](ark_ff::PrimeField) types.
34//!
35//! # Examples
36//!
37//! Example with two simple big-endian byte inputs (converted to field elements)
38//! and BN254-based parameters provided by the library, with
39//! [`PoseidonBytesHasher`](crate::PoseidonHasher) trait and a byte array
40//! result:
41//!
42//! ```rust
43//! use light_poseidon::{Poseidon, PoseidonBytesHasher, parameters::bn254_x5};
44//! use ark_bn254::Fr;
45//! use ark_ff::{BigInteger, PrimeField};
46//!
47//! let mut poseidon = Poseidon::<Fr>::new_circom(2).unwrap();
48//!
49//! let hash = poseidon.hash_bytes_be(&[&[1u8; 32], &[2u8; 32]]).unwrap();
50//!
51//! println!("{:?}", hash);
52//! // Should print:
53//! // [
54//! //     13, 84, 225, 147, 143, 138, 140, 28, 125, 235, 94, 3, 85, 242, 99, 25, 32, 123, 132,
55//! //     254, 156, 162, 206, 27, 38, 231, 53, 200, 41, 130, 25, 144
56//! // ]
57//! ```
58//!
59//! With [`PoseidonHasher`](crate::PoseidonHasher) trait and
60//! [`ark_ff::PrimeField`](ark_ff::PrimeField) result:
61//!
62//! ```rust
63//! use light_poseidon::{Poseidon, PoseidonHasher, parameters::bn254_x5};
64//! use ark_bn254::Fr;
65//! use ark_ff::{BigInteger, PrimeField};
66//!
67//! let mut poseidon = Poseidon::<Fr>::new_circom(2).unwrap();
68//!
69//! let input1 = Fr::from_be_bytes_mod_order(&[1u8; 32]);
70//! let input2 = Fr::from_be_bytes_mod_order(&[2u8; 32]);
71//!
72//! let hash = poseidon.hash(&[input1, input2]).unwrap();
73//!
74//! // Do something with `hash`.
75//! ```
76//!
77//! # Implementation
78//!
79//! The implementation is compatible with the
80//! [original SageMath implementation](https://extgit.iaik.tugraz.at/krypto/hadeshash/-/tree/master/),
81//! but it was also inspired by the following ones:
82//!
83//! * [circomlibjs](https://github.com/iden3/circomlibjs)
84//! * [zero-knowledge-gadgets](https://github.com/webb-tools/zero-knowledge-gadgets)
85//!

144#[derive(Error, Debug, PartialEq)]
145pub enum PoseidonError {
146    #[error("Invalid number of inputs: {inputs}. Maximum allowed is {max_limit} ({width} - 1).")]
147    InvalidNumberOfInputs {
148        inputs: usize,
149        max_limit: usize,
150        width: usize,
151    },
152    #[error("Input is an empty slice.")]
153    EmptyInput,
154    #[error("Invalid length of the input: {len}. The length matching the modulus of the prime field is: {modulus_bytes_len}.")]
155    InvalidInputLength {
156        len: usize,
157        modulus_bytes_len: usize,
158    },
159    #[error("Failed to convert bytes {bytes:?} into a prime field element")]
160    BytesToPrimeFieldElement { bytes: Vec<u8> },
161    #[error("Input is larger than the modulus of the prime field.")]
162    InputLargerThanModulus,
163    #[error("Failed to convert a vector of bytes into an array.")]
164    VecToArray,
165    #[error("Failed to convert the number of inputs from u64 to u8.")]
166    U64Tou8,
167    #[error("Failed to convert bytes to BigInt")]
168    BytesToBigInt,
169    #[error("Invalid width: {width}. Choose a width between 2 and 16 for 1 to 15 inputs.")]
170    InvalidWidthCircom { width: usize, max_limit: usize },
171}

173/// Parameters for the Poseidon hash algorithm.
174pub struct PoseidonParameters<F: PrimeField> {
175    /// Round constants.
176    pub ark: Vec<F>,
177    /// MDS matrix.
178    pub mds: Vec<Vec<F>>,
179    /// Number of full rounds (where S-box is applied to all elements of the
180    /// state).
181    pub full_rounds: usize,
182    /// Number of partial rounds (where S-box is applied only to the first
183    /// element of the state).
184    pub partial_rounds: usize,
185    /// Number of prime fields in the state.
186    pub width: usize,
187    /// Exponential used in S-box to power elements of the state.
188    pub alpha: u64,
189}
190
191impl<F: PrimeField> PoseidonParameters<F> {
192    pub fn new(
193        ark: Vec<F>,
194        mds: Vec<Vec<F>>,
195        full_rounds: usize,
196        partial_rounds: usize,
197        width: usize,
198        alpha: u64,
199    ) -> Self {
200        Self {
201            ark,
202            mds,
203            full_rounds,
204            partial_rounds,
205            width,
206            alpha,
207        }
208    }

217    /// Example with two simple big-endian byte inputs (converted to prime
218    /// fields) and BN254-based parameters provided by the library.
219    ///
220    /// ```rust

234}

236pub trait PoseidonBytesHasher {
237    /// Calculates a Poseidon hash for the given input of big-endian byte slices
238    /// and returns the result as a byte array.
239    ///
240    /// # Examples
241    ///
242    /// Example with two simple big-endian byte inputs and BN254-based
243    /// parameters provided by the library.
244    ///
245    /// ```rust
246    /// use light_poseidon::{Poseidon, PoseidonBytesHasher, parameters::bn254_x5};
247    /// use ark_bn254::Fr;
248    /// use ark_ff::{BigInteger, PrimeField};
249    ///
250    /// let mut poseidon = Poseidon::<Fr>::new_circom(2).unwrap();
251    ///
252    /// let hash = poseidon.hash_bytes_be(&[&[1u8; 32], &[2u8; 32]]).unwrap();
253    ///
254    /// println!("{:?}", hash);
255    /// // Should print:
256    /// // [
257    /// //     13, 84, 225, 147, 143, 138, 140, 28, 125, 235, 94, 3, 85, 242, 99, 25, 32, 123, 132,
258    /// //     254, 156, 162, 206, 27, 38, 231, 53, 200, 41, 130, 25, 144
259    /// // ]
260    /// ```
261    ///
262    /// # Safety
263    ///   
264    /// Unlike the
265    /// [`PrimeField::from_be_bytes_mod_order`](ark_ff::PrimeField::from_be_bytes_mod_order)
266    /// and [`Field::from_random_bytes`](ark_ff::Field::from_random_bytes)
267    /// methods, this function ensures that the input byte slice's length exactly matches
268    /// the modulus size of the prime field. If the size doesn't match, an error is returned.
269    ///
270    /// This strict check is designed to prevent unexpected behaviors and collisions
271    /// that might occur when using `from_be_bytes_mod_order` or `from_random_bytes`,
272    /// which simply take a subslice of the input if it's too large, potentially
273    /// leading to collisions.
274    fn hash_bytes_be(&mut self, inputs: &[&[u8]]) -> Result<[u8; HASH_LEN], PoseidonError>;
275    /// Calculates a Poseidon hash for the given input of little-endian byte
276    /// slices and returns the result as a byte array.
277    ///
278    /// # Examples
279    ///
280    /// Example with two simple little-endian byte inputs and BN254-based
281    /// parameters provided by the library.
282    ///
283    /// ```rust
284    /// use light_poseidon::{Poseidon, PoseidonBytesHasher, parameters::bn254_x5};
285    /// use ark_bn254::Fr;
286    /// use ark_ff::{BigInteger, PrimeField};
287    ///
288    /// let mut poseidon = Poseidon::<Fr>::new_circom(2).unwrap();
289    ///
290    /// let hash = poseidon.hash_bytes_le(&[&[1u8; 32], &[2u8; 32]]).unwrap();
291    ///
292    /// println!("{:?}", hash);
293    /// // Should print:
294    /// // [
295    /// //     144, 25, 130, 41, 200, 53, 231, 38, 27, 206, 162, 156, 254, 132, 123, 32, 25, 99, 242,
296    /// //     85, 3, 94, 235, 125, 28, 140, 138, 143, 147, 225, 84, 13
297    /// // ]
298    /// ```
299    ///
300    /// # Safety
301    ///
302    /// Unlike the
303    /// [`PrimeField::from_le_bytes_mod_order`](ark_ff::PrimeField::from_le_bytes_mod_order)
304    /// and [`Field::from_random_bytes`](ark_ff::Field::from_random_bytes)
305    /// methods, this function ensures that the input byte slice's length exactly matches
306    /// the modulus size of the prime field. If the size doesn't match, an error is returned.
307    ///
308    /// This strict check is designed to prevent unexpected behaviors and collisions
309    /// that might occur when using `from_be_bytes_mod_order` or `from_random_bytes`,
310    /// which simply take a subslice of the input if it's too large, potentially
311    /// leading to collisions.
312    fn hash_bytes_le(&mut self, inputs: &[&[u8]]) -> Result<[u8; HASH_LEN], PoseidonError>;
313}

315/// A stateful sponge performing Poseidon hash computation.
316pub struct Poseidon<F: PrimeField> {
317    params: PoseidonParameters<F>,
318    domain_tag: F,
319    state: Vec<F>,
320}
321
322impl<F: PrimeField> Poseidon<F> {
323    /// Returns a new Poseidon hasher based on the given parameters.
324    ///
325    /// Optionally, a domain tag can be provided. If it is not provided, it
326    /// will be set to zero.
327    pub fn new(params: PoseidonParameters<F>) -> Self {
328        Self::with_domain_tag(params, F::zero())
329    }
330
331    fn with_domain_tag(params: PoseidonParameters<F>, domain_tag: F) -> Self {
332        let width = params.width;
333        Self {
334            domain_tag,
335            params,
336            state: Vec::with_capacity(width),
337        }
338    }
339
340    #[inline(always)]
341    fn apply_ark(&mut self, round: usize) {
342        self.state.iter_mut().enumerate().for_each(|(i, a)| {
343            let c = self.params.ark[round * self.params.width + i];
344            *a += c;
345        });
346    }
347
348    #[inline(always)]
349    fn apply_sbox_full(&mut self) {
350        self.state.iter_mut().for_each(|a| {
351            *a = a.pow([self.params.alpha]);
352        });
353    }
354
355    #[inline(always)]
356    fn apply_sbox_partial(&mut self) {
357        self.state[0] = self.state[0].pow([self.params.alpha]);
358    }
359
360    #[inline(always)]
361    fn apply_mds(&mut self) {
362        self.state = self
363            .state
364            .iter()
365            .enumerate()
366            .map(|(i, _)| {
367                self.state
368                    .iter()
369                    .enumerate()
370                    .fold(F::zero(), |acc, (j, a)| acc + *a * self.params.mds[i][j])
371            })
372            .collect();
373    }
374}
375
376impl<F: PrimeField> PoseidonHasher<F> for Poseidon<F> {
377    fn hash(&mut self, inputs: &[F]) -> Result<F, PoseidonError> {
378        if inputs.len() != self.params.width - 1 {
379            return Err(PoseidonError::InvalidNumberOfInputs {
380                inputs: inputs.len(),
381                max_limit: self.params.width - 1,
382                width: self.params.width,
383            });
384        }
385
386        self.state.push(self.domain_tag);
387
388        for input in inputs {
389            self.state.push(*input);
390        }
391
392        let all_rounds = self.params.full_rounds + self.params.partial_rounds;
393        let half_rounds = self.params.full_rounds / 2;
394
395        // full rounds + partial rounds
396        for round in 0..half_rounds {
397            self.apply_ark(round);
398            self.apply_sbox_full();
399            self.apply_mds();
400        }
401
402        for round in half_rounds..half_rounds + self.params.partial_rounds {
403            self.apply_ark(round);
404            self.apply_sbox_partial();
405            self.apply_mds();
406        }
407
408        for round in half_rounds + self.params.partial_rounds..all_rounds {
409            self.apply_ark(round);
410            self.apply_sbox_full();
411            self.apply_mds();
412        }
413
414        let result = self.state[0];
415        self.state.clear();
416        Ok(result)
417    }
418}

420macro_rules! impl_hash_bytes {
421    ($fn_name:ident, $bytes_to_prime_field_element_fn:ident, $to_bytes_fn:ident) => {
422        fn $fn_name(&mut self, inputs: &[&[u8]]) -> Result<[u8; HASH_LEN], PoseidonError> {
423            let inputs: Result<Vec<_>, _> = inputs
424                .iter()
425                .map(|input| validate_bytes_length::<F>(input))
426                .collect();
427            let inputs = inputs?;
428            let inputs: Result<Vec<_>, _> = inputs
429                .iter()
430                .map(|input| $bytes_to_prime_field_element_fn(input))
431                .collect();
432            let inputs = inputs?;
433            let hash = self.hash(&inputs)?;
434
435            hash.into_bigint()
436                .$to_bytes_fn()
437                .try_into()
438                .map_err(|_| PoseidonError::VecToArray)
439        }
440    };
441}

448/// Checks whether a slice of bytes is not empty or its length does not exceed
449/// the modulus size od the prime field. If it does, an error is returned.
450///
451/// # Safety
452///
453/// [`PrimeField::from_be_bytes_mod_order`](ark_ff::PrimeField::from_be_bytes_mod_order)
454/// just takes a subslice of the input if it's too large, potentially leading
455/// to collisions. The purpose of this function is to prevent them by returning
456/// and error. It should be always used before converting byte slices to
457/// prime field elements.
458pub fn validate_bytes_length<F>(input: &[u8]) -> Result<&[u8], PoseidonError>
459where
460    F: PrimeField,
461{
462    let modulus_bytes_len = ((F::MODULUS_BIT_SIZE + 7) / 8) as usize;
463    if input.is_empty() {
464        return Err(PoseidonError::EmptyInput);
465    }
466    if input.len() > modulus_bytes_len {
467        return Err(PoseidonError::InvalidInputLength {
468            len: input.len(),
469            modulus_bytes_len,
470        });
471    }
472    Ok(input)
473}

475macro_rules! impl_bytes_to_prime_field_element {
476    ($name:ident, $from_bytes_method:ident, $endianess:expr) => {
477        #[doc = "Converts a slice of "]
478        #[doc = $endianess]
479        #[doc = "-endian bytes into a prime field element, \
480                 represented by the [`ark_ff::PrimeField`](ark_ff::PrimeField) trait."]
481        pub fn $name<F>(input: &[u8]) -> Result<F, PoseidonError>
482        where
483            F: PrimeField,
484        {
485            let element = num_bigint::BigUint::$from_bytes_method(input);
486            let element = F::BigInt::try_from(element).map_err(|_| PoseidonError::BytesToBigInt)?;
487
488            // In theory, `F::from_bigint` should also perform a check whether input is
489            // larger than modulus (and return `None` if it is), but it's not reliable...
490            // To be sure, we check it ourselves.
491            if element >= F::MODULUS {
492                return Err(PoseidonError::InputLargerThanModulus);
493            }
494            let element = F::from_bigint(element).ok_or(PoseidonError::InputLargerThanModulus)?;
495
496            Ok(element)
497        }
498    };
499}

504impl<F: PrimeField> Poseidon<F> {
505    pub fn new_circom(nr_inputs: usize) -> Result<Poseidon<Fr>, PoseidonError> {
506        Self::with_domain_tag_circom(nr_inputs, Fr::zero())
507    }
508
509    pub fn with_domain_tag_circom(
510        nr_inputs: usize,
511        domain_tag: Fr,
512    ) -> Result<Poseidon<Fr>, PoseidonError> {
513        let width = nr_inputs + 1;
514        if width > MAX_X5_LEN {
515            return Err(PoseidonError::InvalidWidthCircom {
516                width,
517                max_limit: MAX_X5_LEN,
518            });
519        }
520
521        let params = crate::parameters::bn254_x5::get_poseidon_parameters::<Fr>(
522            (width).try_into().map_err(|_| PoseidonError::U64Tou8)?,
523        )?;
524        Ok(Poseidon::<Fr>::with_domain_tag(params, domain_tag))
525    }
