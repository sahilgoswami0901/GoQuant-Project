182    fn invoke<T, F: std::future::Future<Output = T>>(&self, f: F) -> T {
183        // `block_on()` panics if called within an asynchronous execution context. Whereas
184        // `block_in_place()` only panics if called from a current_thread runtime, which is the
185        // lesser evil.
186        tokio::task::block_in_place(move || self.rpc_client.runtime().block_on(f))
187    }

190// 48 chosen because it's unlikely that 12 leaders in a row will miss their slots
191const MAX_SLOT_SKIP_DISTANCE: u64 = 48;
192
193#[derive(Clone, Debug)]
194pub(crate) struct RecentLeaderSlots(Arc<RwLock<VecDeque<Slot>>>);
195impl RecentLeaderSlots {
196    pub(crate) fn new(current_slot: Slot) -> Self {
197        let mut recent_slots = VecDeque::new();
198        recent_slots.push_back(current_slot);
199        Self(Arc::new(RwLock::new(recent_slots)))
200    }
201
202    pub(crate) fn record_slot(&self, current_slot: Slot) {
203        let mut recent_slots = self.0.write().unwrap();
204        recent_slots.push_back(current_slot);
205        // 12 recent slots should be large enough to avoid a misbehaving
206        // validator from affecting the median recent slot
207        while recent_slots.len() > 12 {
208            recent_slots.pop_front();
209        }
210    }
211
212    // Estimate the current slot from recent slot notifications.
213    pub(crate) fn estimated_current_slot(&self) -> Slot {
214        let mut recent_slots: Vec<Slot> = self.0.read().unwrap().iter().cloned().collect();
215        assert!(!recent_slots.is_empty());
216        recent_slots.sort_unstable();
217
218        // Validators can broadcast invalid blocks that are far in the future
219        // so check if the current slot is in line with the recent progression.
220        let max_index = recent_slots.len() - 1;
221        let median_index = max_index / 2;
222        let median_recent_slot = recent_slots[median_index];
223        let expected_current_slot = median_recent_slot + (max_index - median_index) as u64;
224        let max_reasonable_current_slot = expected_current_slot + MAX_SLOT_SKIP_DISTANCE;
225
226        // Return the highest slot that doesn't exceed what we believe is a
227        // reasonable slot.
228        recent_slots
229            .into_iter()
230            .rev()
231            .find(|slot| *slot <= max_reasonable_current_slot)
232            .unwrap()
233    }
234}
