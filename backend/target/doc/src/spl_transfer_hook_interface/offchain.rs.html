15/// Offchain helper to get all additional required account metas for a mint
16///
17/// To be client-agnostic and to avoid pulling in the full solana-sdk, this
18/// simply takes a function that will return its data as `Future<Vec<u8>>` for
19/// the given address. Can be called in the following way:
20///
21/// ```rust,ignore
22/// use futures_util::TryFutureExt;
23/// use solana_client::nonblocking::rpc_client::RpcClient;
24/// use solana_program::pubkey::Pubkey;
25///
26/// let program_id = Pubkey::new_unique();
27/// let mint = Pubkey::new_unique();
28/// let client = RpcClient::new_mock("succeeds".to_string());
29/// let mut account_metas = vec![];
30///
31/// get_extra_account_metas(
32///     &mut account_metas,
33///     |address| self.client.get_account(&address).map_ok(|opt| opt.map(|acc| acc.data)),
34///     &mint,
35///     &program_id,
36/// ).await?;
37/// ```
38pub async fn resolve_extra_account_metas<F, Fut>(
39    instruction: &mut Instruction,
40    fetch_account_data_fn: F,
41    mint: &Pubkey,
42    permissioned_transfer_program_id: &Pubkey,
43) -> Result<(), AccountFetchError>
44where
45    F: Fn(Pubkey) -> Fut,
46    Fut: Future<Output = AccountDataResult>,
47{
48    let validation_address =
49        get_extra_account_metas_address(mint, permissioned_transfer_program_id);
50    let validation_account_data = fetch_account_data_fn(validation_address)
51        .await?
52        .ok_or(ProgramError::InvalidAccountData)?;
53    ExtraAccountMetaList::add_to_instruction::<ExecuteInstruction, _, _>(
54        instruction,
55        fetch_account_data_fn,
56        &validation_account_data,
57    )
58    .await?;
59    // The onchain helpers pull out the required accounts from an opaque
60    // slice by pubkey, so the order doesn't matter here!
61    instruction.accounts.push(AccountMeta::new_readonly(
62        *permissioned_transfer_program_id,
63        false,
64    ));
65    instruction
66        .accounts
67        .push(AccountMeta::new_readonly(validation_address, false));
68
69    Ok(())
70}
