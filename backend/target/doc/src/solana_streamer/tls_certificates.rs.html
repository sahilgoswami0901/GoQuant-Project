9pub fn new_self_signed_tls_certificate(
10    keypair: &Keypair,
11    san: IpAddr,
12) -> Result<(rustls::Certificate, rustls::PrivateKey), RcgenError> {
13    // TODO(terorie): Is it safe to sign the TLS cert with the identity private key?
14
15    // Unfortunately, rcgen does not accept a "raw" Ed25519 key.
16    // We have to convert it to DER and pass it to the library.
17
18    // Convert private key into PKCS#8 v1 object.
19    // RFC 8410, Section 7: Private Key Format
20    // https://datatracker.ietf.org/doc/html/rfc8410#section-
21
22    // from https://datatracker.ietf.org/doc/html/rfc8410#section-3
23    const ED25519_IDENTIFIER: [u32; 4] = [1, 3, 101, 112];
24    let mut private_key = Vec::<u8>::with_capacity(34);
25    private_key.extend_from_slice(&[0x04, 0x20]); // ASN.1 OCTET STRING
26    private_key.extend_from_slice(keypair.secret().as_bytes());
27    let key_pkcs8 = pkcs8::PrivateKeyInfo {
28        algorithm: AlgorithmIdentifier {
29            oid: ObjectIdentifier::from_arcs(&ED25519_IDENTIFIER).expect("Failed to convert OID"),
30            parameters: None,
31        },
32        private_key: &private_key,
33        public_key: None,
34    };
35    let key_pkcs8_der = key_pkcs8
36        .to_der()
37        .expect("Failed to convert keypair to DER")
38        .to_der();
39
40    let rcgen_keypair = rcgen::KeyPair::from_der(&key_pkcs8_der)?;
41
42    let mut cert_params = CertificateParams::default();
43    cert_params.subject_alt_names = vec![SanType::IpAddress(san)];
44    cert_params.alg = &rcgen::PKCS_ED25519;
45    cert_params.key_pair = Some(rcgen_keypair);
46    cert_params.distinguished_name = DistinguishedName::new();
47    cert_params
48        .distinguished_name
49        .push(DnType::CommonName, "Solana node");
50
51    let cert = rcgen::Certificate::from_params(cert_params)?;
52    let cert_der = cert.serialize_der().unwrap();
53    let priv_key = cert.serialize_private_key_der();
54    let priv_key = rustls::PrivateKey(priv_key);
55    Ok((rustls::Certificate(cert_der), priv_key))
56}

58pub fn get_pubkey_from_tls_certificate(der_cert: &rustls::Certificate) -> Option<Pubkey> {
59    let (_, cert) = X509Certificate::from_der(der_cert.as_ref()).ok()?;
60    match cert.public_key().parsed().ok()? {
61        PublicKey::Unknown(key) => Pubkey::try_from(key).ok(),
62        _ => None,
63    }
64}
