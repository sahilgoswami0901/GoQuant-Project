11/**
12Safe wrapper around dynamic link library handle.
13
14Methods of `Library` return only types that make the library borrowed. Therefore the problem with
15dangling symbols is prevented.
16
17**Note:**: It is recommended that you use certain methods in certain situations:
18
19* `symbol()` - for obtaining functions and pointers (but only if you can't use references
20    instead of pointers and you do not accept null value of a pointer).
21* `reference()` and `reference_mut()` - for obtaining access to
22    statically allocated objects - either constant or mutable.
23* `ptr_or_null()` and `ptr_or_null_mut()` - for obtaining pointers if you accept null values of
24pointers (in 99% of cases you should rather use previously mentioned methods).
25
26#Example
27
28```no_run
29use dlopen2::symbor::Library;
30
31fn main(){
32    let lib = Library::open("libexample.dylib").unwrap();
33    let fun = unsafe{lib.symbol::<unsafe extern "C" fn()>("function")}.unwrap();
34    unsafe{fun()};
35    let glob_val: &mut u32 = unsafe{lib.reference_mut("glob_val")}.unwrap();
36    *glob_val = 42;
37    let ptr_or_null = unsafe{lib.ptr_or_null::<()>("void_ptr")}.unwrap();
38    println!("Pointer is null: {}", ptr_or_null.is_null());
39}
40```
41*/
42pub struct Library {
43    lib: RawLib,
44}
45
46impl Library {
47    ///Open dynamic link library using provided file name or path.
48    pub fn open<S>(name: S) -> Result<Library, Error>
49    where
50        S: AsRef<OsStr>,
51    {
52        Ok(Library {
53            lib: RawLib::open(name)?,
54        })
55    }
56
57    /// Open the program itself as library.
58    ///
59    /// This allows a shared library to load symbols of the program it was
60    /// loaded into.
61    pub fn open_self() -> Result<Library, Error> {
62        Ok(Library {
63            lib: RawLib::open_self()?,
64        })
65    }
66
67    /// Obtain a symbol from library.
68    ///
69    /// This method is the most general one and allows obtaining basically everything assuming
70    /// that the value of the given symbol cannot be null (use `ptr_or_null()` for this case).
71    /// However the `reference()` and `reference_mut()` methods return a native reference and they
72    /// are more programmer friendly when you try accessing statically allocated data in
73    /// the library.
74    pub unsafe fn symbol<T>(&self, name: &str) -> Result<Symbol<T>, Error> {
75        Ok(Symbol::new(self.lib.symbol(name)?))
76    }
77
78    ///Equivalent of the `symbol()` method but takes `CStr` as a argument.
79    pub unsafe fn symbol_cstr<T>(&self, name: &CStr) -> Result<Symbol<T>, Error> {
80        Ok(Symbol::new(self.lib.symbol_cstr(name)?))
81    }
82
83    ///Obtain a const pointer from library.
84    ///
85    /// **Note:** This method is only recommended for data
86    /// that can't be accessed as a reference and that can have a null pointer value
87    /// (so not in 99% of cases).
88    pub unsafe fn ptr_or_null<T>(&self, name: &str) -> Result<PtrOrNull<T>, Error> {
89        let cname = CString::new(name)?;
90        self.ptr_or_null_cstr(cname.as_ref())
91    }
92
93    ///Equivalent of the `pointer()` method but takes `CStr` as a argument.
94    pub unsafe fn ptr_or_null_cstr<T>(&self, name: &CStr) -> Result<PtrOrNull<T>, Error> {
95        let raw_ptr = match self.lib.symbol_cstr(name) {
96            Ok(val) => val,
97            Err(err) => match err {
98                Error::NullSymbol => null(),
99                _ => return Err(err),
100            },
101        };
102        Ok(PtrOrNull::new(raw_ptr))
103    }
104
105    ///Obtain a mutable pointer from library.
106    ///
107    /// **Note:** This method is only recommended for data
108    /// that can't be accessed as a reference and that can have a null pointer value
109    /// (so not in 99% of cases).
110    pub unsafe fn ptr_or_null_mut<T>(&self, name: &str) -> Result<PtrOrNullMut<T>, Error> {
111        let cname = CString::new(name)?;
112        self.ptr_or_null_mut_cstr(cname.as_ref())
113    }
114
115    ///Equivalent of the `pointer_mut()` method but takes `CStr` as a argument.
116    pub unsafe fn ptr_or_null_mut_cstr<T>(&self, name: &CStr) -> Result<PtrOrNullMut<T>, Error> {
117        let raw_ptr = match self.lib.symbol_cstr(name) {
118            Ok(val) => val,
119            Err(err) => match err {
120                Error::NullSymbol => null_mut(),
121                _ => return Err(err),
122            },
123        };
124        Ok(PtrOrNullMut::new(raw_ptr))
125    }
126
127    ///Obtain const reference to statically allocated data in the library.
128    pub unsafe fn reference<T>(&self, name: &str) -> Result<&T, Error> {
129        self.lib.symbol(name)
130    }
131
132    ///Equivalent of the `reference()` method but takes `CStr` as a argument.
133    pub unsafe fn reference_cstr<T>(&self, name: &CStr) -> Result<&T, Error> {
134        self.lib.symbol_cstr(name)
135    }
136
137    ///Obtain mutable reference to statically allocated data in the library.
138    pub unsafe fn reference_mut<T>(&self, name: &str) -> Result<&mut T, Error> {
139        self.lib.symbol(name)
140    }
141
142    ///Equivalent of the `reference_mut()` method but takes `CStr` as a argument.
143    pub unsafe fn reference_mut_cstr<T>(&self, name: &CStr) -> Result<&mut T, Error> {
144        self.lib.symbol_cstr(name)
145    }
146}
