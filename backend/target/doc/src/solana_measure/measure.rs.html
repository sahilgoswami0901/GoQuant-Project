9#[derive(Debug)]
10pub struct Measure {
11    name: &'static str,
12    start: Instant,
13    duration: u64,
14}
15
16impl Measure {
17    pub fn start(name: &'static str) -> Self {
18        Self {
19            name,
20            start: Instant::now(),
21            duration: 0,
22        }
23    }
24
25    pub fn stop(&mut self) {
26        self.duration = duration_as_ns(&self.start.elapsed());
27    }
28
29    pub fn as_ns(&self) -> u64 {
30        self.duration
31    }
32
33    pub fn as_us(&self) -> u64 {
34        self.duration / 1000
35    }
36
37    pub fn as_ms(&self) -> u64 {
38        self.duration / (1000 * 1000)
39    }
40
41    pub fn as_s(&self) -> f32 {
42        self.duration as f32 / (1000.0f32 * 1000.0f32 * 1000.0f32)
43    }
44
45    pub fn as_duration(&self) -> Duration {
46        Duration::from_nanos(self.as_ns())
47    }
48
49    pub fn end_as_ns(self) -> u64 {
50        duration_as_ns(&self.start.elapsed())
51    }
52
53    pub fn end_as_us(self) -> u64 {
54        duration_as_us(&self.start.elapsed())
55    }
56
57    pub fn end_as_ms(self) -> u64 {
58        duration_as_ms(&self.start.elapsed())
59    }
60
61    pub fn end_as_s(self) -> f32 {
62        duration_as_s(&self.start.elapsed())
63    }
64
65    pub fn end_as_duration(self) -> Duration {
66        self.start.elapsed()
67    }
68}

70impl fmt::Display for Measure {
71    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
72        if self.duration == 0 {
73            write!(f, "{} running", self.name)
74        } else if self.as_us() < 1 {
75            write!(f, "{} took {}ns", self.name, self.duration)
76        } else if self.as_ms() < 1 {
77            write!(f, "{} took {}us", self.name, self.as_us())
78        } else if self.as_s() < 1. {
79            write!(f, "{} took {}ms", self.name, self.as_ms())
80        } else {
81            write!(f, "{} took {:.1}s", self.name, self.as_s())
82        }
83    }
84}

99    #[test]

101        let test_duration = Duration::from_millis(100);
102        let measure = Measure {
103            name: "test",
104            start: Instant::now(),
105            duration: test_duration.as_nanos() as u64,
106        };
107
108        assert!(f32::abs(measure.as_s() - 0.1f32) <= f32::EPSILON);
109        assert_eq!(measure.as_ms(), 100);
110        assert_eq!(measure.as_us(), 100_000);
111        assert_eq!(measure.as_ns(), 100_000_000);
112        assert_eq!(measure.as_duration(), test_duration);
