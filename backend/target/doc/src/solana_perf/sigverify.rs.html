110/// Returns true if the signatrue on the packet verifies.
111/// Caller must do packet.set_discard(true) if this returns false.
112#[must_use]
113fn verify_packet(packet: &mut Packet, reject_non_vote: bool) -> bool {
114    // If this packet was already marked as discard, drop it
115    if packet.meta().discard() {
116        return false;
117    }
118
119    let packet_offsets = get_packet_offsets(packet, 0, reject_non_vote);
120    let mut sig_start = packet_offsets.sig_start as usize;
121    let mut pubkey_start = packet_offsets.pubkey_start as usize;
122    let msg_start = packet_offsets.msg_start as usize;
123
124    if packet_offsets.sig_len == 0 {
125        return false;
126    }
127
128    if packet.meta().size <= msg_start {
129        return false;
130    }
131
132    for _ in 0..packet_offsets.sig_len {
133        let pubkey_end = pubkey_start.saturating_add(size_of::<Pubkey>());
134        let Some(sig_end) = sig_start.checked_add(size_of::<Signature>()) else {
135            return false;
136        };
137        let Some(Ok(signature)) = packet.data(sig_start..sig_end).map(Signature::try_from) else {
138            return false;
139        };
140        let Some(pubkey) = packet.data(pubkey_start..pubkey_end) else {
141            return false;
142        };
143        let Some(message) = packet.data(msg_start..) else {
144            return false;
145        };
146        if !signature.verify(pubkey, message) {
147            return false;
148        }
149        pubkey_start = pubkey_end;
150        sig_start = sig_end;
151    }
152    true
153}

187// internal function to be unit-tested; should be used only by get_packet_offsets
188fn do_get_packet_offsets(
189    packet: &Packet,
190    current_offset: usize,
191) -> Result<PacketOffsets, PacketError> {
192    // should have at least 1 signature and sig lengths
193    let _ = 1usize
194        .checked_add(size_of::<Signature>())
195        .filter(|v| *v <= packet.meta().size)
196        .ok_or(PacketError::InvalidLen)?;
197
198    // read the length of Transaction.signatures (serialized with short_vec)
199    let (sig_len_untrusted, sig_size) = packet
200        .data(..)
201        .and_then(|bytes| decode_shortu16_len(bytes).ok())
202        .ok_or(PacketError::InvalidShortVec)?;
203    // Using msg_start_offset which is based on sig_len_untrusted introduces uncertainty.
204    // Ultimately, the actual sigverify will determine the uncertainty.
205    let msg_start_offset = sig_len_untrusted
206        .checked_mul(size_of::<Signature>())
207        .and_then(|v| v.checked_add(sig_size))
208        .ok_or(PacketError::InvalidLen)?;
209
210    // Determine the start of the message header by checking the message prefix bit.
211    let msg_header_offset = {
212        // Packet should have data for prefix bit
213        if msg_start_offset >= packet.meta().size {
214            return Err(PacketError::InvalidSignatureLen);
215        }
216
217        // next byte indicates if the transaction is versioned. If the top bit
218        // is set, the remaining bits encode a version number. If the top bit is
219        // not set, this byte is the first byte of the message header.
220        let message_prefix = *packet
221            .data(msg_start_offset)
222            .ok_or(PacketError::InvalidSignatureLen)?;
223        if message_prefix & MESSAGE_VERSION_PREFIX != 0 {
224            let version = message_prefix & !MESSAGE_VERSION_PREFIX;
225            match version {
226                0 => {
227                    // header begins immediately after prefix byte
228                    msg_start_offset
229                        .checked_add(1)
230                        .ok_or(PacketError::InvalidLen)?
231                }
232
233                // currently only v0 is supported
234                _ => return Err(PacketError::UnsupportedVersion),
235            }
236        } else {
237            msg_start_offset
238        }
239    };
240
241    let msg_header_offset_plus_one = msg_header_offset
242        .checked_add(1)
243        .ok_or(PacketError::InvalidLen)?;
244
245    // Packet should have data at least for MessageHeader and 1 byte for Message.account_keys.len
246    let _ = msg_header_offset_plus_one
247        .checked_add(MESSAGE_HEADER_LENGTH)
248        .filter(|v| *v <= packet.meta().size)
249        .ok_or(PacketError::InvalidSignatureLen)?;
250
251    // read MessageHeader.num_required_signatures (serialized with u8)
252    let sig_len_maybe_trusted = *packet
253        .data(msg_header_offset)
254        .ok_or(PacketError::InvalidSignatureLen)?;
255    let message_account_keys_len_offset = msg_header_offset
256        .checked_add(MESSAGE_HEADER_LENGTH)
257        .ok_or(PacketError::InvalidSignatureLen)?;
258
259    // This reads and compares the MessageHeader num_required_signatures and
260    // num_readonly_signed_accounts bytes. If num_required_signatures is not larger than
261    // num_readonly_signed_accounts, the first account is not debitable, and cannot be charged
262    // required transaction fees.
263    let readonly_signer_offset = msg_header_offset_plus_one;
264    if sig_len_maybe_trusted
265        <= *packet
266            .data(readonly_signer_offset)
267            .ok_or(PacketError::InvalidSignatureLen)?
268    {
269        return Err(PacketError::PayerNotWritable);
270    }
271
272    if usize::from(sig_len_maybe_trusted) != sig_len_untrusted {
273        return Err(PacketError::MismatchSignatureLen);
274    }
275
276    // read the length of Message.account_keys (serialized with short_vec)
277    let (pubkey_len, pubkey_len_size) = packet
278        .data(message_account_keys_len_offset..)
279        .and_then(|bytes| decode_shortu16_len(bytes).ok())
280        .ok_or(PacketError::InvalidShortVec)?;
281    let pubkey_start = message_account_keys_len_offset
282        .checked_add(pubkey_len_size)
283        .ok_or(PacketError::InvalidPubkeyLen)?;
284
285    let _ = pubkey_len
286        .checked_mul(size_of::<Pubkey>())
287        .and_then(|v| v.checked_add(pubkey_start))
288        .filter(|v| *v <= packet.meta().size)
289        .ok_or(PacketError::InvalidPubkeyLen)?;
290
291    if pubkey_len < sig_len_untrusted {
292        return Err(PacketError::InvalidPubkeyLen);
293    }
294
295    let sig_start = current_offset
296        .checked_add(sig_size)
297        .ok_or(PacketError::InvalidLen)?;
298    let msg_start = current_offset
299        .checked_add(msg_start_offset)
300        .ok_or(PacketError::InvalidLen)?;
301    let pubkey_start = current_offset
302        .checked_add(pubkey_start)
303        .ok_or(PacketError::InvalidLen)?;
304
305    Ok(PacketOffsets::new(
306        u32::try_from(sig_len_untrusted)?,
307        u32::try_from(sig_start)?,
308        u32::try_from(msg_start)?,
309        u32::try_from(pubkey_start)?,
310        u32::try_from(pubkey_len)?,
311    ))
312}

314pub fn check_for_tracer_packet(packet: &mut Packet) -> bool {
315    let first_pubkey_start: usize = TRACER_KEY_OFFSET_IN_TRANSACTION;
316    let Some(first_pubkey_end) = first_pubkey_start.checked_add(size_of::<Pubkey>()) else {
317        return false;
318    };
319    // Check for tracer pubkey
320    match packet.data(first_pubkey_start..first_pubkey_end) {
321        Some(pubkey) if pubkey == TRACER_KEY.as_ref() => {
322            packet.meta_mut().set_tracer(true);
323            true
324        }
325        _ => false,
326    }
327}

329fn get_packet_offsets(
330    packet: &mut Packet,
331    current_offset: usize,
332    reject_non_vote: bool,
333) -> PacketOffsets {
334    let unsanitized_packet_offsets = do_get_packet_offsets(packet, current_offset);
335    if let Ok(offsets) = unsanitized_packet_offsets {
336        check_for_simple_vote_transaction(packet, &offsets, current_offset).ok();
337        if !reject_non_vote || packet.meta().is_simple_vote_tx() {
338            return offsets;
339        }
340    }
341    // force sigverify to fail by returning zeros
342    PacketOffsets::new(0, 0, 0, 0, 0)
343}

345fn check_for_simple_vote_transaction(
346    packet: &mut Packet,
347    packet_offsets: &PacketOffsets,
348    current_offset: usize,
349) -> Result<(), PacketError> {
350    // vote could have 1 or 2 sigs; zero sig has already been excluded at
351    // do_get_packet_offsets.
352    if packet_offsets.sig_len > 2 {
353        return Err(PacketError::InvalidSignatureLen);
354    }
355
356    // simple vote should only be legacy message
357    let msg_start = (packet_offsets.msg_start as usize)
358        .checked_sub(current_offset)
359        .ok_or(PacketError::InvalidLen)?;
360    let message_prefix = *packet.data(msg_start).ok_or(PacketError::InvalidLen)?;
361    if message_prefix & MESSAGE_VERSION_PREFIX != 0 {
362        return Ok(());
363    }
364
365    let pubkey_start = (packet_offsets.pubkey_start as usize)
366        .checked_sub(current_offset)
367        .ok_or(PacketError::InvalidLen)?;
368
369    let instructions_len_offset = (packet_offsets.pubkey_len as usize)
370        .checked_mul(size_of::<Pubkey>())
371        .and_then(|v| v.checked_add(pubkey_start))
372        .and_then(|v| v.checked_add(size_of::<Hash>()))
373        .ok_or(PacketError::InvalidLen)?;
374
375    // Packet should have at least 1 more byte for instructions.len
376    let _ = instructions_len_offset
377        .checked_add(1usize)
378        .filter(|v| *v <= packet.meta().size)
379        .ok_or(PacketError::InvalidLen)?;
380
381    let (instruction_len, instruction_len_size) = packet
382        .data(instructions_len_offset..)
383        .and_then(|bytes| decode_shortu16_len(bytes).ok())
384        .ok_or(PacketError::InvalidLen)?;
385    // skip if has more than 1 instruction
386    if instruction_len != 1 {
387        return Err(PacketError::InvalidProgramLen);
388    }
389
390    let instruction_start = instructions_len_offset
391        .checked_add(instruction_len_size)
392        .ok_or(PacketError::InvalidLen)?;
393
394    // Packet should have at least 1 more byte for one instructions_program_id
395    let _ = instruction_start
396        .checked_add(1usize)
397        .filter(|v| *v <= packet.meta().size)
398        .ok_or(PacketError::InvalidLen)?;
399
400    let instruction_program_id_index: usize = usize::from(
401        *packet
402            .data(instruction_start)
403            .ok_or(PacketError::InvalidLen)?,
404    );
405
406    if instruction_program_id_index >= packet_offsets.pubkey_len as usize {
407        return Err(PacketError::InvalidProgramIdIndex);
408    }
409
410    let instruction_program_id_start = instruction_program_id_index
411        .checked_mul(size_of::<Pubkey>())
412        .and_then(|v| v.checked_add(pubkey_start))
413        .ok_or(PacketError::InvalidLen)?;
414    let instruction_program_id_end = instruction_program_id_start
415        .checked_add(size_of::<Pubkey>())
416        .ok_or(PacketError::InvalidLen)?;
417
418    if packet
419        .data(instruction_program_id_start..instruction_program_id_end)
420        .ok_or(PacketError::InvalidLen)?
421        == solana_sdk::vote::program::id().as_ref()
422    {
423        packet.meta_mut().flags |= PacketFlags::SIMPLE_VOTE_TX;
424    }
425    Ok(())
426}

428pub fn generate_offsets(
429    batches: &mut [PacketBatch],
430    recycler: &Recycler<TxOffset>,
431    reject_non_vote: bool,
432) -> TxOffsets {
433    debug!("allocating..");
434    let mut signature_offsets: PinnedVec<_> = recycler.allocate("sig_offsets");
435    signature_offsets.set_pinnable();
436    let mut pubkey_offsets: PinnedVec<_> = recycler.allocate("pubkey_offsets");
437    pubkey_offsets.set_pinnable();
438    let mut msg_start_offsets: PinnedVec<_> = recycler.allocate("msg_start_offsets");
439    msg_start_offsets.set_pinnable();
440    let mut msg_sizes: PinnedVec<_> = recycler.allocate("msg_size_offsets");
441    msg_sizes.set_pinnable();
442    let mut current_offset: usize = 0;
443    let offsets = batches
444        .iter_mut()
445        .map(|batch| {
446            batch
447                .iter_mut()
448                .map(|packet| {
449                    let packet_offsets =
450                        get_packet_offsets(packet, current_offset, reject_non_vote);
451
452                    trace!("pubkey_offset: {}", packet_offsets.pubkey_start);
453
454                    let mut pubkey_offset = packet_offsets.pubkey_start;
455                    let mut sig_offset = packet_offsets.sig_start;
456                    let msg_size = current_offset.saturating_add(packet.meta().size) as u32;
457                    for _ in 0..packet_offsets.sig_len {
458                        signature_offsets.push(sig_offset);
459                        sig_offset = sig_offset.saturating_add(size_of::<Signature>() as u32);
460
461                        pubkey_offsets.push(pubkey_offset);
462                        pubkey_offset = pubkey_offset.saturating_add(size_of::<Pubkey>() as u32);
463
464                        msg_start_offsets.push(packet_offsets.msg_start);
465
466                        let msg_size = msg_size.saturating_sub(packet_offsets.msg_start);
467                        msg_sizes.push(msg_size);
468                    }
469                    current_offset = current_offset.saturating_add(size_of::<Packet>());
470                    packet_offsets.sig_len
471                })
472                .collect()
473        })
474        .collect();
475    (
476        signature_offsets,
477        pubkey_offsets,
478        msg_start_offsets,
479        msg_sizes,
480        offsets,
481    )
482}

484//inplace shrink a batch of packets
485pub fn shrink_batches(batches: &mut Vec<PacketBatch>) {
486    let mut valid_batch_ix = 0;
487    let mut valid_packet_ix = 0;
488    let mut last_valid_batch = 0;
489    for batch_ix in 0..batches.len() {
490        for packet_ix in 0..batches[batch_ix].len() {
491            if batches[batch_ix][packet_ix].meta().discard() {
492                continue;
493            }
494            last_valid_batch = batch_ix.saturating_add(1);
495            let mut found_spot = false;
496            while valid_batch_ix < batch_ix && !found_spot {
497                while valid_packet_ix < batches[valid_batch_ix].len() {
498                    if batches[valid_batch_ix][valid_packet_ix].meta().discard() {
499                        batches[valid_batch_ix][valid_packet_ix] =
500                            batches[batch_ix][packet_ix].clone();
501                        batches[batch_ix][packet_ix].meta_mut().set_discard(true);
502                        last_valid_batch = valid_batch_ix.saturating_add(1);
503                        found_spot = true;
504                        break;
505                    }
506                    valid_packet_ix = valid_packet_ix.saturating_add(1);
507                }
508                if valid_packet_ix >= batches[valid_batch_ix].len() {
509                    valid_packet_ix = 0;
510                    valid_batch_ix = valid_batch_ix.saturating_add(1);
511                }
512            }
513        }
514    }
515    batches.truncate(last_valid_batch);
516}

518pub fn ed25519_verify_cpu(batches: &mut [PacketBatch], reject_non_vote: bool, packet_count: usize) {
519    debug!("CPU ECDSA for {}", packet_count);
520    PAR_THREAD_POOL.install(|| {
521        batches
522            .par_iter_mut()
523            .flatten()
524            .collect::<Vec<&mut Packet>>()
525            .par_chunks_mut(VERIFY_PACKET_CHUNK_SIZE)
526            .for_each(|packets| {
527                for packet in packets.iter_mut() {
528                    if !packet.meta().discard() && !verify_packet(packet, reject_non_vote) {
529                        packet.meta_mut().set_discard(true);
530                    }
531                }
532            });
533    });
534    inc_new_counter_debug!("ed25519_verify_cpu", packet_count);
535}

548pub fn copy_return_values<I, T>(sig_lens: I, out: &PinnedVec<u8>, rvs: &mut [Vec<u8>])
549where
550    I: IntoIterator<Item = T>,
551    T: IntoIterator<Item = u32>,
552{
553    debug_assert!(rvs.iter().flatten().all(|&rv| rv == 0u8));
554    let mut offset = 0usize;
555    let rvs = rvs.iter_mut().flatten();
556    for (k, rv) in sig_lens.into_iter().flatten().zip(rvs) {
557        let out = out[offset..].iter().take(k as usize).all(|&x| x == 1u8);
558        *rv = u8::from(k != 0u32 && out);
559        offset = offset.saturating_add(k as usize);
560    }
561}

601pub fn ed25519_verify(
602    batches: &mut [PacketBatch],
603    recycler: &Recycler<TxOffset>,
604    recycler_out: &Recycler<PinnedVec<u8>>,
605    reject_non_vote: bool,
606    valid_packet_count: usize,
607) {
608    let Some(api) = perf_libs::api() else {
609        return ed25519_verify_cpu(batches, reject_non_vote, valid_packet_count);
610    };
611    let total_packet_count = count_packets_in_batches(batches);
612    // micro-benchmarks show GPU time for smallest batch around 15-20ms
613    // and CPU speed for 64-128 sigverifies around 10-20ms. 64 is a nice
614    // power-of-two number around that accounting for the fact that the CPU
615    // may be busy doing other things while being a real validator
616    // TODO: dynamically adjust this crossover
617    let maybe_valid_percentage = 100usize
618        .wrapping_mul(valid_packet_count)
619        .checked_div(total_packet_count);
620    let Some(valid_percentage) = maybe_valid_percentage else {
621        return;
622    };
623    if valid_percentage < 90 || valid_packet_count < 64 {
624        ed25519_verify_cpu(batches, reject_non_vote, valid_packet_count);
625        return;
626    }
627
628    let (signature_offsets, pubkey_offsets, msg_start_offsets, msg_sizes, sig_lens) =
629        generate_offsets(batches, recycler, reject_non_vote);
630
631    debug!("CUDA ECDSA for {}", valid_packet_count);
632    debug!("allocating out..");
633    let mut out = recycler_out.allocate("out_buffer");
634    out.set_pinnable();
635    let mut elems = Vec::new();
636    let mut rvs = Vec::new();
637
638    let mut num_packets: usize = 0;
639    for batch in batches.iter() {
640        elems.push(perf_libs::Elems {
641            elems: batch.as_ptr().cast::<u8>(),
642            num: batch.len() as u32,
643        });
644        let v = vec![0u8; batch.len()];
645        rvs.push(v);
646        num_packets = num_packets.saturating_add(batch.len());
647    }
648    out.resize(signature_offsets.len(), 0);
649    trace!("Starting verify num packets: {}", num_packets);
650    trace!("elem len: {}", elems.len() as u32);
651    trace!("packet sizeof: {}", size_of::<Packet>() as u32);
652    trace!("len offset: {}", PACKET_DATA_SIZE as u32);
653    const USE_NON_DEFAULT_STREAM: u8 = 1;
654    unsafe {
655        let res = (api.ed25519_verify_many)(
656            elems.as_ptr(),
657            elems.len() as u32,
658            size_of::<Packet>() as u32,
659            num_packets as u32,
660            signature_offsets.len() as u32,
661            msg_sizes.as_ptr(),
662            pubkey_offsets.as_ptr(),
663            signature_offsets.as_ptr(),
664            msg_start_offsets.as_ptr(),
665            out.as_mut_ptr(),
666            USE_NON_DEFAULT_STREAM,
667        );
668        if res != 0 {
669            trace!("RETURN!!!: {}", res);
670        }
671    }
672    trace!("done verify");
673    copy_return_values(sig_lens, &out, &mut rvs);
674    mark_disabled(batches, &rvs);
675    inc_new_counter_debug!("ed25519_verify_gpu", valid_packet_count);
676}
