9// A temporary burst in the workload can cause a large number of allocations,
10// after which they will be recycled and still reside in memory. If the number
11// of recycled objects stays above below limit for long, they will be deemed as
12// redundant since they are not getting reused. The recycler will then shrink
13// by releasing objects above this threshold. This limit aims to maintain a
14// cushion against *normal* variations in the workload while bounding the
15// number of redundant garbage collected objects after temporary bursts.
16const RECYCLER_SHRINK_SIZE: usize = 1024;
17
18// Lookback window for exponential moving averaging number of garbage collected
19// objects in terms of number of allocations. The half-life of the decaying
20// factor based on the window size defined below is 11356. This means a sample
21// of gc.size() that is 11356 allocations ago has half of the weight as the most
22// recent sample of gc.size() at current allocation.
23const RECYCLER_SHRINK_WINDOW: usize = 16384;
24
25#[derive(Debug, Default)]
26struct RecyclerStats {
27    total: AtomicUsize,
28    reuse: AtomicUsize,
29    freed: AtomicUsize,
30    max_gc: AtomicUsize,
31}

38#[derive(Debug)]
39pub struct RecyclerX<T> {
40    gc: Mutex<Vec<T>>,
41    stats: RecyclerStats,
42    id: usize,
43    // Shrink window times the exponential moving average size of gc.len().
44    size_factor: AtomicUsize,
45}
46
47impl<T: Default> Default for RecyclerX<T> {
48    fn default() -> RecyclerX<T> {
49        let id = thread_rng().gen_range(0..1000);
50        trace!("new recycler..{}", id);
51        RecyclerX {
52            gc: Mutex::default(),
53            stats: RecyclerStats::default(),
54            id,
55            size_factor: AtomicUsize::default(),
56        }
57    }
58}

89impl<T: Default + Reset + Sized> Recycler<T> {
90    #[allow(clippy::needless_collect)]
91    pub fn warmed(num: usize, size_hint: usize) -> Self {
92        let new = Self::default();
93        if warm_recyclers() {
94            let warmed_items: Vec<_> = (0..num)
95                .map(|_| {
96                    let mut item = new.allocate("warming");
97                    item.warm(size_hint);
98                    item
99                })
100                .collect();
101            warmed_items
102                .into_iter()
103                .for_each(|i| new.recycler.recycle(i));
104        }
105        new
106    }
107
108    pub fn allocate(&self, name: &'static str) -> T {
109        {
110            const RECYCLER_SHRINK_WINDOW_HALF: usize = RECYCLER_SHRINK_WINDOW / 2;
111            const RECYCLER_SHRINK_WINDOW_SUB_ONE: usize = RECYCLER_SHRINK_WINDOW - 1;
112            let mut gc = self.recycler.gc.lock().unwrap();
113            // Update the exponential moving average of gc.len().
114            // The update equation is:
115            //      a <- a * (n - 1) / n + x / n
116            // To avoid floating point math, define b = n a:
117            //      b <- b * (n - 1) / n + x
118            // To make the remaining division to round (instead of truncate),
119            // add n/2 to the numerator.
120            // Effectively b (size_factor here) is an exponential moving
121            // estimate of the "sum" of x (gc.len()) over the window as opposed
122            // to the "average".
123            self.recycler.size_factor.store(
124                self.recycler
125                    .size_factor
126                    .load(Ordering::Acquire)
127                    .saturating_mul(RECYCLER_SHRINK_WINDOW_SUB_ONE)
128                    .saturating_add(RECYCLER_SHRINK_WINDOW_HALF)
129                    .checked_div(RECYCLER_SHRINK_WINDOW)
130                    .unwrap()
131                    .saturating_add(gc.len()),
132                Ordering::Release,
133            );
134            if let Some(mut x) = gc.pop() {
135                self.recycler.stats.reuse.fetch_add(1, Ordering::Relaxed);
136                x.reset();
137                return x;
138            }
139        }
140        let total = self.recycler.stats.total.fetch_add(1, Ordering::Relaxed);
141        trace!(
142            "allocating new: total {} {:?} id: {} reuse: {} max_gc: {}",
143            total,
144            name,
145            self.recycler.id,
146            self.recycler.stats.reuse.load(Ordering::Relaxed),
147            self.recycler.stats.max_gc.load(Ordering::Relaxed),
148        );
149
150        let mut t = T::default();
151        t.set_recycler(Arc::downgrade(&self.recycler));
152        t
153    }
154}

156impl<T: Default + Reset> RecyclerX<T> {
157    pub fn recycle(&self, x: T) {
158        let len = {
159            let mut gc = self.gc.lock().expect("recycler lock in pub fn recycle");
160            gc.push(x);
161            const SIZE_FACTOR_AFTER_SHRINK: usize = RECYCLER_SHRINK_SIZE * RECYCLER_SHRINK_WINDOW;
162            if gc.len() > RECYCLER_SHRINK_SIZE
163                && self.size_factor.load(Ordering::Acquire) >= SIZE_FACTOR_AFTER_SHRINK
164            {
165                self.stats.freed.fetch_add(
166                    gc.len().saturating_sub(RECYCLER_SHRINK_SIZE),
167                    Ordering::Relaxed,
168                );
169                for mut x in gc.drain(RECYCLER_SHRINK_SIZE..) {
170                    x.set_recycler(Weak::default());
171                }
172                self.size_factor
173                    .store(SIZE_FACTOR_AFTER_SHRINK, Ordering::Release);
174            }
175            gc.len()
176        };
177
178        let max_gc = self.stats.max_gc.load(Ordering::Relaxed);
179        if len > max_gc {
180            // this is not completely accurate, but for most cases should be fine.
181            let _ = self.stats.max_gc.compare_exchange(
182                max_gc,
183                len,
184                Ordering::Relaxed,
185                Ordering::Relaxed,
186            );
187        }
188        let total = self.stats.total.load(Ordering::Relaxed);
189        let reuse = self.stats.reuse.load(Ordering::Relaxed);
190        let freed = self.stats.freed.load(Ordering::Relaxed);
191        datapoint_debug!(
192            "recycler",
193            ("gc_len", len as i64, i64),
194            ("total", total as i64, i64),
195            ("freed", freed as i64, i64),
196            ("reuse", reuse as i64, i64),
197        );
198    }
199}

213    #[test]
214    fn test_recycler() {
215        let recycler = Recycler::default();
216        let mut y: u64 = recycler.allocate("test_recycler1");
217        assert_eq!(y, 0);
218        y = 20;
219        let recycler2 = recycler.clone();
220        recycler2.recycler.recycle(y);
221        assert_eq!(recycler.recycler.gc.lock().unwrap().len(), 1);
222        let z = recycler.allocate("test_recycler2");
223        assert_eq!(z, 10);
224        assert_eq!(recycler.recycler.gc.lock().unwrap().len(), 0);
225    }

227    #[test]
228    fn test_recycler_shrink() {
229        let mut rng = rand::thread_rng();
230        let recycler = PacketBatchRecycler::default();
231        // Allocate a burst of packets.
232        const NUM_PACKETS: usize = RECYCLER_SHRINK_SIZE * 2;
233        {
234            let _packets: Vec<_> = repeat_with(|| recycler.allocate(""))
235                .take(NUM_PACKETS)
236                .collect();
237        }
238        assert_eq!(recycler.recycler.gc.lock().unwrap().len(), NUM_PACKETS);
239        // Process a normal load of packets for a while.
240        for _ in 0..RECYCLER_SHRINK_WINDOW / 16 {
241            let count = rng.gen_range(1..128);
242            let _packets: Vec<_> = repeat_with(|| recycler.allocate("")).take(count).collect();
243        }
244        // Assert that the gc size has shrunk.
245        assert_eq!(
246            recycler.recycler.gc.lock().unwrap().len(),
247            RECYCLER_SHRINK_SIZE
248        );
249    }
