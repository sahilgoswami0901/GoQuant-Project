91impl QuicLazyInitializedEndpoint {
92    pub fn new(
93        client_certificate: Arc<QuicClientCertificate>,
94        client_endpoint: Option<Endpoint>,
95    ) -> Self {
96        Self {
97            endpoint: OnceCell::<Arc<Endpoint>>::new(),
98            client_certificate,
99            client_endpoint,
100        }
101    }
102
103    fn create_endpoint(&self) -> Endpoint {
104        let mut endpoint = if let Some(endpoint) = &self.client_endpoint {
105            endpoint.clone()
106        } else {
107            let client_socket = solana_net_utils::bind_in_range(
108                IpAddr::V4(Ipv4Addr::UNSPECIFIED),
109                VALIDATOR_PORT_RANGE,
110            )
111            .expect("QuicLazyInitializedEndpoint::create_endpoint bind_in_range")
112            .1;
113
114            QuicNewConnection::create_endpoint(EndpointConfig::default(), client_socket)
115        };
116
117        let mut crypto = rustls::ClientConfig::builder()
118            .with_safe_defaults()
119            .with_custom_certificate_verifier(SkipServerVerification::new())
120            .with_client_auth_cert(
121                vec![self.client_certificate.certificate.clone()],
122                self.client_certificate.key.clone(),
123            )
124            .expect("Failed to set QUIC client certificates");
125        crypto.enable_early_data = true;
126        crypto.alpn_protocols = vec![ALPN_TPU_PROTOCOL_ID.to_vec()];
127
128        let mut config = ClientConfig::new(Arc::new(crypto));
129        let mut transport_config = TransportConfig::default();
130
131        let timeout = IdleTimeout::try_from(QUIC_MAX_TIMEOUT).unwrap();
132        transport_config.max_idle_timeout(Some(timeout));
133        transport_config.keep_alive_interval(Some(QUIC_KEEP_ALIVE));
134        config.transport_config(Arc::new(transport_config));
135
136        endpoint.set_default_client_config(config);
137
138        endpoint
139    }
140
141    async fn get_endpoint(&self) -> Arc<Endpoint> {
142        self.endpoint
143            .get_or_init(|| async { Arc::new(self.create_endpoint()) })
144            .await
145            .clone()
146    }
147}

164/// A wrapper over NewConnection with additional capability to create the endpoint as part
165/// of creating a new connection.
166#[derive(Clone)]
167struct QuicNewConnection {
168    endpoint: Arc<Endpoint>,
169    connection: Arc<Connection>,
170}
171
172impl QuicNewConnection {
173    /// Create a QuicNewConnection given the remote address 'addr'.
174    async fn make_connection(
175        endpoint: Arc<QuicLazyInitializedEndpoint>,
176        addr: SocketAddr,
177        stats: &ClientStats,
178    ) -> Result<Self, QuicError> {
179        let mut make_connection_measure = Measure::start("make_connection_measure");
180        let endpoint = endpoint.get_endpoint().await;
181
182        let connecting = endpoint.connect(addr, "connect")?;
183        stats.total_connections.fetch_add(1, Ordering::Relaxed);
184        if let Ok(connecting_result) = timeout(QUIC_CONNECTION_HANDSHAKE_TIMEOUT, connecting).await
185        {
186            if connecting_result.is_err() {
187                stats.connection_errors.fetch_add(1, Ordering::Relaxed);
188            }
189            make_connection_measure.stop();
190            stats
191                .make_connection_ms
192                .fetch_add(make_connection_measure.as_ms(), Ordering::Relaxed);
193
194            let connection = connecting_result?;
195
196            Ok(Self {
197                endpoint,
198                connection: Arc::new(connection),
199            })
200        } else {
201            Err(ConnectionError::TimedOut.into())
202        }
203    }
204
205    fn create_endpoint(config: EndpointConfig, client_socket: UdpSocket) -> Endpoint {
206        quinn::Endpoint::new(config, None, client_socket, Arc::new(TokioRuntime))
207            .expect("QuicNewConnection::create_endpoint quinn::Endpoint::new")
208    }

210    // Attempts to make a faster connection by taking advantage of pre-existing key material.
211    // Only works if connection to this endpoint was previously established.
212    async fn make_connection_0rtt(
213        &mut self,
214        addr: SocketAddr,
215        stats: &ClientStats,
216    ) -> Result<Arc<Connection>, QuicError> {
217        let connecting = self.endpoint.connect(addr, "connect")?;
218        stats.total_connections.fetch_add(1, Ordering::Relaxed);
219        let connection = match connecting.into_0rtt() {
220            Ok((connection, zero_rtt)) => {
221                if let Ok(zero_rtt) = timeout(QUIC_CONNECTION_HANDSHAKE_TIMEOUT, zero_rtt).await {
222                    if zero_rtt {
223                        stats.zero_rtt_accepts.fetch_add(1, Ordering::Relaxed);
224                    } else {
225                        stats.zero_rtt_rejects.fetch_add(1, Ordering::Relaxed);
226                    }
227                    connection
228                } else {
229                    return Err(ConnectionError::TimedOut.into());
230                }
231            }
232            Err(connecting) => {
233                stats.connection_errors.fetch_add(1, Ordering::Relaxed);
234
235                if let Ok(connecting_result) =
236                    timeout(QUIC_CONNECTION_HANDSHAKE_TIMEOUT, connecting).await
237                {
238                    connecting_result?
239                } else {
240                    return Err(ConnectionError::TimedOut.into());
241                }
242            }
243        };
244        self.connection = Arc::new(connection);
245        Ok(self.connection.clone())
246    }
247}

272    async fn _send_buffer_using_conn(
273        data: &[u8],
274        connection: &Connection,
275    ) -> Result<(), QuicError> {
276        let mut send_stream = connection.open_uni().await?;
277
278        send_stream.write_all(data).await?;
279        send_stream.finish().await?;
280        Ok(())
281    }
282
283    // Attempts to send data, connecting/reconnecting as necessary
284    // On success, returns the connection used to successfully send the data
285    async fn _send_buffer(
286        &self,
287        data: &[u8],
288        stats: &ClientStats,
289        connection_stats: Arc<ConnectionCacheStats>,
290    ) -> Result<Arc<Connection>, QuicError> {
291        let mut measure_send_packet = Measure::start("send_packet_us");
292        let mut measure_prepare_connection = Measure::start("prepare_connection");
293        let mut connection_try_count = 0;
294        let mut last_connection_id = 0;
295        let mut last_error = None;
296        while connection_try_count < 2 {
297            let connection = {
298                let mut conn_guard = self.connection.lock().await;
299
300                let maybe_conn = conn_guard.as_mut();
301                match maybe_conn {
302                    Some(conn) => {
303                        if conn.connection.stable_id() == last_connection_id {
304                            // this is the problematic connection we had used before, create a new one
305                            let conn = conn.make_connection_0rtt(self.addr, stats).await;
306                            match conn {
307                                Ok(conn) => {
308                                    info!(
309                                        "Made 0rtt connection to {} with id {} try_count {}, last_connection_id: {}, last_error: {:?}",
310                                        self.addr,
311                                        conn.stable_id(),
312                                        connection_try_count,
313                                        last_connection_id,
314                                        last_error,
315                                    );
316                                    connection_try_count += 1;
317                                    conn
318                                }
319                                Err(err) => {
320                                    info!(
321                                        "Cannot make 0rtt connection to {}, error {:}",
322                                        self.addr, err
323                                    );
324                                    return Err(err);
325                                }
326                            }
327                        } else {
328                            stats.connection_reuse.fetch_add(1, Ordering::Relaxed);
329                            conn.connection.clone()
330                        }
331                    }
332                    None => {
333                        let conn = QuicNewConnection::make_connection(
334                            self.endpoint.clone(),
335                            self.addr,
336                            stats,
337                        )
338                        .await;
339                        match conn {
340                            Ok(conn) => {
341                                *conn_guard = Some(conn.clone());
342                                info!(
343                                    "Made connection to {} id {} try_count {}, from connection cache warming?: {}",
344                                    self.addr,
345                                    conn.connection.stable_id(),
346                                    connection_try_count,
347                                    data.is_empty(),
348                                );
349                                connection_try_count += 1;
350                                conn.connection.clone()
351                            }
352                            Err(err) => {
353                                info!("Cannot make connection to {}, error {:}, from connection cache warming?: {}",
354                                    self.addr, err, data.is_empty());
355                                return Err(err);
356                            }
357                        }
358                    }
359                }
360            };
361
362            let new_stats = connection.stats();
363
364            connection_stats
365                .total_client_stats
366                .congestion_events
367                .update_stat(
368                    &self.stats.congestion_events,
369                    new_stats.path.congestion_events,
370                );
371
372            connection_stats
373                .total_client_stats
374                .streams_blocked_uni
375                .update_stat(
376                    &self.stats.streams_blocked_uni,
377                    new_stats.frame_tx.streams_blocked_uni,
378                );
379
380            connection_stats
381                .total_client_stats
382                .data_blocked
383                .update_stat(&self.stats.data_blocked, new_stats.frame_tx.data_blocked);
384
385            connection_stats
386                .total_client_stats
387                .acks
388                .update_stat(&self.stats.acks, new_stats.frame_tx.acks);
389
390            if data.is_empty() {
391                // no need to send packet as it is only for warming connections
392                return Ok(connection);
393            }
394
395            last_connection_id = connection.stable_id();
396            measure_prepare_connection.stop();
397
398            match Self::_send_buffer_using_conn(data, &connection).await {
399                Ok(()) => {
400                    measure_send_packet.stop();
401                    stats.successful_packets.fetch_add(1, Ordering::Relaxed);
402                    stats
403                        .send_packets_us
404                        .fetch_add(measure_send_packet.as_us(), Ordering::Relaxed);
405                    stats
406                        .prepare_connection_us
407                        .fetch_add(measure_prepare_connection.as_us(), Ordering::Relaxed);
408                    trace!(
409                        "Succcessfully sent to {} with id {}, thread: {:?}, data len: {}, send_packet_us: {} prepare_connection_us: {}",
410                        self.addr,
411                        connection.stable_id(),
412                        thread::current().id(),
413                        data.len(),
414                        measure_send_packet.as_us(),
415                        measure_prepare_connection.as_us(),
416                    );
417
418                    return Ok(connection);
419                }
420                Err(err) => match err {
421                    QuicError::ConnectionError(_) => {
422                        last_error = Some(err);
423                    }
424                    _ => {
425                        info!(
426                            "Error sending to {} with id {}, error {:?} thread: {:?}",
427                            self.addr,
428                            connection.stable_id(),
429                            err,
430                            thread::current().id(),
431                        );
432                        return Err(err);
433                    }
434                },
435            }
436        }
437
438        // if we come here, that means we have exhausted maximum retries, return the error
439        info!(
440            "Ran into an error sending data {:?}, exhausted retries to {}",
441            last_error, self.addr
442        );
443        // If we get here but last_error is None, then we have a logic error
444        // in this function, so panic here with an expect to help debugging
445        Err(last_error.expect("QuicClient::_send_buffer last_error.expect"))
446    }

448    pub async fn send_buffer<T>(
449        &self,
450        data: T,
451        stats: &ClientStats,
452        connection_stats: Arc<ConnectionCacheStats>,
453    ) -> Result<(), ClientErrorKind>
454    where
455        T: AsRef<[u8]>,
456    {
457        self._send_buffer(data.as_ref(), stats, connection_stats)
458            .await
459            .map_err(Into::<ClientErrorKind>::into)?;
460        Ok(())
461    }
462
463    pub async fn send_batch<T>(
464        &self,
465        buffers: &[T],
466        stats: &ClientStats,
467        connection_stats: Arc<ConnectionCacheStats>,
468    ) -> Result<(), ClientErrorKind>
469    where
470        T: AsRef<[u8]>,
471    {
472        // Start off by "testing" the connection by sending the first buffer
473        // This will also connect to the server if not already connected
474        // and reconnect and retry if the first send attempt failed
475        // (for example due to a timed out connection), returning an error
476        // or the connection that was used to successfully send the buffer.
477        // We will use the returned connection to send the rest of the buffers in the batch
478        // to avoid touching the mutex in self, and not bother reconnecting if we fail along the way
479        // since testing even in the ideal GCE environment has found no cases
480        // where reconnecting and retrying in the middle of a batch send
481        // (i.e. we encounter a connection error in the middle of a batch send, which presumably cannot
482        // be due to a timed out connection) has succeeded
483        if buffers.is_empty() {
484            return Ok(());
485        }
486        let connection = self
487            ._send_buffer(buffers[0].as_ref(), stats, connection_stats)
488            .await
489            .map_err(Into::<ClientErrorKind>::into)?;
490
491        // Used to avoid dereferencing the Arc multiple times below
492        // by just getting a reference to the NewConnection once
493        let connection_ref: &Connection = &connection;
494
495        let chunks = buffers[1..buffers.len()].iter().chunks(self.chunk_size);
496
497        let futures: Vec<_> = chunks
498            .into_iter()
499            .map(|buffs| {
500                join_all(
501                    buffs
502                        .into_iter()
503                        .map(|buf| Self::_send_buffer_using_conn(buf.as_ref(), connection_ref)),
504                )
505            })
506            .collect();
507
508        for f in futures {
509            f.await
510                .into_iter()
511                .try_for_each(|res| res)
512                .map_err(Into::<ClientErrorKind>::into)?;
513        }
514        Ok(())
515    }

564#[async_trait]
565impl ClientConnection for QuicClientConnection {
566    fn server_addr(&self) -> &SocketAddr {
567        self.client.server_addr()
568    }
569
570    async fn send_data_batch(&self, buffers: &[Vec<u8>]) -> TransportResult<()> {
571        let stats = ClientStats::default();
572        let len = buffers.len();
573        let res = self
574            .client
575            .send_batch(buffers, &stats, self.connection_stats.clone())
576            .await;
577        self.connection_stats
578            .add_client_stats(&stats, len, res.is_ok());
579        res?;
580        Ok(())
581    }
582
583    async fn send_data(&self, data: &[u8]) -> TransportResult<()> {
584        let stats = Arc::new(ClientStats::default());
585        // When data is empty which is from cache warmer, we are not sending packets actually, do not count it in
586        let num_packets = if data.is_empty() { 0 } else { 1 };
587        self.client
588            .send_buffer(data, &stats, self.connection_stats.clone())
589            .map_ok(|v| {
590                self.connection_stats
591                    .add_client_stats(&stats, num_packets, true);
592                v
593            })
594            .map_err(|e| {
595                warn!(
596                    "Failed to send data async to {}, error: {:?} ",
597                    self.server_addr(),
598                    e
599                );
600                datapoint_warn!("send-wire-async", ("failure", 1, i64),);
601                self.connection_stats
602                    .add_client_stats(&stats, num_packets, false);
603                e.into()
604            })
605            .await
606    }
