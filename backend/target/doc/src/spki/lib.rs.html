3//! # Usage
4//! The following example demonstrates how to use an OID as the `parameters`
5//! of an [`AlgorithmIdentifier`].
6//!
7//! Borrow the [`ObjectIdentifier`] first then use [`der::Any::from`] or `.into()`:
8//!
9//! ```
10//! use spki::{AlgorithmIdentifier, ObjectIdentifier, der::Any};
11//!
12//! let alg_oid = "1.2.840.10045.2.1".parse::<ObjectIdentifier>().unwrap();
13//! let params_oid = "1.2.840.10045.3.1.7".parse::<ObjectIdentifier>().unwrap();
14//!
15//! let alg_id = AlgorithmIdentifier {
16//!     oid: alg_oid,
17//!     parameters: Some(Any::from(&params_oid))
18//! };
19//! ```
