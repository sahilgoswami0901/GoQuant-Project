21/// A QUIC client or server connection.
22#[derive(Debug)]
23pub enum Connection {
24    /// A client connection
25    Client(ClientConnection),
26    /// A server connection
27    Server(ServerConnection),
28}
29
30impl Connection {
31    /// Return the TLS-encoded transport parameters for the session's peer.
32    ///
33    /// See [`ConnectionCommon::quic_transport_parameters()`] for more details.
34    pub fn quic_transport_parameters(&self) -> Option<&[u8]> {
35        match self {
36            Self::Client(conn) => conn.quic_transport_parameters(),
37            Self::Server(conn) => conn.quic_transport_parameters(),
38        }
39    }
40
41    /// Compute the keys for encrypting/decrypting 0-RTT packets, if available
42    pub fn zero_rtt_keys(&self) -> Option<DirectionalKeys> {
43        match self {
44            Self::Client(conn) => conn.zero_rtt_keys(),
45            Self::Server(conn) => conn.zero_rtt_keys(),
46        }
47    }
48
49    /// Consume unencrypted TLS handshake data.
50    ///
51    /// Handshake data obtained from separate encryption levels should be supplied in separate calls.
52    pub fn read_hs(&mut self, plaintext: &[u8]) -> Result<(), Error> {
53        match self {
54            Self::Client(conn) => conn.read_hs(plaintext),
55            Self::Server(conn) => conn.read_hs(plaintext),
56        }
57    }
58
59    /// Emit unencrypted TLS handshake data.
60    ///
61    /// When this returns `Some(_)`, the new keys must be used for future handshake data.
62    pub fn write_hs(&mut self, buf: &mut Vec<u8>) -> Option<KeyChange> {
63        match self {
64            Self::Client(conn) => conn.write_hs(buf),
65            Self::Server(conn) => conn.write_hs(buf),
66        }
67    }
68
69    /// Emit the TLS description code of a fatal alert, if one has arisen.
70    ///
71    /// Check after `read_hs` returns `Err(_)`.
72    pub fn alert(&self) -> Option<AlertDescription> {
73        match self {
74            Self::Client(conn) => conn.alert(),
75            Self::Server(conn) => conn.alert(),
76        }
77    }
78
79    /// Derives key material from the agreed connection secrets.
80    ///
81    /// This function fills in `output` with `output.len()` bytes of key
82    /// material derived from the master session secret using `label`
83    /// and `context` for diversification. Ownership of the buffer is taken
84    /// by the function and returned via the Ok result to ensure no key
85    /// material leaks if the function fails.
86    ///
87    /// See RFC5705 for more details on what this does and is for.
88    ///
89    /// For TLS1.3 connections, this function does not use the
90    /// "early" exporter at any point.
91    ///
92    /// This function fails if called prior to the handshake completing;
93    /// check with [`CommonState::is_handshaking`] first.
94    #[inline]
95    pub fn export_keying_material<T: AsMut<[u8]>>(
96        &self,
97        output: T,
98        label: &[u8],
99        context: Option<&[u8]>,
100    ) -> Result<T, Error> {
101        match self {
102            Self::Client(conn) => conn
103                .core
104                .export_keying_material(output, label, context),
105            Self::Server(conn) => conn
106                .core
107                .export_keying_material(output, label, context),
108        }
109    }
110}

137impl ClientConnection {
138    /// Make a new QUIC ClientConnection. This differs from `ClientConnection::new()`
139    /// in that it takes an extra argument, `params`, which contains the
140    /// TLS-encoded transport parameters to send.
141    pub fn new(
142        config: Arc<ClientConfig>,
143        quic_version: Version,
144        name: ServerName,
145        params: Vec<u8>,
146    ) -> Result<Self, Error> {
147        if !config.supports_version(ProtocolVersion::TLSv1_3) {
148            return Err(Error::General(
149                "TLS 1.3 support is required for QUIC".into(),
150            ));
151        }
152
153        let ext = match quic_version {
154            Version::V1Draft => ClientExtension::TransportParametersDraft(params),
155            Version::V1 | Version::V2 => ClientExtension::TransportParameters(params),
156        };
157
158        let mut inner = ConnectionCore::for_client(config, name, vec![ext], Protocol::Quic)?;
159        inner.common_state.quic.version = quic_version;
160        Ok(Self {
161            inner: inner.into(),
162        })
163    }
164
165    /// Returns True if the server signalled it will process early data.
166    ///
167    /// If you sent early data and this returns false at the end of the
168    /// handshake then the server will not process the data.  This
169    /// is not an error, but you may wish to resend the data.
170    pub fn is_early_data_accepted(&self) -> bool {
171        self.inner.core.is_early_data_accepted()
172    }
173}

204    inner: ConnectionCommon<ServerConnectionData>,
205}

207impl ServerConnection {
208    /// Make a new QUIC ServerConnection. This differs from `ServerConnection::new()`
209    /// in that it takes an extra argument, `params`, which contains the
210    /// TLS-encoded transport parameters to send.
211    pub fn new(
212        config: Arc<ServerConfig>,
213        quic_version: Version,
214        params: Vec<u8>,
215    ) -> Result<Self, Error> {
216        if !config.supports_version(ProtocolVersion::TLSv1_3) {
217            return Err(Error::General(
218                "TLS 1.3 support is required for QUIC".into(),
219            ));
220        }
221
222        if config.max_early_data_size != 0 && config.max_early_data_size != 0xffff_ffff {
223            return Err(Error::General(
224                "QUIC sessions must set a max early data of 0 or 2^32-1".into(),
225            ));
226        }
227
228        let ext = match quic_version {
229            Version::V1Draft => ServerExtension::TransportParametersDraft(params),
230            Version::V1 | Version::V2 => ServerExtension::TransportParameters(params),
231        };
232
233        let mut core = ConnectionCore::for_server(config, vec![ext])?;
234        core.common_state.protocol = Protocol::Quic;
235        core.common_state.quic.version = quic_version;
236        Ok(Self { inner: core.into() })
237    }
238
239    /// Explicitly discard early data, notifying the client
240    ///
241    /// Useful if invariants encoded in `received_resumption_data()` cannot be respected.
242    ///
243    /// Must be called while `is_handshaking` is true.
244    pub fn reject_early_data(&mut self) {
245        self.inner.core.reject_early_data()
246    }
247
248    /// Retrieves the server name, if any, used to select the certificate and
249    /// private key.
250    ///
251    /// This returns `None` until some time after the client's server name indication
252    /// (SNI) extension value is processed during the handshake. It will never be
253    /// `None` when the connection is ready to send or process application data,
254    /// unless the client does not support SNI.
255    ///
256    /// This is useful for application protocols that need to enforce that the
257    /// server name matches an application layer protocol hostname. For
258    /// example, HTTP/1.1 servers commonly expect the `Host:` header field of
259    /// every request on a connection to match the hostname in the SNI extension
260    /// when the client provides the SNI extension.
261    ///
262    /// The server name is also used to match sessions during session resumption.
263    pub fn server_name(&self) -> Option<&str> {
264        self.inner.core.get_sni_str()
265    }
266}

295/// A shared interface for QUIC connections.
296pub struct ConnectionCommon<Data> {
297    core: ConnectionCore<Data>,
298}
299
300impl<Data: SideData> ConnectionCommon<Data> {
301    /// Return the TLS-encoded transport parameters for the session's peer.
302    ///
303    /// While the transport parameters are technically available prior to the
304    /// completion of the handshake, they cannot be fully trusted until the
305    /// handshake completes, and reliance on them should be minimized.
306    /// However, any tampering with the parameters will cause the handshake
307    /// to fail.
308    pub fn quic_transport_parameters(&self) -> Option<&[u8]> {
309        self.core
310            .common_state
311            .quic
312            .params
313            .as_ref()
314            .map(|v| v.as_ref())
315    }
316
317    /// Compute the keys for encrypting/decrypting 0-RTT packets, if available
318    pub fn zero_rtt_keys(&self) -> Option<DirectionalKeys> {
319        Some(DirectionalKeys::new(
320            self.core
321                .common_state
322                .suite
323                .and_then(|suite| suite.tls13())?,
324            self.core
325                .common_state
326                .quic
327                .early_secret
328                .as_ref()?,
329            self.core.common_state.quic.version,
330        ))
331    }
332
333    /// Consume unencrypted TLS handshake data.
334    ///
335    /// Handshake data obtained from separate encryption levels should be supplied in separate calls.
336    pub fn read_hs(&mut self, plaintext: &[u8]) -> Result<(), Error> {
337        self.core
338            .message_deframer
339            .push(ProtocolVersion::TLSv1_3, plaintext)?;
340        self.core.process_new_packets()?;
341        Ok(())
342    }
343
344    /// Emit unencrypted TLS handshake data.
345    ///
346    /// When this returns `Some(_)`, the new keys must be used for future handshake data.
347    pub fn write_hs(&mut self, buf: &mut Vec<u8>) -> Option<KeyChange> {
348        self.core
349            .common_state
350            .quic
351            .write_hs(buf)
352    }
353
354    /// Emit the TLS description code of a fatal alert, if one has arisen.
355    ///
356    /// Check after `read_hs` returns `Err(_)`.
357    pub fn alert(&self) -> Option<AlertDescription> {
358        self.core.common_state.quic.alert
359    }
360}

363    type Target = CommonState;

366        &self.core.common_state
367    }
368}

382#[cfg(feature = "quic")]
383#[derive(Default)]
384pub(crate) struct Quic {
385    /// QUIC transport parameters received from the peer during the handshake
386    pub(crate) params: Option<Vec<u8>>,
387    pub(crate) alert: Option<AlertDescription>,
388    pub(crate) hs_queue: VecDeque<(bool, Vec<u8>)>,
389    pub(crate) early_secret: Option<hkdf::Prk>,
390    pub(crate) hs_secrets: Option<Secrets>,
391    pub(crate) traffic_secrets: Option<Secrets>,
392    /// Whether keys derived from traffic_secrets have been passed to the QUIC implementation
393    pub(crate) returned_traffic_keys: bool,
394    pub(crate) version: Version,
395}
396
397impl Quic {
398    pub(crate) fn write_hs(&mut self, buf: &mut Vec<u8>) -> Option<KeyChange> {
399        while let Some((_, msg)) = self.hs_queue.pop_front() {
400            buf.extend_from_slice(&msg);
401            if let Some(&(true, _)) = self.hs_queue.front() {
402                if self.hs_secrets.is_some() {
403                    // Allow the caller to switch keys before proceeding.
404                    break;
405                }
406            }
407        }
408
409        if let Some(secrets) = self.hs_secrets.take() {
410            return Some(KeyChange::Handshake {
411                keys: Keys::new(&secrets),
412            });
413        }
414
415        if let Some(mut secrets) = self.traffic_secrets.take() {
416            if !self.returned_traffic_keys {
417                self.returned_traffic_keys = true;
418                let keys = Keys::new(&secrets);
419                secrets.update();
420                return Some(KeyChange::OneRtt {
421                    keys,
422                    next: secrets,
423                });
424            }
425        }
426
427        None
428    }
429}

431/// Secrets used to encrypt/decrypt traffic
432#[derive(Clone, Debug)]
433pub struct Secrets {
434    /// Secret used to encrypt packets transmitted by the client
435    client: hkdf::Prk,
436    /// Secret used to encrypt packets transmitted by the server
437    server: hkdf::Prk,
438    /// Cipher suite used with these secrets
439    suite: &'static Tls13CipherSuite,
440    side: Side,
441    version: Version,
442}
443
444impl Secrets {
445    pub(crate) fn new(
446        client: hkdf::Prk,
447        server: hkdf::Prk,
448        suite: &'static Tls13CipherSuite,
449        side: Side,
450        version: Version,
451    ) -> Self {
452        Self {
453            client,
454            server,
455            suite,
456            side,
457            version,
458        }
459    }
460
461    /// Derive the next set of packet keys
462    pub fn next_packet_keys(&mut self) -> PacketKeySet {
463        let keys = PacketKeySet::new(self);
464        self.update();
465        keys
466    }
467
468    fn update(&mut self) {
469        let hkdf_alg = self.suite.hkdf_algorithm;
470        self.client = hkdf_expand(&self.client, hkdf_alg, self.version.key_update_label(), &[]);
471        self.server = hkdf_expand(&self.server, hkdf_alg, self.version.key_update_label(), &[]);
472    }
473
474    fn local_remote(&self) -> (&hkdf::Prk, &hkdf::Prk) {
475        match self.side {
476            Side::Client => (&self.client, &self.server),
477            Side::Server => (&self.server, &self.client),
478        }
479    }
480}

503/// A QUIC header protection key
504pub struct HeaderProtectionKey(aead::quic::HeaderProtectionKey);
505
506impl HeaderProtectionKey {
507    fn new(suite: &'static Tls13CipherSuite, secret: &hkdf::Prk, version: Version) -> Self {
508        let alg = match suite.common.bulk {
509            BulkAlgorithm::Aes128Gcm => &aead::quic::AES_128,
510            BulkAlgorithm::Aes256Gcm => &aead::quic::AES_256,
511            BulkAlgorithm::Chacha20Poly1305 => &aead::quic::CHACHA20,
512        };
513
514        Self(hkdf_expand(secret, alg, version.header_key_label(), &[]))
515    }
516
517    /// Adds QUIC Header Protection.
518    ///
519    /// `sample` must contain the sample of encrypted payload; see
520    /// [Header Protection Sample].
521    ///
522    /// `first` must reference the first byte of the header, referred to as
523    /// `packet[0]` in [Header Protection Application].
524    ///
525    /// `packet_number` must reference the Packet Number field; this is
526    /// `packet[pn_offset:pn_offset+pn_length]` in [Header Protection Application].
527    ///
528    /// Returns an error without modifying anything if `sample` is not
529    /// the correct length (see [Header Protection Sample] and [`Self::sample_len()`]),
530    /// or `packet_number` is longer than allowed (see [Packet Number Encoding and Decoding]).
531    ///
532    /// Otherwise, `first` and `packet_number` will have the header protection added.
533    ///
534    /// [Header Protection Application]: https://datatracker.ietf.org/doc/html/rfc9001#section-5.4.1
535    /// [Header Protection Sample]: https://datatracker.ietf.org/doc/html/rfc9001#section-5.4.2
536    /// [Packet Number Encoding and Decoding]: https://datatracker.ietf.org/doc/html/rfc9000#section-17.1
537    #[inline]
538    pub fn encrypt_in_place(
539        &self,
540        sample: &[u8],
541        first: &mut u8,
542        packet_number: &mut [u8],
543    ) -> Result<(), Error> {
544        self.xor_in_place(sample, first, packet_number, false)
545    }
546
547    /// Removes QUIC Header Protection.
548    ///
549    /// `sample` must contain the sample of encrypted payload; see
550    /// [Header Protection Sample].
551    ///
552    /// `first` must reference the first byte of the header, referred to as
553    /// `packet[0]` in [Header Protection Application].
554    ///
555    /// `packet_number` must reference the Packet Number field; this is
556    /// `packet[pn_offset:pn_offset+pn_length]` in [Header Protection Application].
557    ///
558    /// Returns an error without modifying anything if `sample` is not
559    /// the correct length (see [Header Protection Sample] and [`Self::sample_len()`]),
560    /// or `packet_number` is longer than allowed (see
561    /// [Packet Number Encoding and Decoding]).
562    ///
563    /// Otherwise, `first` and `packet_number` will have the header protection removed.
564    ///
565    /// [Header Protection Application]: https://datatracker.ietf.org/doc/html/rfc9001#section-5.4.1
566    /// [Header Protection Sample]: https://datatracker.ietf.org/doc/html/rfc9001#section-5.4.2
567    /// [Packet Number Encoding and Decoding]: https://datatracker.ietf.org/doc/html/rfc9000#section-17.1
568    #[inline]
569    pub fn decrypt_in_place(
570        &self,
571        sample: &[u8],
572        first: &mut u8,
573        packet_number: &mut [u8],
574    ) -> Result<(), Error> {
575        self.xor_in_place(sample, first, packet_number, true)
576    }
577
578    fn xor_in_place(
579        &self,
580        sample: &[u8],
581        first: &mut u8,
582        packet_number: &mut [u8],
583        masked: bool,
584    ) -> Result<(), Error> {
585        // This implements [Header Protection Application] almost verbatim.
586
587        let mask = self
588            .0
589            .new_mask(sample)
590            .map_err(|_| Error::General("sample of invalid length".into()))?;
591
592        // The `unwrap()` will not panic because `new_mask` returns a
593        // non-empty result.
594        let (first_mask, pn_mask) = mask.split_first().unwrap();
595
596        // It is OK for the `mask` to be longer than `packet_number`,
597        // but a valid `packet_number` will never be longer than `mask`.
598        if packet_number.len() > pn_mask.len() {
599            return Err(Error::General("packet number too long".into()));
600        }
601
602        // Infallible from this point on. Before this point, `first` and
603        // `packet_number` are unchanged.
604
605        const LONG_HEADER_FORM: u8 = 0x80;
606        let bits = match *first & LONG_HEADER_FORM == LONG_HEADER_FORM {
607            true => 0x0f,  // Long header: 4 bits masked
608            false => 0x1f, // Short header: 5 bits masked
609        };
610
611        let first_plain = match masked {
612            // When unmasking, use the packet length bits after unmasking
613            true => *first ^ (first_mask & bits),
614            // When masking, use the packet length bits before masking
615            false => *first,
616        };
617        let pn_len = (first_plain & 0x03) as usize + 1;
618
619        *first ^= first_mask & bits;
620        for (dst, m) in packet_number
621            .iter_mut()
622            .zip(pn_mask)
623            .take(pn_len)
624        {
625            *dst ^= m;
626        }
627
628        Ok(())
629    }
630
631    /// Expected sample length for the key's algorithm
632    #[inline]
633    pub fn sample_len(&self) -> usize {
634        self.0.algorithm().sample_len()
635    }
636}

638/// Keys to encrypt or decrypt the payload of a packet
639pub struct PacketKey {
640    /// Encrypts or decrypts a packet's payload
641    key: aead::LessSafeKey,
642    /// Computes unique nonces for each packet
643    iv: Iv,
644    /// The cipher suite used for this packet key
645    suite: &'static Tls13CipherSuite,
646}
647
648impl PacketKey {
649    fn new(suite: &'static Tls13CipherSuite, secret: &hkdf::Prk, version: Version) -> Self {
650        Self {
651            key: aead::LessSafeKey::new(hkdf_expand(
652                secret,
653                suite.common.aead_algorithm,
654                version.packet_key_label(),
655                &[],
656            )),
657            iv: hkdf_expand(secret, IvLen, version.packet_iv_label(), &[]),
658            suite,
659        }
660    }
661
662    /// Encrypt a QUIC packet
663    ///
664    /// Takes a `packet_number`, used to derive the nonce; the packet `header`, which is used as
665    /// the additional authenticated data; and the `payload`. The authentication tag is returned if
666    /// encryption succeeds.
667    ///
668    /// Fails iff the payload is longer than allowed by the cipher suite's AEAD algorithm.
669    pub fn encrypt_in_place(
670        &self,
671        packet_number: u64,
672        header: &[u8],
673        payload: &mut [u8],
674    ) -> Result<Tag, Error> {
675        let aad = aead::Aad::from(header);
676        let nonce = nonce_for(packet_number, &self.iv);
677        let tag = self
678            .key
679            .seal_in_place_separate_tag(nonce, aad, payload)
680            .map_err(|_| Error::EncryptError)?;
681        Ok(Tag(tag))
682    }
683
684    /// Decrypt a QUIC packet
685    ///
686    /// Takes the packet `header`, which is used as the additional authenticated data, and the
687    /// `payload`, which includes the authentication tag.
688    ///
689    /// If the return value is `Ok`, the decrypted payload can be found in `payload`, up to the
690    /// length found in the return value.
691    pub fn decrypt_in_place<'a>(
692        &self,
693        packet_number: u64,
694        header: &[u8],
695        payload: &'a mut [u8],
696    ) -> Result<&'a [u8], Error> {
697        let payload_len = payload.len();
698        let aad = aead::Aad::from(header);
699        let nonce = nonce_for(packet_number, &self.iv);
700        self.key
701            .open_in_place(nonce, aad, payload)
702            .map_err(|_| Error::DecryptError)?;
703
704        let plain_len = payload_len - self.key.algorithm().tag_len();
705        Ok(&payload[..plain_len])
706    }
707
708    /// Number of times the packet key can be used without sacrificing confidentiality
709    ///
710    /// See <https://www.rfc-editor.org/rfc/rfc9001.html#name-confidentiality-limit>.
711    #[inline]
712    pub fn confidentiality_limit(&self) -> u64 {
713        self.suite.confidentiality_limit
714    }
715
716    /// Number of times the packet key can be used without sacrificing integrity
717    ///
718    /// See <https://www.rfc-editor.org/rfc/rfc9001.html#name-integrity-limit>.
719    #[inline]
720    pub fn integrity_limit(&self) -> u64 {
721        self.suite.integrity_limit
722    }
723
724    /// Tag length for the underlying AEAD algorithm
725    #[inline]
726    pub fn tag_len(&self) -> usize {
727        self.key.algorithm().tag_len()
728    }
729}

759/// Complete set of keys used to communicate with the peer
760pub struct Keys {
761    /// Encrypts outgoing packets
762    pub local: DirectionalKeys,
763    /// Decrypts incoming packets
764    pub remote: DirectionalKeys,
765}
766
767impl Keys {
768    /// Construct keys for use with initial packets
769    pub fn initial(version: Version, client_dst_connection_id: &[u8], side: Side) -> Self {
770        const CLIENT_LABEL: &[u8] = b"client in";
771        const SERVER_LABEL: &[u8] = b"server in";
772        let salt = version.initial_salt();
773        let hs_secret = hkdf::Salt::new(hkdf::HKDF_SHA256, salt).extract(client_dst_connection_id);
774
775        let secrets = Secrets {
776            version,
777            client: hkdf_expand(&hs_secret, hkdf::HKDF_SHA256, CLIENT_LABEL, &[]),
778            server: hkdf_expand(&hs_secret, hkdf::HKDF_SHA256, SERVER_LABEL, &[]),
779            suite: TLS13_AES_128_GCM_SHA256_INTERNAL,
780            side,
781        };
782        Self::new(&secrets)
783    }
784
785    fn new(secrets: &Secrets) -> Self {
786        let (local, remote) = secrets.local_remote();
787        Self {
788            local: DirectionalKeys::new(secrets.suite, local, secrets.version),
789            remote: DirectionalKeys::new(secrets.suite, remote, secrets.version),
790        }
791    }

794/// Key material for use in QUIC packet spaces
795///
796/// QUIC uses 4 different sets of keys (and progressive key updates for long-running connections):
797///
798/// * Initial: these can be created from [`Keys::initial()`]
799/// * 0-RTT keys: can be retrieved from [`ConnectionCommon::zero_rtt_keys()`]
800/// * Handshake: these are returned from [`ConnectionCommon::write_hs()`] after `ClientHello` and
801///   `ServerHello` messages have been exchanged
802/// * 1-RTT keys: these are returned from [`ConnectionCommon::write_hs()`] after the handshake is done
803///
804/// Once the 1-RTT keys have been exchanged, either side may initiate a key update. Progressive
805/// update keys can be obtained from the [`Secrets`] returned in [`KeyChange::OneRtt`]. Note that
806/// only packet keys are updated by key updates; header protection keys remain the same.
807#[allow(clippy::large_enum_variant)]
808pub enum KeyChange {
809    /// Keys for the handshake space
810    Handshake {
811        /// Header and packet keys for the handshake space
812        keys: Keys,
813    },
814    /// Keys for 1-RTT data
815    OneRtt {
816        /// Header and packet keys for 1-RTT data
817        keys: Keys,
818        /// Secrets to derive updated keys from
819        next: Secrets,
820    },
821}
822
823/// Compute the nonce to use for encrypting or decrypting `packet_number`
824fn nonce_for(packet_number: u64, iv: &Iv) -> aead::Nonce {
825    let mut out = [0; aead::NONCE_LEN];
826    out[4..].copy_from_slice(&packet_number.to_be_bytes());
827    for (out, inp) in out.iter_mut().zip(iv.0.iter()) {
828        *out ^= inp;
829    }
830    aead::Nonce::assume_unique_for_key(out)
831}

833/// QUIC protocol version
834///
835/// Governs version-specific behavior in the TLS layer
836#[non_exhaustive]
837#[derive(Clone, Copy, Debug)]
838pub enum Version {
839    /// Draft versions 29, 30, 31 and 32
840    V1Draft,
841    /// First stable RFC
842    V1,
843    /// Anti-ossification variant of V1
844    V2,
845}
846
847impl Version {
848    fn initial_salt(self) -> &'static [u8; 20] {
849        match self {
850            Self::V1Draft => &[
851                // https://datatracker.ietf.org/doc/html/draft-ietf-quic-tls-32#section-5.2
852                0xaf, 0xbf, 0xec, 0x28, 0x99, 0x93, 0xd2, 0x4c, 0x9e, 0x97, 0x86, 0xf1, 0x9c, 0x61,
853                0x11, 0xe0, 0x43, 0x90, 0xa8, 0x99,
854            ],
855            Self::V1 => &[
856                // https://www.rfc-editor.org/rfc/rfc9001.html#name-initial-secrets
857                0x38, 0x76, 0x2c, 0xf7, 0xf5, 0x59, 0x34, 0xb3, 0x4d, 0x17, 0x9a, 0xe6, 0xa4, 0xc8,
858                0x0c, 0xad, 0xcc, 0xbb, 0x7f, 0x0a,
859            ],
860            Self::V2 => &[
861                // https://www.ietf.org/archive/id/draft-ietf-quic-v2-10.html#name-initial-salt-2
862                0x0d, 0xed, 0xe3, 0xde, 0xf7, 0x00, 0xa6, 0xdb, 0x81, 0x93, 0x81, 0xbe, 0x6e, 0x26,
863                0x9d, 0xcb, 0xf9, 0xbd, 0x2e, 0xd9,
864            ],
865        }
866    }
867
868    fn packet_key_label(&self) -> &'static [u8] {
869        match self {
870            Self::V1Draft | Self::V1 => b"quic key",
871            Self::V2 => b"quicv2 key",
872        }
873    }
874
875    fn packet_iv_label(&self) -> &'static [u8] {
876        match self {
877            Self::V1Draft | Self::V1 => b"quic iv",
878            Self::V2 => b"quicv2 iv",
879        }
880    }
881
882    fn header_key_label(&self) -> &'static [u8] {
883        match self {
884            Self::V1Draft | Self::V1 => b"quic hp",
885            Self::V2 => b"quicv2 hp",
886        }
887    }
888
889    fn key_update_label(&self) -> &'static [u8] {
890        match self {
891            Self::V1Draft | Self::V1 => b"quic ku",
892            Self::V2 => b"quicv2 ku",
893        }
894    }
895}

908        const PN: u64 = 654360564;

914

916        use crate::tls13::TLS13_CHACHA20_POLY1305_SHA256_INTERNAL;
917        let hpk =
