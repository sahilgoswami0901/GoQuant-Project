14//! # Using Ed25519 generically over algorithm implementations/providers
15//!
16//! By using the `ed25519` crate, you can write code which signs and verifies
17//! messages using the Ed25519 signature algorithm generically over any
18//! supported Ed25519 implementation (see the next section for available
19//! providers).
20//!
21//! This allows consumers of your code to plug in whatever implementation they
22//! want to use without having to add all potential Ed25519 libraries you'd
23//! like to support as optional dependencies.
24//!
25//! ## Example
26//!
27//! ```
28//! use ed25519::signature::{Signer, Verifier};
29//!
30//! pub struct HelloSigner<S>
31//! where
32//!     S: Signer<ed25519::Signature>
33//! {
34//!     pub signing_key: S
35//! }
36//!
37//! impl<S> HelloSigner<S>
38//! where
39//!     S: Signer<ed25519::Signature>
40//! {
41//!     pub fn sign(&self, person: &str) -> ed25519::Signature {
42//!         // NOTE: use `try_sign` if you'd like to be able to handle
43//!         // errors from external signing services/devices (e.g. HSM/KMS)
44//!         // <https://docs.rs/signature/latest/signature/trait.Signer.html#tymethod.try_sign>
45//!         self.signing_key.sign(format_message(person).as_bytes())
46//!     }
47//! }
48//!
49//! pub struct HelloVerifier<V> {
50//!     pub verify_key: V
51//! }
52//!
53//! impl<V> HelloVerifier<V>
54//! where
55//!     V: Verifier<ed25519::Signature>
56//! {
57//!     pub fn verify(
58//!         &self,
59//!         person: &str,
60//!         signature: &ed25519::Signature
61//!     ) -> Result<(), ed25519::Error> {
62//!         self.verify_key.verify(format_message(person).as_bytes(), signature)
63//!     }
64//! }
65//!
66//! fn format_message(person: &str) -> String {
67//!     format!("Hello, {}!", person)
68//! }
69//! ```
70//!
71//! ## Using above example with `ed25519-dalek`
72//!
73//! The [`ed25519-dalek`] crate natively supports the [`ed25519::Signature`][`Signature`]
74//! type defined in this crate along with the [`signature::Signer`] and
75//! [`signature::Verifier`] traits.
76//!
77//! Below is an example of how a hypothetical consumer of the code above can
78//! instantiate and use the previously defined `HelloSigner` and `HelloVerifier`
79//! types with [`ed25519-dalek`] as the signing/verification provider:
80//!
81//! ```
82//! use ed25519_dalek::{Signer, Verifier, Signature};
83//! #
84//! # pub struct HelloSigner<S>
85//! # where
86//! #     S: Signer<Signature>
87//! # {
88//! #     pub signing_key: S
89//! # }
90//! #
91//! # impl<S> HelloSigner<S>
92//! # where
93//! #     S: Signer<Signature>
94//! # {
95//! #     pub fn sign(&self, person: &str) -> Signature {
96//! #         // NOTE: use `try_sign` if you'd like to be able to handle
97//! #         // errors from external signing services/devices (e.g. HSM/KMS)
98//! #         // <https://docs.rs/signature/latest/signature/trait.Signer.html#tymethod.try_sign>
99//! #         self.signing_key.sign(format_message(person).as_bytes())
100//! #     }
101//! # }
102//! #
103//! # pub struct HelloVerifier<V> {
104//! #     pub verify_key: V
105//! # }
106//! #
107//! # impl<V> HelloVerifier<V>
108//! # where
109//! #     V: Verifier<Signature>
110//! # {
111//! #     pub fn verify(
112//! #         &self,
113//! #         person: &str,
114//! #         signature: &Signature
115//! #     ) -> Result<(), ed25519::Error> {
116//! #         self.verify_key.verify(format_message(person).as_bytes(), signature)
117//! #     }
118//! # }
119//! #
120//! # fn format_message(person: &str) -> String {
121//! #     format!("Hello, {}!", person)
122//! # }
123//! use rand_core::OsRng; // Requires the `std` feature of `rand_core`
124//!
125//! /// `HelloSigner` defined above instantiated with `ed25519-dalek` as
126//! /// the signing provider.
127//! pub type DalekHelloSigner = HelloSigner<ed25519_dalek::Keypair>;
128//!
129//! let signing_key = ed25519_dalek::Keypair::generate(&mut OsRng);
130//! let signer = DalekHelloSigner { signing_key };
131//! let person = "Joe"; // Message to sign
132//! let signature = signer.sign(person);
133//!
134//! /// `HelloVerifier` defined above instantiated with `ed25519-dalek`
135//! /// as the signature verification provider.
136//! pub type DalekHelloVerifier = HelloVerifier<ed25519_dalek::PublicKey>;
137//!
138//! let verify_key: ed25519_dalek::PublicKey = signer.signing_key.public;
139//! let verifier = DalekHelloVerifier { verify_key };
140//! assert!(verifier.verify(person, &signature).is_ok());
141//! ```
142//!
143//! ## Using above example with `ring-compat`
144//!
145//! The [`ring-compat`] crate provides wrappers for [*ring*] which implement
146//! the [`signature::Signer`] and [`signature::Verifier`] traits for
147//! [`ed25519::Signature`][`Signature`].
148//!
149//! Below is an example of how a hypothetical consumer of the code above can
150//! instantiate and use the previously defined `HelloSigner` and `HelloVerifier`
151//! types with [`ring-compat`] as the signing/verification provider:
152//!
153//! ```ignore
154//! use ring_compat::signature::{
155//!     ed25519::{Signature, SigningKey, VerifyingKey},
156//!     Signer, Verifier
157//! };
158//! #
159//! # pub struct HelloSigner<S>
160//! # where
161//! #     S: Signer<Signature>
162//! # {
163//! #     pub signing_key: S
164//! # }
165//! #
166//! # impl<S> HelloSigner<S>
167//! # where
168//! #     S: Signer<Signature>
169//! # {
170//! #     pub fn sign(&self, person: &str) -> Signature {
171//! #         // NOTE: use `try_sign` if you'd like to be able to handle
172//! #         // errors from external signing services/devices (e.g. HSM/KMS)
173//! #         // <https://docs.rs/signature/latest/signature/trait.Signer.html#tymethod.try_sign>
174//! #         self.signing_key.sign(format_message(person).as_bytes())
175//! #     }
176//! # }
177//! #
178//! # pub struct HelloVerifier<V> {
179//! #     pub verify_key: V
180//! # }
181//! #
182//! # impl<V> HelloVerifier<V>
183//! # where
184//! #     V: Verifier<Signature>
185//! # {
186//! #     pub fn verify(
187//! #         &self,
188//! #         person: &str,
189//! #         signature: &Signature
190//! #     ) -> Result<(), ed25519::Error> {
191//! #         self.verify_key.verify(format_message(person).as_bytes(), signature)
192//! #     }
193//! # }
194//! #
195//! # fn format_message(person: &str) -> String {
196//! #     format!("Hello, {}!", person)
197//! # }
198//! use rand_core::{OsRng, RngCore}; // Requires the `std` feature of `rand_core`
199//!
200//! /// `HelloSigner` defined above instantiated with *ring* as
201//! /// the signing provider.
202//! pub type RingHelloSigner = HelloSigner<SigningKey>;
203//!
204//! let mut ed25519_seed = [0u8; 32];
205//! OsRng.fill_bytes(&mut ed25519_seed);
206//!
207//! let signing_key = SigningKey::from_seed(&ed25519_seed).unwrap();
208//! let verify_key = signing_key.verify_key();
209//!
210//! let signer = RingHelloSigner { signing_key };
211//! let person = "Joe"; // Message to sign
212//! let signature = signer.sign(person);
213//!
214//! /// `HelloVerifier` defined above instantiated with *ring*
215//! /// as the signature verification provider.
216//! pub type RingHelloVerifier = HelloVerifier<VerifyingKey>;
217//!
218//! let verifier = RingHelloVerifier { verify_key };
219//! assert!(verifier.verify(person, &signature).is_ok());
220//! ```
221//!
222//! # Available Ed25519 providers
223//!
224//! The following libraries support the types/traits from the `ed25519` crate:
225//!
226//! - [`ed25519-dalek`] - mature pure Rust implementation of Ed25519
227//! - [`ring-compat`] - compatibility wrapper for [*ring*]
228//! - [`yubihsm`] - host-side client library for YubiHSM2 devices from Yubico
229//!
230//! [`ed25519-dalek`]: https://docs.rs/ed25519-dalek
231//! [`ring-compat`]: https://docs.rs/ring-compat
232//! [*ring*]: https://github.com/briansmith/ring
233//! [`yubihsm`]: https://github.com/iqlusioninc/yubihsm.rs/blob/develop/README.md
234//!
235//! # Features
236//!
237//! The following features are presently supported:
238//!
239//! - `pkcs8`: support for decoding/encoding PKCS#8-formatted private keys using the
240//!   [`KeypairBytes`] type.
241//! - `std` *(default)*: Enable `std` support in [`signature`], which currently only affects whether
242//!   [`signature::Error`] implements `std::error::Error`.
243//! - `serde`: Implement `serde::Deserialize` and `serde::Serialize` for [`Signature`]. Signatures
244//!   are serialized as their bytes.
245//! - `serde_bytes`: Implement `serde_bytes::Deserialize` and `serde_bytes::Serialize` for
246//!   [`Signature`]. This enables more compact representations for formats with an efficient byte
247//!   array representation. As per the `serde_bytes` documentation, this can most easily be realised
248//!   using the `#[serde(with = "serde_bytes")]` annotation, e.g.:
249//!
250//!   ```ignore
251//!   # use ed25519::Signature;
252//!   # use serde::{Deserialize, Serialize};
253//!   #[derive(Deserialize, Serialize)]
254//!   #[serde(transparent)]
255//!   struct SignatureAsBytes(#[serde(with = "serde_bytes")] Signature);
256//!   ```

282/// Ed25519 signature.
283#[derive(Copy, Clone, Eq, PartialEq)]
284pub struct Signature([u8; Signature::BYTE_SIZE]);
285
286impl Signature {
287    /// Size of an encoded Ed25519 signature in bytes.
288    pub const BYTE_SIZE: usize = 64;
289
290    /// Parse an Ed25519 signature from a byte slice.
291    pub fn from_bytes(bytes: &[u8]) -> signature::Result<Self> {
292        let result = bytes.try_into().map(Self).map_err(|_| Error::new())?;
293
294        // Perform a partial reduction check on the signature's `s` scalar.
295        // When properly reduced, at least the three highest bits of the scalar
296        // will be unset so as to fit within the order of ~2^(252.5).
297        //
298        // This doesn't ensure that `s` is fully reduced (which would require a
299        // full reduction check in the event that the 4th most significant bit
300        // is set), however it will catch a number of invalid signatures
301        // relatively inexpensively.
302        if result.0[Signature::BYTE_SIZE - 1] & 0b1110_0000 != 0 {
303            return Err(Error::new());
304        }
305
306        Ok(result)
307    }
308
309    /// Return the inner byte array.
310    pub fn to_bytes(self) -> [u8; Self::BYTE_SIZE] {
311        self.0
312    }
313
314    /// Convert this signature into a byte vector.
315    #[cfg(feature = "alloc")]
316    #[cfg_attr(docsrs, doc(cfg(feature = "alloc")))]
317    pub fn to_vec(&self) -> Vec<u8> {
318        self.0.to_vec()
319    }
320
321    /// DEPRECATED: Create a new signature from a byte array.
322    ///
323    /// # Panics
324    ///
325    /// This method will panic if an invalid signature is encountered.
326    ///
327    /// Use [`Signature::from_bytes`] or [`Signature::try_from`] instead for
328    /// a fallible conversion.
329    #[deprecated(since = "1.3.0", note = "use ed25519::Signature::from_bytes instead")]
330    pub fn new(bytes: [u8; Self::BYTE_SIZE]) -> Self {
331        Self::from_bytes(&bytes[..]).expect("invalid signature")
332    }
333}

410/// Decode a signature from hexadecimal.
411///
412/// Upper and lower case hexadecimal are both accepted, however mixed case is
413/// rejected.
414// TODO(tarcieri): use `base16ct`?
415impl str::FromStr for Signature {
416    type Err = Error;
417
418    fn from_str(hex: &str) -> signature::Result<Self> {
419        if hex.as_bytes().len() != Signature::BYTE_SIZE * 2 {
420            return Err(Error::new());
421        }
422
423        let mut upper_case = None;
424
425        // Ensure all characters are valid and case is not mixed
426        for &byte in hex.as_bytes() {
427            match byte {
428                b'0'..=b'9' => (),
429                b'a'..=b'z' => match upper_case {
430                    Some(true) => return Err(Error::new()),
431                    Some(false) => (),
432                    None => upper_case = Some(false),
433                },
434                b'A'..=b'Z' => match upper_case {
435                    Some(true) => (),
436                    Some(false) => return Err(Error::new()),
437                    None => upper_case = Some(true),
438                },
439                _ => return Err(Error::new()),
440            }
441        }
442
443        let mut result = [0u8; Self::BYTE_SIZE];
444        for (digit, byte) in hex.as_bytes().chunks_exact(2).zip(result.iter_mut()) {
445            *byte = str::from_utf8(digit)
446                .ok()
447                .and_then(|s| u8::from_str_radix(s, 16).ok())
448                .ok_or_else(Error::new)?;
449        }
450
451        Self::try_from(&result[..])
452    }
