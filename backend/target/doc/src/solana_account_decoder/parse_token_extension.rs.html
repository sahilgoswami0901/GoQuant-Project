43pub fn parse_extension<S: BaseState>(
44    extension_type: &ExtensionType,
45    account: &StateWithExtensions<S>,
46) -> UiExtension {
47    match extension_type {
48        ExtensionType::Uninitialized => UiExtension::Uninitialized,
49        ExtensionType::TransferFeeConfig => account
50            .get_extension::<extension::transfer_fee::TransferFeeConfig>()
51            .map(|&extension| UiExtension::TransferFeeConfig(extension.into()))
52            .unwrap_or(UiExtension::UnparseableExtension),
53        ExtensionType::TransferFeeAmount => account
54            .get_extension::<extension::transfer_fee::TransferFeeAmount>()
55            .map(|&extension| UiExtension::TransferFeeAmount(extension.into()))
56            .unwrap_or(UiExtension::UnparseableExtension),
57        ExtensionType::MintCloseAuthority => account
58            .get_extension::<extension::mint_close_authority::MintCloseAuthority>()
59            .map(|&extension| UiExtension::MintCloseAuthority(extension.into()))
60            .unwrap_or(UiExtension::UnparseableExtension),
61        ExtensionType::ConfidentialTransferMint => account
62            .get_extension::<extension::confidential_transfer::ConfidentialTransferMint>()
63            .map(|&extension| UiExtension::ConfidentialTransferMint(extension.into()))
64            .unwrap_or(UiExtension::UnparseableExtension),
65        ExtensionType::ConfidentialTransferFeeConfig => account
66            .get_extension::<extension::confidential_transfer_fee::ConfidentialTransferFeeConfig>()
67            .map(|&extension| UiExtension::ConfidentialTransferFeeConfig(extension.into()))
68            .unwrap_or(UiExtension::UnparseableExtension),
69        ExtensionType::ConfidentialTransferAccount => account
70            .get_extension::<extension::confidential_transfer::ConfidentialTransferAccount>()
71            .map(|&extension| UiExtension::ConfidentialTransferAccount(extension.into()))
72            .unwrap_or(UiExtension::UnparseableExtension),
73        ExtensionType::ConfidentialTransferFeeAmount => account
74            .get_extension::<extension::confidential_transfer_fee::ConfidentialTransferFeeAmount>()
75            .map(|&extension| UiExtension::ConfidentialTransferFeeAmount(extension.into()))
76            .unwrap_or(UiExtension::UnparseableExtension),
77        ExtensionType::DefaultAccountState => account
78            .get_extension::<extension::default_account_state::DefaultAccountState>()
79            .map(|&extension| UiExtension::DefaultAccountState(extension.into()))
80            .unwrap_or(UiExtension::UnparseableExtension),
81        ExtensionType::ImmutableOwner => UiExtension::ImmutableOwner,
82        ExtensionType::MemoTransfer => account
83            .get_extension::<extension::memo_transfer::MemoTransfer>()
84            .map(|&extension| UiExtension::MemoTransfer(extension.into()))
85            .unwrap_or(UiExtension::UnparseableExtension),
86        ExtensionType::NonTransferable => UiExtension::NonTransferable,
87        ExtensionType::InterestBearingConfig => account
88            .get_extension::<extension::interest_bearing_mint::InterestBearingConfig>()
89            .map(|&extension| UiExtension::InterestBearingConfig(extension.into()))
90            .unwrap_or(UiExtension::UnparseableExtension),
91        ExtensionType::CpiGuard => account
92            .get_extension::<extension::cpi_guard::CpiGuard>()
93            .map(|&extension| UiExtension::CpiGuard(extension.into()))
94            .unwrap_or(UiExtension::UnparseableExtension),
95        ExtensionType::PermanentDelegate => account
96            .get_extension::<extension::permanent_delegate::PermanentDelegate>()
97            .map(|&extension| UiExtension::PermanentDelegate(extension.into()))
98            .unwrap_or(UiExtension::UnparseableExtension),
99        ExtensionType::NonTransferableAccount => UiExtension::NonTransferableAccount,
100        ExtensionType::MetadataPointer => account
101            .get_extension::<extension::metadata_pointer::MetadataPointer>()
102            .map(|&extension| UiExtension::MetadataPointer(extension.into()))
103            .unwrap_or(UiExtension::UnparseableExtension),
104        ExtensionType::TokenMetadata => account
105            .get_variable_len_extension::<TokenMetadata>()
106            .map(|extension| UiExtension::TokenMetadata(extension.into()))
107            .unwrap_or(UiExtension::UnparseableExtension),
108        ExtensionType::TransferHook => account
109            .get_extension::<extension::transfer_hook::TransferHook>()
110            .map(|&extension| UiExtension::TransferHook(extension.into()))
111            .unwrap_or(UiExtension::UnparseableExtension),
112        ExtensionType::TransferHookAccount => account
113            .get_extension::<extension::transfer_hook::TransferHookAccount>()
114            .map(|&extension| UiExtension::TransferHookAccount(extension.into()))
115            .unwrap_or(UiExtension::UnparseableExtension),
116        ExtensionType::GroupPointer => account
117            .get_extension::<extension::group_pointer::GroupPointer>()
118            .map(|&extension| UiExtension::GroupPointer(extension.into()))
119            .unwrap_or(UiExtension::UnparseableExtension),
120        ExtensionType::GroupMemberPointer => account
121            .get_extension::<extension::group_member_pointer::GroupMemberPointer>()
122            .map(|&extension| UiExtension::GroupMemberPointer(extension.into()))
123            .unwrap_or(UiExtension::UnparseableExtension),
124        ExtensionType::TokenGroup => account
125            .get_extension::<TokenGroup>()
126            .map(|&extension| UiExtension::TokenGroup(extension.into()))
127            .unwrap_or(UiExtension::UnparseableExtension),
128        ExtensionType::TokenGroupMember => account
129            .get_extension::<TokenGroupMember>()
130            .map(|&extension| UiExtension::TokenGroupMember(extension.into()))
131            .unwrap_or(UiExtension::UnparseableExtension),
132    }
133}

163impl From<extension::transfer_fee::TransferFeeConfig> for UiTransferFeeConfig {
164    fn from(transfer_fee_config: extension::transfer_fee::TransferFeeConfig) -> Self {
165        let transfer_fee_config_authority: Option<Pubkey> =

172                .map(|pubkey| pubkey.to_string()),
173            withdraw_withheld_authority: withdraw_withheld_authority
174                .map(|pubkey| pubkey.to_string()),
175            withheld_amount: u64::from(transfer_fee_config.withheld_amount),
176            older_transfer_fee: transfer_fee_config.older_transfer_fee.into(),
177            newer_transfer_fee: transfer_fee_config.newer_transfer_fee.into(),
178        }
179    }
