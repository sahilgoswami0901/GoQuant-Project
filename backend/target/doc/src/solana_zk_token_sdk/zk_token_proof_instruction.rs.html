1//! Instructions provided by the [`ZK Token proof`] program.
2//!
3//! There are two types of instructions in the proof program: proof verification instructions
4//! and the `CloseContextState` instruction.
5//!
6//! Each proof verification instruction verifies a certain type of zero-knowledge proof. These
7//! instructions are processed by the program in two steps:
8//!   1. The program verifies the zero-knowledge proof.
9//!   2. The program optionally stores the context component of the zero-knowledge proof to a
10//!      dedicated [`context-state`] account.
11//!
12//! In step 1, the zero-knowledge proof can be included directly as the instruction data or
13//! pre-written to an account. The program determines whether the proof is provided as instruction
14//! data or pre-written to an account by inspecting the length of the data. If the instruction data
15//! is exactly 5 bytes (instruction disciminator + unsigned 32-bit integer), then the program
16//! assumes that the first account provided with the instruction contains the zero-knowledge proof
17//! and verifies the account data at the offset specified in the instruction data. Otherwise, the
18//! program assumes that the zero-knowledge proof is provided as part of the instruction data.
19//!
20//! In step 2, the program determines whether to create a context-state account by inspecting the
21//! number of accounts provided with the instruction. If two additional accounts are provided with
22//! the instruction after verifying the zero-knowledge proof, then the program writes the context data to
23//! the specified context-state account.
24//!
25//! NOTE: A context-state account must be pre-allocated to the exact size of the context data that
26//! is expected for a proof type before it is included in a proof verification instruction.
27//!
28//! The `CloseContextState` instruction closes a context state account. A transaction containing
29//! this instruction must be signed by the context account's owner. This instruction can be used by
30//! the account owner to reclaim lamports for storage.
31//!
32//! [`ZK Token proof`]: https://docs.solanalabs.com/runtime/zk-token-proof
33//! [`context-state`]: https://docs.solanalabs.com/runtime/zk-token-proof#context-data

311/// Pubkeys associated with a context state account to be used as parameters to functions.
312#[derive(Clone, Copy, Debug, PartialEq)]
313pub struct ContextStateInfo<'a> {
314    pub context_state_account: &'a Pubkey,
315    pub context_state_authority: &'a Pubkey,
316}
317
318/// Create a `CloseContextState` instruction.
319pub fn close_context_state(
320    context_state_info: ContextStateInfo,
321    destination_account: &Pubkey,
322) -> Instruction {
323    let accounts = vec![
324        AccountMeta::new(*context_state_info.context_state_account, false),
325        AccountMeta::new(*destination_account, false),
326        AccountMeta::new_readonly(*context_state_info.context_state_authority, true),
327    ];
328
329    let data = vec![ToPrimitive::to_u8(&ProofInstruction::CloseContextState).unwrap()];
330
331    Instruction {
332        program_id: crate::zk_token_proof_program::id(),
333        accounts,
334        data,
335    }
336}

430impl ProofInstruction {
431    pub fn encode_verify_proof<T, U>(
432        &self,
433        context_state_info: Option<ContextStateInfo>,
434        proof_data: &T,
435    ) -> Instruction
436    where
437        T: Pod + ZkProofData<U>,
438        U: Pod,
439    {
440        let accounts = if let Some(context_state_info) = context_state_info {
441            vec![
442                AccountMeta::new(*context_state_info.context_state_account, false),
443                AccountMeta::new_readonly(*context_state_info.context_state_authority, false),
444            ]
445        } else {
446            vec![]
447        };
448
449        let mut data = vec![ToPrimitive::to_u8(self).unwrap()];
450        data.extend_from_slice(bytes_of(proof_data));
451
452        Instruction {
453            program_id: crate::zk_token_proof_program::id(),
454            accounts,
455            data,
456        }
457    }
458
459    pub fn encode_verify_proof_from_account(
460        &self,
461        context_state_info: Option<ContextStateInfo>,
462        proof_account: &Pubkey,
463        offset: u32,
464    ) -> Instruction {
465        let accounts = if let Some(context_state_info) = context_state_info {
466            vec![
467                AccountMeta::new(*proof_account, false),
468                AccountMeta::new(*context_state_info.context_state_account, false),
469                AccountMeta::new_readonly(*context_state_info.context_state_authority, false),
470            ]
471        } else {
472            vec![AccountMeta::new(*proof_account, false)]
473        };
474
475        let mut data = vec![ToPrimitive::to_u8(self).unwrap()];
476        data.extend_from_slice(&offset.to_le_bytes());
477
478        Instruction {
479            program_id: crate::zk_token_proof_program::id(),
480            accounts,
481            data,
482        }
483    }
484
485    pub fn instruction_type(input: &[u8]) -> Option<Self> {
486        input
487            .first()
488            .and_then(|instruction| FromPrimitive::from_u8(*instruction))
489    }
490
491    pub fn proof_data<T, U>(input: &[u8]) -> Option<&T>
492    where
493        T: Pod + ZkProofData<U>,
494        U: Pod,
495    {
496        input
497            .get(1..)
498            .and_then(|data| bytemuck::try_from_bytes(data).ok())
499    }
