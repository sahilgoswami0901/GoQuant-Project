33/// Takes in a 64-bit number `amount` and a bit length `bit_length`. It returns:
34///  - the `bit_length` low bits of `amount` interpreted as u64
35///  - the (64 - `bit_length`) high bits of `amount` interpreted as u64
36#[cfg(not(target_os = "solana"))]
37pub fn split_u64(amount: u64, bit_length: usize) -> (u64, u64) {
38    if bit_length == 64 {
39        (amount, 0)
40    } else {
41        let lo = amount << (64 - bit_length) >> (64 - bit_length);
42        let hi = amount >> bit_length;
43        (lo, hi)
44    }
45}

47#[cfg(not(target_os = "solana"))]
48pub fn combine_lo_hi_u64(amount_lo: u64, amount_hi: u64, bit_length: usize) -> u64 {
49    if bit_length == 64 {
50        amount_lo
51    } else {
52        amount_lo + (amount_hi << bit_length)
53    }
54}

56#[cfg(not(target_os = "solana"))]
57fn combine_lo_hi_ciphertexts(
58    ciphertext_lo: &ElGamalCiphertext,
59    ciphertext_hi: &ElGamalCiphertext,
60    bit_length: usize,
61) -> ElGamalCiphertext {
62    let two_power = (1_u64) << bit_length;
63    ciphertext_lo + &(ciphertext_hi * &Scalar::from(two_power))
64}
