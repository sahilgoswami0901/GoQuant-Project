63/// The context data needed to verify a transfer proof.

66pub struct TransferProofContext {
67    /// Group encryption of the low 16 bits of the transfer amount
68    pub ciphertext_lo: pod::TransferAmountCiphertext, // 128 bytes

70    /// Group encryption of the high 48 bits of the transfer amount

72
73    /// The public encryption keys associated with the transfer: source, dest, and auditor
74    pub transfer_pubkeys: TransferPubkeys, // 96 bytes
75
76    /// The final spendable ciphertext after the transfer
77    pub new_source_ciphertext: pod::ElGamalCiphertext, // 64 bytes

89#[cfg(not(target_os = "solana"))]
90impl TransferData {
91    #[allow(clippy::too_many_arguments)]
92    pub fn new(
93        transfer_amount: u64,
94        (spendable_balance, ciphertext_old_source): (u64, &ElGamalCiphertext),
95        source_keypair: &ElGamalKeypair,
96        (destination_pubkey, auditor_pubkey): (&ElGamalPubkey, &ElGamalPubkey),
97    ) -> Result<Self, ProofGenerationError> {
98        // split and encrypt transfer amount
99        let (amount_lo, amount_hi) = split_u64(transfer_amount, TRANSFER_AMOUNT_LO_BITS);
100
101        let (ciphertext_lo, opening_lo) = TransferAmountCiphertext::new(
102            amount_lo,
103            source_keypair.pubkey(),
104            destination_pubkey,
105            auditor_pubkey,
106        );
107
108        let (ciphertext_hi, opening_hi) = TransferAmountCiphertext::new(
109            amount_hi,
110            source_keypair.pubkey(),
111            destination_pubkey,
112            auditor_pubkey,
113        );
114
115        // subtract transfer amount from the spendable ciphertext
116        let new_spendable_balance = spendable_balance
117            .checked_sub(transfer_amount)
118            .ok_or(ProofGenerationError::NotEnoughFunds)?;
119
120        let transfer_amount_lo_source = ElGamalCiphertext {
121            commitment: *ciphertext_lo.get_commitment(),
122            handle: *ciphertext_lo.get_source_handle(),
123        };
124
125        let transfer_amount_hi_source = ElGamalCiphertext {
126            commitment: *ciphertext_hi.get_commitment(),
127            handle: *ciphertext_hi.get_source_handle(),
128        };
129
130        let new_source_ciphertext = ciphertext_old_source
131            - combine_lo_hi_ciphertexts(
132                &transfer_amount_lo_source,
133                &transfer_amount_hi_source,
134                TRANSFER_AMOUNT_LO_BITS,
135            );
136
137        // generate transcript and append all public inputs
138        let pod_transfer_pubkeys = TransferPubkeys {
139            source: (*source_keypair.pubkey()).into(),
140            destination: (*destination_pubkey).into(),
141            auditor: (*auditor_pubkey).into(),
142        };
143        let pod_ciphertext_lo: pod::TransferAmountCiphertext = ciphertext_lo.into();
144        let pod_ciphertext_hi: pod::TransferAmountCiphertext = ciphertext_hi.into();
145        let pod_new_source_ciphertext: pod::ElGamalCiphertext = new_source_ciphertext.into();
146
147        let context = TransferProofContext {
148            ciphertext_lo: pod_ciphertext_lo,
149            ciphertext_hi: pod_ciphertext_hi,
150            transfer_pubkeys: pod_transfer_pubkeys,
151            new_source_ciphertext: pod_new_source_ciphertext,
152        };
153
154        let mut transcript = context.new_transcript();
155        let proof = TransferProof::new(
156            (amount_lo, amount_hi),
157            source_keypair,
158            (destination_pubkey, auditor_pubkey),
159            &opening_lo,
160            &opening_hi,
161            (new_spendable_balance, &new_source_ciphertext),
162            &mut transcript,
163        )?;
164
165        Ok(Self { context, proof })
166    }
167
168    /// Extracts the lo ciphertexts associated with a transfer data
169    fn ciphertext_lo(&self, role: Role) -> Result<ElGamalCiphertext, InstructionError> {
170        let ciphertext_lo: TransferAmountCiphertext = self
171            .context
172            .ciphertext_lo
173            .try_into()
174            .map_err(|_| InstructionError::Decryption)?;
175
176        let handle_lo = match role {
177            Role::Source => Some(ciphertext_lo.get_source_handle()),
178            Role::Destination => Some(ciphertext_lo.get_destination_handle()),
179            Role::Auditor => Some(ciphertext_lo.get_auditor_handle()),
180            Role::WithdrawWithheldAuthority => None,
181        };
182
183        if let Some(handle) = handle_lo {
184            Ok(ElGamalCiphertext {
185                commitment: *ciphertext_lo.get_commitment(),
186                handle: *handle,
187            })
188        } else {
189            Err(InstructionError::MissingCiphertext)
190        }
191    }
192
193    /// Extracts the lo ciphertexts associated with a transfer data
194    fn ciphertext_hi(&self, role: Role) -> Result<ElGamalCiphertext, InstructionError> {
195        let ciphertext_hi: TransferAmountCiphertext = self
196            .context
197            .ciphertext_hi
198            .try_into()
199            .map_err(|_| InstructionError::Decryption)?;
200
201        let handle_hi = match role {
202            Role::Source => Some(ciphertext_hi.get_source_handle()),
203            Role::Destination => Some(ciphertext_hi.get_destination_handle()),
204            Role::Auditor => Some(ciphertext_hi.get_auditor_handle()),
205            Role::WithdrawWithheldAuthority => None,
206        };
207
208        if let Some(handle) = handle_hi {
209            Ok(ElGamalCiphertext {
210                commitment: *ciphertext_hi.get_commitment(),
211                handle: *handle,
212            })
213        } else {
214            Err(InstructionError::MissingCiphertext)
215        }
216    }
217
218    /// Decrypts transfer amount from transfer data
219    pub fn decrypt_amount(
220        &self,
221        role: Role,
222        sk: &ElGamalSecretKey,
223    ) -> Result<u64, InstructionError> {
224        let ciphertext_lo = self.ciphertext_lo(role)?;
225        let ciphertext_hi = self.ciphertext_hi(role)?;
226
227        let amount_lo = ciphertext_lo.decrypt_u32(sk);
228        let amount_hi = ciphertext_hi.decrypt_u32(sk);
229
230        if let (Some(amount_lo), Some(amount_hi)) = (amount_lo, amount_hi) {
231            let two_power = 1 << TRANSFER_AMOUNT_LO_BITS;
232            Ok(amount_lo + two_power * amount_hi)
233        } else {
234            Err(InstructionError::Decryption)
235        }
236    }
237}

246    #[cfg(not(target_os = "solana"))]
247    fn verify_proof(&self) -> Result<(), ProofVerificationError> {
248        // generate transcript and append all public inputs
249        let mut transcript = self.context.new_transcript();
250
251        let source_pubkey = self.context.transfer_pubkeys.source.try_into()?;
252        let destination_pubkey = self.context.transfer_pubkeys.destination.try_into()?;
253        let auditor_pubkey = self.context.transfer_pubkeys.auditor.try_into()?;
254
255        let ciphertext_lo = self.context.ciphertext_lo.try_into()?;
256        let ciphertext_hi = self.context.ciphertext_hi.try_into()?;
257        let new_spendable_ciphertext = self.context.new_source_ciphertext.try_into()?;
258
259        self.proof.verify(
260            &source_pubkey,
261            &destination_pubkey,
262            &auditor_pubkey,
263            &ciphertext_lo,
264            &ciphertext_hi,
265            &new_spendable_ciphertext,
266            &mut transcript,
267        )
268    }

287#[allow(non_snake_case)]
288#[derive(Clone, Copy, Pod, Zeroable)]
289#[repr(C)]
290pub struct TransferProof {
291    /// New Pedersen commitment for the remaining balance in source
292    pub new_source_commitment: pod::PedersenCommitment,
293
294    /// Associated equality proof
295    pub equality_proof: pod::CiphertextCommitmentEqualityProof,
296
297    /// Associated ciphertext validity proof
298    pub validity_proof: pod::BatchedGroupedCiphertext2HandlesValidityProof,
299
300    // Associated range proof
301    pub range_proof: pod::RangeProofU128,
302}
303
304#[allow(non_snake_case)]
305#[cfg(not(target_os = "solana"))]
306impl TransferProof {
307    pub fn new(
308        (transfer_amount_lo, transfer_amount_hi): (u64, u64),
309        source_keypair: &ElGamalKeypair,
310        (destination_pubkey, auditor_pubkey): (&ElGamalPubkey, &ElGamalPubkey),
311        opening_lo: &PedersenOpening,
312        opening_hi: &PedersenOpening,
313        (source_new_balance, new_source_ciphertext): (u64, &ElGamalCiphertext),
314        transcript: &mut Transcript,
315    ) -> Result<Self, ProofGenerationError> {
316        // generate a Pedersen commitment for the remaining balance in source
317        let (new_source_commitment, source_opening) = Pedersen::new(source_new_balance);
318
319        let pod_new_source_commitment: pod::PedersenCommitment = new_source_commitment.into();
320        transcript.append_commitment(b"commitment-new-source", &pod_new_source_commitment);
321
322        // generate equality_proof
323        let equality_proof = CiphertextCommitmentEqualityProof::new(
324            source_keypair,
325            new_source_ciphertext,
326            &source_opening,
327            source_new_balance,
328            transcript,
329        );
330
331        // generate ciphertext validity proof
332        let validity_proof = BatchedGroupedCiphertext2HandlesValidityProof::new(
333            (destination_pubkey, auditor_pubkey),
334            (transfer_amount_lo, transfer_amount_hi),
335            (opening_lo, opening_hi),
336            transcript,
337        );
338
339        // generate the range proof
340        let range_proof = if TRANSFER_AMOUNT_LO_BITS == 32 {
341            RangeProof::new(
342                vec![source_new_balance, transfer_amount_lo, transfer_amount_hi],
343                vec![
344                    TRANSFER_SOURCE_AMOUNT_BITS,
345                    TRANSFER_AMOUNT_LO_BITS,
346                    TRANSFER_AMOUNT_HI_BITS,
347                ],
348                vec![&source_opening, opening_lo, opening_hi],
349                transcript,
350            )
351        } else {
352            let transfer_amount_lo_negated =
353                (1 << TRANSFER_AMOUNT_LO_NEGATED_BITS) - 1 - transfer_amount_lo;
354            let opening_lo_negated = &PedersenOpening::default() - opening_lo;
355
356            RangeProof::new(
357                vec![
358                    source_new_balance,
359                    transfer_amount_lo,
360                    transfer_amount_lo_negated,
361                    transfer_amount_hi,
362                ],
363                vec![
364                    TRANSFER_SOURCE_AMOUNT_BITS,
365                    TRANSFER_AMOUNT_LO_BITS,
366                    TRANSFER_AMOUNT_LO_NEGATED_BITS,
367                    TRANSFER_AMOUNT_HI_BITS,
368                ],
369                vec![&source_opening, opening_lo, &opening_lo_negated, opening_hi],
370                transcript,
371            )
372        }?;
373
374        Ok(Self {
375            new_source_commitment: pod_new_source_commitment,
376            equality_proof: equality_proof.into(),
377            validity_proof: validity_proof.into(),
378            range_proof: range_proof
379                .try_into()
380                .map_err(|_| ProofGenerationError::ProofLength)?,
381        })
382    }
383
384    pub fn verify(
385        &self,
386        source_pubkey: &ElGamalPubkey,
387        destination_pubkey: &ElGamalPubkey,
388        auditor_pubkey: &ElGamalPubkey,
389        ciphertext_lo: &TransferAmountCiphertext,
390        ciphertext_hi: &TransferAmountCiphertext,
391        ciphertext_new_spendable: &ElGamalCiphertext,
392        transcript: &mut Transcript,
393    ) -> Result<(), ProofVerificationError> {
394        transcript.append_commitment(b"commitment-new-source", &self.new_source_commitment);
395
396        let commitment: PedersenCommitment = self.new_source_commitment.try_into()?;
397        let equality_proof: CiphertextCommitmentEqualityProof = self.equality_proof.try_into()?;
398        let aggregated_validity_proof: BatchedGroupedCiphertext2HandlesValidityProof =
399            self.validity_proof.try_into()?;
400        let range_proof: RangeProof = self.range_proof.try_into()?;
401
402        // verify equality proof
403        equality_proof.verify(
404            source_pubkey,
405            ciphertext_new_spendable,
406            &commitment,
407            transcript,
408        )?;
409
410        // verify validity proof
411        aggregated_validity_proof.verify(
412            (destination_pubkey, auditor_pubkey),
413            (
414                ciphertext_lo.get_commitment(),
415                ciphertext_hi.get_commitment(),
416            ),
417            (
418                ciphertext_lo.get_destination_handle(),
419                ciphertext_hi.get_destination_handle(),
420            ),
421            (
422                ciphertext_lo.get_auditor_handle(),
423                ciphertext_hi.get_auditor_handle(),
424            ),
425            transcript,
426        )?;
427
428        // verify range proof
429        let new_source_commitment = self.new_source_commitment.try_into()?;
430        if TRANSFER_AMOUNT_LO_BITS == 32 {
431            range_proof.verify(
432                vec![
433                    &new_source_commitment,
434                    ciphertext_lo.get_commitment(),
435                    ciphertext_hi.get_commitment(),
436                ],
437                vec![
438                    TRANSFER_SOURCE_AMOUNT_BITS,
439                    TRANSFER_AMOUNT_LO_BITS,
440                    TRANSFER_AMOUNT_HI_BITS,
441                ],
442                transcript,
443            )?;
444        } else {
445            let commitment_lo_negated = &(*COMMITMENT_MAX) - ciphertext_lo.get_commitment();
446
447            range_proof.verify(
448                vec![
449                    &new_source_commitment,
450                    ciphertext_lo.get_commitment(),
451                    &commitment_lo_negated,
452                    ciphertext_hi.get_commitment(),
453                ],
454                vec![
455                    TRANSFER_SOURCE_AMOUNT_BITS,
456                    TRANSFER_AMOUNT_LO_BITS,
457                    TRANSFER_AMOUNT_LO_NEGATED_BITS,
458                    TRANSFER_AMOUNT_HI_BITS,
459                ],
460                transcript,
461            )?;
462        }
463
464        Ok(())
465    }
466}

473    fn test_transfer_correctness() {
474        // ElGamalKeypair keys for source, destination, and auditor accounts
475        let source_keypair = ElGamalKeypair::new_rand();
476
477        let dest_keypair = ElGamalKeypair::new_rand();
478        let dest_pk = dest_keypair.pubkey();
479
480        let auditor_keypair = ElGamalKeypair::new_rand();
481        let auditor_pk = auditor_keypair.pubkey();
482
483        // Case 1: transfer 0 amount
484
485        // create source account spendable ciphertext
486        let spendable_balance: u64 = 0;
487        let spendable_ciphertext = source_keypair.pubkey().encrypt(spendable_balance);
488
489        // transfer amount
490        let transfer_amount: u64 = 0;
491
492        // create transfer data
493        let transfer_data = TransferData::new(
494            transfer_amount,
495            (spendable_balance, &spendable_ciphertext),
496            &source_keypair,
497            (dest_pk, auditor_pk),
498        )
499        .unwrap();
500
501        assert!(transfer_data.verify_proof().is_ok());
502
503        // Case 2: transfer max amount
504
505        // create source account spendable ciphertext
506        let spendable_balance: u64 = u64::max_value();
507        let spendable_ciphertext = source_keypair.pubkey().encrypt(spendable_balance);
508
509        // transfer amount
510        let transfer_amount: u64 =
511            (1u64 << (TRANSFER_AMOUNT_LO_BITS + TRANSFER_AMOUNT_HI_BITS)) - 1;
512
513        // create transfer data
514        let transfer_data = TransferData::new(
515            transfer_amount,
516            (spendable_balance, &spendable_ciphertext),
517            &source_keypair,
518            (dest_pk, auditor_pk),
519        )
520        .unwrap();
521
522        assert!(transfer_data.verify_proof().is_ok());
523
524        // Case 3: general success case
525
526        // create source account spendable ciphertext
527        let spendable_balance: u64 = 77;
528        let spendable_ciphertext = source_keypair.pubkey().encrypt(spendable_balance);
529
530        // transfer amount
531        let transfer_amount: u64 = 55;
532
533        // create transfer data
534        let transfer_data = TransferData::new(
535            transfer_amount,
536            (spendable_balance, &spendable_ciphertext),
537            &source_keypair,
538            (dest_pk, auditor_pk),
539        )
540        .unwrap();

542        assert!(transfer_data.verify_proof().is_ok());

544        // Case 4: destination pubkey is invalid
