54pub enum ProofType {

56    Uninitialized,
57    ZeroBalance,
58    Withdraw,
59    CiphertextCiphertextEquality,
60    Transfer,
61    TransferWithFee,
62    PubkeyValidity,
63    RangeProofU64,
64    BatchedRangeProofU64,
65    BatchedRangeProofU128,
66    BatchedRangeProofU256,
67    CiphertextCommitmentEquality,
68    GroupedCiphertext2HandlesValidity,
69    BatchedGroupedCiphertext2HandlesValidity,
70    FeeSigma,
71}

73pub trait ZkProofData<T: Pod> {
