1//! The twisted ElGamal encryption implementation.
2//!
3//! The message space consists of any number that is representable as a scalar (a.k.a. "exponent")
4//! for Curve25519.
5//!
6//! A twisted ElGamal ciphertext consists of two components:
7//! - A Pedersen commitment that encodes a message to be encrypted
8//! - A "decryption handle" that binds the Pedersen opening to a specific public key
9//! In contrast to the traditional ElGamal encryption scheme, the twisted ElGamal encodes messages
10//! directly as a Pedersen commitment. Therefore, proof systems that are designed specifically for
11//! Pedersen commitments can be used on the twisted ElGamal ciphertexts.
12//!
13//! As the messages are encrypted as scalar elements (a.k.a. in the "exponent"), one must solve the
14//! discrete log to recover the originally encrypted value.
15

58/// Byte length of a decrypt handle
59const DECRYPT_HANDLE_LEN: usize = RISTRETTO_POINT_LEN;
60
61/// Byte length of an ElGamal ciphertext
62const ELGAMAL_CIPHERTEXT_LEN: usize = PEDERSEN_COMMITMENT_LEN + DECRYPT_HANDLE_LEN;
63
64/// Byte length of an ElGamal public key
65const ELGAMAL_PUBKEY_LEN: usize = RISTRETTO_POINT_LEN;
66
67/// Byte length of an ElGamal secret key
68const ELGAMAL_SECRET_KEY_LEN: usize = SCALAR_LEN;
69
70/// Byte length of an ElGamal keypair
71const ELGAMAL_KEYPAIR_LEN: usize = ELGAMAL_PUBKEY_LEN + ELGAMAL_SECRET_KEY_LEN;

87/// Algorithm handle for the twisted ElGamal encryption scheme
88pub struct ElGamal;
89impl ElGamal {
90    /// Generates an ElGamal keypair.
91    ///
92    /// This function is randomized. It internally samples a scalar element using `OsRng`.
93    #[cfg(not(target_os = "solana"))]
94    #[allow(non_snake_case)]
95    fn keygen() -> ElGamalKeypair {
96        // secret scalar should be non-zero except with negligible probability
97        let mut s = Scalar::random(&mut OsRng);
98        let keypair = Self::keygen_with_scalar(&s);
99
100        s.zeroize();
101        keypair
102    }
103
104    /// Generates an ElGamal keypair from a scalar input that determines the ElGamal private key.
105    ///
106    /// This function panics if the input scalar is zero, which is not a valid key.
107    #[cfg(not(target_os = "solana"))]
108    #[allow(non_snake_case)]
109    fn keygen_with_scalar(s: &Scalar) -> ElGamalKeypair {
110        let secret = ElGamalSecretKey(*s);
111        let public = ElGamalPubkey::new(&secret);
112
113        ElGamalKeypair { public, secret }
114    }
115
116    /// On input an ElGamal public key and an amount to be encrypted, the function returns a
117    /// corresponding ElGamal ciphertext.
118    ///
119    /// This function is randomized. It internally samples a scalar element using `OsRng`.
120    #[cfg(not(target_os = "solana"))]
121    fn encrypt<T: Into<Scalar>>(public: &ElGamalPubkey, amount: T) -> ElGamalCiphertext {
122        let (commitment, opening) = Pedersen::new(amount);
123        let handle = public.decrypt_handle(&opening);
124
125        ElGamalCiphertext { commitment, handle }
126    }
127
128    /// On input a public key, amount, and Pedersen opening, the function returns the corresponding
129    /// ElGamal ciphertext.
130    #[cfg(not(target_os = "solana"))]
131    fn encrypt_with<T: Into<Scalar>>(
132        amount: T,
133        public: &ElGamalPubkey,
134        opening: &PedersenOpening,
135    ) -> ElGamalCiphertext {
136        let commitment = Pedersen::with(amount, opening);
137        let handle = public.decrypt_handle(opening);
138
139        ElGamalCiphertext { commitment, handle }
140    }
141
142    /// On input an amount, the function returns a twisted ElGamal ciphertext where the associated
143    /// Pedersen opening is always zero. Since the opening is zero, any twisted ElGamal ciphertext
144    /// of this form is a valid ciphertext under any ElGamal public key.
145    #[cfg(not(target_os = "solana"))]
146    pub fn encode<T: Into<Scalar>>(amount: T) -> ElGamalCiphertext {
147        let commitment = Pedersen::encode(amount);
148        let handle = DecryptHandle(RistrettoPoint::identity());
149
150        ElGamalCiphertext { commitment, handle }
151    }
152
153    /// On input a secret key and a ciphertext, the function returns the discrete log encoding of
154    /// original amount.
155    ///
156    /// The output of this function is of type `DiscreteLog`. To recover, the originally encrypted
157    /// amount, use `DiscreteLog::decode`.
158    #[cfg(not(target_os = "solana"))]
159    fn decrypt(secret: &ElGamalSecretKey, ciphertext: &ElGamalCiphertext) -> DiscreteLog {
160        DiscreteLog::new(
161            *G,
162            ciphertext.commitment.get_point() - &(&secret.0 * &ciphertext.handle.0),
163        )
164    }
165
166    /// On input a secret key and a ciphertext, the function returns the decrypted amount
167    /// interpreted as a positive 32-bit number (but still of type `u64`).
168    ///
169    /// If the originally encrypted amount is not a positive 32-bit number, then the function
170    /// returns `None`.
171    #[cfg(not(target_os = "solana"))]
172    fn decrypt_u32(secret: &ElGamalSecretKey, ciphertext: &ElGamalCiphertext) -> Option<u64> {
173        let discrete_log_instance = Self::decrypt(secret, ciphertext);
174        discrete_log_instance.decode_u32()
175    }
176}

189impl ElGamalKeypair {
190    /// Create an ElGamal keypair from an ElGamal public key and an ElGamal secret key.
191    ///
192    /// An ElGamal keypair should never be instantiated manually; `ElGamalKeypair::new_rand` or
193    /// `ElGamalKeypair::new_from_signer` should be used instead. This function exists to create
194    /// custom ElGamal keypairs for tests.
195    pub fn new_for_tests(public: ElGamalPubkey, secret: ElGamalSecretKey) -> Self {
196        Self { public, secret }
197    }

199    /// Deterministically derives an ElGamal keypair from a Solana signer and a public seed.
200    ///
201    /// This function exists for applications where a user may not wish to maintain a Solana signer
202    /// and an ElGamal keypair separately. Instead, a user can derive the ElGamal keypair
203    /// on-the-fly whenever encryption/decryption is needed.
204    ///
205    /// For the spl-token-2022 confidential extension, the ElGamal public key is specified in a
206    /// token account. A natural way to derive an ElGamal keypair is to define it from the hash of
207    /// a Solana keypair and a Solana address as the public seed. However, for general hardware
208    /// wallets, the signing key is not exposed in the API. Therefore, this function uses a signer
209    /// to sign a public seed and the resulting signature is then hashed to derive an ElGamal
210    /// keypair.
211    #[cfg(not(target_os = "solana"))]
212    #[allow(non_snake_case)]
213    pub fn new_from_signer(
214        signer: &dyn Signer,
215        public_seed: &[u8],
216    ) -> Result<Self, Box<dyn error::Error>> {
217        let secret = ElGamalSecretKey::new_from_signer(signer, public_seed)?;
218        let public = ElGamalPubkey::new(&secret);
219        Ok(ElGamalKeypair { public, secret })
220    }

222    /// Generates the public and secret keys for ElGamal encryption.
223    ///
224    /// This function is randomized. It internally samples a scalar element using `OsRng`.
225    #[cfg(not(target_os = "solana"))]
226    pub fn new_rand() -> Self {
227        ElGamal::keygen()
228    }
229
230    pub fn pubkey(&self) -> &ElGamalPubkey {
231        &self.public
232    }
233
234    pub fn secret(&self) -> &ElGamalSecretKey {
235        &self.secret
236    }
237
238    pub fn to_bytes(&self) -> [u8; ELGAMAL_KEYPAIR_LEN] {
239        let mut bytes = [0u8; ELGAMAL_KEYPAIR_LEN];
240        bytes[..ELGAMAL_PUBKEY_LEN].copy_from_slice(&self.public.to_bytes());
241        bytes[ELGAMAL_PUBKEY_LEN..].copy_from_slice(self.secret.as_bytes());
242        bytes
243    }
244
245    pub fn from_bytes(bytes: &[u8]) -> Option<Self> {
246        if bytes.len() != ELGAMAL_KEYPAIR_LEN {
247            return None;
248        }
249
250        Some(Self {
251            public: ElGamalPubkey::from_bytes(&bytes[..ELGAMAL_PUBKEY_LEN])?,
252            secret: ElGamalSecretKey::from_bytes(bytes[ELGAMAL_PUBKEY_LEN..].try_into().ok()?)?,
253        })
254    }

257    pub fn read_json<R: Read>(reader: &mut R) -> Result<Self, Box<dyn error::Error>> {
258        let bytes: Vec<u8> = serde_json::from_reader(reader)?;

264    /// Reads keypair from a file
265    pub fn read_json_file<F: AsRef<Path>>(path: F) -> Result<Self, Box<dyn error::Error>> {

267    }

269    /// Writes to a `Write` implementer with JSON-encoding
270    pub fn write_json<W: Write>(&self, writer: &mut W) -> Result<String, Box<dyn error::Error>> {
271        let bytes = self.to_bytes();

273        writer.write_all(&json.clone().into_bytes())?;

406/// Secret key for the ElGamal encryption scheme.
407///
408/// Instances of ElGamal secret key are zeroized on drop.
409#[derive(Clone, Debug, Deserialize, Serialize, Zeroize)]
410#[zeroize(drop)]
411pub struct ElGamalSecretKey(Scalar);
412impl ElGamalSecretKey {
413    /// Deterministically derives an ElGamal secret key from a Solana signer and a public seed.
414    ///
415    /// See `ElGamalKeypair::new_from_signer` for more context on the key derivation.
416    pub fn new_from_signer(
417        signer: &dyn Signer,
418        public_seed: &[u8],
419    ) -> Result<Self, Box<dyn error::Error>> {
420        let seed = Self::seed_from_signer(signer, public_seed)?;
421        let key = Self::from_seed(&seed)?;
422        Ok(key)
423    }
424
425    /// Derive a seed from a Solana signer used to generate an ElGamal secret key.
426    ///
427    /// The seed is derived as the hash of the signature of a public seed.
428    pub fn seed_from_signer(
429        signer: &dyn Signer,
430        public_seed: &[u8],
431    ) -> Result<Vec<u8>, SignerError> {
432        let message = [b"ElGamalSecretKey", public_seed].concat();
433        let signature = signer.try_sign_message(&message)?;
434
435        // Some `Signer` implementations return the default signature, which is not suitable for
436        // use as key material
437        if bool::from(signature.as_ref().ct_eq(Signature::default().as_ref())) {
438            return Err(SignerError::Custom("Rejecting default signatures".into()));
439        }
440
441        let mut hasher = Sha3_512::new();
442        hasher.update(signature.as_ref());
443        let result = hasher.finalize();
444
445        Ok(result.to_vec())
446    }
447
448    /// Randomly samples an ElGamal secret key.
449    ///
450    /// This function is randomized. It internally samples a scalar element using `OsRng`.
451    pub fn new_rand() -> Self {
452        ElGamalSecretKey(Scalar::random(&mut OsRng))
453    }
454
455    /// Derive an ElGamal secret key from an entropy seed.
456    pub fn from_seed(seed: &[u8]) -> Result<Self, ElGamalError> {
457        const MINIMUM_SEED_LEN: usize = ELGAMAL_SECRET_KEY_LEN;
458        const MAXIMUM_SEED_LEN: usize = 65535;
459
460        if seed.len() < MINIMUM_SEED_LEN {
461            return Err(ElGamalError::SeedLengthTooShort);
462        }
463        if seed.len() > MAXIMUM_SEED_LEN {
464            return Err(ElGamalError::SeedLengthTooLong);
465        }
466        Ok(ElGamalSecretKey(Scalar::hash_from_bytes::<Sha3_512>(seed)))
467    }
468
469    pub fn get_scalar(&self) -> &Scalar {
470        &self.0
471    }

473    /// Decrypts a ciphertext using the ElGamal secret key.
474    ///
475    /// The output of this function is of type `DiscreteLog`. To recover, the originally encrypted
476    /// message, use `DiscreteLog::decode`.
477    pub fn decrypt(&self, ciphertext: &ElGamalCiphertext) -> DiscreteLog {
478        ElGamal::decrypt(self, ciphertext)
479    }
480
481    /// Decrypts a ciphertext using the ElGamal secret key interpretting the message as type `u32`.
482    pub fn decrypt_u32(&self, ciphertext: &ElGamalCiphertext) -> Option<u64> {
483        ElGamal::decrypt_u32(self, ciphertext)
484    }
485
486    pub fn as_bytes(&self) -> &[u8; ELGAMAL_SECRET_KEY_LEN] {
487        self.0.as_bytes()
488    }
489
490    pub fn to_bytes(&self) -> [u8; ELGAMAL_SECRET_KEY_LEN] {
491        self.0.to_bytes()
492    }
493
494    pub fn from_bytes(bytes: &[u8]) -> Option<ElGamalSecretKey> {
495        match bytes.try_into() {
496            Ok(bytes) => Scalar::from_canonical_bytes(bytes).map(ElGamalSecretKey),
497            _ => None,
498        }
499    }

561/// Ciphertext for the ElGamal encryption scheme.
562#[allow(non_snake_case)]
563#[derive(Clone, Copy, Debug, Default, Deserialize, Eq, PartialEq, Serialize)]
564pub struct ElGamalCiphertext {
565    pub commitment: PedersenCommitment,
566    pub handle: DecryptHandle,
567}
568impl ElGamalCiphertext {
569    pub fn add_amount<T: Into<Scalar>>(&self, amount: T) -> Self {
570        let point = amount.into() * &(*G);
571        let commitment_to_add = PedersenCommitment::new(point);
572        ElGamalCiphertext {
573            commitment: &self.commitment + &commitment_to_add,
574            handle: self.handle,
575        }
576    }
577
578    pub fn subtract_amount<T: Into<Scalar>>(&self, amount: T) -> Self {
579        let point = amount.into() * &(*G);
580        let commitment_to_subtract = PedersenCommitment::new(point);
581        ElGamalCiphertext {
582            commitment: &self.commitment - &commitment_to_subtract,
583            handle: self.handle,
584        }
585    }
586
587    pub fn to_bytes(&self) -> [u8; ELGAMAL_CIPHERTEXT_LEN] {
588        let mut bytes = [0u8; ELGAMAL_CIPHERTEXT_LEN];
589        bytes[..PEDERSEN_COMMITMENT_LEN].copy_from_slice(&self.commitment.to_bytes());
590        bytes[PEDERSEN_COMMITMENT_LEN..].copy_from_slice(&self.handle.to_bytes());
591        bytes
592    }
593
594    pub fn from_bytes(bytes: &[u8]) -> Option<ElGamalCiphertext> {
595        if bytes.len() != ELGAMAL_CIPHERTEXT_LEN {
596            return None;
597        }
598
599        Some(ElGamalCiphertext {
600            commitment: PedersenCommitment::from_bytes(&bytes[..PEDERSEN_COMMITMENT_LEN])?,
601            handle: DecryptHandle::from_bytes(&bytes[PEDERSEN_COMMITMENT_LEN..])?,
602        })
603    }

605    /// Decrypts the ciphertext using an ElGamal secret key.
606    ///
607    /// The output of this function is of type `DiscreteLog`. To recover, the originally encrypted
608    /// amount, use `DiscreteLog::decode`.
609    pub fn decrypt(&self, secret: &ElGamalSecretKey) -> DiscreteLog {
610        ElGamal::decrypt(secret, self)
611    }
612
613    /// Decrypts the ciphertext using an ElGamal secret key assuming that the message is a positive
614    /// 32-bit number.
615    ///
616    /// If the originally encrypted amount is not a positive 32-bit number, then the function
617    /// returns `None`.
618    pub fn decrypt_u32(&self, secret: &ElGamalSecretKey) -> Option<u64> {
619        ElGamal::decrypt_u32(secret, self)
620    }
621}

697/// Decryption handle for Pedersen commitment.
698#[derive(Clone, Copy, Debug, Default, Deserialize, Eq, PartialEq, Serialize)]
699pub struct DecryptHandle(RistrettoPoint);
700impl DecryptHandle {
701    pub fn new(public: &ElGamalPubkey, opening: &PedersenOpening) -> Self {
702        Self(&public.0 * opening.get_scalar())
703    }
704
705    pub fn get_point(&self) -> &RistrettoPoint {
706        &self.0
707    }
708
709    pub fn to_bytes(&self) -> [u8; DECRYPT_HANDLE_LEN] {
710        self.0.compress().to_bytes()
711    }
712
713    pub fn from_bytes(bytes: &[u8]) -> Option<DecryptHandle> {
714        if bytes.len() != DECRYPT_HANDLE_LEN {
715            return None;
716        }
717
718        Some(DecryptHandle(
719            CompressedRistretto::from_slice(bytes).decompress()?,
720        ))
721    }
722}
723

725    type Output = DecryptHandle;
726
727    fn add(self, handle: &'b DecryptHandle) -> DecryptHandle {
728        DecryptHandle(&self.0 + &handle.0)
729    }
730}
