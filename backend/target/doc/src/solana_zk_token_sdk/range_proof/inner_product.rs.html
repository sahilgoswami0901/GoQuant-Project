20#[derive(Clone)]
21pub struct InnerProductProof {
22    pub L_vec: Vec<CompressedRistretto>, // 32 * log(bit_length)
23    pub R_vec: Vec<CompressedRistretto>, // 32 * log(bit_length)
24    pub a: Scalar,                       // 32 bytes
25    pub b: Scalar,                       // 32 bytes
26}

28#[allow(non_snake_case)]
29impl InnerProductProof {
30    /// Create an inner-product proof.
31    ///
32    /// The proof is created with respect to the bases \\(G\\), \\(H'\\),
33    /// where \\(H'\_i = H\_i \cdot \texttt{Hprime\\_factors}\_i\\).
34    ///
35    /// The `verifier` is passed in as a parameter so that the
36    /// challenges depend on the *entire* transcript (including parent
37    /// protocols).
38    ///
39    /// The lengths of the vectors must all be the same, and must all be
40    /// a power of 2.
41    #[allow(clippy::too_many_arguments)]
42    pub fn new(
43        Q: &RistrettoPoint,
44        G_factors: &[Scalar],
45        H_factors: &[Scalar],
46        mut G_vec: Vec<RistrettoPoint>,
47        mut H_vec: Vec<RistrettoPoint>,
48        mut a_vec: Vec<Scalar>,
49        mut b_vec: Vec<Scalar>,
50        transcript: &mut Transcript,
51    ) -> Result<Self, RangeProofGenerationError> {
52        // Create slices G, H, a, b backed by their respective
53        // vectors.  This lets us reslice as we compress the lengths
54        // of the vectors in the main loop below.
55        let mut G = &mut G_vec[..];
56        let mut H = &mut H_vec[..];
57        let mut a = &mut a_vec[..];
58        let mut b = &mut b_vec[..];
59
60        let mut n = G.len();
61
62        // All of the input vectors must have the same length.
63        if G.len() != n
64            || H.len() != n
65            || a.len() != n
66            || b.len() != n
67            || G_factors.len() != n
68            || H_factors.len() != n
69        {
70            return Err(RangeProofGenerationError::GeneratorLengthMismatch);
71        }
72
73        // All of the input vectors must have a length that is a power of two.
74        if !n.is_power_of_two() {
75            return Err(RangeProofGenerationError::InvalidBitSize);
76        }
77
78        transcript.innerproduct_domain_separator(n as u64);
79
80        let lg_n = n.next_power_of_two().trailing_zeros() as usize;
81        let mut L_vec = Vec::with_capacity(lg_n);
82        let mut R_vec = Vec::with_capacity(lg_n);
83
84        // If it's the first iteration, unroll the Hprime = H*y_inv scalar mults
85        // into multiscalar muls, for performance.
86        if n != 1 {
87            n = n.checked_div(2).unwrap();
88            let (a_L, a_R) = a.split_at_mut(n);
89            let (b_L, b_R) = b.split_at_mut(n);
90            let (G_L, G_R) = G.split_at_mut(n);
91            let (H_L, H_R) = H.split_at_mut(n);
92
93            let c_L = util::inner_product(a_L, b_R)
94                .ok_or(RangeProofGenerationError::InnerProductLengthMismatch)?;
95            let c_R = util::inner_product(a_R, b_L)
96                .ok_or(RangeProofGenerationError::InnerProductLengthMismatch)?;
97
98            let L = RistrettoPoint::multiscalar_mul(
99                a_L.iter()
100                    // `n` was previously divided in half and therefore, it cannot overflow.
101                    .zip(G_factors[n..n.checked_mul(2).unwrap()].iter())
102                    .map(|(a_L_i, g)| a_L_i * g)
103                    .chain(
104                        b_R.iter()
105                            .zip(H_factors[0..n].iter())
106                            .map(|(b_R_i, h)| b_R_i * h),
107                    )
108                    .chain(iter::once(c_L)),
109                G_R.iter().chain(H_L.iter()).chain(iter::once(Q)),
110            )
111            .compress();
112
113            let R = RistrettoPoint::multiscalar_mul(
114                a_R.iter()
115                    .zip(G_factors[0..n].iter())
116                    .map(|(a_R_i, g)| a_R_i * g)
117                    .chain(
118                        b_L.iter()
119                            .zip(H_factors[n..n.checked_mul(2).unwrap()].iter())
120                            .map(|(b_L_i, h)| b_L_i * h),
121                    )
122                    .chain(iter::once(c_R)),
123                G_L.iter().chain(H_R.iter()).chain(iter::once(Q)),
124            )
125            .compress();
126
127            L_vec.push(L);
128            R_vec.push(R);
129
130            transcript.append_point(b"L", &L);
131            transcript.append_point(b"R", &R);
132
133            let u = transcript.challenge_scalar(b"u");
134            let u_inv = u.invert();
135
136            for i in 0..n {
137                a_L[i] = a_L[i] * u + u_inv * a_R[i];
138                b_L[i] = b_L[i] * u_inv + u * b_R[i];
139                G_L[i] = RistrettoPoint::multiscalar_mul(
140                    &[
141                        u_inv * G_factors[i],
142                        u * G_factors[n.checked_add(i).unwrap()],
143                    ],
144                    &[G_L[i], G_R[i]],
145                );
146                H_L[i] = RistrettoPoint::multiscalar_mul(
147                    &[
148                        u * H_factors[i],
149                        u_inv * H_factors[n.checked_add(i).unwrap()],
150                    ],
151                    &[H_L[i], H_R[i]],
152                )
153            }
154
155            a = a_L;
156            b = b_L;
157            G = G_L;
158            H = H_L;
159        }
160
161        while n != 1 {
162            n = n.checked_div(2).unwrap();
163            let (a_L, a_R) = a.split_at_mut(n);
164            let (b_L, b_R) = b.split_at_mut(n);
165            let (G_L, G_R) = G.split_at_mut(n);
166            let (H_L, H_R) = H.split_at_mut(n);
167
168            let c_L = util::inner_product(a_L, b_R)
169                .ok_or(RangeProofGenerationError::InnerProductLengthMismatch)?;
170            let c_R = util::inner_product(a_R, b_L)
171                .ok_or(RangeProofGenerationError::InnerProductLengthMismatch)?;
172
173            let L = RistrettoPoint::multiscalar_mul(
174                a_L.iter().chain(b_R.iter()).chain(iter::once(&c_L)),
175                G_R.iter().chain(H_L.iter()).chain(iter::once(Q)),
176            )
177            .compress();
178
179            let R = RistrettoPoint::multiscalar_mul(
180                a_R.iter().chain(b_L.iter()).chain(iter::once(&c_R)),
181                G_L.iter().chain(H_R.iter()).chain(iter::once(Q)),
182            )
183            .compress();
184
185            L_vec.push(L);
186            R_vec.push(R);
187
188            transcript.append_point(b"L", &L);
189            transcript.append_point(b"R", &R);
190
191            let u = transcript.challenge_scalar(b"u");
192            let u_inv = u.invert();
193
194            for i in 0..n {
195                a_L[i] = a_L[i] * u + u_inv * a_R[i];
196                b_L[i] = b_L[i] * u_inv + u * b_R[i];
197                G_L[i] = RistrettoPoint::multiscalar_mul(&[u_inv, u], &[G_L[i], G_R[i]]);
198                H_L[i] = RistrettoPoint::multiscalar_mul(&[u, u_inv], &[H_L[i], H_R[i]]);
199            }
200
201            a = a_L;
202            b = b_L;
203            G = G_L;
204            H = H_L;
205        }
206
207        Ok(InnerProductProof {
208            L_vec,
209            R_vec,
210            a: a[0],
211            b: b[0],
212        })
213    }

215    /// Computes three vectors of verification scalars \\([u\_{i}^{2}]\\), \\([u\_{i}^{-2}]\\) and
216    /// \\([s\_{i}]\\) for combined multiscalar multiplication in a parent protocol. See [inner
217    /// product protocol notes](index.html#verification-equation) for details. The verifier must
218    /// provide the input length \\(n\\) explicitly to avoid unbounded allocation within the inner
219    /// product proof.
220    #[allow(clippy::type_complexity)]
221    pub(crate) fn verification_scalars(
222        &self,
223        n: usize,
224        transcript: &mut Transcript,
225    ) -> Result<(Vec<Scalar>, Vec<Scalar>, Vec<Scalar>), RangeProofVerificationError> {
226        let lg_n = self.L_vec.len();
227        if lg_n == 0 || lg_n >= 32 {
228            // 4 billion multiplications should be enough for anyone
229            // and this check prevents overflow in 1<<lg_n below.
230            return Err(RangeProofVerificationError::InvalidBitSize);
231        }
232        if n != (1_usize.checked_shl(lg_n as u32).unwrap()) {
233            return Err(RangeProofVerificationError::InvalidBitSize);
234        }
235
236        transcript.innerproduct_domain_separator(n as u64);
237
238        // 1. Recompute x_k,...,x_1 based on the proof transcript
239
240        let mut challenges = Vec::with_capacity(lg_n);
241        for (L, R) in self.L_vec.iter().zip(self.R_vec.iter()) {
242            transcript.validate_and_append_point(b"L", L)?;
243            transcript.validate_and_append_point(b"R", R)?;
244            challenges.push(transcript.challenge_scalar(b"u"));
245        }
246
247        // 2. Compute 1/(u_k...u_1) and 1/u_k, ..., 1/u_1
248
249        let mut challenges_inv = challenges.clone();
250        let allinv = Scalar::batch_invert(&mut challenges_inv);
251
252        // 3. Compute u_i^2 and (1/u_i)^2
253
254        for i in 0..lg_n {
255            challenges[i] = challenges[i] * challenges[i];
256            challenges_inv[i] = challenges_inv[i] * challenges_inv[i];
257        }
258        let challenges_sq = challenges;
259        let challenges_inv_sq = challenges_inv;
260
261        // 4. Compute s values inductively.
262
263        let mut s = Vec::with_capacity(n);
264        s.push(allinv);
265        for i in 1..n {
266            let lg_i = 31_u32.checked_sub((i as u32).leading_zeros()).unwrap() as usize;
267            let k = 1_usize.checked_shl(lg_i as u32).unwrap();
268            // The challenges are stored in "creation order" as [u_k,...,u_1],
269            // so u_{lg(i)+1} = is indexed by (lg_n-1) - lg_i
270            let u_lg_i_sq = challenges_sq[lg_n
271                .checked_sub(1)
272                .and_then(|x| x.checked_sub(lg_i))
273                .unwrap()];
274            s.push(s[i - k] * u_lg_i_sq);
275        }
276
277        Ok((challenges_sq, challenges_inv_sq, s))
278    }

280    /// This method is for testing that proof generation work, but for efficiency the actual
281    /// protocols would use `verification_scalars` method to combine inner product verification
282    /// with other checks in a single multiscalar multiplication.
283    #[allow(clippy::too_many_arguments)]
284    pub fn verify<IG, IH>(
285        &self,
286        n: usize,
287        G_factors: IG,
288        H_factors: IH,
289        P: &RistrettoPoint,
290        Q: &RistrettoPoint,
291        G: &[RistrettoPoint],
292        H: &[RistrettoPoint],
293        transcript: &mut Transcript,
294    ) -> Result<(), RangeProofVerificationError>
295    where
296        IG: IntoIterator,
297        IG::Item: Borrow<Scalar>,
298        IH: IntoIterator,
299        IH::Item: Borrow<Scalar>,
300    {
301        let (u_sq, u_inv_sq, s) = self.verification_scalars(n, transcript)?;
302
303        let g_times_a_times_s = G_factors
304            .into_iter()
305            .zip(s.iter())
306            .map(|(g_i, s_i)| (self.a * s_i) * g_i.borrow())
307            .take(G.len());
308
309        // 1/s[i] is s[!i], and !i runs from n-1 to 0 as i runs from 0 to n-1
310        let inv_s = s.iter().rev();
311
312        let h_times_b_div_s = H_factors
313            .into_iter()
314            .zip(inv_s)
315            .map(|(h_i, s_i_inv)| (self.b * s_i_inv) * h_i.borrow());
316
317        let neg_u_sq = u_sq.iter().map(|ui| -ui);
318        let neg_u_inv_sq = u_inv_sq.iter().map(|ui| -ui);
319
320        let Ls = self
321            .L_vec
322            .iter()
323            .map(|p| {
324                p.decompress()
325                    .ok_or(RangeProofVerificationError::Deserialization)
326            })
327            .collect::<Result<Vec<_>, _>>()?;
328
329        let Rs = self
330            .R_vec
331            .iter()
332            .map(|p| {
333                p.decompress()
334                    .ok_or(RangeProofVerificationError::Deserialization)
335            })
336            .collect::<Result<Vec<_>, _>>()?;
337
338        let expect_P = RistrettoPoint::vartime_multiscalar_mul(
339            iter::once(self.a * self.b)
340                .chain(g_times_a_times_s)
341                .chain(h_times_b_div_s)
342                .chain(neg_u_sq)
343                .chain(neg_u_inv_sq),
344            iter::once(Q)
345                .chain(G.iter())
346                .chain(H.iter())
347                .chain(Ls.iter())
348                .chain(Rs.iter()),
349        );
350
351        if expect_P == *P {
352            Ok(())
353        } else {
354            Err(RangeProofVerificationError::AlgebraicRelation)
355        }
356    }

358    /// Returns the size in bytes required to serialize the inner
359    /// product proof.
360    ///
361    /// For vectors of length `n` the proof size is
362    /// \\(32 \cdot (2\lg n+2)\\) bytes.
363    pub fn serialized_size(&self) -> usize {
364        (self.L_vec.len() * 2 + 2) * 32
365    }

367    /// Serializes the proof into a byte array of \\(2n+2\\) 32-byte elements.
368    /// The layout of the inner product proof is:
369    /// * \\(n\\) pairs of compressed Ristretto points \\(L_0, R_0 \dots, L_{n-1}, R_{n-1}\\),
370    /// * two scalars \\(a, b\\).
371    pub fn to_bytes(&self) -> Vec<u8> {
372        let mut buf = Vec::with_capacity(self.serialized_size());
373        for (l, r) in self.L_vec.iter().zip(self.R_vec.iter()) {
374            buf.extend_from_slice(l.as_bytes());
375            buf.extend_from_slice(r.as_bytes());
376        }
377        buf.extend_from_slice(self.a.as_bytes());
378        buf.extend_from_slice(self.b.as_bytes());
379        buf
380    }

382    /// Deserializes the proof from a byte slice.
383    /// Returns an error in the following cases:
384    /// * the slice does not have \\(2n+2\\) 32-byte elements,
385    /// * \\(n\\) is larger or equal to 32 (proof is too big),
386    /// * any of \\(2n\\) points are not valid compressed Ristretto points,
387    /// * any of 2 scalars are not canonical scalars modulo Ristretto group order.
388    pub fn from_bytes(slice: &[u8]) -> Result<InnerProductProof, RangeProofVerificationError> {
389        let b = slice.len();
390        if b % 32 != 0 {
391            return Err(RangeProofVerificationError::Deserialization);
392        }
393        let num_elements = b / 32;
394        if num_elements < 2 {
395            return Err(RangeProofVerificationError::Deserialization);
396        }
397        if (num_elements - 2) % 2 != 0 {
398            return Err(RangeProofVerificationError::Deserialization);
399        }
400        let lg_n = (num_elements - 2) / 2;
401        if lg_n >= 32 {
402            return Err(RangeProofVerificationError::Deserialization);
403        }
404
405        let mut L_vec: Vec<CompressedRistretto> = Vec::with_capacity(lg_n);
406        let mut R_vec: Vec<CompressedRistretto> = Vec::with_capacity(lg_n);
407        for i in 0..lg_n {
408            let pos = 2 * i * 32;
409            L_vec.push(CompressedRistretto(util::read32(&slice[pos..])));
410            R_vec.push(CompressedRistretto(util::read32(&slice[pos + 32..])));
411        }
412
413        let pos = 2 * lg_n * 32;
414        let a = Scalar::from_canonical_bytes(util::read32(&slice[pos..]))
415            .ok_or(RangeProofVerificationError::Deserialization)?;
416        let b = Scalar::from_canonical_bytes(util::read32(&slice[pos + 32..]))
417            .ok_or(RangeProofVerificationError::Deserialization)?;
418
419        Ok(InnerProductProof { L_vec, R_vec, a, b })
420    }

424mod tests {
425    use {
426        super::*, crate::range_proof::generators::BulletproofGens, rand::rngs::OsRng,
427        sha3::Sha3_512,

430    #[test]
431    #[allow(non_snake_case)]
432    fn test_basic_correctness() {
433        let n = 32;
434
435        let bp_gens = BulletproofGens::new(n).unwrap();
436        let G: Vec<RistrettoPoint> = bp_gens.G(n).cloned().collect();
437        let H: Vec<RistrettoPoint> = bp_gens.H(n).cloned().collect();
438
439        let Q = RistrettoPoint::hash_from_bytes::<Sha3_512>(b"test point");
440
441        let a: Vec<_> = (0..n).map(|_| Scalar::random(&mut OsRng)).collect();
442        let b: Vec<_> = (0..n).map(|_| Scalar::random(&mut OsRng)).collect();
443        let c = util::inner_product(&a, &b).unwrap();
444
445        let G_factors: Vec<Scalar> = iter::repeat(Scalar::one()).take(n).collect();
446
447        let y_inv = Scalar::random(&mut OsRng);
448        let H_factors: Vec<Scalar> = util::exp_iter(y_inv).take(n).collect();
449
450        // P would be determined upstream, but we need a correct P to check the proof.
451        //
452        // To generate P = <a,G> + <b,H'> + <a,b> Q, compute
453        //             P = <a,G> + <b',H> + <a,b> Q,
454        // where b' = b \circ y^(-n)
455        let b_prime = b.iter().zip(util::exp_iter(y_inv)).map(|(bi, yi)| bi * yi);
456        // a.iter() has Item=&Scalar, need Item=Scalar to chain with b_prime
457        let a_prime = a.iter().cloned();
458
459        let P = RistrettoPoint::vartime_multiscalar_mul(
460            a_prime.chain(b_prime).chain(iter::once(c)),
461            G.iter().chain(H.iter()).chain(iter::once(&Q)),
462        );
463
464        let mut prover_transcript = Transcript::new(b"innerproducttest");
465        let mut verifier_transcript = Transcript::new(b"innerproducttest");
466
467        let proof = InnerProductProof::new(
468            &Q,
469            &G_factors,
470            &H_factors,
471            G.clone(),
472            H.clone(),
473            a.clone(),
474            b.clone(),
475            &mut prover_transcript,
476        )
477        .unwrap();
478
479        assert!(proof
480            .verify(
481                n,
482                iter::repeat(Scalar::one()).take(n),
483                util::exp_iter(y_inv).take(n),
484                &P,
485                &Q,
486                &G,
487                &H,
488                &mut verifier_transcript,
489            )
490            .is_ok());
491
492        let proof = InnerProductProof::from_bytes(proof.to_bytes().as_slice()).unwrap();
493        let mut verifier_transcript = Transcript::new(b"innerproducttest");
494        assert!(proof
495            .verify(
496                n,
497                iter::repeat(Scalar::one()).take(n),
498                util::exp_iter(y_inv).take(n),
499                &P,
500                &Q,
501                &G,
502                &H,
503                &mut verifier_transcript,
504            )
505            .is_ok());
506    }
