1//! Plain Old Data types for the AES128-GCM-SIV authenticated encryption scheme.
2
3#[cfg(not(target_os = "solana"))]
4use crate::encryption::auth_encryption::{self as decoded, AuthenticatedEncryptionError};
5use {
6    crate::zk_token_elgamal::pod::{Pod, Zeroable},
7    base64::{prelude::BASE64_STANDARD, Engine},
8    std::fmt,
9};
10
11/// Byte length of an authenticated encryption ciphertext
12const AE_CIPHERTEXT_LEN: usize = 36;
13
14/// The `AeCiphertext` type as a `Pod`.
15#[derive(Clone, Copy, PartialEq, Eq)]
16#[repr(transparent)]
17pub struct AeCiphertext(pub [u8; AE_CIPHERTEXT_LEN]);
18
19// `AeCiphertext` is a wrapper type for a byte array, which is both `Pod` and `Zeroable`. However,
20// the marker traits `bytemuck::Pod` and `bytemuck::Zeroable` can only be derived for power-of-two
21// length byte arrays. Directly implement these traits for `AeCiphertext`.
22unsafe impl Zeroable for AeCiphertext {}
23unsafe impl Pod for AeCiphertext {}
24
25impl fmt::Debug for AeCiphertext {
26    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
27        write!(f, "{:?}", self.0)
28    }
29}
30
31impl fmt::Display for AeCiphertext {
32    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
33        write!(f, "{}", BASE64_STANDARD.encode(self.0))
34    }
35}
36
37impl Default for AeCiphertext {
38    fn default() -> Self {
39        Self::zeroed()
40    }
41}
42
43#[cfg(not(target_os = "solana"))]
44impl From<decoded::AeCiphertext> for AeCiphertext {
45    fn from(decoded_ciphertext: decoded::AeCiphertext) -> Self {
46        Self(decoded_ciphertext.to_bytes())
47    }
48}
49
50#[cfg(not(target_os = "solana"))]
51impl TryFrom<AeCiphertext> for decoded::AeCiphertext {
52    type Error = AuthenticatedEncryptionError;
53
54    fn try_from(pod_ciphertext: AeCiphertext) -> Result<Self, Self::Error> {
55        Self::from_bytes(&pod_ciphertext.0).ok_or(AuthenticatedEncryptionError::Deserialization)
56    }
57}
