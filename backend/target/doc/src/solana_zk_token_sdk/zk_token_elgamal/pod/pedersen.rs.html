1//! Plain Old Data type for the Pedersen commitment scheme.
2
3#[cfg(not(target_os = "solana"))]
4use {
5    crate::encryption::{elgamal::ElGamalError, pedersen as decoded},
6    curve25519_dalek::ristretto::CompressedRistretto,
7};
8use {
9    crate::{
10        zk_token_elgamal::pod::{Pod, Zeroable},
11        RISTRETTO_POINT_LEN,
12    },
13    std::fmt,
14};
15
16/// Byte length of a Pedersen commitment
17pub(crate) const PEDERSEN_COMMITMENT_LEN: usize = RISTRETTO_POINT_LEN;
18
19/// The `PedersenCommitment` type as a `Pod`.
20#[derive(Clone, Copy, Default, Pod, Zeroable, PartialEq, Eq)]
21#[repr(transparent)]
22pub struct PedersenCommitment(pub [u8; PEDERSEN_COMMITMENT_LEN]);
23
24impl fmt::Debug for PedersenCommitment {
25    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
26        write!(f, "{:?}", self.0)
27    }
28}
29
30#[cfg(not(target_os = "solana"))]
31impl From<decoded::PedersenCommitment> for PedersenCommitment {
32    fn from(decoded_commitment: decoded::PedersenCommitment) -> Self {
33        Self(decoded_commitment.to_bytes())
34    }
35}
36
37// For proof verification, interpret pod::PedersenCommitment directly as CompressedRistretto
38#[cfg(not(target_os = "solana"))]
39impl From<PedersenCommitment> for CompressedRistretto {
40    fn from(pod_commitment: PedersenCommitment) -> Self {
41        Self(pod_commitment.0)
42    }
43}
44
45#[cfg(not(target_os = "solana"))]
46impl TryFrom<PedersenCommitment> for decoded::PedersenCommitment {
47    type Error = ElGamalError;
48
49    fn try_from(pod_commitment: PedersenCommitment) -> Result<Self, Self::Error> {
50        Self::from_bytes(&pod_commitment.0).ok_or(ElGamalError::CiphertextDeserialization)
51    }
52}
