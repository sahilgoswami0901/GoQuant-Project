1//! The ciphertext validity sigma proof system.
2//!
3//! The ciphertext validity proof is defined with respect to a Pedersen commitment and two
4//! decryption handles. The proof certifies that a given Pedersen commitment can be decrypted using
5//! ElGamal private keys that are associated with each of the two decryption handles. To generate
6//! the proof, a prover must provide the Pedersen opening associated with the commitment.
7//!
8//! The protocol guarantees computational soundness (by the hardness of discrete log) and perfect
9//! zero-knowledge in the random oracle model.
10
11#[cfg(not(target_os = "solana"))]
12use crate::encryption::{
13    elgamal::{DecryptHandle, ElGamalPubkey},
14    pedersen::{PedersenCommitment, PedersenOpening},
15};
16use {
17    crate::{
18        sigma_proofs::{
19            errors::ValidityProofVerificationError,
20            grouped_ciphertext_validity_proof::GroupedCiphertext2HandlesValidityProof,
21        },
22        transcript::TranscriptProtocol,
23    },
24    curve25519_dalek::scalar::Scalar,
25    merlin::Transcript,
26};
27
28/// Batched grouped ciphertext validity proof with two handles.
29///
30/// A batched grouped ciphertext validity proof certifies the validity of two instances of a
31/// standard ciphertext validity proof. An instance of a standard validity proof consists of one
32/// ciphertext and two decryption handles: `(commitment, destination_handle, auditor_handle)`. An
33/// instance of a batched ciphertext validity proof is a pair `(commitment_0,
34/// destination_handle_0, auditor_handle_0)` and `(commitment_1, destination_handle_1,
35/// auditor_handle_1)`. The proof certifies the analogous decryptable properties for each one of
36/// these pairs of commitment and decryption handles.
37#[allow(non_snake_case)]
38#[derive(Clone)]
39pub struct BatchedGroupedCiphertext2HandlesValidityProof(GroupedCiphertext2HandlesValidityProof);
40
41#[allow(non_snake_case)]
42#[cfg(not(target_os = "solana"))]
43impl BatchedGroupedCiphertext2HandlesValidityProof {
44    /// Creates a batched grouped ciphertext validity proof.
45    ///
46    /// The function simply batches the input openings and invokes the standard grouped ciphertext
47    /// validity proof constructor.
48    ///
49    /// This function is randomized. It uses `OsRng` internally to generate random scalars.
50    pub fn new<T: Into<Scalar>>(
51        (destination_pubkey, auditor_pubkey): (&ElGamalPubkey, &ElGamalPubkey),
52        (amount_lo, amount_hi): (T, T),
53        (opening_lo, opening_hi): (&PedersenOpening, &PedersenOpening),
54        transcript: &mut Transcript,
55    ) -> Self {
56        transcript.batched_grouped_ciphertext_validity_proof_domain_separator();
57
58        let t = transcript.challenge_scalar(b"t");
59
60        let batched_message = amount_lo.into() + amount_hi.into() * t;
61        let batched_opening = opening_lo + &(opening_hi * &t);
62
63        BatchedGroupedCiphertext2HandlesValidityProof(GroupedCiphertext2HandlesValidityProof::new(
64            (destination_pubkey, auditor_pubkey),
65            batched_message,
66            &batched_opening,
67            transcript,
68        ))
69    }
70
71    /// Verifies a batched grouped ciphertext validity proof.
72    ///
73    /// The function does *not* hash the public keys, commitment, or decryption handles into the
74    /// transcript. For security, the caller (the main protocol) should hash these public
75    /// components prior to invoking this constructor.
76    pub fn verify(
77        self,
78        (destination_pubkey, auditor_pubkey): (&ElGamalPubkey, &ElGamalPubkey),
79        (commitment_lo, commitment_hi): (&PedersenCommitment, &PedersenCommitment),
80        (destination_handle_lo, destination_handle_hi): (&DecryptHandle, &DecryptHandle),
81        (auditor_handle_lo, auditor_handle_hi): (&DecryptHandle, &DecryptHandle),
82        transcript: &mut Transcript,
83    ) -> Result<(), ValidityProofVerificationError> {
84        transcript.batched_grouped_ciphertext_validity_proof_domain_separator();
85
86        let t = transcript.challenge_scalar(b"t");
87
88        let batched_commitment = commitment_lo + commitment_hi * t;
89        let destination_batched_handle = destination_handle_lo + destination_handle_hi * t;
90        let auditor_batched_handle = auditor_handle_lo + auditor_handle_hi * t;
91
92        let BatchedGroupedCiphertext2HandlesValidityProof(validity_proof) = self;
93
94        validity_proof.verify(
95            &batched_commitment,
96            (destination_pubkey, auditor_pubkey),
97            (&destination_batched_handle, &auditor_batched_handle),
98            transcript,
99        )
100    }
101
102    pub fn to_bytes(&self) -> [u8; 160] {
103        self.0.to_bytes()
104    }
105
106    pub fn from_bytes(bytes: &[u8]) -> Result<Self, ValidityProofVerificationError> {
107        GroupedCiphertext2HandlesValidityProof::from_bytes(bytes).map(Self)
108    }
109}

111#[cfg(test)]
112mod test {
113    use {
114        super::*,
115        crate::encryption::{elgamal::ElGamalKeypair, pedersen::Pedersen},
116    };
117
118    #[test]
119    fn test_batched_grouped_ciphertext_validity_proof() {
120        let destination_keypair = ElGamalKeypair::new_rand();
121        let destination_pubkey = destination_keypair.pubkey();
122
123        let auditor_keypair = ElGamalKeypair::new_rand();
124        let auditor_pubkey = auditor_keypair.pubkey();
125
126        let amount_lo: u64 = 55;
127        let amount_hi: u64 = 77;
128
129        let (commitment_lo, open_lo) = Pedersen::new(amount_lo);
130        let (commitment_hi, open_hi) = Pedersen::new(amount_hi);
131
132        let destination_handle_lo = destination_pubkey.decrypt_handle(&open_lo);
133        let destination_handle_hi = destination_pubkey.decrypt_handle(&open_hi);
134
135        let auditor_handle_lo = auditor_pubkey.decrypt_handle(&open_lo);
136        let auditor_handle_hi = auditor_pubkey.decrypt_handle(&open_hi);
137
138        let mut prover_transcript = Transcript::new(b"Test");
139        let mut verifier_transcript = Transcript::new(b"Test");
140
141        let proof = BatchedGroupedCiphertext2HandlesValidityProof::new(
142            (destination_pubkey, auditor_pubkey),
143            (amount_lo, amount_hi),
144            (&open_lo, &open_hi),
145            &mut prover_transcript,
146        );
147
148        assert!(proof
149            .verify(
150                (destination_pubkey, auditor_pubkey),
151                (&commitment_lo, &commitment_hi),
152                (&destination_handle_lo, &destination_handle_hi),
153                (&auditor_handle_lo, &auditor_handle_hi),
154                &mut verifier_transcript,
155            )
156            .is_ok());
157    }
