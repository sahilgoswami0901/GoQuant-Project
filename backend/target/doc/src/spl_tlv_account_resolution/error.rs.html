3use spl_program_error::*;

5/// Errors that may be returned by the Account Resolution library.
6#[spl_program_error(hash_error_code_start = 2_724_315_840)]
7pub enum AccountResolutionError {
8    /// Incorrect account provided
9    #[error("Incorrect account provided")]
10    IncorrectAccount,
11    /// Not enough accounts provided
12    #[error("Not enough accounts provided")]
13    NotEnoughAccounts,
14    /// No value initialized in TLV data
15    #[error("No value initialized in TLV data")]
16    TlvUninitialized,
17    /// Some value initialized in TLV data
18    #[error("Some value initialized in TLV data")]
19    TlvInitialized,
20    /// Too many pubkeys provided
21    #[error("Too many pubkeys provided")]
22    TooManyPubkeys,
23    /// Failed to parse `Pubkey` from bytes
24    #[error("Failed to parse `Pubkey` from bytes")]
25    InvalidPubkey,
26    /// Attempted to deserialize an `AccountMeta` but the underlying type has
27    /// PDA configs rather than a fixed address
28    #[error(
29        "Attempted to deserialize an `AccountMeta` but the underlying type has PDA configs rather \
30         than a fixed address"
31    )]
32    AccountTypeNotAccountMeta,
33    /// Provided list of seed configurations too large for a validation account
34    #[error("Provided list of seed configurations too large for a validation account")]
35    SeedConfigsTooLarge,
36    /// Not enough bytes available to pack seed configuration
37    #[error("Not enough bytes available to pack seed configuration")]
38    NotEnoughBytesForSeed,
39    /// The provided bytes are not valid for a seed configuration
40    #[error("The provided bytes are not valid for a seed configuration")]
41    InvalidBytesForSeed,
42    /// Tried to pack an invalid seed configuration
43    #[error("Tried to pack an invalid seed configuration")]
44    InvalidSeedConfig,
45    /// Instruction data too small for seed configuration
46    #[error("Instruction data too small for seed configuration")]
47    InstructionDataTooSmall,
48    /// Could not find account at specified index
49    #[error("Could not find account at specified index")]
50    AccountNotFound,
51    /// Error in checked math operation
52    #[error("Error in checked math operation")]
53    CalculationFailure,
54    /// Could not find account data at specified index
55    #[error("Could not find account data at specified index")]
56    AccountDataNotFound,
57    /// Account data too small for requested seed configuration
58    #[error("Account data too small for requested seed configuration")]
59    AccountDataTooSmall,
60    /// Failed to fetch account
61    #[error("Failed to fetch account")]
62    AccountFetchFailed,
63}
