13//! This “verifier” performs simple checks when the eBPF program is loaded into the VM (before it is
14//! interpreted or JIT-compiled). It has nothing to do with the much more elaborated verifier inside
15//! Linux kernel. There is no verification regarding the program flow control (should be a Direct
16//! Acyclic Graph) or the consistency for registers usage (the verifier of the kernel assigns types
17//! to the registers and is much stricter).
18//!
19//! On the other hand, rbpf is not expected to run in kernel space.
20//!
21//! Improving the verifier would be nice, but this is not trivial (and Linux kernel is under GPL
22//! license, so we cannot copy it).
23//!
24//! Contrary to the verifier of the Linux kernel, this one does not modify the bytecode at all.
25

33/// Error definitions
34#[derive(Debug, Error, Eq, PartialEq)]
35pub enum VerifierError {
36    /// ProgramLengthNotMultiple
37    #[error("program length must be a multiple of {} octets", ebpf::INSN_SIZE)]
38    ProgramLengthNotMultiple,
39    /// Deprecated
40    #[error("Deprecated")]
41    ProgramTooLarge(usize),
42    /// NoProgram
43    #[error("no program set, call prog_set() to load one")]
44    NoProgram,
45    /// Division by zero
46    #[error("division by 0 (insn #{0})")]
47    DivisionByZero(usize),
48    /// UnsupportedLEBEArgument
49    #[error("unsupported argument for LE/BE (insn #{0})")]
50    UnsupportedLEBEArgument(usize),
51    /// LDDWCannotBeLast
52    #[error("LD_DW instruction cannot be last in program")]
53    LDDWCannotBeLast,
54    /// IncompleteLDDW
55    #[error("incomplete LD_DW instruction (insn #{0})")]
56    IncompleteLDDW(usize),
57    /// InfiniteLoop
58    #[error("infinite loop (insn #{0})")]
59    InfiniteLoop(usize),
60    /// JumpOutOfCode
61    #[error("jump out of code to #{0} (insn #{1})")]
62    JumpOutOfCode(usize, usize),
63    /// JumpToMiddleOfLDDW
64    #[error("jump to middle of LD_DW at #{0} (insn #{1})")]
65    JumpToMiddleOfLDDW(usize, usize),
66    /// InvalidSourceRegister
67    #[error("invalid source register (insn #{0})")]
68    InvalidSourceRegister(usize),
69    /// CannotWriteR10
70    #[error("cannot write into register r10 (insn #{0})")]
71    CannotWriteR10(usize),
72    /// InvalidDestinationRegister
73    #[error("invalid destination register (insn #{0})")]
74    InvalidDestinationRegister(usize),
75    /// UnknownOpCode
76    #[error("unknown eBPF opcode {0:#2x} (insn #{1:?})")]
77    UnknownOpCode(u8, usize),
78    /// Shift with overflow
79    #[error("Shift with overflow of {0}-bit value by {1} (insn #{2:?})")]
80    ShiftWithOverflow(u64, u64, usize),
81    /// Invalid register specified
82    #[error("Invalid register specified at instruction {0}")]
83    InvalidRegister(usize),
84    /// Invalid function
85    #[error("Invalid function at instruction {0}")]
86    InvalidFunction(usize),
87}

89/// eBPF Verifier
90pub trait Verifier {
91    /// eBPF verification function that returns an error if the program does not meet its requirements.
92    ///
93    /// Some examples of things the verifier may reject the program for:
94    ///
95    ///   - Program does not terminate.
96    ///   - Unknown instructions.
97    ///   - Bad formed instruction.
98    ///   - Unknown eBPF syscall index.
99    fn verify(
100        prog: &[u8],
101        config: &Config,
102        sbpf_version: &SBPFVersion,
103        function_registry: &FunctionRegistry<usize>,
104    ) -> Result<(), VerifierError>;
105}
106
107fn check_prog_len(prog: &[u8]) -> Result<(), VerifierError> {
108    if prog.len() % ebpf::INSN_SIZE != 0 {
109        return Err(VerifierError::ProgramLengthNotMultiple);
110    }
111    if prog.is_empty() {
112        return Err(VerifierError::NoProgram);
113    }
114    Ok(())
115}
116
117fn check_imm_nonzero(insn: &ebpf::Insn, insn_ptr: usize) -> Result<(), VerifierError> {
118    if insn.imm == 0 {
119        return Err(VerifierError::DivisionByZero(insn_ptr));
120    }
121    Ok(())
122}
123
124fn check_imm_endian(insn: &ebpf::Insn, insn_ptr: usize) -> Result<(), VerifierError> {
125    match insn.imm {
126        16 | 32 | 64 => Ok(()),
127        _ => Err(VerifierError::UnsupportedLEBEArgument(insn_ptr)),
128    }
129}
130
131fn check_load_dw(prog: &[u8], insn_ptr: usize) -> Result<(), VerifierError> {
132    if (insn_ptr + 1) * ebpf::INSN_SIZE >= prog.len() {
133        // Last instruction cannot be LD_DW because there would be no 2nd DW
134        return Err(VerifierError::LDDWCannotBeLast);
135    }
136    let next_insn = ebpf::get_insn(prog, insn_ptr + 1);
137    if next_insn.opc != 0 {
138        return Err(VerifierError::IncompleteLDDW(insn_ptr));
139    }
140    Ok(())
141}
142
143fn check_jmp_offset(
144    prog: &[u8],
145    insn_ptr: usize,
146    function_range: &std::ops::Range<usize>,
147) -> Result<(), VerifierError> {
148    let insn = ebpf::get_insn(prog, insn_ptr);
149
150    let dst_insn_ptr = insn_ptr as isize + 1 + insn.off as isize;
151    if dst_insn_ptr < 0 || !function_range.contains(&(dst_insn_ptr as usize)) {
152        return Err(VerifierError::JumpOutOfCode(
153            dst_insn_ptr as usize,
154            insn_ptr,
155        ));
156    }
157    let dst_insn = ebpf::get_insn(prog, dst_insn_ptr as usize);
158    if dst_insn.opc == 0 {
159        return Err(VerifierError::JumpToMiddleOfLDDW(
160            dst_insn_ptr as usize,
161            insn_ptr,
162        ));
163    }
164    Ok(())
165}
166
167fn check_registers(
168    insn: &ebpf::Insn,
169    store: bool,
170    insn_ptr: usize,
171    sbpf_version: &SBPFVersion,
172) -> Result<(), VerifierError> {
173    if insn.src > 10 {
174        return Err(VerifierError::InvalidSourceRegister(insn_ptr));
175    }
176
177    match (insn.dst, store) {
178        (0..=9, _) | (10, true) => Ok(()),
179        (11, _) if sbpf_version.dynamic_stack_frames() && insn.opc == ebpf::ADD64_IMM => Ok(()),
180        (10, false) => Err(VerifierError::CannotWriteR10(insn_ptr)),
181        (_, _) => Err(VerifierError::InvalidDestinationRegister(insn_ptr)),
182    }
183}
184
185/// Check that the imm is a valid shift operand
186fn check_imm_shift(insn: &ebpf::Insn, insn_ptr: usize, imm_bits: u64) -> Result<(), VerifierError> {
187    let shift_by = insn.imm as u64;
188    if insn.imm < 0 || shift_by >= imm_bits {
189        return Err(VerifierError::ShiftWithOverflow(
190            shift_by, imm_bits, insn_ptr,
191        ));
192    }
193    Ok(())
194}
195
196/// Check that callx has a valid register number
197fn check_callx_register(
198    insn: &ebpf::Insn,
199    insn_ptr: usize,
200    config: &Config,
201    sbpf_version: &SBPFVersion,
202) -> Result<(), VerifierError> {
203    let reg = if sbpf_version.callx_uses_src_reg() {
204        insn.src as i64
205    } else {
206        insn.imm
207    };
208    if !(0..=10).contains(&reg) || (reg == 10 && config.reject_callx_r10) {
209        return Err(VerifierError::InvalidRegister(insn_ptr));
210    }
211    Ok(())
212}

214/// Mandatory verifier for solana programs to run on-chain
215#[derive(Debug)]
216pub struct RequisiteVerifier {}
217impl Verifier for RequisiteVerifier {
218    /// Check the program against the verifier's rules
219    #[rustfmt::skip]
220    fn verify(prog: &[u8], config: &Config, sbpf_version: &SBPFVersion, function_registry: &FunctionRegistry<usize>) -> Result<(), VerifierError> {
221        check_prog_len(prog)?;
222
223        let program_range = 0..prog.len() / ebpf::INSN_SIZE;
224        let mut function_iter = function_registry.keys().map(|insn_ptr| insn_ptr as usize).peekable();
225        let mut function_range = program_range.start..program_range.end;
226        let mut insn_ptr: usize = 0;
227        while (insn_ptr + 1) * ebpf::INSN_SIZE <= prog.len() {
228            let insn = ebpf::get_insn(prog, insn_ptr);
229            let mut store = false;
230
231            if sbpf_version.static_syscalls() && function_iter.peek() == Some(&insn_ptr) {
232                function_range.start = function_iter.next().unwrap_or(0);
233                function_range.end = *function_iter.peek().unwrap_or(&program_range.end);
234                let insn = ebpf::get_insn(prog, function_range.end.saturating_sub(1));
235                match insn.opc {
236                    ebpf::JA | ebpf::EXIT => {},
237                    _ => return Err(VerifierError::InvalidFunction(
238                        function_range.end.saturating_sub(1),
239                    )),
240                }
241            }
242
243            match insn.opc {
244                ebpf::LD_DW_IMM if !sbpf_version.disable_lddw() => {
245                    check_load_dw(prog, insn_ptr)?;
246                    insn_ptr += 1;
247                },
248
249                // BPF_LDX class
250                ebpf::LD_B_REG   => {},
251                ebpf::LD_H_REG   => {},
252                ebpf::LD_W_REG   => {},
253                ebpf::LD_DW_REG  => {},
254
255                // BPF_ST class
256                ebpf::ST_B_IMM   => store = true,
257                ebpf::ST_H_IMM   => store = true,
258                ebpf::ST_W_IMM   => store = true,
259                ebpf::ST_DW_IMM  => store = true,
260
261                // BPF_STX class
262                ebpf::ST_B_REG   => store = true,
263                ebpf::ST_H_REG   => store = true,
264                ebpf::ST_W_REG   => store = true,
265                ebpf::ST_DW_REG  => store = true,
266
267                // BPF_ALU class
268                ebpf::ADD32_IMM  => {},
269                ebpf::ADD32_REG  => {},
270                ebpf::SUB32_IMM  => {},
271                ebpf::SUB32_REG  => {},
272                ebpf::MUL32_IMM  if !sbpf_version.enable_pqr() => {},
273                ebpf::MUL32_REG  if !sbpf_version.enable_pqr() => {},
274                ebpf::DIV32_IMM  if !sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
275                ebpf::DIV32_REG  if !sbpf_version.enable_pqr() => {},
276                ebpf::OR32_IMM   => {},
277                ebpf::OR32_REG   => {},
278                ebpf::AND32_IMM  => {},
279                ebpf::AND32_REG  => {},
280                ebpf::LSH32_IMM  => { check_imm_shift(&insn, insn_ptr, 32)?; },
281                ebpf::LSH32_REG  => {},
282                ebpf::RSH32_IMM  => { check_imm_shift(&insn, insn_ptr, 32)?; },
283                ebpf::RSH32_REG  => {},
284                ebpf::NEG32      if sbpf_version.enable_neg() => {},
285                ebpf::MOD32_IMM  if !sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
286                ebpf::MOD32_REG  if !sbpf_version.enable_pqr() => {},
287                ebpf::XOR32_IMM  => {},
288                ebpf::XOR32_REG  => {},
289                ebpf::MOV32_IMM  => {},
290                ebpf::MOV32_REG  => {},
291                ebpf::ARSH32_IMM => { check_imm_shift(&insn, insn_ptr, 32)?; },
292                ebpf::ARSH32_REG => {},
293                ebpf::LE         if sbpf_version.enable_le() => { check_imm_endian(&insn, insn_ptr)?; },
294                ebpf::BE         => { check_imm_endian(&insn, insn_ptr)?; },
295
296                // BPF_ALU64 class
297                ebpf::ADD64_IMM  => {},
298                ebpf::ADD64_REG  => {},
299                ebpf::SUB64_IMM  => {},
300                ebpf::SUB64_REG  => {},
301                ebpf::MUL64_IMM  if !sbpf_version.enable_pqr() => {},
302                ebpf::MUL64_REG  if !sbpf_version.enable_pqr() => {},
303                ebpf::DIV64_IMM  if !sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
304                ebpf::DIV64_REG  if !sbpf_version.enable_pqr() => {},
305                ebpf::OR64_IMM   => {},
306                ebpf::OR64_REG   => {},
307                ebpf::AND64_IMM  => {},
308                ebpf::AND64_REG  => {},
309                ebpf::LSH64_IMM  => { check_imm_shift(&insn, insn_ptr, 64)?; },
310                ebpf::LSH64_REG  => {},
311                ebpf::RSH64_IMM  => { check_imm_shift(&insn, insn_ptr, 64)?; },
312                ebpf::RSH64_REG  => {},
313                ebpf::NEG64      if sbpf_version.enable_neg() => {},
314                ebpf::MOD64_IMM  if !sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
315                ebpf::MOD64_REG  if !sbpf_version.enable_pqr() => {},
316                ebpf::XOR64_IMM  => {},
317                ebpf::XOR64_REG  => {},
318                ebpf::MOV64_IMM  => {},
319                ebpf::MOV64_REG  => {},
320                ebpf::ARSH64_IMM => { check_imm_shift(&insn, insn_ptr, 64)?; },
321                ebpf::ARSH64_REG => {},
322                ebpf::HOR64_IMM  if sbpf_version.disable_lddw() => {},
323
324                // BPF_PQR class
325                ebpf::LMUL32_IMM if sbpf_version.enable_pqr() => {},
326                ebpf::LMUL32_REG if sbpf_version.enable_pqr() => {},
327                ebpf::LMUL64_IMM if sbpf_version.enable_pqr() => {},
328                ebpf::LMUL64_REG if sbpf_version.enable_pqr() => {},
329                ebpf::UHMUL64_IMM if sbpf_version.enable_pqr() => {},
330                ebpf::UHMUL64_REG if sbpf_version.enable_pqr() => {},
331                ebpf::SHMUL64_IMM if sbpf_version.enable_pqr() => {},
332                ebpf::SHMUL64_REG if sbpf_version.enable_pqr() => {},
333                ebpf::UDIV32_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
334                ebpf::UDIV32_REG if sbpf_version.enable_pqr() => {},
335                ebpf::UDIV64_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
336                ebpf::UDIV64_REG if sbpf_version.enable_pqr() => {},
337                ebpf::UREM32_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
338                ebpf::UREM32_REG if sbpf_version.enable_pqr() => {},
339                ebpf::UREM64_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
340                ebpf::UREM64_REG if sbpf_version.enable_pqr() => {},
341                ebpf::SDIV32_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
342                ebpf::SDIV32_REG if sbpf_version.enable_pqr() => {},
343                ebpf::SDIV64_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
344                ebpf::SDIV64_REG if sbpf_version.enable_pqr() => {},
345                ebpf::SREM32_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
346                ebpf::SREM32_REG if sbpf_version.enable_pqr() => {},
347                ebpf::SREM64_IMM if sbpf_version.enable_pqr() => { check_imm_nonzero(&insn, insn_ptr)?; },
348                ebpf::SREM64_REG if sbpf_version.enable_pqr() => {},
349
350                // BPF_JMP class
351                ebpf::JA         => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
352                ebpf::JEQ_IMM    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
353                ebpf::JEQ_REG    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
354                ebpf::JGT_IMM    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
355                ebpf::JGT_REG    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
356                ebpf::JGE_IMM    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
357                ebpf::JGE_REG    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
358                ebpf::JLT_IMM    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
359                ebpf::JLT_REG    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
360                ebpf::JLE_IMM    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
361                ebpf::JLE_REG    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
362                ebpf::JSET_IMM   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
363                ebpf::JSET_REG   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
364                ebpf::JNE_IMM    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
365                ebpf::JNE_REG    => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
366                ebpf::JSGT_IMM   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
367                ebpf::JSGT_REG   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
368                ebpf::JSGE_IMM   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
369                ebpf::JSGE_REG   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
370                ebpf::JSLT_IMM   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
371                ebpf::JSLT_REG   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
372                ebpf::JSLE_IMM   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
373                ebpf::JSLE_REG   => { check_jmp_offset(prog, insn_ptr, &function_range)?; },
374                ebpf::CALL_IMM   if sbpf_version.static_syscalls() && insn.src != 0 => { check_jmp_offset(prog, insn_ptr, &program_range)?; },
375                ebpf::CALL_IMM   => {},
376                ebpf::CALL_REG   => { check_callx_register(&insn, insn_ptr, config, sbpf_version)?; },
377                ebpf::EXIT       => {},
378
379                _                => {
380                    return Err(VerifierError::UnknownOpCode(insn.opc, insn_ptr));
381                }
382            }
383
384            check_registers(&insn, store, insn_ptr, sbpf_version)?;
385
386            insn_ptr += 1;
387        }
388
389        // insn_ptr should now be equal to number of instructions.
390        if insn_ptr != prog.len() / ebpf::INSN_SIZE {
391            return Err(VerifierError::JumpOutOfCode(insn_ptr, insn_ptr));
392        }
393
394        Ok(())
395    }
