27/// Shift the RUNTIME_ENVIRONMENT_KEY by this many bits to the LSB
28///
29/// 3 bits for 8 Byte alignment, and 1 bit to have encoding space for the RuntimeEnvironment.
30const PROGRAM_ENVIRONMENT_KEY_SHIFT: u32 = 4;
31static RUNTIME_ENVIRONMENT_KEY: std::sync::OnceLock<i32> = std::sync::OnceLock::<i32>::new();
32
33/// Returns (and if not done before generates) the encryption key for the VM pointer
34pub fn get_runtime_environment_key() -> i32 {
35    *RUNTIME_ENVIRONMENT_KEY
36        .get_or_init(|| rand::thread_rng().gen::<i32>() >> PROGRAM_ENVIRONMENT_KEY_SHIFT)
37}

39/// VM configuration settings
40#[derive(Debug, Clone, Copy, PartialEq, Eq)]
41pub struct Config {
42    /// Maximum call depth
43    pub max_call_depth: usize,
44    /// Size of a stack frame in bytes, must match the size specified in the LLVM BPF backend
45    pub stack_frame_size: usize,
46    /// Enables the use of MemoryMapping and MemoryRegion for address translation
47    pub enable_address_translation: bool,
48    /// Enables gaps in VM address space between the stack frames
49    pub enable_stack_frame_gaps: bool,
50    /// Maximal pc distance after which a new instruction meter validation is emitted by the JIT
51    pub instruction_meter_checkpoint_distance: usize,
52    /// Enable instruction meter and limiting
53    pub enable_instruction_meter: bool,
54    /// Enable instruction tracing
55    pub enable_instruction_tracing: bool,
56    /// Enable dynamic string allocation for labels
57    pub enable_symbol_and_section_labels: bool,
58    /// Reject ELF files containing issues that the verifier did not catch before (up to v0.2.21)
59    pub reject_broken_elfs: bool,
60    /// Ratio of native host instructions per random no-op in JIT (0 = OFF)
61    pub noop_instruction_rate: u32,
62    /// Enable disinfection of immediate values and offsets provided by the user in JIT
63    pub sanitize_user_provided_values: bool,
64    /// Throw ElfError::SymbolHashCollision when a BPF function collides with a registered syscall
65    pub external_internal_function_hash_collision: bool,
66    /// Have the verifier reject "callx r10"
67    pub reject_callx_r10: bool,
68    /// Avoid copying read only sections when possible
69    pub optimize_rodata: bool,
70    /// Use the new ELF parser
71    pub new_elf_parser: bool,
72    /// Use aligned memory mapping
73    pub aligned_memory_mapping: bool,
74    /// Allow ExecutableCapability::V1
75    pub enable_sbpf_v1: bool,
76    /// Allow ExecutableCapability::V2
77    pub enable_sbpf_v2: bool,
78}

80impl Config {
81    /// Returns the size of the stack memory region
82    pub fn stack_size(&self) -> usize {
83        self.stack_frame_size * self.max_call_depth
84    }

112/// Static constructors for Executable
113impl<C: ContextObject> Executable<C> {
114    /// Creates an executable from an ELF file
115    pub fn from_elf(elf_bytes: &[u8], loader: Arc<BuiltinProgram<C>>) -> Result<Self, EbpfError> {
116        let executable = Executable::load(elf_bytes, loader)?;
117        Ok(executable)
118    }
119    /// Creates an executable from machine code
120    pub fn from_text_bytes(
121        text_bytes: &[u8],
122        loader: Arc<BuiltinProgram<C>>,
123        sbpf_version: SBPFVersion,
124        function_registry: FunctionRegistry<usize>,
125    ) -> Result<Self, EbpfError> {
126        Executable::new_from_text_bytes(text_bytes, loader, sbpf_version, function_registry)
127            .map_err(EbpfError::ElfError)
128    }

131/// Runtime context
132pub trait ContextObject {
133    /// Called for every instruction executed when tracing is enabled
134    fn trace(&mut self, state: [u64; 12]);
135    /// Consume instructions from meter
136    fn consume(&mut self, amount: u64);
137    /// Get the number of remaining instructions allowed
138    fn get_remaining(&self) -> u64;
139}

141/// Simple instruction meter for testing
142#[derive(Debug, Clone, Default)]
143pub struct TestContextObject {
144    /// Contains the register state at every instruction in order of execution
145    pub trace_log: Vec<TraceLogEntry>,
146    /// Maximal amount of instructions which still can be executed
147    pub remaining: u64,
148}
149
150impl ContextObject for TestContextObject {
151    fn trace(&mut self, state: [u64; 12]) {
152        self.trace_log.push(state);
153    }
154
155    fn consume(&mut self, amount: u64) {
156        self.remaining = self.remaining.saturating_sub(amount);
157    }
158
159    fn get_remaining(&self) -> u64 {
160        self.remaining
161    }
162}
163
164impl TestContextObject {
165    /// Initialize with instruction meter
166    pub fn new(remaining: u64) -> Self {
167        Self {
168            trace_log: Vec::new(),
169            remaining,
170        }
171    }
172
173    /// Compares an interpreter trace and a JIT trace.
174    ///
175    /// The log of the JIT can be longer because it only validates the instruction meter at branches.
176    pub fn compare_trace_log(interpreter: &Self, jit: &Self) -> bool {
177        let interpreter = interpreter.trace_log.as_slice();
178        let mut jit = jit.trace_log.as_slice();
179        if jit.len() > interpreter.len() {
180            jit = &jit[0..interpreter.len()];
181        }
182        interpreter == jit
183    }
184}

186/// Statistic of taken branches (from a recorded trace)
187pub struct DynamicAnalysis {
188    /// Maximal edge counter value
189    pub edge_counter_max: usize,
190    /// src_node, dst_node, edge_counter
191    pub edges: BTreeMap<usize, BTreeMap<usize, usize>>,
192}
193
194impl DynamicAnalysis {
195    /// Accumulates a trace
196    pub fn new(trace_log: &[[u64; 12]], analysis: &Analysis) -> Self {
197        let mut result = Self {
198            edge_counter_max: 0,
199            edges: BTreeMap::new(),
200        };
201        let mut last_basic_block = usize::MAX;
202        for traced_instruction in trace_log.iter() {
203            let pc = traced_instruction[11] as usize;
204            if analysis.cfg_nodes.contains_key(&pc) {
205                let counter = result
206                    .edges
207                    .entry(last_basic_block)
208                    .or_default()
209                    .entry(pc)
210                    .or_insert(0);
211                *counter += 1;
212                result.edge_counter_max = result.edge_counter_max.max(*counter);
213                last_basic_block = pc;
214            }
215        }
216        result
217    }
218}

231/// A virtual machine to run eBPF programs.
232///
233/// # Examples
234///
235/// ```
236/// use solana_rbpf::{
237///     aligned_memory::AlignedMemory,
238///     ebpf,
239///     elf::Executable,
240///     memory_region::{MemoryMapping, MemoryRegion},
241///     program::{BuiltinProgram, FunctionRegistry, SBPFVersion},
242///     verifier::RequisiteVerifier,
243///     vm::{Config, EbpfVm, TestContextObject},
244/// };
245///
246/// let prog = &[
247///     0x95, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00  // exit
248/// ];
249/// let mem = &mut [
250///     0xaa, 0xbb, 0x11, 0x22, 0xcc, 0xdd
251/// ];
252///
253/// let loader = std::sync::Arc::new(BuiltinProgram::new_mock());
254/// let function_registry = FunctionRegistry::default();
255/// let mut executable = Executable::<TestContextObject>::from_text_bytes(prog, loader.clone(), SBPFVersion::V2, function_registry).unwrap();
256/// executable.verify::<RequisiteVerifier>().unwrap();
257/// let mut context_object = TestContextObject::new(1);
258/// let sbpf_version = executable.get_sbpf_version();
259///
260/// let mut stack = AlignedMemory::<{ebpf::HOST_ALIGN}>::zero_filled(executable.get_config().stack_size());
261/// let stack_len = stack.len();
262/// let mut heap = AlignedMemory::<{ebpf::HOST_ALIGN}>::with_capacity(0);
263///
264/// let regions: Vec<MemoryRegion> = vec![
265///     executable.get_ro_region(),
266///     MemoryRegion::new_writable(
267///         stack.as_slice_mut(),
268///         ebpf::MM_STACK_START,
269///     ),
270///     MemoryRegion::new_writable(heap.as_slice_mut(), ebpf::MM_HEAP_START),
271///     MemoryRegion::new_writable(mem, ebpf::MM_INPUT_START),
272/// ];
273///
274/// let memory_mapping = MemoryMapping::new(regions, executable.get_config(), sbpf_version).unwrap();
275///
276/// let mut vm = EbpfVm::new(loader, sbpf_version, &mut context_object, memory_mapping, stack_len);
277///
278/// let (instruction_count, result) = vm.execute_program(&executable, true);
279/// assert_eq!(instruction_count, 1);
280/// assert_eq!(result.unwrap(), 0);
281/// ```
282#[repr(C)]
283pub struct EbpfVm<'a, C: ContextObject> {
284    /// Needed to exit from the guest back into the host
285    pub host_stack_pointer: *mut u64,
286    /// The current call depth.
287    ///
288    /// Incremented on calls and decremented on exits. It's used to enforce
289    /// config.max_call_depth and to know when to terminate execution.
290    pub call_depth: u64,
291    /// Guest stack pointer (r11).
292    ///
293    /// The stack pointer isn't exposed as an actual register. Only sub and add
294    /// instructions (typically generated by the LLVM backend) are allowed to
295    /// access it when sbpf_version.dynamic_stack_frames()=true. Its value is only
296    /// stored here and therefore the register is not tracked in REGISTER_MAP.
297    pub stack_pointer: u64,
298    /// Pointer to ContextObject
299    pub context_object_pointer: &'a mut C,
300    /// Last return value of instruction_meter.get_remaining()
301    pub previous_instruction_meter: u64,
302    /// Outstanding value to instruction_meter.consume()
303    pub due_insn_count: u64,
304    /// CPU cycles accumulated by the stop watch
305    pub stopwatch_numerator: u64,
306    /// Number of times the stop watch was used
307    pub stopwatch_denominator: u64,
308    /// Registers inlined
309    pub registers: [u64; 12],
310    /// ProgramResult inlined
311    pub program_result: ProgramResult,
312    /// MemoryMapping inlined
313    pub memory_mapping: MemoryMapping<'a>,
314    /// Stack of CallFrames used by the Interpreter
315    pub call_frames: Vec<CallFrame>,
316    /// Loader built-in program
317    pub loader: Arc<BuiltinProgram<C>>,
318    /// TCP port for the debugger interface
319    #[cfg(feature = "debugger")]
320    pub debug_port: Option<u16>,
321}
322
323impl<'a, C: ContextObject> EbpfVm<'a, C> {
324    /// Creates a new virtual machine instance.
325    pub fn new(
326        loader: Arc<BuiltinProgram<C>>,
327        sbpf_version: &SBPFVersion,
328        context_object: &'a mut C,
329        mut memory_mapping: MemoryMapping<'a>,
330        stack_len: usize,
331    ) -> Self {
332        let config = loader.get_config();
333        let stack_pointer =
334            ebpf::MM_STACK_START.saturating_add(if sbpf_version.dynamic_stack_frames() {
335                // the stack is fully descending, frames start as empty and change size anytime r11 is modified
336                stack_len
337            } else {
338                // within a frame the stack grows down, but frames are ascending
339                config.stack_frame_size
340            } as u64);
341        if !config.enable_address_translation {
342            memory_mapping = MemoryMapping::new_identity();
343        }
344        EbpfVm {
345            host_stack_pointer: std::ptr::null_mut(),
346            call_depth: 0,
347            stack_pointer,
348            context_object_pointer: context_object,
349            previous_instruction_meter: 0,
350            due_insn_count: 0,
351            stopwatch_numerator: 0,
352            stopwatch_denominator: 0,
353            registers: [0u64; 12],
354            program_result: ProgramResult::Ok(0),
355            memory_mapping,
356            call_frames: vec![CallFrame::default(); config.max_call_depth],
357            loader,
358            #[cfg(feature = "debugger")]
359            debug_port: None,
360        }
361    }
362
363    /// Execute the program
364    ///
365    /// If interpreted = `false` then the JIT compiled executable is used.
366    pub fn execute_program(
367        &mut self,
368        executable: &Executable<C>,
369        interpreted: bool,
370    ) -> (u64, ProgramResult) {
371        debug_assert!(Arc::ptr_eq(&self.loader, executable.get_loader()));
372        // R1 points to beginning of input memory, R10 to the stack of the first frame, R11 is the pc (hidden)
373        self.registers[1] = ebpf::MM_INPUT_START;
374        self.registers[ebpf::FRAME_PTR_REG] = self.stack_pointer;
375        self.registers[11] = executable.get_entrypoint_instruction_offset() as u64;
376        let config = executable.get_config();
377        let initial_insn_count = if config.enable_instruction_meter {
378            self.context_object_pointer.get_remaining()
379        } else {
380            0
381        };
382        self.previous_instruction_meter = initial_insn_count;
383        self.due_insn_count = 0;
384        self.program_result = ProgramResult::Ok(0);
385        if interpreted {
386            #[cfg(feature = "debugger")]
387            let debug_port = self.debug_port.clone();
388            let mut interpreter = Interpreter::new(self, executable, self.registers);
389            #[cfg(feature = "debugger")]
390            if let Some(debug_port) = debug_port {
391                crate::debugger::execute(&mut interpreter, debug_port);
392            } else {
393                while interpreter.step() {}
394            }
395            #[cfg(not(feature = "debugger"))]
396            while interpreter.step() {}
397        } else {
398            #[cfg(all(feature = "jit", not(target_os = "windows"), target_arch = "x86_64"))]
399            {
400                let compiled_program = match executable
401                    .get_compiled_program()
402                    .ok_or_else(|| EbpfError::JitNotCompiled)
403                {
404                    Ok(compiled_program) => compiled_program,
405                    Err(error) => return (0, ProgramResult::Err(error)),
406                };
407                compiled_program.invoke(config, self, self.registers);
408            }
409            #[cfg(not(all(feature = "jit", not(target_os = "windows"), target_arch = "x86_64")))]
410            {
411                return (0, ProgramResult::Err(EbpfError::JitNotCompiled));
412            }
413        };
414        let instruction_count = if config.enable_instruction_meter {
415            self.context_object_pointer.consume(self.due_insn_count);
416            initial_insn_count.saturating_sub(self.context_object_pointer.get_remaining())
417        } else {
418            0
419        };
420        let mut result = ProgramResult::Ok(0);
421        std::mem::swap(&mut result, &mut self.program_result);
422        (instruction_count, result)
423    }
424
425    /// Invokes a built-in function
426    pub fn invoke_function(&mut self, function: BuiltinFunction<C>) {
427        function(
428            unsafe {
429                (self as *mut _ as *mut u64).offset(get_runtime_environment_key() as isize)
430                    as *mut _
431            },
432            self.registers[1],
433            self.registers[2],
434            self.registers[3],
435            self.registers[4],
436            self.registers[5],
437        );
438    }
