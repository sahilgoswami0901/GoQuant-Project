43// The order can't align with release lifecycle only to remain ABI-compatible...
44#[derive(Serialize, Deserialize, Debug, Clone, Copy, PartialEq, Eq, AbiEnumVisitor, AbiExample)]
45pub enum ClusterType {
46    Testnet,
47    MainnetBeta,
48    Devnet,
49    Development,
50}
51
52impl ClusterType {
53    pub const STRINGS: [&'static str; 4] = ["development", "devnet", "testnet", "mainnet-beta"];
54
55    /// Get the known genesis hash for this ClusterType
56    pub fn get_genesis_hash(&self) -> Option<Hash> {
57        match self {
58            Self::MainnetBeta => {
59                Some(Hash::from_str("5eykt4UsFv8P8NJdTREpY1vzqKqZKvdpKuc147dw2N9d").unwrap())
60            }
61            Self::Testnet => {
62                Some(Hash::from_str("4uhcVJyU9pJkvQyS88uRDiswHXSCkY3zQawwpjk2NsNY").unwrap())
63            }
64            Self::Devnet => {
65                Some(Hash::from_str("EtWTRABZaYq6iMfeYKouRu166VU2xqa1wcaWoxPkrZBG").unwrap())
66            }
67            Self::Development => None,
68        }
69    }
70}

72impl FromStr for ClusterType {
73    type Err = String;
74
75    fn from_str(s: &str) -> Result<Self, Self::Err> {
76        match s {
77            "development" => Ok(ClusterType::Development),
78            "devnet" => Ok(ClusterType::Devnet),
79            "testnet" => Ok(ClusterType::Testnet),
80            "mainnet-beta" => Ok(ClusterType::MainnetBeta),
81            _ => Err(format!("{s} is unrecognized for cluster type")),
82        }
83    }
84}

86#[frozen_abi(digest = "3V3ZVRyzNhRfe8RJwDeGpeTP8xBWGGFBEbwTkvKKVjEa")]
87#[derive(Serialize, Deserialize, Debug, Clone, AbiExample, PartialEq)]
88pub struct GenesisConfig {
89    /// when the network (bootstrap validator) was started relative to the UNIX Epoch
90    pub creation_time: UnixTimestamp,
91    /// initial accounts
92    pub accounts: BTreeMap<Pubkey, Account>,
93    /// built-in programs
94    pub native_instruction_processors: Vec<(String, Pubkey)>,
95    /// accounts for network rewards, these do not count towards capitalization
96    pub rewards_pools: BTreeMap<Pubkey, Account>,
97    pub ticks_per_slot: u64,
98    pub unused: u64,
99    /// network speed configuration
100    pub poh_config: PohConfig,
101    /// this field exists only to ensure that the binary layout of GenesisConfig remains compatible
102    /// with the Solana v0.23 release line
103    pub __backwards_compat_with_v0_23: u64,
104    /// transaction fee config
105    pub fee_rate_governor: FeeRateGovernor,
106    /// rent config
107    pub rent: Rent,
108    /// inflation config
109    pub inflation: Inflation,
110    /// how slots map to epochs
111    pub epoch_schedule: EpochSchedule,
112    /// network runlevel
113    pub cluster_type: ClusterType,
114}

116// useful for basic tests
117pub fn create_genesis_config(lamports: u64) -> (GenesisConfig, Keypair) {
118    let faucet_keypair = Keypair::new();
119    (
120        GenesisConfig::new(
121            &[(
122                faucet_keypair.pubkey(),
123                AccountSharedData::new(lamports, 0, &system_program::id()),
124            )],
125            &[],
126        ),
127        faucet_keypair,
128    )
129}

154impl GenesisConfig {
155    pub fn new(
156        accounts: &[(Pubkey, AccountSharedData)],
157        native_instruction_processors: &[(String, Pubkey)],
158    ) -> Self {
159        Self {
160            accounts: accounts
161                .iter()
162                .cloned()
163                .map(|(key, account)| (key, Account::from(account)))
164                .collect::<BTreeMap<Pubkey, Account>>(),
165            native_instruction_processors: native_instruction_processors.to_vec(),
166            ..GenesisConfig::default()
167        }
168    }
169
170    pub fn hash(&self) -> Hash {
171        let serialized = serialize(&self).unwrap();
172        hash(&serialized)
173    }
174
175    fn genesis_filename(ledger_path: &Path) -> PathBuf {
176        Path::new(ledger_path).join(DEFAULT_GENESIS_FILE)
177    }
178
179    pub fn load(ledger_path: &Path) -> Result<Self, std::io::Error> {
180        let filename = Self::genesis_filename(ledger_path);
181        let file = OpenOptions::new()
182            .read(true)
183            .open(&filename)
184            .map_err(|err| {
185                std::io::Error::new(
186                    std::io::ErrorKind::Other,
187                    format!("Unable to open {filename:?}: {err:?}"),
188                )
189            })?;
190
191        //UNSAFE: Required to create a Mmap
192        let mem = unsafe { Mmap::map(&file) }.map_err(|err| {
193            std::io::Error::new(
194                std::io::ErrorKind::Other,
195                format!("Unable to map {filename:?}: {err:?}"),
196            )
197        })?;
198
199        let genesis_config = deserialize(&mem).map_err(|err| {
200            std::io::Error::new(
201                std::io::ErrorKind::Other,
202                format!("Unable to deserialize {filename:?}: {err:?}"),
203            )
204        })?;
205        Ok(genesis_config)
206    }
207
208    pub fn write(&self, ledger_path: &Path) -> Result<(), std::io::Error> {
209        let serialized = serialize(&self).map_err(|err| {
210            std::io::Error::new(
211                std::io::ErrorKind::Other,
212                format!("Unable to serialize: {err:?}"),
213            )
214        })?;
215
216        std::fs::create_dir_all(ledger_path)?;
217
218        let mut file = File::create(Self::genesis_filename(ledger_path))?;
219        file.write_all(&serialized)
220    }
221
222    pub fn add_account(&mut self, pubkey: Pubkey, account: AccountSharedData) {
223        self.accounts.insert(pubkey, Account::from(account));
224    }
225
226    pub fn add_native_instruction_processor(&mut self, name: String, program_id: Pubkey) {
227        self.native_instruction_processors.push((name, program_id));
228    }
229
230    pub fn hashes_per_tick(&self) -> Option<u64> {
231        self.poh_config.hashes_per_tick
232    }
233
234    pub fn ticks_per_slot(&self) -> u64 {
235        self.ticks_per_slot
236    }
237
238    pub fn ns_per_slot(&self) -> u128 {
239        self.poh_config
240            .target_tick_duration
241            .as_nanos()
242            .saturating_mul(self.ticks_per_slot() as u128)
243    }
244
245    pub fn slots_per_year(&self) -> f64 {
246        years_as_slots(
247            1.0,
248            &self.poh_config.target_tick_duration,
249            self.ticks_per_slot(),
250        )
251    }
252}

254impl fmt::Display for GenesisConfig {
255    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
256        write!(
257            f,
258            "\
259             Creation time: {}\n\
260             Cluster type: {:?}\n\
261             Genesis hash: {}\n\
262             Shred version: {}\n\
263             Ticks per slot: {:?}\n\
264             Hashes per tick: {:?}\n\
265             Target tick duration: {:?}\n\
266             Slots per epoch: {}\n\
267             Warmup epochs: {}abled\n\
268             Slots per year: {}\n\
269             {:?}\n\
270             {:?}\n\
271             {:?}\n\
272             Capitalization: {} SOL in {} accounts\n\
273             Native instruction processors: {:#?}\n\
274             Rewards pool: {:#?}\n\
275             ",
276            Utc.timestamp_opt(self.creation_time, 0)
277                .unwrap()
278                .to_rfc3339(),
279            self.cluster_type,
280            self.hash(),
281            compute_shred_version(&self.hash(), None),
282            self.ticks_per_slot,
283            self.poh_config.hashes_per_tick,
284            self.poh_config.target_tick_duration,
285            self.epoch_schedule.slots_per_epoch,
286            if self.epoch_schedule.warmup {
287                "en"
288            } else {
289                "dis"
290            },
291            self.slots_per_year(),
292            self.inflation,
293            self.rent,
294            self.fee_rate_governor,
295            lamports_to_sol(
296                self.accounts
297                    .iter()
298                    .map(|(pubkey, account)| {
299                        assert!(account.lamports > 0, "{:?}", (pubkey, account));
300                        account.lamports
301                    })
302                    .sum::<u64>()
303            ),
304            self.accounts.len(),
305            self.native_instruction_processors,
306            self.rewards_pools,
307        )
308    }

319    fn make_tmp_path(name: &str) -> PathBuf {
320        let out_dir = std::env::var("FARF_DIR").unwrap_or_else(|_| "farf".to_string());
321        let keypair = Keypair::new();
322
323        let path = [
324            out_dir,
325            "tmp".to_string(),
326            format!("{}-{}", name, keypair.pubkey()),
327        ]
328        .iter()
329        .collect();
330
331        // whack any possible collision
332        let _ignored = std::fs::remove_dir_all(&path);
333        // whack any possible collision
334        let _ignored = std::fs::remove_file(&path);
335
336        path
337    }
338
339    #[test]
340    fn test_genesis_config() {
341        let faucet_keypair = Keypair::new();
342        let mut config = GenesisConfig::default();
343        config.add_account(
344            faucet_keypair.pubkey(),
345            AccountSharedData::new(10_000, 0, &Pubkey::default()),
346        );
347        config.add_account(
348            solana_sdk::pubkey::new_rand(),
349            AccountSharedData::new(1, 0, &Pubkey::default()),
350        );
351        config.add_native_instruction_processor("hi".to_string(), solana_sdk::pubkey::new_rand());
352
353        assert_eq!(config.accounts.len(), 2);
354        assert!(config
355            .accounts
356            .iter()
357            .any(|(pubkey, account)| *pubkey == faucet_keypair.pubkey()
358                && account.lamports == 10_000));
359
360        let path = &make_tmp_path("genesis_config");
361        config.write(path).expect("write");
362        let loaded_config = GenesisConfig::load(path).expect("load");
363        assert_eq!(config.hash(), loaded_config.hash());
364        let _ignored = std::fs::remove_file(path);
365    }
