18/// Number of bytes in a signature
19pub const SIGNATURE_BYTES: usize = 64;
20/// Maximum string length of a base58 encoded signature
21const MAX_BASE58_SIGNATURE_LEN: usize = 88;

23#[repr(transparent)]
24#[derive(
25    Serialize, Deserialize, Clone, Copy, Default, Eq, PartialEq, Ord, PartialOrd, Hash, AbiExample,
26)]
27pub struct Signature(GenericArray<u8, U64>);

29impl crate::sanitize::Sanitize for Signature {}

31impl Signature {
32    #[deprecated(
33        since = "1.16.4",
34        note = "Please use 'Signature::from' or 'Signature::try_from' instead"
35    )]
36    pub fn new(signature_slice: &[u8]) -> Self {
37        Self(GenericArray::clone_from_slice(signature_slice))
38    }
39
40    pub fn new_unique() -> Self {
41        Self::from(std::array::from_fn(|_| rand::random()))
42    }
43
44    pub(self) fn verify_verbose(
45        &self,
46        pubkey_bytes: &[u8],
47        message_bytes: &[u8],
48    ) -> Result<(), ed25519_dalek::SignatureError> {
49        let publickey = ed25519_dalek::PublicKey::from_bytes(pubkey_bytes)?;
50        let signature = self.0.as_slice().try_into()?;
51        publickey.verify_strict(message_bytes, &signature)
52    }
53
54    pub fn verify(&self, pubkey_bytes: &[u8], message_bytes: &[u8]) -> bool {
55        self.verify_verbose(pubkey_bytes, message_bytes).is_ok()
56    }
57}

59pub trait Signable {
60    fn sign(&mut self, keypair: &Keypair) {
61        let signature = keypair.sign_message(self.signable_data().borrow());
62        self.set_signature(signature);
63    }
64    fn verify(&self) -> bool {
65        self.get_signature()
66            .verify(self.pubkey().as_ref(), self.signable_data().borrow())
67    }
68
69    fn pubkey(&self) -> Pubkey;
70    fn signable_data(&self) -> Cow<[u8]>;
71    fn get_signature(&self) -> Signature;
72    fn set_signature(&mut self, signature: Signature);
73}

124#[derive(Debug, Clone, PartialEq, Eq, Error)]
125pub enum ParseSignatureError {
126    #[error("string decoded to wrong size for signature")]
127    WrongSize,
128    #[error("failed to decode string to signature")]
129    Invalid,
130}
131
132impl FromStr for Signature {
133    type Err = ParseSignatureError;
134
135    fn from_str(s: &str) -> Result<Self, Self::Err> {
136        if s.len() > MAX_BASE58_SIGNATURE_LEN {
137            return Err(ParseSignatureError::WrongSize);
138        }
139        let bytes = bs58::decode(s)
140            .into_vec()
141            .map_err(|_| ParseSignatureError::Invalid)?;
142        Signature::try_from(bytes).map_err(|_| ParseSignatureError::WrongSize)
143    }
144}

146#[cfg(test)]
147mod tests {
148    use super::*;
149    #[test]
150    fn test_signature_fromstr() {
151        let signature = Keypair::new().sign_message(&[0u8]);
152
153        let mut signature_base58_str = bs58::encode(signature).into_string();
154
155        assert_eq!(signature_base58_str.parse::<Signature>(), Ok(signature));
156
157        signature_base58_str.push_str(&bs58::encode(signature.0).into_string());
158        assert_eq!(
159            signature_base58_str.parse::<Signature>(),
160            Err(ParseSignatureError::WrongSize)
161        );
162
163        signature_base58_str.truncate(signature_base58_str.len() / 2);
164        assert_eq!(signature_base58_str.parse::<Signature>(), Ok(signature));
165
166        signature_base58_str.truncate(signature_base58_str.len() / 2);
167        assert_eq!(
168            signature_base58_str.parse::<Signature>(),
169            Err(ParseSignatureError::WrongSize)
170        );
171
172        let mut signature_base58_str = bs58::encode(signature.0).into_string();
173        assert_eq!(signature_base58_str.parse::<Signature>(), Ok(signature));
174
175        // throw some non-base58 stuff in there
176        signature_base58_str.replace_range(..1, "I");
177        assert_eq!(
178            signature_base58_str.parse::<Signature>(),
179            Err(ParseSignatureError::Invalid)
180        );
181
182        // too long input string
183        // longest valid encoding
184        let mut too_long = bs58::encode(&[255u8; SIGNATURE_BYTES]).into_string();
185        // and one to grow on
186        too_long.push('1');
187        assert_eq!(
188            too_long.parse::<Signature>(),
189            Err(ParseSignatureError::WrongSize)
190        );
191    }

193    #[test]
194    fn test_off_curve_pubkey_verify_fails() {
195        // Golden point off the ed25519 curve
196        let off_curve_bytes = bs58::decode("9z5nJyQar1FUxVJxpBXzon6kHehbomeYiDaLi9WAMhCq")
197            .into_vec()
198            .unwrap();
199
200        // Confirm golden's off-curvedness
201        let mut off_curve_bits = [0u8; 32];
202        off_curve_bits.copy_from_slice(&off_curve_bytes);
203        let off_curve_point = curve25519_dalek::edwards::CompressedEdwardsY(off_curve_bits);
204        assert_eq!(off_curve_point.decompress(), None);
205
206        let pubkey = Pubkey::try_from(off_curve_bytes).unwrap();
207        let signature = Signature::default();
208        // Unfortunately, ed25519-dalek doesn't surface the internal error types that we'd ideally
209        // `source()` out of the `SignatureError` returned by `verify_strict()`.  So the best we
210        // can do is `is_err()` here.
211        assert!(signature.verify_verbose(pubkey.as_ref(), &[0u8]).is_err());
212    }
