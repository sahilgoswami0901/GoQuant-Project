1//! The compute budget native program.
2
3#![cfg(feature = "full")]
4
5use {
6    crate::instruction::Instruction,
7    borsh::{BorshDeserialize, BorshSerialize},
8};
9
10crate::declare_id!("ComputeBudget111111111111111111111111111111");
11
12/// Compute Budget Instructions
13#[derive(
14    AbiExample,
15    AbiEnumVisitor,
16    BorshDeserialize,
17    BorshSerialize,
18    Clone,
19    Debug,
20    Deserialize,
21    PartialEq,
22    Eq,
23    Serialize,
24)]
25pub enum ComputeBudgetInstruction {
26    Unused, // deprecated variant, reserved value.
27    /// Request a specific transaction-wide program heap region size in bytes.
28    /// The value requested must be a multiple of 1024. This new heap region
29    /// size applies to each program executed in the transaction, including all
30    /// calls to CPIs.
31    RequestHeapFrame(u32),
32    /// Set a specific compute unit limit that the transaction is allowed to consume.
33    SetComputeUnitLimit(u32),
34    /// Set a compute unit price in "micro-lamports" to pay a higher transaction
35    /// fee for higher transaction prioritization.
36    SetComputeUnitPrice(u64),
37    /// Set a specific transaction-wide account data size limit, in bytes, is allowed to load.
38    SetLoadedAccountsDataSizeLimit(u32),
39}
40
41impl ComputeBudgetInstruction {
42    /// Create a `ComputeBudgetInstruction::RequestHeapFrame` `Instruction`
43    pub fn request_heap_frame(bytes: u32) -> Instruction {
44        Instruction::new_with_borsh(id(), &Self::RequestHeapFrame(bytes), vec![])
45    }
46
47    /// Create a `ComputeBudgetInstruction::SetComputeUnitLimit` `Instruction`
48    pub fn set_compute_unit_limit(units: u32) -> Instruction {
49        Instruction::new_with_borsh(id(), &Self::SetComputeUnitLimit(units), vec![])
50    }
51
52    /// Create a `ComputeBudgetInstruction::SetComputeUnitPrice` `Instruction`
53    pub fn set_compute_unit_price(micro_lamports: u64) -> Instruction {
54        Instruction::new_with_borsh(id(), &Self::SetComputeUnitPrice(micro_lamports), vec![])
55    }
56
57    /// Serialize Instruction using borsh, this is only used in runtime::cost_model::tests but compilation
58    /// can't be restricted as it's used across packages
59    #[cfg(feature = "dev-context-only-utils")]
60    pub fn pack(self) -> Result<Vec<u8>, borsh::io::Error> {
61        borsh::to_vec(&self)
62    }
63
64    /// Create a `ComputeBudgetInstruction::SetLoadedAccountsDataSizeLimit` `Instruction`
65    pub fn set_loaded_accounts_data_size_limit(bytes: u32) -> Instruction {
66        Instruction::new_with_borsh(id(), &Self::SetLoadedAccountsDataSizeLimit(bytes), vec![])
67    }
68}
