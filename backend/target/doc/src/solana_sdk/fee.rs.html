3use crate::native_token::sol_to_lamports;
4#[cfg(not(target_os = "solana"))]
5use solana_program::message::SanitizedMessage;

7/// A fee and its associated compute unit limit
8#[derive(Debug, Default, Clone, Eq, PartialEq)]
9pub struct FeeBin {
10    /// maximum compute units for which this fee will be charged
11    pub limit: u64,
12    /// fee in lamports
13    pub fee: u64,
14}
15
16pub struct FeeBudgetLimits {
17    pub loaded_accounts_data_size_limit: usize,
18    pub heap_cost: u64,
19    pub compute_unit_limit: u64,
20    pub prioritization_fee: u64,
21}
22
23/// Information used to calculate fees
24#[derive(Debug, Clone, Eq, PartialEq)]
25pub struct FeeStructure {
26    /// lamports per signature
27    pub lamports_per_signature: u64,
28    /// lamports_per_write_lock
29    pub lamports_per_write_lock: u64,
30    /// Compute unit fee bins
31    pub compute_fee_bins: Vec<FeeBin>,
32}
33
34pub const ACCOUNT_DATA_COST_PAGE_SIZE: u64 = 32_u64.saturating_mul(1024);
35
36impl FeeStructure {
37    pub fn new(
38        sol_per_signature: f64,
39        sol_per_write_lock: f64,
40        compute_fee_bins: Vec<(u64, f64)>,
41    ) -> Self {
42        let compute_fee_bins = compute_fee_bins
43            .iter()
44            .map(|(limit, sol)| FeeBin {
45                limit: *limit,
46                fee: sol_to_lamports(*sol),
47            })
48            .collect::<Vec<_>>();
49        FeeStructure {
50            lamports_per_signature: sol_to_lamports(sol_per_signature),
51            lamports_per_write_lock: sol_to_lamports(sol_per_write_lock),
52            compute_fee_bins,
53        }
54    }
55
56    pub fn get_max_fee(&self, num_signatures: u64, num_write_locks: u64) -> u64 {
57        num_signatures
58            .saturating_mul(self.lamports_per_signature)
59            .saturating_add(num_write_locks.saturating_mul(self.lamports_per_write_lock))
60            .saturating_add(
61                self.compute_fee_bins
62                    .last()
63                    .map(|bin| bin.fee)
64                    .unwrap_or_default(),
65            )
66    }
67
68    pub fn calculate_memory_usage_cost(
69        loaded_accounts_data_size_limit: usize,
70        heap_cost: u64,
71    ) -> u64 {
72        (loaded_accounts_data_size_limit as u64)
73            .saturating_add(ACCOUNT_DATA_COST_PAGE_SIZE.saturating_sub(1))
74            .saturating_div(ACCOUNT_DATA_COST_PAGE_SIZE)
75            .saturating_mul(heap_cost)
76    }
77
78    /// Calculate fee for `SanitizedMessage`
79    #[cfg(not(target_os = "solana"))]
80    pub fn calculate_fee(
81        &self,
82        message: &SanitizedMessage,
83        lamports_per_signature: u64,
84        budget_limits: &FeeBudgetLimits,
85        include_loaded_account_data_size_in_fee: bool,
86    ) -> u64 {
87        // Fee based on compute units and signatures
88        let congestion_multiplier = if lamports_per_signature == 0 {
89            0.0 // test only
90        } else {
91            1.0 // multiplier that has no effect
92        };
93
94        let signature_fee = message
95            .num_signatures()
96            .saturating_mul(self.lamports_per_signature);
97        let write_lock_fee = message
98            .num_write_locks()
99            .saturating_mul(self.lamports_per_write_lock);
100
101        // `compute_fee` covers costs for both requested_compute_units and
102        // requested_loaded_account_data_size
103        let loaded_accounts_data_size_cost = if include_loaded_account_data_size_in_fee {
104            FeeStructure::calculate_memory_usage_cost(
105                budget_limits.loaded_accounts_data_size_limit,
106                budget_limits.heap_cost,
107            )
108        } else {
109            0_u64
110        };
111        let total_compute_units =
112            loaded_accounts_data_size_cost.saturating_add(budget_limits.compute_unit_limit);
113        let compute_fee = self
114            .compute_fee_bins
115            .iter()
116            .find(|bin| total_compute_units <= bin.limit)
117            .map(|bin| bin.fee)
118            .unwrap_or_else(|| {
119                self.compute_fee_bins
120                    .last()
121                    .map(|bin| bin.fee)
122                    .unwrap_or_default()
123            });
124
125        ((budget_limits
126            .prioritization_fee
127            .saturating_add(signature_fee)
128            .saturating_add(write_lock_fee)
129            .saturating_add(compute_fee) as f64)
130            * congestion_multiplier)
131            .round() as u64
132    }
133}
134
135impl Default for FeeStructure {

137        Self::new(0.000005, 0.0, vec![(1_400_000, 0.0)])
138    }

148#[cfg(test)]
149mod tests {
150    use super::*;
151
152    #[test]
153    fn test_calculate_memory_usage_cost() {
154        let heap_cost = 99;
155        const K: usize = 1024;
156
157        // accounts data size are priced in block of 32K, ...
158
159        // ... requesting less than 32K should still be charged as one block
160        assert_eq!(
161            heap_cost,
162            FeeStructure::calculate_memory_usage_cost(31 * K, heap_cost)
163        );
164
165        // ... requesting exact 32K should be charged as one block
166        assert_eq!(
167            heap_cost,
168            FeeStructure::calculate_memory_usage_cost(32 * K, heap_cost)
169        );
170
171        // ... requesting slightly above 32K should be charged as 2 block
172        assert_eq!(
173            heap_cost * 2,
174            FeeStructure::calculate_memory_usage_cost(33 * K, heap_cost)
175        );
176
177        // ... requesting exact 64K should be charged as 2 block
178        assert_eq!(
179            heap_cost * 2,
180            FeeStructure::calculate_memory_usage_cost(64 * K, heap_cost)
181        );
182    }
