1//! Collection of all runtime features.
2//!
3//! Steps to add a new feature are outlined below. Note that these steps only cover
4//! the process of getting a feature into the core Solana code.
5//! - For features that are unambiguously good (ie bug fixes), these steps are sufficient.
6//! - For features that should go up for community vote (ie fee structure changes), more
7//!   information on the additional steps to follow can be found at:
8//!   <https://spl.solana.com/feature-proposal#feature-proposal-life-cycle>
9//!
10//! 1. Generate a new keypair with `solana-keygen new --outfile feature.json --no-passphrase`
11//!    - Keypairs should be held by core contributors only. If you're a non-core contributor going
12//!      through these steps, the PR process will facilitate a keypair holder being picked. That
13//!      person will generate the keypair, provide pubkey for PR, and ultimately enable the feature.
14//! 2. Add a public module for the feature, specifying keypair pubkey as the id with
15//!    `solana_sdk::declare_id!()` within the module.
16//!    Additionally, add an entry to `FEATURE_NAMES` map.
17//! 3. Add desired logic to check for and switch on feature availability.
18//!
19//! For more information on how features are picked up, see comments for `Feature`.

1026/// `FeatureSet` holds the set of currently active/inactive runtime features
1027#[derive(AbiExample, Debug, Clone, Eq, PartialEq)]
1028pub struct FeatureSet {
1029    pub active: HashMap<Pubkey, Slot>,
1030    pub inactive: HashSet<Pubkey>,
1031}
1032impl Default for FeatureSet {
1033    fn default() -> Self {
1034        // All features disabled
1035        Self {
1036            active: HashMap::new(),
1037            inactive: FEATURE_NAMES.keys().cloned().collect(),
1038        }
1039    }
1040}
1041impl FeatureSet {
1042    pub fn is_active(&self, feature_id: &Pubkey) -> bool {
1043        self.active.contains_key(feature_id)
1044    }
1045
1046    pub fn activated_slot(&self, feature_id: &Pubkey) -> Option<Slot> {
1047        self.active.get(feature_id).copied()
1048    }
1049
1050    /// List of enabled features that trigger full inflation
1051    pub fn full_inflation_features_enabled(&self) -> HashSet<Pubkey> {
1052        let mut hash_set = FULL_INFLATION_FEATURE_PAIRS
1053            .iter()
1054            .filter_map(|pair| {
1055                if self.is_active(&pair.vote_id) && self.is_active(&pair.enable_id) {
1056                    Some(pair.enable_id)
1057                } else {
1058                    None
1059                }
1060            })
1061            .collect::<HashSet<_>>();
1062
1063        if self.is_active(&full_inflation::devnet_and_testnet::id()) {
1064            hash_set.insert(full_inflation::devnet_and_testnet::id());
1065        }
1066        hash_set
1067    }
1068
1069    /// All features enabled, useful for testing
1070    pub fn all_enabled() -> Self {
1071        Self {
1072            active: FEATURE_NAMES.keys().cloned().map(|key| (key, 0)).collect(),
1073            inactive: HashSet::new(),
1074        }
1075    }
1076
1077    /// Activate a feature
1078    pub fn activate(&mut self, feature_id: &Pubkey, slot: u64) {
1079        self.inactive.remove(feature_id);
1080        self.active.insert(*feature_id, slot);
1081    }
1082
1083    /// Deactivate a feature
1084    pub fn deactivate(&mut self, feature_id: &Pubkey) {
1085        self.active.remove(feature_id);
1086        self.inactive.insert(*feature_id);
1087    }
1088
1089    pub fn new_warmup_cooldown_rate_epoch(&self, epoch_schedule: &EpochSchedule) -> Option<Epoch> {
1090        self.activated_slot(&reduce_stake_warmup_cooldown::id())
1091            .map(|slot| epoch_schedule.get_epoch(slot))
1092    }
1093}
