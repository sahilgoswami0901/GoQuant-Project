15/// Common interface for different congestion controllers
16pub trait Controller: Send {
17    /// One or more packets were just sent
18    #[allow(unused_variables)]
19    fn on_sent(&mut self, now: Instant, bytes: u64, last_packet_number: u64) {}
20
21    /// Packet deliveries were confirmed
22    ///
23    /// `app_limited` indicates whether the connection was blocked on outgoing
24    /// application data prior to receiving these acknowledgements.
25    #[allow(unused_variables)]
26    fn on_ack(
27        &mut self,
28        now: Instant,
29        sent: Instant,
30        bytes: u64,
31        app_limited: bool,
32        rtt: &RttEstimator,
33    ) {
34    }
35
36    /// Packets are acked in batches, all with the same `now` argument. This indicates one of those batches has completed.
37    #[allow(unused_variables)]
38    fn on_end_acks(
39        &mut self,
40        now: Instant,
41        in_flight: u64,
42        app_limited: bool,
43        largest_packet_num_acked: Option<u64>,
44    ) {
45    }
46
47    /// Packets were deemed lost or marked congested
48    ///
49    /// `in_persistent_congestion` indicates whether all packets sent within the persistent
50    /// congestion threshold period ending when the most recent packet in this batch was sent were
51    /// lost.
52    /// `lost_bytes` indicates how many bytes were lost. This value will be 0 for ECN triggers.
53    fn on_congestion_event(
54        &mut self,
55        now: Instant,
56        sent: Instant,
57        is_persistent_congestion: bool,
58        lost_bytes: u64,
59    );
60
61    /// The known MTU for the current network path has been updated
62    fn on_mtu_update(&mut self, new_mtu: u16);
63
64    /// Number of ack-eliciting bytes that may be in flight
65    fn window(&self) -> u64;
66
67    /// Duplicate the controller's state
68    fn clone_box(&self) -> Box<dyn Controller>;
69
70    /// Initial congestion window
71    fn initial_window(&self) -> u64;
72
73    /// Returns Self for use in down-casting to extract implementation details
74    fn into_any(self: Box<Self>) -> Box<dyn Any>;
75}
76
77/// Constructs controllers on demand
78pub trait ControllerFactory {
79    /// Construct a fresh `Controller`
80    fn build(&self, now: Instant, current_mtu: u16) -> Box<dyn Controller>;
81}
