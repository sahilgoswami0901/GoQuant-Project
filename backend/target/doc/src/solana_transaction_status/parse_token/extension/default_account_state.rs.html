8pub(in crate::parse_token) fn parse_default_account_state_instruction(
9    instruction_data: &[u8],
10    account_indexes: &[u8],
11    account_keys: &AccountKeys,
12) -> Result<ParsedInstructionEnum, ParseInstructionError> {
13    let (default_account_state_instruction, account_state) = decode_instruction(instruction_data)
14        .map_err(|_| {
15        ParseInstructionError::InstructionNotParsable(ParsableProgram::SplToken)
16    })?;
17    let instruction_type = "DefaultAccountState";
18    match default_account_state_instruction {
19        DefaultAccountStateInstruction::Initialize => {
20            check_num_token_accounts(account_indexes, 1)?;
21            Ok(ParsedInstructionEnum {
22                instruction_type: format!("initialize{instruction_type}"),
23                info: json!({
24                    "mint": account_keys[account_indexes[0] as usize].to_string(),
25                    "accountState": UiAccountState::from(account_state),
26                }),
27            })
28        }
29        DefaultAccountStateInstruction::Update => {
30            check_num_token_accounts(account_indexes, 2)?;
31            let mut value = json!({
32                "mint": account_keys[account_indexes[0] as usize].to_string(),
33                "accountState": UiAccountState::from(account_state),
34            });
35            let map = value.as_object_mut().unwrap();
36            parse_signers(
37                map,
38                1,
39                account_keys,
40                account_indexes,
41                "freezeAuthority",
42                "multisigFreezeAuthority",
43            );
44            Ok(ParsedInstructionEnum {
45                instruction_type: format!("update{instruction_type}"),
46                info: value,
47            })
48        }
49    }
50}

52#[cfg(test)]
53mod test {
54    use {
55        super::*,
56        solana_sdk::pubkey::Pubkey,
57        spl_token_2022::{
58            extension::default_account_state::instruction::{
59                initialize_default_account_state, update_default_account_state,
60            },
61            solana_program::message::Message,
62            state::AccountState,
63        },
64    };
65
66    #[test]
67    fn test_parse_default_account_state_instruction() {
68        let mint_pubkey = Pubkey::new_unique();
69        let init_default_account_state_ix = initialize_default_account_state(
70            &spl_token_2022::id(),
71            &mint_pubkey,
72            &AccountState::Frozen,
73        )
74        .unwrap();
75        let message = Message::new(&[init_default_account_state_ix], None);
76        let compiled_instruction = &message.instructions[0];
77        assert_eq!(
78            parse_token(
79                compiled_instruction,
80                &AccountKeys::new(&message.account_keys, None)
81            )
82            .unwrap(),
83            ParsedInstructionEnum {
84                instruction_type: "initializeDefaultAccountState".to_string(),
85                info: json!({
86                    "mint": mint_pubkey.to_string(),
87                    "accountState": "frozen",
88                })
89            }
90        );
91
92        // Single mint freeze_authority
93        let mint_freeze_authority = Pubkey::new_unique();
94        let update_default_account_state_ix = update_default_account_state(
95            &spl_token_2022::id(),
96            &mint_pubkey,
97            &mint_freeze_authority,
98            &[],
99            &AccountState::Initialized,
100        )
101        .unwrap();
102        let message = Message::new(&[update_default_account_state_ix], None);
103        let compiled_instruction = &message.instructions[0];
104        assert_eq!(
105            parse_token(
106                compiled_instruction,
107                &AccountKeys::new(&message.account_keys, None)
108            )
109            .unwrap(),
110            ParsedInstructionEnum {
111                instruction_type: "updateDefaultAccountState".to_string(),
112                info: json!({
113                    "mint": mint_pubkey.to_string(),
114                    "accountState": "initialized",
115                    "freezeAuthority": mint_freeze_authority.to_string(),
116                })
117            }
118        );
119
120        // Multisig mint freeze_authority
121        let multisig_pubkey = Pubkey::new_unique();
122        let multisig_signer0 = Pubkey::new_unique();
123        let multisig_signer1 = Pubkey::new_unique();
124        let update_default_account_state_ix = update_default_account_state(
125            &spl_token_2022::id(),
126            &mint_pubkey,
127            &multisig_pubkey,
128            &[&multisig_signer0, &multisig_signer1],
129            &AccountState::Initialized,
130        )
131        .unwrap();
132        let message = Message::new(&[update_default_account_state_ix], None);
133        let compiled_instruction = &message.instructions[0];
134        assert_eq!(
135            parse_token(
136                compiled_instruction,
137                &AccountKeys::new(&message.account_keys, None)
138            )
139            .unwrap(),
140            ParsedInstructionEnum {
141                instruction_type: "updateDefaultAccountState".to_string(),
142                info: json!({
143                    "mint": mint_pubkey.to_string(),
144                    "accountState": "initialized",
145                    "multisigFreezeAuthority": multisig_pubkey.to_string(),
146                    "signers": vec![
147                        multisig_signer0.to_string(),
148                        multisig_signer1.to_string(),
149                    ],
150                })
151            }
152        );
153    }
