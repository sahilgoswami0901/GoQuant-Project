12/// The primary type in this crate, most tasks require creating or using one.
13///
14/// To create a *new* [`Mnemonic`][Mnemonic] from a randomly generated key, call [`Mnemonic::new()`][Mnemonic::new()].
15///
16/// To get a [`Mnemonic`][Mnemonic] instance for an existing mnemonic phrase, including
17/// those generated by other software or hardware wallets, use [`Mnemonic::from_phrase()`][Mnemonic::from_phrase()].
18///
19/// You can get the HD wallet [`Seed`][Seed] from a [`Mnemonic`][Mnemonic] by calling [`Seed::new()`][Seed::new()].
20/// From there you can either get the raw byte value with [`Seed::as_bytes()`][Seed::as_bytes()], or the hex
21/// representation using Rust formatting: `format!("{:X}", seed)`.
22///
23/// You can also get the original entropy value back from a [`Mnemonic`][Mnemonic] with [`Mnemonic::entropy()`][Mnemonic::entropy()],
24/// but beware that the entropy value is **not the same thing** as an HD wallet seed, and should
25/// *never* be used that way.
26///
27/// [`Mnemonic`][Mnemonic] implements [`Zeroize`][Zeroize], so it's bytes will be zeroed when it's dropped.
28///
29/// [Mnemonic]: ./mnemonic/struct.Mnemonic.html
30/// [Mnemonic::new()]: ./mnemonic/struct.Mnemonic.html#method.new
31/// [Mnemonic::from_phrase()]: ./mnemonic/struct.Mnemonic.html#method.from_phrase
32/// [Mnemonic::entropy()]: ./mnemonic/struct.Mnemonic.html#method.entropy
33/// [Seed]: ./seed/struct.Seed.html
34/// [Seed::new()]: ./seed/struct.Seed.html#method.new
35/// [Seed::as_bytes()]: ./seed/struct.Seed.html#method.as_bytes
36///
37#[derive(Clone)]
38pub struct Mnemonic {
39    phrase: Zeroizing<String>,
40    lang: Language,
41    entropy: Zeroizing<Vec<u8>>,
42}

44impl Mnemonic {
45    /// Generates a new [`Mnemonic`][Mnemonic]
46    ///
47    /// Use [`Mnemonic::phrase()`][Mnemonic::phrase()] to get an `str` slice of the generated phrase.
48    ///
49    /// # Example
50    ///
51    /// ```
52    /// use bip39::{Mnemonic, MnemonicType, Language};
53    ///
54    /// let mnemonic = Mnemonic::new(MnemonicType::Words12, Language::English);
55    /// let phrase = mnemonic.phrase();
56    ///
57    /// println!("phrase: {}", phrase);
58    ///
59    /// assert_eq!(phrase.split(" ").count(), 12);
60    /// ```
61    ///
62    /// [Mnemonic]: ./mnemonic/struct.Mnemonic.html
63    /// [Mnemonic::phrase()]: ./mnemonic/struct.Mnemonic.html#method.phrase
64    pub fn new(mtype: MnemonicType, lang: Language) -> Mnemonic {
65        let entropy = gen_random_bytes(mtype.entropy_bits() / 8);
66
67        Mnemonic::from_entropy_unchecked(entropy, lang)
68    }

70    /// Create a [`Mnemonic`][Mnemonic] from pre-generated entropy
71    ///
72    /// # Example
73    ///
74    /// ```
75    /// use bip39::{Mnemonic, MnemonicType, Language};
76    ///
77    /// let entropy = &[0x33, 0xE4, 0x6B, 0xB1, 0x3A, 0x74, 0x6E, 0xA4, 0x1C, 0xDD, 0xE4, 0x5C, 0x90, 0x84, 0x6A, 0x79];
78    /// let mnemonic = Mnemonic::from_entropy(entropy, Language::English).unwrap();
79    ///
80    /// assert_eq!("crop cash unable insane eight faith inflict route frame loud box vibrant", mnemonic.phrase());
81    /// assert_eq!("33E46BB13A746EA41CDDE45C90846A79", format!("{:X}", mnemonic));
82    /// ```
83    ///
84    /// [Mnemonic]: ../mnemonic/struct.Mnemonic.html
85    pub fn from_entropy(entropy: &[u8], lang: Language) -> Result<Mnemonic, Error> {
86        // Validate entropy size
87        MnemonicType::for_key_size(entropy.len() * 8)?;
88
89        Ok(Self::from_entropy_unchecked(entropy, lang))
90    }
91
92    fn from_entropy_unchecked<E>(entropy: E, lang: Language) -> Mnemonic
93    where
94        E: Into<Vec<u8>>,
95    {
96        let entropy = Zeroizing::new(entropy.into());
97        let wordlist = lang.wordlist();
98
99        let checksum_byte = sha256_first_byte(&entropy);
100
101        // First, create a byte iterator for the given entropy and the first byte of the
102        // hash of the entropy that will serve as the checksum (up to 8 bits for biggest
103        // entropy source).
104        //
105        // Then we transform that into a bits iterator that returns 11 bits at a
106        // time (as u16), which we can map to the words on the `wordlist`.
107        //
108        // Given the entropy is of correct size, this ought to give us the correct word
109        // count.
110        let phrase = Zeroizing::new(entropy
111            .iter()
112            .chain(Some(&checksum_byte))
113            .bits()
114            .map(|bits| wordlist.get_word(bits))
115            .join(" "));
116
117        Mnemonic {
118            phrase,
119            lang,
120            entropy,
121        }
122    }

124    /// Create a [`Mnemonic`][Mnemonic] from an existing mnemonic phrase
125    ///
126    /// The phrase supplied will be checked for word length and validated according to the checksum
127    /// specified in BIP0039
128    ///
129    /// # Example
130    ///
131    /// ```
132    /// use bip39::{Mnemonic, Language};
133    ///
134    /// let phrase = "park remain person kitchen mule spell knee armed position rail grid ankle";
135    /// let mnemonic = Mnemonic::from_phrase(phrase, Language::English).unwrap();
136    ///
137    /// assert_eq!(phrase, mnemonic.phrase());
138    /// ```
139    ///
140    /// [Mnemonic]: ../mnemonic/struct.Mnemonic.html
141    pub fn from_phrase(phrase: &str, lang: Language) -> Result<Mnemonic, Error> {
142        let phrase = Zeroizing::new(phrase
143            .split_whitespace()
144            .map(|w| w.nfkd())
145            .join::<String>(" "));
146
147        // this also validates the checksum and phrase length before returning the entropy so we
148        // can store it. We don't use the validate function here to avoid having a public API that
149        // takes a phrase string and returns the entropy directly.
150        let entropy = Zeroizing::new(Mnemonic::phrase_to_entropy(&phrase, lang)?);
151
152        let mnemonic = Mnemonic {
153            phrase,
154            lang,
155            entropy,
156        };
157
158        Ok(mnemonic)
159    }
160
161    /// Validate a mnemonic phrase
162    ///
163    /// The phrase supplied will be checked for word length and validated according to the checksum
164    /// specified in BIP0039.
165    ///
166    /// # Example
167    ///
168    /// ```
169    /// use bip39::{Mnemonic, Language};
170    ///
171    /// let test_mnemonic = "park remain person kitchen mule spell knee armed position rail grid ankle";
172    ///
173    /// assert!(Mnemonic::validate(test_mnemonic, Language::English).is_ok());
174    /// ```
175    pub fn validate(phrase: &str, lang: Language) -> Result<(), Error> {
176        Mnemonic::phrase_to_entropy(phrase, lang)?;
177
178        Ok(())
179    }

181    /// Calculate the checksum, verify it and return the entropy
182    ///
183    /// Only intended for internal use, as returning a `Vec<u8>` that looks a bit like it could be
184    /// used as the seed is likely to cause problems for someone eventually. All the other functions
185    /// that return something like that are explicit about what it is and what to use it for.
186    fn phrase_to_entropy(phrase: &str, lang: Language) -> Result<Vec<u8>, Error> {
187        let wordmap = lang.wordmap();
188
189        // Preallocate enough space for the longest possible word list
190        let mut bits = BitWriter::with_capacity(264);
191
192        for word in phrase.split(" ") {
193            bits.push(wordmap.get_bits(&word)?);
194        }
195
196        let mtype = MnemonicType::for_word_count(bits.len() / 11)?;
197
198        debug_assert!(
199            bits.len() == mtype.total_bits(),
200            "Insufficient amount of bits to validate"
201        );
202
203        let mut entropy = bits.into_bytes();
204        let entropy_bytes = mtype.entropy_bits() / 8;
205
206        let actual_checksum = checksum(entropy[entropy_bytes], mtype.checksum_bits());
207
208        // Truncate to get rid of the byte containing the checksum
209        entropy.truncate(entropy_bytes);
210
211        let checksum_byte = sha256_first_byte(&entropy);
212        let expected_checksum = checksum(checksum_byte, mtype.checksum_bits());
213
214        if actual_checksum != expected_checksum {
215            Err(ErrorKind::InvalidChecksum)?;
216        }
217
218        Ok(entropy)
219    }

226    /// Consume the `Mnemonic` and return the phrase as a `String`.
227    pub fn into_phrase(mut self) -> String {
228        // Create an empty string and swap values with the mnemonic's phrase.
229        // This allows `Mnemonic` to implement `Drop`, while still returning the phrase.
230        mem::replace(&mut self.phrase, String::new())
231    }

280    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
281        if f.alternate() {
282            f.write_str("0x")?;
283        }
284
