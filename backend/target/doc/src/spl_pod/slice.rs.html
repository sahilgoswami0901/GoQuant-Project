15const LENGTH_SIZE: usize = std::mem::size_of::<PodU32>();
16/// Special type for using a slice of `Pod`s in a zero-copy way
17pub struct PodSlice<'data, T: Pod> {
18    length: &'data PodU32,
19    data: &'data [T],
20}
21impl<'data, T: Pod> PodSlice<'data, T> {
22    /// Unpack the buffer into a slice
23    pub fn unpack<'a>(data: &'a [u8]) -> Result<Self, ProgramError>
24    where
25        'a: 'data,
26    {
27        if data.len() < LENGTH_SIZE {
28            return Err(PodSliceError::BufferTooSmall.into());
29        }
30        let (length, data) = data.split_at(LENGTH_SIZE);
31        let length = pod_from_bytes::<PodU32>(length)?;
32        let _max_length = max_len_for_type::<T>(data.len())?;
33        let data = pod_slice_from_bytes(data)?;
34        Ok(Self { length, data })
35    }
36
37    /// Get the slice data
38    pub fn data(&self) -> &[T] {
39        let length = u32::from(*self.length) as usize;
40        &self.data[..length]
41    }
42
43    /// Get the amount of bytes used by `num_items`
44    pub fn size_of(num_items: usize) -> Result<usize, ProgramError> {
45        std::mem::size_of::<T>()
46            .checked_mul(num_items)
47            .and_then(|len| len.checked_add(LENGTH_SIZE))
48            .ok_or_else(|| PodSliceError::CalculationFailure.into())
49    }
50}

52/// Special type for using a slice of mutable `Pod`s in a zero-copy way
53pub struct PodSliceMut<'data, T: Pod> {
54    length: &'data mut PodU32,
55    data: &'data mut [T],
56    max_length: usize,
57}
58impl<'data, T: Pod> PodSliceMut<'data, T> {
59    /// Unpack the mutable buffer into a mutable slice, with the option to
60    /// initialize the data
61    fn unpack_internal<'a>(data: &'a mut [u8], init: bool) -> Result<Self, ProgramError>
62    where
63        'a: 'data,
64    {
65        if data.len() < LENGTH_SIZE {
66            return Err(PodSliceError::BufferTooSmall.into());
67        }
68        let (length, data) = data.split_at_mut(LENGTH_SIZE);
69        let length = pod_from_bytes_mut::<PodU32>(length)?;
70        if init {
71            *length = 0.into();
72        }
73        let max_length = max_len_for_type::<T>(data.len())?;
74        let data = pod_slice_from_bytes_mut(data)?;
75        Ok(Self {
76            length,
77            data,
78            max_length,
79        })
80    }
81
82    /// Unpack the mutable buffer into a mutable slice
83    pub fn unpack<'a>(data: &'a mut [u8]) -> Result<Self, ProgramError>
84    where
85        'a: 'data,
86    {
87        Self::unpack_internal(data, /* init */ false)
88    }
89
90    /// Unpack the mutable buffer into a mutable slice, and initialize the
91    /// slice to 0-length
92    pub fn init<'a>(data: &'a mut [u8]) -> Result<Self, ProgramError>
93    where
94        'a: 'data,
95    {
96        Self::unpack_internal(data, /* init */ true)
97    }
98
99    /// Add another item to the slice
100    pub fn push(&mut self, t: T) -> Result<(), ProgramError> {
101        let length = u32::from(*self.length);
102        if length as usize == self.max_length {
103            Err(PodSliceError::BufferTooSmall.into())
104        } else {
105            self.data[length as usize] = t;
106            *self.length = length.saturating_add(1).into();
107            Ok(())
108        }
109    }
110}
111
112fn max_len_for_type<T>(data_len: usize) -> Result<usize, ProgramError> {
113    let size: usize = std::mem::size_of::<T>();
114    let max_len = data_len
115        .checked_div(size)
116        .ok_or(PodSliceError::CalculationFailure)?;
117    // check that it isn't over or under allocated
118    if max_len.saturating_mul(size) != data_len {
119        if max_len == 0 {
120            // Size of T is greater than buffer size
121            Err(PodSliceError::BufferTooSmall.into())
122        } else {
123            Err(PodSliceError::BufferTooLarge.into())
124        }
125    } else {
126        Ok(max_len)
127    }
128}

130#[cfg(test)]
131mod tests {
132    use {super::*, crate::bytemuck::pod_slice_to_bytes, bytemuck::Zeroable};
133
134    #[repr(C)]
135    #[derive(Clone, Copy, Debug, Default, PartialEq, Pod, Zeroable)]
136    struct TestStruct {
137        test_field: u8,
138        test_pubkey: [u8; 32],
139    }
140
141    #[test]
142    fn test_pod_slice() {
143        let test_field_bytes = [0];
144        let test_pubkey_bytes = [1; 32];
145        let len_bytes = [2, 0, 0, 0];
146
147        // Slice will contain 2 `TestStruct`
148        let mut data_bytes = [0; 66];
149        data_bytes[0..1].copy_from_slice(&test_field_bytes);
150        data_bytes[1..33].copy_from_slice(&test_pubkey_bytes);
151        data_bytes[33..34].copy_from_slice(&test_field_bytes);
152        data_bytes[34..66].copy_from_slice(&test_pubkey_bytes);
153
154        let mut pod_slice_bytes = [0; 70];
155        pod_slice_bytes[0..4].copy_from_slice(&len_bytes);
156        pod_slice_bytes[4..70].copy_from_slice(&data_bytes);
157
158        let pod_slice = PodSlice::<TestStruct>::unpack(&pod_slice_bytes).unwrap();
159        let pod_slice_data = pod_slice.data();
160
161        assert_eq!(*pod_slice.length, PodU32::from(2));
162        assert_eq!(pod_slice_to_bytes(pod_slice.data()), data_bytes);
163        assert_eq!(pod_slice_data[0].test_field, test_field_bytes[0]);
164        assert_eq!(pod_slice_data[0].test_pubkey, test_pubkey_bytes);
165        assert_eq!(PodSlice::<TestStruct>::size_of(1).unwrap(), 37);
166    }
167
168    #[test]
169    fn test_pod_slice_buffer_too_large() {
170        // 1 `TestStruct` + length = 37 bytes
171        // we pass 38 to trigger BufferTooLarge
172        let pod_slice_bytes = [1; 38];
173        let err = PodSlice::<TestStruct>::unpack(&pod_slice_bytes)
174            .err()
175            .unwrap();
176        assert_eq!(
177            err,
178            PodSliceError::BufferTooLarge.into(),
179            "Expected an `PodSliceError::BufferTooLarge` error"
180        );
181    }
182
183    #[test]
184    fn test_pod_slice_buffer_too_small() {
185        // 1 `TestStruct` + length = 37 bytes
186        // we pass 36 to trigger BufferTooSmall
187        let pod_slice_bytes = [1; 36];
188        let err = PodSlice::<TestStruct>::unpack(&pod_slice_bytes)
189            .err()
190            .unwrap();
191        assert_eq!(
192            err,
193            PodSliceError::BufferTooSmall.into(),
194            "Expected an `PodSliceError::BufferTooSmall` error"
195        );
196    }
197
198    #[test]
199    fn test_pod_slice_mut() {
200        // slice can fit 2 `TestStruct`
201        let mut pod_slice_bytes = [0; 70];
202        // set length to 1, so we have room to push 1 more item
203        let len_bytes = [1, 0, 0, 0];
204        pod_slice_bytes[0..4].copy_from_slice(&len_bytes);
205
206        let mut pod_slice = PodSliceMut::<TestStruct>::unpack(&mut pod_slice_bytes).unwrap();
207
208        assert_eq!(*pod_slice.length, PodU32::from(1));
209        pod_slice.push(TestStruct::default()).unwrap();
210        assert_eq!(*pod_slice.length, PodU32::from(2));
211        let err = pod_slice
212            .push(TestStruct::default())
213            .expect_err("Expected an `PodSliceError::BufferTooSmall` error");
214        assert_eq!(err, PodSliceError::BufferTooSmall.into());
215    }
