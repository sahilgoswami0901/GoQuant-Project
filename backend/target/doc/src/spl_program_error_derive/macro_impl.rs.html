14const SPL_ERROR_HASH_NAMESPACE: &str = "spl_program_error";
15const SPL_ERROR_HASH_MIN_VALUE: u32 = 7_000;
16
17/// The type of macro being called, thus directing which tokens to generate
18#[allow(clippy::enum_variant_names)]
19pub enum MacroType {
20    IntoProgramError {
21        ident: Ident,
22    },
23    DecodeError {
24        ident: Ident,
25    },
26    PrintProgramError {
27        ident: Ident,
28        variants: Punctuated<Variant, Comma>,
29    },
30    SplProgramError {
31        args: SplProgramErrorArgs,
32        item_enum: ItemEnum,
33    },
34}
35
36impl MacroType {
37    /// Generates the corresponding tokens based on variant selection
38    pub fn generate_tokens(&mut self) -> proc_macro2::TokenStream {
39        match self {
40            Self::IntoProgramError { ident } => into_program_error(ident),
41            Self::DecodeError { ident } => decode_error(ident),
42            Self::PrintProgramError { ident, variants } => print_program_error(ident, variants),
43            Self::SplProgramError { args, item_enum } => spl_program_error(args, item_enum),
44        }
45    }
46}

48/// Builds the implementation of
49/// `Into<solana_program::program_error::ProgramError>` More specifically,
50/// implements `From<Self> for solana_program::program_error::ProgramError`
51pub fn into_program_error(ident: &Ident) -> proc_macro2::TokenStream {
52    quote! {
53        impl From<#ident> for solana_program::program_error::ProgramError {
54            fn from(e: #ident) -> Self {
55                solana_program::program_error::ProgramError::Custom(e as u32)
56            }
57        }
58    }
59}

61/// Builds the implementation of `solana_program::decode_error::DecodeError<T>`
62pub fn decode_error(ident: &Ident) -> proc_macro2::TokenStream {
63    quote! {
64        impl<T> solana_program::decode_error::DecodeError<T> for #ident {
65            fn type_of() -> &'static str {
66                stringify!(#ident)
67            }
68        }
69    }
70}

72/// Builds the implementation of
73/// `solana_program::program_error::PrintProgramError`
74pub fn print_program_error(
75    ident: &Ident,
76    variants: &Punctuated<Variant, Comma>,
77) -> proc_macro2::TokenStream {
78    let ppe_match_arms = variants.iter().map(|variant| {
79        let variant_ident = &variant.ident;
80        let error_msg = get_error_message(variant)
81            .unwrap_or_else(|| String::from("Unknown custom program error"));
82        quote! {
83            #ident::#variant_ident => {
84                solana_program::msg!(#error_msg)
85            }
86        }
87    });
88    quote! {
89        impl solana_program::program_error::PrintProgramError for #ident {
90            fn print<E>(&self)
91            where
92                E: 'static
93                    + std::error::Error
94                    + solana_program::decode_error::DecodeError<E>
95                    + solana_program::program_error::PrintProgramError
96                    + num_traits::FromPrimitive,
97            {
98                match self {
99                    #(#ppe_match_arms),*
100                }
101            }
102        }
103    }
104}

106/// Helper to parse out the string literal from the `#[error(..)]` attribute
107fn get_error_message(variant: &Variant) -> Option<String> {
108    let attrs = &variant.attrs;
109    for attr in attrs {
110        if attr.path().is_ident("error") {
111            if let Ok(lit_str) = attr.parse_args::<LitStr>() {
112                return Some(lit_str.value());
113            }
114        }
115    }
116    None
117}

119/// The main function that produces the tokens required to turn your
120/// error enum into a Solana Program Error
121pub fn spl_program_error(
122    args: &SplProgramErrorArgs,
123    item_enum: &mut ItemEnum,
124) -> proc_macro2::TokenStream {
125    if let Some(error_code_start) = args.hash_error_code_start {
126        set_first_discriminant(item_enum, error_code_start);
127    }
128
129    let ident = &item_enum.ident;
130    let variants = &item_enum.variants;
131    let into_program_error = into_program_error(ident);
132    let decode_error = decode_error(ident);
133    let print_program_error = print_program_error(ident, variants);
134
135    quote! {
136        #[repr(u32)]
137        #[derive(Clone, Debug, Eq, thiserror::Error, num_derive::FromPrimitive, PartialEq)]
138        #[num_traits = "num_traits"]
139        #item_enum
140
141        #into_program_error
142
143        #decode_error
144
145        #print_program_error
146    }
147}

149/// This function adds a discriminant to the first enum variant based on the
150/// hash of the `SPL_ERROR_HASH_NAMESPACE` constant, the enum name and variant
151/// name.
152/// It will then check to make sure the provided `hash_error_code_start` is
153/// equal to the hash-produced `u32`.
154///
155/// See https://docs.rs/syn/latest/syn/struct.Variant.html
156fn set_first_discriminant(item_enum: &mut ItemEnum, error_code_start: u32) {
157    let enum_ident = &item_enum.ident;
158    if item_enum.variants.is_empty() {
159        panic!("Enum must have at least one variant");
160    }
161    let first_variant = &mut item_enum.variants[0];
162    let discriminant = u32_from_hash(enum_ident);
163    if discriminant == error_code_start {
164        let eq = Token![=](Span::call_site());
165        let expr = Expr::Lit(ExprLit {
166            attrs: Vec::new(),
167            lit: Lit::Int(LitInt::new(&discriminant.to_string(), Span::call_site())),
168        });
169        first_variant.discriminant = Some((eq, expr));
170    } else {
171        panic!(
172            "Error code start value from hash must be {0}. Update your macro attribute to \
173             `#[spl_program_error(hash_error_code_start = {0})]`.",
174            discriminant
175        );
176    }
177}

179/// Hashes the `SPL_ERROR_HASH_NAMESPACE` constant, the enum name and variant
180/// name and returns four middle bytes (13 through 16) as a u32.
181fn u32_from_hash(enum_ident: &Ident) -> u32 {
182    let hash_input = format!("{}:{}", SPL_ERROR_HASH_NAMESPACE, enum_ident);
183
184    // We don't want our error code to start at any number below
185    // `SPL_ERROR_HASH_MIN_VALUE`!
186    let mut nonce: u32 = 0;
187    loop {
188        let mut hasher = Sha256::new_with_prefix(hash_input.as_bytes());
189        hasher.update(nonce.to_le_bytes());
190        let d = u32::from_le_bytes(
191            hasher.finalize()[13..17]
192                .try_into()
193                .expect("Unable to convert hash to u32"),
194        );
195        if d >= SPL_ERROR_HASH_MIN_VALUE {
196            return d;
197        }
198        nonce += 1;
199    }
