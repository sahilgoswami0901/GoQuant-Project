6/// A very generic, contextual pwrite interface in Rust.
7///
8/// Like [Pread](trait.Pread.html) — but for writing!
9///
10/// Implementing `Pwrite` on a data store allows you to then write almost arbitrarily complex types
11/// efficiently.
12///
13/// To this end the Pwrite trait works in conjuction with the [TryIntoCtx](ctx/trait.TryIntoCtx.html);
14/// The `TryIntoCtx` trait implemented on a type defines how to convert said type into data that
15/// an implementation of Pwrite can … well … write.
16///
17/// As with [Pread](trait.Pread.html) 'data' does not necessarily mean `&[u8]` but can be any
18/// indexable type. In fact much of the documentation of `Pread` applies to `Pwrite` as well just
19/// with 'read' switched for 'write' and 'From' switched with 'Into' so if you haven't yet you
20/// should read the documentation of `Pread` first.
21///
22/// Unless you need to implement your own data store — that is either can't convert to `&[u8]` or
23/// have a data that does not expose a `&mut [u8]` — you will probably want to implement
24/// [TryIntoCtx](ctx/trait.TryIntoCtx.html) on your Rust types to be written.
25///
26pub trait Pwrite<Ctx: Copy, E> {
27    #[inline]
28    fn pwrite<N: TryIntoCtx<Ctx, Self, Error = E>>(
29        &mut self,
30        n: N,
31        offset: usize,
32    ) -> result::Result<usize, E>
33    where
34        Ctx: Default,
35    {
36        self.pwrite_with(n, offset, Ctx::default())
37    }
38
39    /// Write `N` at offset `I` with context `Ctx`
40    /// # Example
41    /// ```
42    /// use scroll::{Pwrite, Pread, LE};
43    /// let mut bytes: [u8; 8] = [0, 0, 0, 0, 0, 0, 0, 0];
44    /// bytes.pwrite_with::<u32>(0xbeefbeef, 0, LE).unwrap();
45    /// assert_eq!(bytes.pread_with::<u32>(0, LE).unwrap(), 0xbeefbeef);
46    fn pwrite_with<N: TryIntoCtx<Ctx, Self, Error = E>>(
47        &mut self,
48        n: N,
49        offset: usize,
50        ctx: Ctx,
51    ) -> result::Result<usize, E>;
52
53    /// Write `n` into `self` at `offset`, with a default `Ctx`. Updates the offset.
54    #[inline]
55    fn gwrite<N: TryIntoCtx<Ctx, Self, Error = E>>(
56        &mut self,
57        n: N,
58        offset: &mut usize,
59    ) -> result::Result<usize, E>
60    where
61        Ctx: Default,
62    {
63        let ctx = Ctx::default();
64        self.gwrite_with(n, offset, ctx)
65    }
66
67    /// Write `n` into `self` at `offset`, with the `ctx`. Updates the offset.
68    #[inline]
69    fn gwrite_with<N: TryIntoCtx<Ctx, Self, Error = E>>(
70        &mut self,
71        n: N,
72        offset: &mut usize,
73        ctx: Ctx,
74    ) -> result::Result<usize, E> {
75        let o = *offset;
76        self.pwrite_with(n, o, ctx).map(|size| {
77            *offset += size;
78            size
79        })
80    }
81}
82
83impl<Ctx: Copy, E: From<error::Error>> Pwrite<Ctx, E> for [u8] {
84    fn pwrite_with<N: TryIntoCtx<Ctx, Self, Error = E>>(
85        &mut self,
86        n: N,
87        offset: usize,
88        ctx: Ctx,
89    ) -> result::Result<usize, E> {
90        if offset >= self.len() {
91            return Err(error::Error::BadOffset(offset).into());
92        }
93        let dst = &mut self[offset..];
94        n.try_into_ctx(dst, ctx)
95    }
96}
