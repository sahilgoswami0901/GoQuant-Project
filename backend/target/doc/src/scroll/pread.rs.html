6/// A very generic, contextual pread interface in Rust.
7///
8/// Like [Pwrite](trait.Pwrite.html) — but for reading!
9///
10/// Implementing `Pread` on a data store allows you to then read almost arbitrarily complex types
11/// efficiently.
12///
13/// To this end the Pread trait works in conjuction with the [TryFromCtx](ctx/trait.TryFromCtx.html);
14/// The `TryFromCtx` trait implemented on a type defines how to convert data to an object of that
15/// type, the Pread trait implemented on a data store defines how to extract said data from that
16/// store.
17///
18/// It should be noted though that in this context, data does not necessarily mean `&[u8]` —
19/// `Pread` and `TryFromCtx` are generic over what 'data' means and could be implemented instead
20/// over chunks of memory or any other indexable type — but scroll does come with a set of powerful
21/// blanket implementations for data being a continous block of byte-addressable memory.
22///
23/// Pread provides two main groups of functions: pread and gread.
24///
25/// `pread` is the basic function that simply extracts a given type from a given data store - either
26/// using a provided Context in the case of [pread_with](trait.Pread.html#method.pread_with) or
27/// with the default context for the given type in the case of [pread](trait.Pread.html#method.pread)
28///
29/// `gread` does in addition to that update the offset it's currently at, allowing for a cursored
30/// read — `gread_inout` expands on that and reads a number of continous types from the data store.
31/// gread again comes with `_with` variants to allow using a specific context.
32///
33/// Since pread and friends are very generic functions their types are rather complex, but very
34/// much understandable; `TryFromCtx` is generic over `Ctx` ([described
35/// here](ctx/index.html#context)), `Output` and `Error`. The Error type is hopefully
36/// self-explanatory, however the `Output` type is rather important; it defines what Pread extracts
37/// from the data store and has to match up with what `TryFromCtx` expects as input to convert into
38/// the resulting type. scroll defaults to `&[u8]` here.
39///
40/// Unless you need to implement your own data store — that is either can't convert to `&[u8]` or
41/// have a data that does not expose a `&[u8]` — you will probably want to implement
42/// [TryFromCtx](ctx/trait.TryFromCtx.html) on your Rust types to be extracted.
43///
44pub trait Pread<Ctx: Copy, E> {
45    #[inline]
46    /// Reads a value from `self` at `offset` with a default `Ctx`. For the primitive numeric values, this will read at the machine's endianness.
47    /// # Example
48    /// ```rust
49    /// use scroll::Pread;
50    /// let bytes = [0x7fu8; 0x01];
51    /// let byte = bytes.pread::<u8>(0).unwrap();
52    fn pread<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
53        &'a self,
54        offset: usize,
55    ) -> result::Result<N, E>
56    where
57        Ctx: Default,
58    {
59        self.pread_with(offset, Ctx::default())
60    }
61
62    #[inline]
63    /// Reads a value from `self` at `offset` with the given `ctx`
64    /// # Example
65    /// ```rust
66    /// use scroll::Pread;
67    /// let bytes: [u8; 2] = [0xde, 0xad];
68    /// let dead: u16 = bytes.pread_with(0, scroll::BE).unwrap();
69    /// assert_eq!(dead, 0xdeadu16);
70    fn pread_with<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
71        &'a self,
72        offset: usize,
73        ctx: Ctx,
74    ) -> result::Result<N, E> {
75        let mut ignored = offset;
76        self.gread_with(&mut ignored, ctx)
77    }
78
79    #[inline]
80    /// Reads a value from `self` at `offset` with a default `Ctx`. For the primitive numeric values, this will read at the machine's endianness. Updates the offset
81    /// # Example
82    /// ```rust
83    /// use scroll::Pread;
84    /// let offset = &mut 0;
85    /// let bytes = [0x7fu8; 0x01];
86    /// let byte = bytes.gread::<u8>(offset).unwrap();
87    /// assert_eq!(*offset, 1);
88    fn gread<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
89        &'a self,
90        offset: &mut usize,
91    ) -> result::Result<N, E>
92    where
93        Ctx: Default,
94    {
95        let ctx = Ctx::default();
96        self.gread_with(offset, ctx)
97    }
98
99    /// Reads a value from `self` at `offset` with the given `ctx`, and updates the offset.
100    /// # Example
101    /// ```rust
102    /// use scroll::Pread;
103    /// let offset = &mut 0;
104    /// let bytes: [u8; 2] = [0xde, 0xad];
105    /// let dead: u16 = bytes.gread_with(offset, scroll::BE).unwrap();
106    /// assert_eq!(dead, 0xdeadu16);
107    /// assert_eq!(*offset, 2);
108    fn gread_with<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
109        &'a self,
110        offset: &mut usize,
111        ctx: Ctx,
112    ) -> result::Result<N, E>;
113
114    /// Tries to write `inout.len()` `N`s into `inout` from `Self` starting at `offset`, using the default context for `N`, and updates the offset.
115    /// # Example
116    /// ```rust
117    /// use scroll::Pread;
118    /// let mut bytes: Vec<u8> = vec![0, 0];
119    /// let offset = &mut 0;
120    /// let bytes_from: [u8; 2] = [0x48, 0x49];
121    /// bytes_from.gread_inout(offset, &mut bytes).unwrap();
122    /// assert_eq!(&bytes, &bytes_from);
123    /// assert_eq!(*offset, 2);
124    #[inline]
125    fn gread_inout<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
126        &'a self,
127        offset: &mut usize,
128        inout: &mut [N],
129    ) -> result::Result<(), E>
130    where
131        Ctx: Default,
132    {
133        for i in inout.iter_mut() {
134            *i = self.gread(offset)?;
135        }
136        Ok(())
137    }
138
139    /// Tries to write `inout.len()` `N`s into `inout` from `Self` starting at `offset`, using the context `ctx`
140    /// # Example
141    /// ```rust
142    /// use scroll::{ctx, LE, Pread};
143    /// let mut bytes: Vec<u8> = vec![0, 0];
144    /// let offset = &mut 0;
145    /// let bytes_from: [u8; 2] = [0x48, 0x49];
146    /// bytes_from.gread_inout_with(offset, &mut bytes, LE).unwrap();
147    /// assert_eq!(&bytes, &bytes_from);
148    /// assert_eq!(*offset, 2);
149    #[inline]
150    fn gread_inout_with<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
151        &'a self,
152        offset: &mut usize,
153        inout: &mut [N],
154        ctx: Ctx,
155    ) -> result::Result<(), E> {
156        for i in inout.iter_mut() {
157            *i = self.gread_with(offset, ctx)?;
158        }
159        Ok(())
160    }
161}

163impl<Ctx: Copy, E: From<error::Error>> Pread<Ctx, E> for [u8] {
164    fn gread_with<'a, N: TryFromCtx<'a, Ctx, Self, Error = E>>(
165        &'a self,
166        offset: &mut usize,
167        ctx: Ctx,
168    ) -> result::Result<N, E> {
169        let start = *offset;
170        if start >= self.len() {
171            return Err(error::Error::BadOffset(start).into());
172        }
173        N::try_from_ctx(&self[start..], ctx).map(|(n, size)| {
174            *offset += size;
175            n
176        })
177    }
178}
