1///
2/// Emulates eager expansion of macros.
3///
4/// # Example
5/// ```
6/// #[macro_use]
7/// extern crate eager;
8///
9/// //Declare an eager macro
10/// eager_macro_rules!{ $eager_1
11///     macro_rules! plus_1{
12///         ()=>{+ 1};
13///     }
14/// }
15///
16/// fn main(){
17/// 	// Use the macro inside an eager! call to expand it eagerly
18/// 	assert_eq!(4, eager!{2 plus_1!() plus_1!()});
19/// }
20/// ```
21///
22/// # Usage
23///
24/// `eager!` can wrap any code, and if that code contains a macro call, that macro will be
25/// expanded before its consumer. This means:
26///
27/// * If a macro call is given as an argument to another macro, the first macro will be expanded
28/// first.
29/// * All macros will be fully expanded before `eager!` expands. Therefore, otherwise illegal
30/// intermediate expansion steps are possible.
31///
32/// `eager!` does not work with any macro; only macros declared using [`eager_macro_rules!`] may be
33/// used. Such macros are said to be `eager!`-enabled.
34///
35/// To enable the use of non-`eager!`-enabled macros inside an `eager!` call,
36/// a `lazy!` block can be inserted. Everything inside the `lazy!` block will be lazily expanded,
37/// while everything outside it will continue to be eagerly expanded. Since, `lazy!` reverts
38/// to the usual rules for macro expansion, an `eager!` block can be inserted inside the `lazy!`
39/// block, to re-enable eager expansion for some subset of it.
40///
41/// [`eager_macro_rules!`]: macro.eager_macro_rules.html
42/// [`lazy!`]: macro.lazy.html
43/// # Cons
44///
45/// * Because of the way `eager!` is implemented - being a hack of recursive macros - the compiler's
46/// default macro recursion limit is quickly exceeded. Therefore, `#![recursion_limit="256"]`
47/// must be used in most situations - potentially with a higher limit -
48/// such that expansion can happen.
49///
50/// * Debugging an eagerly expanded macro is very difficult and requires intimate knowledge
51/// of the implementation of `eager!`. There is no way to mitigate this, except to try and
52/// recreate the bug without using `eager!`. Likewise, the error messages the compiler will
53/// emit are exponentially more cryptic than they already would have been.
54///
55/// * Only `eager!`-enabled macros can be eagerly expanded, so existing macros do not gain much.
56/// The `lazy!` block alleviates this a bit, by allowing the use of existing macros in it,
57/// while eager expansion can be done around them.
58/// Luckily, `eager!`-enabling an existing macro should not be too much
59/// trouble using [`eager_macro_rules!`].
60///
61/// ---
62/// # Macro expansions
63///
64/// Rust is lazy when it comes to macro expansion. When the compiler sees a macro call, it will
65/// try to expand the macro without looking at its arguments or what the expansion becomes.
66/// Using `eager!`, previously illegal macro expansions can be made possible.
67///
68/// The following is a non-exhaustive list of illegal macro patterns that can be used with `eager!`.
69///
70/// ### The arguments to a macro usually cannot be the resulting expansion of another macro call:
71/// Say you have a macro that adds two numbers:
72/// ```ignore
73/// macro_rules! add{
74///     ($e1:expr, $e2:expr)=> {$e1 + $e2}
75/// }
76/// ```
77/// And a macro that expands to two comma-separated numbers:
78///
79/// ```ignore
80/// macro_rules! two_and_three{
81///     ()=>{2,3}
82/// }
83/// ```
84///
85/// You cannot use the expansion of `two_and_three!` as an argument to `add!`:
86/// ```ignore
87/// let x = add!(two_and_three!()); // error
88/// ```
89/// The compiler will complain about no rule in `add!` accepting `two_and_three`, since it does not
90/// get expanded before the `add!`, who requires two expressions and not just one.
91///
92/// With eager expansion, this can be made possible:
93/// ```
94/// #[macro_use]
95/// extern crate eager;
96///
97/// eager_macro_rules!{ $eager_1
98///     macro_rules! add{
99///         ($e1:expr, $e2:expr)=> {$e1 + $e2}
100///     }
101///
102///     macro_rules! two_and_three{
103///     	()=>{2,3}
104///     }
105/// }
106///
107/// fn main(){
108/// 	let x = eager!{add!(two_and_three!())};
109/// 	assert_eq!(5, x);
110/// }
111/// ```
112///
113/// ### Macros are illegal in some contexts (e.g. as an identifier)
114///
115/// Say you have a macro that expands to an identifier:
116/// ```ignore
117/// macro_rules! id{
118///     ()=> {SomeStruct}
119/// }
120/// ```
121/// And want to use it to declare a struct:
122/// ```ignore
123/// struct id!(){
124///     v: u32
125/// }
126/// ```
127///
128/// This will not compile since macros are illegal in identifier position. The compiler does
129/// not check whether the expansion of the macro will result in valid Rust code.
130///
131/// With eager expansion, `id!` will expand before the `eager!` block , making it possible to use it
132/// in an identifier position:
133/// ```
134/// #[macro_use]
135/// extern crate eager;
136///
137/// eager_macro_rules!{ $eager_1
138///     macro_rules! id{
139///         ()=> {SomeStruct}
140/// 	}
141/// }
142///
143/// eager!{
144///     struct id!(){
145///         v: u32
146///     }
147/// }
148///
149/// fn main(){
150/// 	let some_struct = SomeStruct{v: 4};
151///     assert_eq!(4, some_struct.v);
152/// }
153/// ```
154/// To circumvent any restriction on where macros can be used, we can therefore just wrap
155/// the code surrounding the macro call with `eager!`. The `eager!` must still be in a valid position,
156/// but in the worst case it can be put around the whole item
157/// (struct, trait, implement, function, etc.).
158///
159///
160/// ### No intermediate expansion step can include invalid syntax
161///
162/// Say we want to create a macro that interprets natural language, converting it into an expression.
163///
164/// We start by declaring a macro that interprets operator words:
165/// ```ignore
166/// macro_rules! op{
167///     ( plus ) => { + };
168///     ( minus ) => { - };
169/// }
170/// ```
171///
172/// We then declare a macro that interprets integer words:
173/// ```ignore
174/// macro_rules! integer{
175///     ( one ) => { 1 };
176///     ( two ) => { 2 };
177/// }
178/// ```
179///
180/// Lastly, we declare the top-level macro that uses the previous two macros to
181/// expand into an expression:
182/// ```ignore
183/// macro_rules! calculate{
184///     ( $lhs:tt $op:tt $rhs:tt ) => {
185///          integer!{$lhs} op!{$op} integer!{$rhs}
186///     };
187/// }
188/// ```
189///
190/// Using this macro will fail to compile:
191/// ```ignore
192/// let x = calculate!(one plus two); //Error
193/// ```
194///
195/// Looking at the first expansion step we can see that three macro calls in a sequence
196/// are not a valid expression:
197/// ```ignore
198/// let x = integer!(one) op!{plus} integer!(two); //Error
199/// ```
200///
201/// We can circumvent this restriction, by having `calculate!` wrap its output in an `eager!`:
202///
203/// ```
204/// #[macro_use]
205/// extern crate eager;
206///
207/// eager_macro_rules!{ $eager_1
208///     macro_rules! op{
209///         ( plus ) => { + };
210///         ( minus ) => { - };
211///     }
212///
213///     macro_rules! integer{
214///         ( one ) => { 1 };
215///         ( two ) => { 2 };
216///     }
217///
218///     macro_rules! calculate{
219///         ( $lhs:tt $op:tt $rhs:tt ) => {
220///              eager!{integer!{$lhs} op!{$op} integer!{$rhs}}
221///         };
222/// 	}
223/// }
224///
225/// fn main(){
226/// 	let x = calculate!(one plus two);
227/// 	assert_eq!(3, x);
228/// }
229/// ```
230/// In this case, `calculate!` does not actually have to be `eager!`-enabled, since it is not inserted
231/// into an `eager!` block. Though - as per [the conventions](#conventions) - we do enable it such
232/// that others may later use it inside an `eager!` block.
233///
234///
235/// # Conventions
236///
237/// Since we expect the use of this macro to be broadly applicable, we propose the following
238/// conventions for the Rust community to use, to ease interoperability.
239///
240/// ### Documentation
241///
242/// To make it clearly visible that a given macro is `eager!`-enabled, its short rustdoc description
243/// must start with a pair of brackets, within which a link to the official `eager!` macro documentation
244/// must be provided. The link's visible text must be 'eager!' and
245/// the brackets must not be part of the link.
246///
247/// ### Auxiliary variable
248///
249/// The auxiliary variable that must always be provided to `eager_macro_rules!`
250/// must use the identifier `eager_1`. This makes it easier for everyone to
251/// get used to its presence and ignore it. By having it be the same in every project,
252/// no one has to think about why a given project uses some specific identifier.
253///
254/// # Trivia
255///
256/// * Ironically, `eager!` is not technically `eager!`-enabled. Instead, it ignores itself if
257/// it is nested or a macro expands into an `eager!` block.
258/// Likewise, `eager_macro_rules!` is not `eager!`-enabled, though this might be possible.
259///
260/// * `lazy!` is treated by `eager!` as a keyword and not a macro.
261///
262/// * `eager_macro_rules!`'s auxiliary variable is affectionately called `Simon`.
263/// This nickname should probably not be used as the identifier in production code.
264/// Before reaching production, though...
265///
266/// * Simon once had a brother called `Garkel`.
267///
268/// * It requires continuous effort from [Emoun](http://github.com/Emoun) to not
269/// forcibly rename `eager_macro_rules!` to `eager_macros_rule`.
270///
271///
272#[macro_export]
273macro_rules! eager{
274	(
275		$($all:tt)*
276	)=>{
277		eager_internal!{
278			@check_expansion[
279				[[][][][]]
280			]
281			$($all)*
282		}
283	};
284}

286/*
287Decoded format:
288[ [] [] [] [] {} ]
289  1  2  3  4  5
290
2911. The mode, either `[]` for eager or `[@lazy]` for lazy. Specifies whether the
292current decode mode is eager or lazy. If there is more input, then that input must be
293decoded in the mode. If there is no input, then the modefix must be decoded in the opposite mode.
294
2952. The modefix (mode postfix). Contains input that has yet to be decoded, and that would have
296to be decoded in the opposite mode to what is in 1.
297
2983. The block prefix, i.e. input that came before the current block (if there is one).
299Contains input that has been decoded and expanded completely. It is in
300reverse order that what it should be in the final result.
301
3024. The block postfix, i.e. input that came after the current block. If there is no
303block, then this must be empty.
304
3055. The current block, optional, and can be either `{}`, `[]`, or `()` which specifies the type
306of the block in the input. While the contents of the block are being decoded, it is empty.
307When the content has been decoded, checked, and expanded where appropriate it is input into
308the block.
309
310## Decoding workflow
311
312The decoding starts with an empty level that is by default in eager mode: `[[] [] [] []]`.
313
314Any input token that is not a block needs no work, therefore it is immediately put
315in the prefix as it can be output as is. We call these tokens simple tokens.
316We always token munch, which means the prfix is always in reverse order.
317We will unreverse it in the end using the `reverse!` macro.
318So if the input starts with the simple tokens `1 2 3`, the level will look like this:
319`[[] [] [3 2 1] []]`.
320
321Say the rest of the input is `{4 5 6} 7 8`. The block could contain a macro call that needs
322to be eagerly expanded, so we cannot just add the block to the prefix yet. To check the contents
323of the block we add the block to the current level, add the rest of the input (after the block)
324to the postfix, and make the contents of the block as our input. We also add a new level, which
325we will use to decode the blocks content.
326The levels are a stack, always using the top to decode, and putting new one on the top.
327The result is our decoded input becomes:
328```
329[[] [] [] []] // Used to decode the block contents
330[[] [] [3 2 1] [7 8] {}]
331```
332And our input is `4 5 6`. We can now ignore the second level (the original level) and trivially
333decode the simple tokens that were in the block:
334```
335[[] [] [6 5 4] []]
336[[] [] [3 2 1] [7 8] {}]
337```
338At this point we have no more input, which means the content of the block have been decoded
339and checked for any expansion needs. To signal that the block is done, we promote the prefix
340to the block in the second level, and pop the first level. Since the prefix is in reverse order,
341we unreverse it when putting it in the block. This is done with token munching:
342```
343[[] [] [3 2 1] [7 8] {4 5 6}]
344```
345Since we still have no more input to decode, but we can see that we have checked a block, we
346promote the block to prefix. At the same time, we take the blocks postfix and put it as input,
347such that it can be decoded:
348```
349[[] [] [{4 5 6} 3 2 1] []]
350```
351The block contents do not need to be reversed since the prefix is only reversed on the token tree
352level. Since the input after the block was just simple tokens we get:
353```
354[[] [] [8 7 {4 5 6} 3 2 1] []]
355```
356Now that we have no more input, no block, and no postfix, we know we have decoded everything
357and the contents of the prefix are our result. so we output the prefix in reverse order:
358`1 2 3 4 5 6 7 8`.
359
360To see how we handle macros, say our input has a macro invocation instead of the blocks:
361`1 2 some_macro!{t1 t2} 5 6`. We start, as usual, by decoding the first simple tokens.
362```
363[[] [] [! some_macro 2 1] []]
364```
365Note how the macro invocation is also put in the prefix. We then decode the block, checking its
366contents. When they have been checked and promoted into the block we will have:
367```
368[[] [] [! some_macro 2 1] [5 6]{t1 t2}]
369```
370At this point we would previously have promoted the block to the prefix, but we can now see
371that the prefix contains a macro invocation. Since we have checked the contents of the block,
372we know that we can safely call the macro with it. We do so, removing the invocation from the
373prefix, and removing the block. When the macro returns it will put its result as input, so the
374first thing we do is extract the postfix to the input too, putting it after the macros result,
375where it belongs. Our level will no look like:
376```
377[[] [] [2 1] []]
378```
379And say the macro expands to `3 4`, we will have the input `3 4 5 6`.
380Using our previous rules, the result will be `1 2 3 4 5 6`.
381
382Say we have a lazy block: `eager_macro_1!{} lazy!{ lazy_macro!{}} eager_macro_2!{}`
383Say `eager_macro_1!` expands to `1 2`, and `eager_macro_2!` expands to `3 4`, and both are
384`eager!`-enabled. `lazy_macro!` on the other hand is not `eager!`enabled.
385
386We startin the usual way, and after expanding the first macro we will have the levels:
387```
388[[] [] [2 1] []]
389```
390and the input `lazy!{ lazy_macro!{}} eager_macro_2!{}`. We can now see that the `lazy!` block
391is the opposite of the current mode (eager), so we will have to do a mode change. We take all
392input that is after the `lazy!` block, and put it is modefix as it still needs to be eagerly
393expanded. Then we change the mode to lazy. Lastly, we extract the content of the `lazy!` block
394and put it as input. This will result in the levels:
395```
396[[@lazy] [eager_macro_2!{}] [2 1] []]
397```
398and the input `lazy_macro!{}`. At this point we continue as we would have previously done,
399resulting in the block being checked (trivial since its empty) and no more input:
400```
401[[@lazy] [eager_macro_2!{}] [! lazy_macro 2 1] []{}]
402```
403Previously, we would have invoked `lazy_macro!`, since its block has been checked, but this time
404we are in lazy expansion mode. Therefore, there is no need to call the macro and we just promote
405the block to prefix immediately:
406```
407[[@lazy] [eager_macro_2!{}] [{}! lazy_macro 2 1] []]
408```
409At this point we have no more input and no block, but we still have something in the modefix.
410So, we extract that into input, and switch the mode, since the input in modefix always needs to
411be decoded in the opposite mode. So we get the levels:
412```
413[[] [] [{}! lazy_macro 2 1] []]
414```
415and input `eager_macro_2!{}`. The decoding now proceeds as previously described, resulting in
416`1 2 lazy_macro!{} 3 4` as output.
417
418Notes:
419
420* When decoding blocks, managing the type of the block is critical. This is the reason the block
421is not always `[]`. Therefore, when promoting blocks to prefix, make sure the block type is maintained.
422The following are examples of the same promotion, except the input uses the different blocks:
423`[[] [] [] [] {something}]` to `[[] [] [{something}] []]`.
424`[[] [] [] [] (something)]` to `[[] [] [(something)] []]`.
425`[[] [] [] [] [something]]` to `[[] [] [[something]] []]`.
426
427* Promoting modefix to input (the last step above) must only be done after all other input and
428blocks have been decoded fully.
429*/

432macro_rules! eager_internal{
433// Handle return from eager macro expansion
434	(
435		@from_macro[
436			[$lazy:tt $modefix:tt $prefix:tt[$($postfix:tt)*]]
437			$($rest_decoded:tt)*
438		]
439		$($expanded:tt)*
440	) => {
441		eager_internal!{
442			@check_expansion[
443				[$lazy $modefix $prefix []]
444				$($rest_decoded)*
445			]
446			$($expanded)* $($postfix)*
447		}
448	};
449// Decode input stream
450	(	// If the next token is a block, check it (brace type)
451		@check_expansion[
452			[$lazy:tt $modefix:tt [$($prefix:tt)*][]]
453			$($rest_decoded:tt)*
454		]
455		{$($body:tt)*} $($rest:tt)*
456	)=>{
457		eager_internal!{
458			@check_expansion[
459				[$lazy [][][]]
460				[$lazy $modefix [$($prefix)*][$($rest)*]{}]
461				$($rest_decoded)*
462			]
463			$($body)*
464		}
465	};
466	(	// If the next token is a block, check it (parenthesis type)
467		@check_expansion[
468			[$lazy:tt $modefix:tt [$($prefix:tt)*][]]
469			$($rest_decoded:tt)*
470		]
471		($($body:tt)*) $($rest:tt)*
472	)=>{
473		eager_internal!{
474			@check_expansion[
475				[$lazy [][][]]
476				[$lazy $modefix [$($prefix)*][$($rest)*]()]
477				$($rest_decoded)*
478			]
479			$($body)*
480		}
481	};
482	(	// If the next token is a block, check it (bracket type)
483		@check_expansion[
484			[$lazy:tt $modefix:tt [$($prefix:tt)*][]]
485			$($rest_decoded:tt)*
486		]
487		[$($body:tt)*] $($rest:tt)*
488	)=>{
489		eager_internal!{
490			@check_expansion[
491				[$lazy [][][]]
492				[$lazy $modefix [$($prefix)*][$($rest)*][]]
493				$($rest_decoded)*
494			]
495			$($body)*
496		}
497	};
498// eager/lazy mode changes
499	(	// If the next token is an 'eager!' macro call and we are already
500		// in eager mode, ignore it, extracting the body. (brace type)
501		@check_expansion[
502			[[]$modefix:tt[$($prefix:tt)*][]]
503			$($rest_decoded:tt)*
504		]
505		eager!{$($body:tt)*} $($rest:tt)*
506	)=>{
507		eager_internal!{
508			@check_expansion[
509				[[]$modefix[$($prefix)*][]]
510				$($rest_decoded)*
511			]
512			$($body)* $($rest)*
513		}
514	};
515	(	// If the next token is an 'eager!' macro call and we are already
516		// in eager mode, ignore it, extracting the body. (parenthesis type)
517		@check_expansion[
518			[[]$modefix:tt[$($prefix:tt)*][]]
519			$($rest_decoded:tt)*
520		]
521		eager!($($body:tt)*) $($rest:tt)*
522	)=>{
523		eager_internal!{
524			@check_expansion[
525				[[]$modefix[$($prefix)*][]]
526				$($rest_decoded)*
527			]
528			$($body)* $($rest)*
529		}
530	};
531	(	// If the next token is an 'eager!' macro call and we are already
532		// in eager mode, ignore it, extracting the body. (bracket type)
533		@check_expansion[
534			[[]$modefix:tt[$($prefix:tt)*][]]
535			$($rest_decoded:tt)*
536		]
537		eager![$($body:tt)*] $($rest:tt)*
538	)=>{
539		eager_internal!{
540			@check_expansion[
541				[[]$modefix[$($prefix)*][]]
542				$($rest_decoded)*
543			]
544			$($body)* $($rest)*
545		}
546	};
547	(	// If the next token is an 'lazy!' macro call and we are already
548		// in lazy mode, ignore it, extracting the body. (brace type)
549		@check_expansion[
550			[[@lazy]$modefix:tt[$($prefix:tt)*][]]
551			$($rest_decoded:tt)*
552		]
553		lazy!{$($body:tt)*} $($rest:tt)*
554	)=>{
555		eager_internal!{
556			@check_expansion[
557				[[@lazy]$modefix[$($prefix)*][]]
558				$($rest_decoded)*
559			]
560			$($body)* $($rest)*
561		}
562	};
563	(	// If the next token is an 'lazy!' macro call and we are already
564		// in lazy mode, ignore it, extracting the body. (parenthesis type)
565		@check_expansion[
566			[[@lazy]$modefix:tt[$($prefix:tt)*][]]
567			$($rest_decoded:tt)*
568		]
569		lazy!($($body:tt)*) $($rest:tt)*
570	)=>{
571		eager_internal!{
572			@check_expansion[
573				[[@lazy]$modefix[$($prefix)*][]]
574				$($rest_decoded)*
575			]
576			$($body)* $($rest)*
577		}
578	};
579	(	// If the next token is an 'lazy!' macro call and we are already
580		// in lazy mode, ignore it, extracting the body. (bracket type)
581		@check_expansion[
582			[[@lazy]$modefix:tt[$($prefix:tt)*][]]
583			$($rest_decoded:tt)*
584		]
585		lazy![$($body:tt)*] $($rest:tt)*
586	)=>{
587		eager_internal!{
588			@check_expansion[
589				[[@lazy]$modefix[$($prefix)*][]]
590				$($rest_decoded)*
591			]
592			$($body)* $($rest)*
593		}
594	};
595	(	// If the next token is an 'eager!' macro call and we are
596		// in lazy mode (brace type)
597		@check_expansion[
598			[[@lazy][][$($prefix:tt)*][]]
599			$($rest_decoded:tt)*
600		]
601		eager!{$($body:tt)*} $($rest:tt)*
602	)=>{
603		eager_internal!{
604			@check_expansion[
605				[[][$($rest)*][$($prefix)*][]]
606				$($rest_decoded)*
607			]
608			$($body)*
609		}
610	};
611	(	// If the next token is an 'eager!' macro call and we are
612		// in lazy mode, ignore it, extracting the body. (parenthesis type)
613		@check_expansion[
614			[[@lazy][][$($prefix:tt)*][]]
615			$($rest_decoded:tt)*
616		]
617		eager!($($body:tt)*) $($rest:tt)*
618	)=>{
619		eager_internal!{
620			@check_expansion[
621				[[][$($rest)*][$($prefix)*][]]
622				$($rest_decoded)*
623			]
624			$($body)*
625		}
626	};
627	(	// If the next token is an 'eager!' macro call and we are
628		// in lazy mode, ignore it, extracting the body. (bracket type)
629		@check_expansion[
630			[[@lazy][][$($prefix:tt)*][]]
631			$($rest_decoded:tt)*
632		]
633		eager![$($body:tt)*] $($rest:tt)*
634	)=>{
635		eager_internal!{
636			@check_expansion[
637				[[][$($rest)*][$($prefix)*][]]
638				$($rest_decoded)*
639			]
640			$($body)*
641		}
642	};
643	(	// If the next token is an 'lazy!' macro call and we are
644		// in eager mode, ignore it, extracting the body. (brace type)
645		@check_expansion[
646			[[][][$($prefix:tt)*][]]
647			$($rest_decoded:tt)*
648		]
649		lazy!{$($body:tt)*} $($rest:tt)*
650	)=>{
651		eager_internal!{
652			@check_expansion[
653				[[@lazy][$($rest)*][$($prefix)*][]]
654				$($rest_decoded)*
655			]
656			$($body)*
657		}
658	};
659	(	// If the next token is an 'eager!' macro call and we are already
660		// in eager mode, ignore it, extracting the body. (parenthesis type)
661		@check_expansion[
662			[[][][$($prefix:tt)*][]]
663			$($rest_decoded:tt)*
664		]
665		lazy!($($body:tt)*) $($rest:tt)*
666	)=>{
667		eager_internal!{
668			@check_expansion[
669				[[@lazy][$($rest)*][$($prefix)*][]]
670				$($rest_decoded)*
671			]
672			$($body)*
673		}
674	};
675	(	// If the next token is an 'eager!' macro call and we are already
676		// in eager mode, ignore it, extracting the body. (bracket type)
677		@check_expansion[
678			[[][][$($prefix:tt)*][]]
679			$($rest_decoded:tt)*
680		]
681		lazy![$($body:tt)*] $($rest:tt)*
682	)=>{
683		eager_internal!{
684			@check_expansion[
685				[[@lazy][$($rest)*][$($prefix)*][]]
686				$($rest_decoded)*
687			]
688			$($body)*
689		}
690	};
691// end eager/lazy mode switches
692	(	// If the next token isn't any of the above
693		// it is safe to add it to the prefix
694		@check_expansion[
695			[$lazy:tt $modefix:tt [$($prefix:tt)*][]]
696			$($rest_decoded:tt)*
697		]
698		$next:tt $($rest:tt)*
699	)=>{
700		eager_internal!{
701			@check_expansion[
702				[$lazy $modefix[$next $($prefix)*][]]
703				$($rest_decoded)*
704			]
705			$($rest)*
706		}
707	};
708// Done decoding input
709// Expanding macros in eager mode
710	(	// When there is no more input, the last input was a macro call,
711		// and we are in eager mode, call the macro eagerly
712		// (brace type)
713		@check_expansion[
714			[[]$modefix:tt[! $macro_name:tt $($prefix:tt)*][$($postfix:tt)*]{$($body:tt)*}]
715			$($rest_decoded:tt)*
716		]
717	)=>{
718		$macro_name!{
719			@eager[
720				[[]$modefix[$($prefix)*][$($postfix)*]]
721				$($rest_decoded)*
722			]
723			$($body)*
724		}
725	};
726	(	// When there is no more input and the last input was a macro call,
727		// and we are in eager mode, call the macro eagerly
728		// (parenthesis type)
729		@check_expansion[
730			[[]$modefix:tt[! $macro_name:tt $($prefix:tt)*][$($postfix:tt)*]($($body:tt)*)]
731			$($rest_decoded:tt)*
732		]
733	)=>{
734		$macro_name!{
735			@eager[
736				[[]$modefix[$($prefix)*][$($postfix)*]]
737				$($rest_decoded)*
738			]
739			$($body)*
740		}
741	};
742	(	// When there is no more input and the last input was a macro call,
743		// and we are in eager mode, call the macro eagerly
744		// (bracket type)
745		@check_expansion[
746			[[]$modefix:tt[! $macro_name:tt $($prefix:tt)*][$($postfix:tt)*][$($body:tt)*]]
747			$($rest_decoded:tt)*
748		]
749	)=>{
750		$macro_name!{
751			@eager[
752				[[]$modefix[$($prefix)*][$($postfix)*]]
753				$($rest_decoded)*
754			]
755			$($body)*
756		}
757	};
758// Promote modefix to input
759	(	// When there is no more input, but there is some postfix,
760		// if the current mode is eager, redecode the postfix in lazy mode
761		@check_expansion[
762			[[][$($modefix:tt)+] $prefix:tt []]
763			$($rest:tt)*
764		]
765	)=>{
766		eager_internal!{
767			@check_expansion[
768				[[@lazy][] $prefix []]
769				$($rest)*
770			]
771			$($modefix)+
772		}
773	};
774	(	// When there is no more input, but there is some postfix,
775		// if the current mode is lazy, redecode the postfix in eager mode
776		@check_expansion[
777			[[@lazy][$($modefix:tt)+] $prefix:tt []]
778			$($rest:tt)*
779		]
780	)=>{
781		eager_internal!{
782			@check_expansion[
783				[[][] $prefix []]
784				$($rest)*
785			]
786			$($modefix)+
787		}
788	};
789// end Promote modefix to input
790// Promote prefix
791	(	// When there is no more input and the last input wasn't a macro call in eager mode
792		// insert it into the previous block (brace type)
793		@check_expansion[
794			[$lazy_0:tt $modefix_0:tt [$last:tt $($last_rest:tt)*] []]
795			[$lazy:tt $modefix:tt $prefix:tt $postfix:tt {$($body:tt)*}]
796			$($rest:tt)*
797		]
798	)=>{
799		eager_internal!{
800			@check_expansion[
801				[$lazy_0 $modefix_0 [$($last_rest)*] []]
802				[$lazy $modefix $prefix $postfix {$last $($body)*}]
803				$($rest)*
804			]
805		}
806	};
807	(	// When there is no more input and the last input wasn't a macro call in eager mode
808		// insert it into the previous block (parenthesis type)
809		@check_expansion[
810			[$lazy_0:tt $modefix_0:tt[$last:tt $($last_rest:tt)*] []]
811			[$lazy:tt $modefix:tt $prefix:tt $postfix:tt ($($body:tt)*)]
812			$($rest:tt)*
813		]
814	)=>{
815		eager_internal!{
816			@check_expansion[
817				[$lazy_0 $modefix_0 [$($last_rest)*] []]
818				[$lazy $modefix $prefix $postfix ($last $($body)*)]
819				$($rest)*
820			]
821		}
822	};
823	(	// When there is no more input and the last input wasn't a macro call in eager mode
824		// insert it into the previous block (bracket type)
825		@check_expansion[
826			[$lazy_0:tt $modefix_0:tt[$last:tt $($last_rest:tt)*] []]
827			[$lazy:tt $modefix:tt $prefix:tt $postfix:tt [$($body:tt)*]]
828			$($rest:tt)*
829		]
830	)=>{
831		eager_internal!{
832			@check_expansion[
833				[$lazy_0 $modefix_0 [$($last_rest)*] []]
834				[$lazy $modefix $prefix $postfix [$last $($body)*]]
835				$($rest)*
836			]
837		}
838	};
839	(	// When there is no more input, prefix or postfix,
840		// but there is a previous block, remove the input catcher
841		@check_expansion[
842			[$lazy_0:tt[][][]]
843			$([$lazy:tt $modefix:tt $prefix:tt $postfix:tt $body:tt])+
844			
845		]
846	)=>{
847		eager_internal!{
848			@check_expansion[
849				$([$lazy $modefix $prefix $postfix $body])+
850			]
851		}
852	};
853// end promote prefix
854// Promote block to prefix
855	(	// When there is no more input but a block,
856		// the block must have already been checked,
857		// therefore, begin promoting to prefix (brace type)
858		@check_expansion[
859			[$lazy:tt $modefix:tt [$($prefix:tt)*][$($postfix:tt)*]{$($body:tt)*}]
860			$($rest:tt)*
861		]
862	)=>{
863		eager_internal!{
864			@check_expansion[
865				[$lazy $modefix [{$($body)*} $($prefix)*][]]
866				$($rest)*
867			]
868			$($postfix)*
869		}
870	};
871	(	// When there is no more input and but a block
872		// the block must have already been checked,
873		// so output everything (parenthesis type)
874		@check_expansion[
875			[$lazy:tt $modefix:tt [$($prefix:tt)*][$($postfix:tt)*]($($body:tt)*)]
876			$($rest:tt)*
877		]
878	)=>{
879		eager_internal!{
880			@check_expansion[
881				[$lazy $modefix [($($body)*) $($prefix)*][]]
882				$($rest)*
883			]
884			$($postfix)*
885		}
886	};
887	(	// When there is no more input and but a block
888		// the block must have already been checked,
889		// so output everything (bracket type)
890		@check_expansion[
891			[$lazy:tt $modefix:tt [$($prefix:tt)*][$($postfix:tt)*][$($body:tt)*]]
892			$($rest:tt)*
893		]
894	)=>{
895		eager_internal!{
896			@check_expansion[
897				[$lazy $modefix [[$($body)*] $($prefix)*][]]
898				$($rest)*
899			]
900			$($postfix)*
901		}
902	};
903// End Promote block to prefix
904// Finished
905	(	// When there is no more input and no block
906		// output the result, reversing it to ensure correct order
907		@check_expansion[
908			[$lazy:tt [][$($result:tt)*][]]
909		]
910	)=>{
911		eager_internal!{
912			@reverse_tt[
913				[$($result)*]
914				[]
915			]
916		}
917	};
918	
919// To finish, reverse-output the result
920	(
921		// While there is more to reverse
922		@reverse_tt[
923			[$to_reverse_next:tt $($to_reverse_rest:tt)+]
924			[$($reversed:tt)*]
925		]
926	) => {
927		eager_internal!{
928			@reverse_tt[
929				[$($to_reverse_rest)+]
930				[$to_reverse_next $($reversed)*]
931			]
932		}
933	};
934	(
935		// Done reversing
936		@reverse_tt[
937			[$to_reverse_last:tt]
938			[$($reversed:tt)*]
939		]
940	) => {
941		$to_reverse_last $($reversed)*
942	};
943}
