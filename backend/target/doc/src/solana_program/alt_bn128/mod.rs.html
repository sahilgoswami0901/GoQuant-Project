44// AltBn128Error must be removed once the
45// simplify_alt_bn128_syscall_error_codes feature gets activated
46#[derive(Debug, Error, Clone, PartialEq, Eq)]
47pub enum AltBn128Error {
48    #[error("The input data is invalid")]
49    InvalidInputData,
50    #[error("Invalid group data")]
51    GroupError,
52    #[error("Slice data is going out of input data bounds")]
53    SliceOutOfBounds,
54    #[error("Unexpected error")]
55    UnexpectedError,
56    #[error("Failed to convert a byte slice into a vector {0:?}")]
57    TryIntoVecError(Vec<u8>),
58    #[error("Failed to convert projective to affine g1")]
59    ProjectiveToG1Failed,
60}
61
62impl From<u64> for AltBn128Error {
63    fn from(v: u64) -> AltBn128Error {
64        match v {
65            1 => AltBn128Error::InvalidInputData,
66            2 => AltBn128Error::GroupError,
67            3 => AltBn128Error::SliceOutOfBounds,
68            4 => AltBn128Error::TryIntoVecError(Vec::new()),
69            5 => AltBn128Error::ProjectiveToG1Failed,
70            _ => AltBn128Error::UnexpectedError,
71        }
72    }
73}
74
75impl From<AltBn128Error> for u64 {
76    fn from(v: AltBn128Error) -> u64 {
77        // note: should never return 0, as it risks to be confused with syscall success
78        match v {
79            AltBn128Error::InvalidInputData => 1,
80            AltBn128Error::GroupError => 2,
81            AltBn128Error::SliceOutOfBounds => 3,
82            AltBn128Error::TryIntoVecError(_) => 4,
83            AltBn128Error::ProjectiveToG1Failed => 5,
84            AltBn128Error::UnexpectedError => 6,
85        }
86    }
87}

110    impl TryFrom<PodG1> for G1 {
111        type Error = AltBn128Error;
112
113        fn try_from(bytes: PodG1) -> Result<Self, Self::Error> {
114            if bytes.0 == [0u8; 64] {
115                return Ok(G1::zero());
116            }
117            let g1 = Self::deserialize_with_mode(
118                &*[&bytes.0[..], &[0u8][..]].concat(),
119                Compress::No,
120                Validate::Yes,
121            );
122
123            match g1 {
124                Ok(g1) => {
125                    if !g1.is_on_curve() {
126                        Err(AltBn128Error::GroupError)
127                    } else {
128                        Ok(g1)
129                    }
130                }
131                Err(_) => Err(AltBn128Error::InvalidInputData),
132            }
133        }
134    }

162    pub fn alt_bn128_addition(input: &[u8]) -> Result<Vec<u8>, AltBn128Error> {
163        if input.len() > ALT_BN128_ADDITION_INPUT_LEN {
164            return Err(AltBn128Error::InvalidInputData);
165        }
166
167        let mut input = input.to_vec();
168        input.resize(ALT_BN128_ADDITION_INPUT_LEN, 0);
169
170        let p: G1 = PodG1(
171            convert_edianness_64(&input[..64])
172                .try_into()
173                .map_err(AltBn128Error::TryIntoVecError)?,
174        )
175        .try_into()?;
176        let q: G1 = PodG1(
177            convert_edianness_64(&input[64..ALT_BN128_ADDITION_INPUT_LEN])
178                .try_into()
179                .map_err(AltBn128Error::TryIntoVecError)?,
180        )
181        .try_into()?;
182
183        #[allow(clippy::arithmetic_side_effects)]
184        let result_point = p + q;
185
186        let mut result_point_data = [0u8; ALT_BN128_ADDITION_OUTPUT_LEN];
187        let result_point_affine: G1 = result_point.into();
188        result_point_affine
189            .x
190            .serialize_with_mode(&mut result_point_data[..32], Compress::No)
191            .map_err(|_| AltBn128Error::InvalidInputData)?;
192        result_point_affine
193            .y
194            .serialize_with_mode(&mut result_point_data[32..], Compress::No)
195            .map_err(|_| AltBn128Error::InvalidInputData)?;
196
197        Ok(convert_edianness_64(&result_point_data[..]).to_vec())
198    }

200    pub fn alt_bn128_multiplication(input: &[u8]) -> Result<Vec<u8>, AltBn128Error> {
201        if input.len() > ALT_BN128_MULTIPLICATION_INPUT_LEN {
202            return Err(AltBn128Error::InvalidInputData);
203        }
204
205        let mut input = input.to_vec();
206        input.resize(ALT_BN128_MULTIPLICATION_INPUT_LEN, 0);
207
208        let p: G1 = PodG1(
209            convert_edianness_64(&input[..64])
210                .try_into()
211                .map_err(AltBn128Error::TryIntoVecError)?,
212        )
213        .try_into()?;
214        let fr = BigInteger256::deserialize_uncompressed_unchecked(
215            &convert_edianness_64(&input[64..96])[..],
216        )
217        .map_err(|_| AltBn128Error::InvalidInputData)?;
218
219        let result_point: G1 = p.mul_bigint(fr).into();
220
221        let mut result_point_data = [0u8; ALT_BN128_MULTIPLICATION_OUTPUT_LEN];
222
223        result_point
224            .x
225            .serialize_with_mode(&mut result_point_data[..32], Compress::No)
226            .map_err(|_| AltBn128Error::InvalidInputData)?;
227        result_point
228            .y
229            .serialize_with_mode(&mut result_point_data[32..], Compress::No)
230            .map_err(|_| AltBn128Error::InvalidInputData)?;
231
232        Ok(
233            convert_edianness_64(&result_point_data[..ALT_BN128_MULTIPLICATION_OUTPUT_LEN])
234                .to_vec(),
235        )
236    }

238    pub fn alt_bn128_pairing(input: &[u8]) -> Result<Vec<u8>, AltBn128Error> {
239        if input
240            .len()
241            .checked_rem(consts::ALT_BN128_PAIRING_ELEMENT_LEN)
242            .is_none()
243        {
244            return Err(AltBn128Error::InvalidInputData);
245        }
246
247        let ele_len = input.len().saturating_div(ALT_BN128_PAIRING_ELEMENT_LEN);
248
249        let mut vec_pairs: Vec<(G1, G2)> = Vec::new();
250        for i in 0..ele_len {
251            vec_pairs.push((
252                PodG1(
253                    convert_edianness_64(
254                        &input[i.saturating_mul(ALT_BN128_PAIRING_ELEMENT_LEN)
255                            ..i.saturating_mul(ALT_BN128_PAIRING_ELEMENT_LEN)
256                                .saturating_add(ALT_BN128_POINT_SIZE)],
257                    )
258                    .try_into()
259                    .map_err(AltBn128Error::TryIntoVecError)?,
260                )
261                .try_into()?,
262                PodG2(
263                    convert_edianness_128(
264                        &input[i
265                            .saturating_mul(ALT_BN128_PAIRING_ELEMENT_LEN)
266                            .saturating_add(ALT_BN128_POINT_SIZE)
267                            ..i.saturating_mul(ALT_BN128_PAIRING_ELEMENT_LEN)
268                                .saturating_add(ALT_BN128_PAIRING_ELEMENT_LEN)],
269                    )
270                    .try_into()
271                    .map_err(AltBn128Error::TryIntoVecError)?,
272                )
273                .try_into()?,
274            ));
275        }
276
277        let mut result = BigInteger256::from(0u64);
278        let res = <Bn<Config> as Pairing>::multi_pairing(
279            vec_pairs.iter().map(|pair| pair.0),
280            vec_pairs.iter().map(|pair| pair.1),
281        );
282
283        if res.0 == ark_bn254::Fq12::one() {
284            result = BigInteger256::from(1u64);
285        }
286
287        let output = result.to_bytes_be();
288        Ok(output)
289    }

291    fn convert_edianness_64(bytes: &[u8]) -> Vec<u8> {
292        bytes
293            .chunks(32)
294            .flat_map(|b| b.iter().copied().rev().collect::<Vec<u8>>())
295            .collect::<Vec<u8>>()
296    }

299        bytes
300            .chunks(64)
301            .flat_map(|b| b.iter().copied().rev().collect::<Vec<u8>>())
302            .collect::<Vec<u8>>()
303    }
