10const MAX_DATA_SIZE: usize = 128;
11const MAX_DATA_BASE58_SIZE: usize = 175;
12const MAX_DATA_BASE64_SIZE: usize = 172;

17    DataSize(u64),
18    Memcmp(Memcmp),
19    TokenAccountState,
20}

22impl RpcFilterType {
23    pub fn verify(&self) -> Result<(), RpcFilterError> {
24        match self {
25            RpcFilterType::DataSize(_) => Ok(()),
26            RpcFilterType::Memcmp(compare) => {
27                let encoding = compare.encoding.as_ref().unwrap_or(&MemcmpEncoding::Binary);
28                match encoding {
29                    MemcmpEncoding::Binary => {
30                        use MemcmpEncodedBytes::*;
31                        match &compare.bytes {
32                            // DEPRECATED
33                            Binary(bytes) => {
34                                if bytes.len() > MAX_DATA_BASE58_SIZE {
35                                    return Err(RpcFilterError::Base58DataTooLarge);
36                                }
37                                let bytes = bs58::decode(&bytes)
38                                    .into_vec()
39                                    .map_err(RpcFilterError::DecodeError)?;
40                                if bytes.len() > MAX_DATA_SIZE {
41                                    Err(RpcFilterError::Base58DataTooLarge)
42                                } else {
43                                    Ok(())
44                                }
45                            }
46                            Base58(bytes) => {
47                                if bytes.len() > MAX_DATA_BASE58_SIZE {
48                                    return Err(RpcFilterError::DataTooLarge);
49                                }
50                                let bytes = bs58::decode(&bytes).into_vec()?;
51                                if bytes.len() > MAX_DATA_SIZE {
52                                    Err(RpcFilterError::DataTooLarge)
53                                } else {
54                                    Ok(())
55                                }
56                            }
57                            Base64(bytes) => {
58                                if bytes.len() > MAX_DATA_BASE64_SIZE {
59                                    return Err(RpcFilterError::DataTooLarge);
60                                }
61                                let bytes = base64::decode(bytes)?;
62                                if bytes.len() > MAX_DATA_SIZE {
63                                    Err(RpcFilterError::DataTooLarge)
64                                } else {
65                                    Ok(())
66                                }
67                            }
68                            Bytes(bytes) => {
69                                if bytes.len() > MAX_DATA_SIZE {
70                                    return Err(RpcFilterError::DataTooLarge);
71                                }
72                                Ok(())
73                            }
74                        }
75                    }
76                }
77            }
78            RpcFilterType::TokenAccountState => Ok(()),
79        }
80    }
81
82    pub fn allows(&self, account: &AccountSharedData) -> bool {
83        match self {
84            RpcFilterType::DataSize(size) => account.data().len() as u64 == *size,
85            RpcFilterType::Memcmp(compare) => compare.bytes_match(account.data()),
86            RpcFilterType::TokenAccountState => Account::valid_account_data(account.data()),
87        }
88    }
89}

134pub struct Memcmp {

136    #[deprecated(
137        since = "1.15.0",

139    )]
140    pub offset: usize,

142    #[deprecated(
143        since = "1.15.0",
144        note = "Field will be made private in future. Please use a constructor method instead."
145    )]

148    #[deprecated(
149        since = "1.11.2",

151            Field will be made private in future. Please use a constructor method instead."

154}

156impl Memcmp {
157    pub fn new(offset: usize, encoded_bytes: MemcmpEncodedBytes) -> Self {
158        Self {
159            offset,
160            bytes: encoded_bytes,
161            encoding: None,
162        }
163    }
164
165    pub fn new_raw_bytes(offset: usize, bytes: Vec<u8>) -> Self {
166        Self {
167            offset,
168            bytes: MemcmpEncodedBytes::Bytes(bytes),
169            encoding: None,
170        }
171    }
172
173    pub fn new_base58_encoded(offset: usize, bytes: &[u8]) -> Self {
174        Self {
175            offset,
176            bytes: MemcmpEncodedBytes::Base58(bs58::encode(bytes).into_string()),
177            encoding: None,
178        }
179    }
180
181    pub fn bytes(&self) -> Option<Cow<Vec<u8>>> {
182        use MemcmpEncodedBytes::*;
183        match &self.bytes {
184            Binary(bytes) | Base58(bytes) => bs58::decode(bytes).into_vec().ok().map(Cow::Owned),
185            Base64(bytes) => base64::decode(bytes).ok().map(Cow::Owned),
186            Bytes(bytes) => Some(Cow::Borrowed(bytes)),
187        }
188    }
189
190    pub fn convert_to_raw_bytes(&mut self) -> Result<(), RpcFilterError> {
191        use MemcmpEncodedBytes::*;
192        match &self.bytes {
193            Binary(bytes) | Base58(bytes) => {
194                let bytes = bs58::decode(bytes).into_vec()?;
195                self.bytes = Bytes(bytes);
196                Ok(())
197            }
198            Base64(bytes) => {
199                let bytes = base64::decode(bytes)?;
200                self.bytes = Bytes(bytes);
201                Ok(())
202            }
203            _ => Ok(()),
204        }
205    }
206
207    pub fn bytes_match(&self, data: &[u8]) -> bool {
208        match self.bytes() {
209            Some(bytes) => {
210                if self.offset > data.len() {
211                    return false;
212                }
213                if data[self.offset..].len() < bytes.len() {
214                    return false;
215                }
216                data[self.offset..self.offset + bytes.len()] == bytes[..]
217            }
218            None => false,
219        }
220    }
221}

242// Internal struct to enable Memcmp filters with explicit Base58 and Base64 encoding. The From
243// implementations emulate `#[serde(tag = "encoding", content = "bytes")]` for
244// `MemcmpEncodedBytes`. On the next major version, all these internal elements should be removed
245// and replaced with adjacent tagging of `MemcmpEncodedBytes`.
246#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
247struct RpcMemcmp {
248    offset: usize,
249    bytes: DataType,
250    encoding: Option<RpcMemcmpEncoding>,
251}
252
253impl From<Memcmp> for RpcMemcmp {
254    fn from(memcmp: Memcmp) -> RpcMemcmp {
255        let (bytes, encoding) = match memcmp.bytes {
256            MemcmpEncodedBytes::Binary(string) => {
257                (DataType::Encoded(string), Some(RpcMemcmpEncoding::Binary))
258            }
259            MemcmpEncodedBytes::Base58(string) => {
260                (DataType::Encoded(string), Some(RpcMemcmpEncoding::Base58))
261            }
262            MemcmpEncodedBytes::Base64(string) => {
263                (DataType::Encoded(string), Some(RpcMemcmpEncoding::Base64))
264            }
265            MemcmpEncodedBytes::Bytes(vector) => (DataType::Raw(vector), None),
266        };
267        RpcMemcmp {
268            offset: memcmp.offset,
269            bytes,
270            encoding,
271        }
272    }
273}
274
275impl From<RpcMemcmp> for Memcmp {
276    fn from(memcmp: RpcMemcmp) -> Memcmp {
277        let encoding = memcmp.encoding.unwrap_or(RpcMemcmpEncoding::Binary);
278        let bytes = match (encoding, memcmp.bytes) {
279            (RpcMemcmpEncoding::Binary, DataType::Encoded(string))
280            | (RpcMemcmpEncoding::Base58, DataType::Encoded(string)) => {
281                MemcmpEncodedBytes::Base58(string)
282            }
283            (RpcMemcmpEncoding::Base64, DataType::Encoded(string)) => {
284                MemcmpEncodedBytes::Base64(string)
285            }
286            (_, DataType::Raw(vector)) => MemcmpEncodedBytes::Bytes(vector),
287        };
288        Memcmp {
289            offset: memcmp.offset,
290            bytes,
291            encoding: None,
292        }
293    }
294}

296pub fn maybe_map_filters(
297    node_version: Option<semver::Version>,
298    filters: &mut [RpcFilterType],
299) -> Result<(), String> {
300    let version_reqs = VersionReq::from_strs(&["<1.11.2", "~1.13"])?;
301    let needs_mapping = node_version
302        .map(|version| version_reqs.matches_any(&version))
303        .unwrap_or(true);
304    if needs_mapping {
305        for filter in filters.iter_mut() {
306            if let RpcFilterType::Memcmp(memcmp) = filter {
307                match &memcmp.bytes {
308                    MemcmpEncodedBytes::Base58(string) => {
309                        memcmp.bytes = MemcmpEncodedBytes::Binary(string.clone());
310                    }
311                    MemcmpEncodedBytes::Base64(_) => {
312                        return Err("RPC node on old version does not support base64 \
313                            encoding for memcmp filters"
314                            .to_string());
315                    }
316                    _ => {}
317                }
318            }
319        }
320    }
321    Ok(())
322}
