1//! A simple BIP32 implementation for ed25519 public keys. Although there exists [another very good
2//! library that does this](https://docs.rs/ed25519-bip32), this library preserves 32 byte secret
3//! keys and doesn't allow for extended public keys or "normal" child indexes, so that it can be as
4//! close to the BIP32 specifications as possible, allowing for compatibility with libraries like
5//! `trezor-crypto`
6
7#![cfg_attr(not(feature = "std"), no_std)]
8
9pub extern crate derivation_path;
10pub extern crate ed25519_dalek;
11
12pub use derivation_path::{ChildIndex, DerivationPath};
13pub use ed25519_dalek::{PublicKey, SecretKey};

15use core::fmt;
16use hmac::{Hmac, Mac};
17use sha2::Sha512;

19const ED25519_BIP32_NAME: &str = "ed25519 seed";
20
21/// Errors thrown while deriving secret keys
22#[derive(Debug)]
23pub enum Error {
24    Ed25519,
25    ExpectedHardenedIndex(ChildIndex),
26}
27
28impl fmt::Display for Error {
29    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
30        match self {
31            Self::Ed25519 => f.write_str("ed25519 error"),
32            Self::ExpectedHardenedIndex(index) => {
33                f.write_fmt(format_args!("expected hardened child index: {}", index))
34            }
35        }
36    }
37}
38
39#[cfg(feature = "std")]
40impl std::error::Error for Error {}

42/// An expanded secret key with chain code and meta data
43#[derive(Debug)]
44pub struct ExtendedSecretKey {
45    /// How many derivations this key is from the root (0 for root)
46    pub depth: u8,
47    /// Child index of the key used to derive from parent (`Normal(0)` for root)
48    pub child_index: ChildIndex,
49    /// Secret Key
50    pub secret_key: SecretKey,
51    /// Chain code
52    pub chain_code: [u8; 32],
53}
54
55type HmacSha512 = Hmac<Sha512>;
56
57/// A convenience wrapper for a [`core::result::Result`] with an [`Error`]
58pub type Result<T, E = Error> = core::result::Result<T, E>;
59
60impl ExtendedSecretKey {
61    /// Create a new extended secret key from a seed
62    pub fn from_seed(seed: &[u8]) -> Result<Self> {
63        let mut mac = HmacSha512::new_from_slice(ED25519_BIP32_NAME.as_ref()).unwrap();
64        mac.update(seed);
65        let bytes = mac.finalize().into_bytes();
66
67        let secret_key = SecretKey::from_bytes(&bytes[..32])?;
68        let mut chain_code = [0; 32];
69        chain_code.copy_from_slice(&bytes[32..]);
70
71        Ok(Self {
72            depth: 0,
73            child_index: ChildIndex::Normal(0),
74            secret_key,
75            chain_code,
76        })
77    }
78
79    /// Derive an extended secret key fom the current using a derivation path
80    pub fn derive<P: AsRef<[ChildIndex]>>(&self, path: &P) -> Result<Self> {
81        let mut path = path.as_ref().iter();
82        let mut next = match path.next() {
83            Some(index) => self.derive_child(*index)?,
84            None => self.clone(),
85        };
86        for index in path {
87            next = next.derive_child(*index)?;
88        }
89        Ok(next)
90    }
91
92    /// Derive a child extended secret key with an index
93    pub fn derive_child(&self, index: ChildIndex) -> Result<Self> {
94        if index.is_normal() {
95            return Err(Error::ExpectedHardenedIndex(index));
96        }
97
98        let mut mac = HmacSha512::new_from_slice(&self.chain_code).unwrap();
99        mac.update(&[0u8]);
100        mac.update(self.secret_key.to_bytes().as_ref());
101        mac.update(index.to_bits().to_be_bytes().as_ref());
102        let bytes = mac.finalize().into_bytes();
103
104        let secret_key = SecretKey::from_bytes(&bytes[..32])?;
105        let mut chain_code = [0; 32];
106        chain_code.copy_from_slice(&bytes[32..]);
107
108        Ok(Self {
109            depth: self.depth + 1,
110            child_index: index,
111            secret_key,
112            chain_code,
113        })
114    }
115
116    /// Get the associated public key
117    #[inline]
118    pub fn public_key(&self) -> PublicKey {
119        PublicKey::from(&self.secret_key)
120    }
121
122    #[inline]
123    fn clone(&self) -> Self {
124        Self {
125            depth: self.depth,
126            child_index: self.child_index,
127            secret_key: SecretKey::from_bytes(&self.secret_key.to_bytes()).unwrap(),
128            chain_code: self.chain_code,
129        }
130    }
131}
132
133impl From<ed25519_dalek::SignatureError> for Error {
134    fn from(_: ed25519_dalek::SignatureError) -> Self {
135        Self::Ed25519
136    }
137}

155    fn derivation_path() {
156        let vector1_path: DerivationPath = "m/0'/1'/2'/2'/1000000000'".parse().unwrap();

198    #[test]
199    fn normal_errs() {
200        let node = root("000102030405060708090a0b0c0d0e0f");
201
202        let res = node.derive_child(ChildIndex::Normal(0));
203        assert!(matches!(
204            res,
205            Err(Error::ExpectedHardenedIndex(ChildIndex::Normal(0)))
206        ));
207
208        let res = node.derive_child(ChildIndex::Normal(100000));
209        assert!(matches!(
210            res,
211            Err(Error::ExpectedHardenedIndex(ChildIndex::Normal(100000)))
212        ));
213
214        let soft_path: DerivationPath = "m/0'/1'/2'/3/4'".parse().unwrap();
215        let res = node.derive(&soft_path);
216        assert!(matches!(
217            res,
218            Err(Error::ExpectedHardenedIndex(ChildIndex::Normal(3)))
219        ));
220    }
